33 share_type asset_bitasset_data_object::max_force_settlement_volume(share_type current_supply) const
34 {
35  if( options.maximum_force_settlement_volume == 0 )
36  return 0;
37  if( options.maximum_force_settlement_volume == GRAPHENE_100_PERCENT )
38  return current_supply + force_settled_volume;
39 
40  fc::uint128 volume = current_supply.value + force_settled_volume.value;
41  volume *= options.maximum_force_settlement_volume;
42  volume /= GRAPHENE_100_PERCENT;
43  return volume.to_uint64();

46 /******
47  * @brief calculate the median feed
48  *
49  * This calculates the median feed. It sets the current_feed_publication_time
50  * and current_feed member variables
51  *
52  * @param current_time the time to use in the calculations
53  */
54 void graphene::chain::asset_bitasset_data_object::update_median_feeds(time_point_sec current_time)
55 {
56  current_feed_publication_time = current_time;
57  vector<std::reference_wrapper<const price_feed>> current_feeds;
58  // find feeds that were alive at current_time
59  for( const pair<account_id_type, pair<time_point_sec,price_feed>>& f : feeds )
60  {
61  if( (current_time - f.second.first).to_seconds() < options.feed_lifetime_sec &&
62  f.second.first != time_point_sec() )
63  {
64  current_feeds.emplace_back(f.second.second);
65  current_feed_publication_time = std::min(current_feed_publication_time, f.second.first);
66  }
67  }
68 
69  // If there are no valid feeds, or the number available is less than the minimum to calculate a median...
70  if( current_feeds.size() < options.minimum_feeds )
71  {
72  //... don't calculate a median, and set a null feed
73  feed_cer_updated = false; // new median cer is null, won't update asset_object anyway, set to false for better performance
74  current_feed_publication_time = current_time;
75  current_feed = price_feed();
76  return;
77  }
78  if( current_feeds.size() == 1 )
79  {
80  if( current_feed.core_exchange_rate != current_feeds.front().get().core_exchange_rate )
81  feed_cer_updated = true;
82  current_feed = std::move(current_feeds.front());
83  return;
84  }
85 
86  // *** Begin Median Calculations ***
87  price_feed median_feed;
88  const auto median_itr = current_feeds.begin() + current_feeds.size() / 2;
89 #define CALCULATE_MEDIAN_VALUE(r, data, field_name) \
90  std::nth_element( current_feeds.begin(), median_itr, current_feeds.end(), \
91  [](const price_feed& a, const price_feed& b) { \
92  return a.field_name < b.field_name; \
93  }); \
94  median_feed.field_name = median_itr->get().field_name;
95 
96  BOOST_PP_SEQ_FOR_EACH( CALCULATE_MEDIAN_VALUE, ~, GRAPHENE_PRICE_FEED_FIELDS )
97 #undef CALCULATE_MEDIAN_VALUE
98  // *** End Median Calculations ***
99 
100  if( current_feed.core_exchange_rate != median_feed.core_exchange_rate )
101  feed_cer_updated = true;
102  current_feed = median_feed;
103 }

107 asset asset_object::amount_from_string(string amount_string) const
108 { try {
109  bool negative_found = false;
110  bool decimal_found = false;
111  for( const char c : amount_string )
112  {
113  if( isdigit( c ) )
114  continue;
115 
116  if( c == '-' && !negative_found )
117  {
118  negative_found = true;
119  continue;
120  }
121 
122  if( c == '.' && !decimal_found )
123  {
124  decimal_found = true;
125  continue;
126  }
127 
128  FC_THROW( (amount_string) );
129  }
130 
131  share_type satoshis = 0;
132 
133  share_type scaled_precision = asset::scaled_precision( precision );
134 
135  const auto decimal_pos = amount_string.find( '.' );
136  const string lhs = amount_string.substr( negative_found, decimal_pos );
137  if( !lhs.empty() )
138  satoshis += fc::safe<int64_t>(std::stoll(lhs)) *= scaled_precision;
139 
140  if( decimal_found )
141  {
142  const size_t max_rhs_size = std::to_string( scaled_precision.value ).substr( 1 ).size();
143 
144  string rhs = amount_string.substr( decimal_pos + 1 );
145  FC_ASSERT( rhs.size() <= max_rhs_size );
146 
147  while( rhs.size() < max_rhs_size )
148  rhs += '0';
149 
150  if( !rhs.empty() )
151  satoshis += std::stoll( rhs );
152  }
153 
154  FC_ASSERT( satoshis <= GRAPHENE_MAX_SHARE_SUPPLY );
155 
156  if( negative_found )
157  satoshis *= -1;
158 
159  return amount(satoshis);
160  } FC_CAPTURE_AND_RETHROW( (amount_string) ) }

162 string asset_object::amount_to_string(share_type amount) const
163 {
164  share_type scaled_precision = asset::scaled_precision( precision );
165 
166  string result = fc::to_string(amount.value / scaled_precision.value);
167  auto decimals = abs( amount.value % scaled_precision.value );
168  if( decimals )
169  {
170  if( amount < 0 && result == "0" )
171  result = "-0";
172  result += "." + fc::to_string(scaled_precision.value + decimals).erase(0,1);
173  }
174  return result;
