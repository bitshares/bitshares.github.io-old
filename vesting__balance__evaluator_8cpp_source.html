81 object_id_type vesting_balance_create_evaluator::do_apply( const vesting_balance_create_operation& op )
82 { try {
83  database& d = db();
84  const time_point_sec now = d.head_block_time();
85 
86  FC_ASSERT( d.get_balance( op.creator, op.amount.asset_id ) >= op.amount );
87  d.adjust_balance( op.creator, -op.amount );
88 
89  const vesting_balance_object& vbo = d.create< vesting_balance_object >( [&]( vesting_balance_object& obj )
90  {
91  //WARNING: The logic to create a vesting balance object is replicated in vesting_balance_worker_type::initializer::init.
92  // If making changes to this logic, check if those changes should also be made there as well.
93  obj.owner = op.owner;
94  obj.balance = op.amount;
95  op.policy.visit( init_policy_visitor( obj.policy, op.amount.amount, now ) );
96  } );
97 
98 
99  return vbo.id;
100 } FC_CAPTURE_AND_RETHROW( (op) ) }

117 void_result vesting_balance_withdraw_evaluator::do_apply( const vesting_balance_withdraw_operation& op )
118 { try {
119  database& d = db();
120  const time_point_sec now = d.head_block_time();
121 
122  const vesting_balance_object& vbo = op.vesting_balance( d );
123 
124  // Allow zero balance objects to stick around, (1) to comply
125  // with the chain's "objects live forever" design principle, (2)
126  // if it's cashback or worker, it'll be filled up again.
127 
128  d.modify( vbo, [&]( vesting_balance_object& vbo )
129  {
130  vbo.withdraw( now, op.amount );
131  } );
132 
133  d.adjust_balance( op.owner, op.amount );
134 
135  // TODO: Check asset authorizations and withdrawals
136  return void_result();
137 } FC_CAPTURE_AND_RETHROW( (op) ) }
