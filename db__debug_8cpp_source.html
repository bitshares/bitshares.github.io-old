113 void debug_apply_update( database& db, const fc::variant_object& vo )
114 {
115  static const uint8_t
116  db_action_nil = 0,
117  db_action_create = 1,
118  db_action_write = 2,
119  db_action_update = 3,
120  db_action_delete = 4;
121 
122  // "_action" : "create" object must not exist, unspecified fields take defaults
123  // "_action" : "write" object may exist, is replaced entirely, unspecified fields take defaults
124  // "_action" : "update" object must exist, unspecified fields don't change
125  // "_action" : "delete" object must exist, will be deleted
126 
127  // if _action is unspecified:
128  // - delete if object contains only ID field
129  // - otherwise, write
130 
131  object_id_type oid;
132  uint8_t action = db_action_nil;
133  auto it_id = vo.find("id");
134  FC_ASSERT( it_id != vo.end() );
135 
136  from_variant( it_id->value(), oid );
137  action = ( vo.size() == 1 ) ? db_action_delete : db_action_write;
138 
139  from_variant( vo["id"], oid );
140  if( vo.size() == 1 )
141  action = db_action_delete;
142  auto it_action = vo.find("_action" );
143  if( it_action != vo.end() )
144  {
145  const std::string& str_action = it_action->value().get_string();
146  if( str_action == "create" )
147  action = db_action_create;
148  else if( str_action == "write" )
149  action = db_action_write;
150  else if( str_action == "update" )
151  action = db_action_update;
152  else if( str_action == "delete" )
153  action = db_action_delete;
154  }
155 
156  auto& idx = db.get_index( oid );
157 
158  switch( action )
159  {
160  case db_action_create:
161  FC_ASSERT( false );
162  break;
163  case db_action_write:
164  db.modify( db.get_object( oid ), [&]( object& obj )
165  {
166  idx.object_default( obj );
167  idx.object_from_variant( vo, obj, GRAPHENE_MAX_NESTED_OBJECTS );
168  } );
169  break;
170  case db_action_update:
171  db.modify( db.get_object( oid ), [&]( object& obj )
172  {
173  idx.object_from_variant( vo, obj, GRAPHENE_MAX_NESTED_OBJECTS );
174  } );
175  break;
176  case db_action_delete:
177  db.remove( db.get_object( oid ) );
178  break;
179  default:
180  FC_ASSERT( false );
181  }
182 }

184 void database::apply_debug_updates()
185 {
186  block_id_type head_id = head_block_id();
187  auto it = _node_property_object.debug_updates.find( head_id );
188  if( it == _node_property_object.debug_updates.end() )
189  return;
190  for( const fc::variant_object& update : it->second )
191  debug_apply_update( *this, update );

194 void database::debug_update( const fc::variant_object& update )
195 {
196  block_id_type head_id = head_block_id();
197  auto it = _node_property_object.debug_updates.find( head_id );
198  if( it == _node_property_object.debug_updates.end() )
199  it = _node_property_object.debug_updates.emplace( head_id, std::vector< fc::variant_object >() ).first;
200  it->second.emplace_back( update );
201 
202  optional<signed_block> head_block = fetch_block_by_id( head_id );
203  FC_ASSERT( head_block.valid() );
204 
205  // What the last block does has been changed by adding to node_property_object, so we have to re-apply it
206  pop_block();
207  push_block( *head_block );
208 }
