39 void_result asset_create_evaluator::do_evaluate( const asset_create_operation& op )
40 { try {
41 
42  database& d = db();
43 
44  const auto& chain_parameters = d.get_global_properties().parameters;
45  FC_ASSERT( op.common_options.whitelist_authorities.size() <= chain_parameters.maximum_asset_whitelist_authorities );
46  FC_ASSERT( op.common_options.blacklist_authorities.size() <= chain_parameters.maximum_asset_whitelist_authorities );
47 
48  // Check that all authorities do exist
49  for( auto id : op.common_options.whitelist_authorities )
50  d.get_object(id);
51  for( auto id : op.common_options.blacklist_authorities )
52  d.get_object(id);
53 
54  auto& asset_indx = d.get_index_type<asset_index>().indices().get<by_symbol>();
55  auto asset_symbol_itr = asset_indx.find( op.symbol );
56  FC_ASSERT( asset_symbol_itr == asset_indx.end() );
57 
58  if( d.head_block_time() > HARDFORK_385_TIME )
59  {
60  auto dotpos = op.symbol.rfind( '.' );
61  if( dotpos != std::string::npos )
62  {
63  auto prefix = op.symbol.substr( 0, dotpos );
64  auto asset_symbol_itr = asset_indx.find( prefix );
65  FC_ASSERT( asset_symbol_itr != asset_indx.end(), "Asset ${s} may only be created by issuer of ${p}, but ${p} has not been registered",
66  ("s",op.symbol)("p",prefix) );
67  FC_ASSERT( asset_symbol_itr->issuer == op.issuer, "Asset ${s} may only be created by issuer of ${p}, ${i}",
68  ("s",op.symbol)("p",prefix)("i", op.issuer(d).name) );
69  }
70 
71  if(d.head_block_time() <= HARDFORK_CORE_620_TIME ) { // TODO: remove this check after hf_620
72  static const std::locale& loc = std::locale::classic();
73  FC_ASSERT(isalpha(op.symbol.back(), loc), "Asset ${s} must end with alpha character before hardfork 620", ("s",op.symbol));
74  }
75  }
76  else
77  {
78  auto dotpos = op.symbol.find( '.' );
79  if( dotpos != std::string::npos )
80  wlog( "Asset ${s} has a name which requires hardfork 385", ("s",op.symbol) );
81  }
82 
83  if( op.bitasset_opts )
84  {
85  const asset_object& backing = op.bitasset_opts->short_backing_asset(d);
86  if( backing.is_market_issued() )
87  {
88  const asset_bitasset_data_object& backing_bitasset_data = backing.bitasset_data(d);
89  const asset_object& backing_backing = backing_bitasset_data.options.short_backing_asset(d);
90  FC_ASSERT( !backing_backing.is_market_issued(),
91  "May not create a bitasset backed by a bitasset backed by a bitasset." );
92  FC_ASSERT( op.issuer != GRAPHENE_COMMITTEE_ACCOUNT || backing_backing.get_id() == asset_id_type(),
93  "May not create a blockchain-controlled market asset which is not backed by CORE.");
94  } else
95  FC_ASSERT( op.issuer != GRAPHENE_COMMITTEE_ACCOUNT || backing.get_id() == asset_id_type(),
96  "May not create a blockchain-controlled market asset which is not backed by CORE.");
97  FC_ASSERT( op.bitasset_opts->feed_lifetime_sec > chain_parameters.block_interval &&
98  op.bitasset_opts->force_settlement_delay_sec > chain_parameters.block_interval );
99  }
100  if( op.is_prediction_market )
101  {
102  FC_ASSERT( op.bitasset_opts );
103  FC_ASSERT( op.precision == op.bitasset_opts->short_backing_asset(d).precision );
104  }
105 
106  return void_result();
107 } FC_CAPTURE_AND_RETHROW( (op) ) }

379 /****************
380  * Loop through assets, looking for ones that are backed by the asset being changed. When found,
381  * perform checks to verify validity
382  *
383  * @param d the database
384  * @param op the bitasset update operation being performed
385  * @param new_backing_asset
386  * @param true if after hf 922/931 (if nothing triggers, this and the logic that depends on it
387  * should be removed).
388  */
389 void check_children_of_bitasset(database& d, const asset_update_bitasset_operation& op,
390  const asset_object& new_backing_asset, bool after_hf_922_931)
391 {
392  // no need to do these checks if the new backing asset is CORE
393  if ( new_backing_asset.get_id() == asset_id_type() )
394  return;
395 
396  // loop through all assets that have this asset as a backing asset
397  const auto& idx = d.get_index_type<graphene::chain::asset_bitasset_data_index>()
398  .indices()
399  .get<by_short_backing_asset>();
400  auto backed_range = idx.equal_range(op.asset_to_update);
401  std::for_each( backed_range.first, backed_range.second,
402  [after_hf_922_931, &new_backing_asset, &d, &op](const asset_bitasset_data_object& bitasset_data)
403  {
404  const auto& child = bitasset_data.asset_id(d);
405  if ( after_hf_922_931 )
406  {
407  FC_ASSERT( child.get_id() != op.new_options.short_backing_asset,
408  "A BitAsset would be invalidated by changing this backing asset ('A' backed by 'B' backed by 'A')." );
409 
410  FC_ASSERT( child.issuer != GRAPHENE_COMMITTEE_ACCOUNT,
411  "A blockchain-controlled market asset would be invalidated by changing this backing asset." );
412 
413  FC_ASSERT( !new_backing_asset.is_market_issued(),
414  "A non-blockchain controlled BitAsset would be invalidated by changing this backing asset.");
415 
416  }
417  else
418  {
419  if( child.get_id() == op.new_options.short_backing_asset )
420  {
421  wlog( "Before hf-922-931, modified an asset to be backed by another, but would cause a continuous "
422  "loop. A cannot be backed by B which is backed by A." );
423  return;
424  }
425 
426  if( child.issuer == GRAPHENE_COMMITTEE_ACCOUNT )
427  {
428  wlog( "before hf-922-931, modified an asset to be backed by a non-CORE, but this asset "
429  "is a backing asset for a committee-issued asset. This occurred at block ${b}",
430  ("b", d.head_block_num()));
431  return;
432  }
433  else
434  {
435  if ( new_backing_asset.is_market_issued() ) // a.k.a. !UIA
436  {
437  wlog( "before hf-922-931, modified an asset to be backed by an MPA, but this asset "
438  "is a backing asset for another MPA, which would cause MPA backed by MPA backed by MPA. "
439  "This occurred at block ${b}",
440  ("b", d.head_block_num()));
441  return;
442  }
443  } // if child.issuer
444  } // if hf 922/931
445  } ); // end of lambda and std::for_each()
446 } // check_children_of_bitasset

448 void_result asset_update_bitasset_evaluator::do_evaluate(const asset_update_bitasset_operation& op)
449 { try {
450  database& d = db();
451 
452  const asset_object& asset_obj = op.asset_to_update(d);
453 
454  FC_ASSERT( asset_obj.is_market_issued(), "Cannot update BitAsset-specific settings on a non-BitAsset." );
455 
456  FC_ASSERT( op.issuer == asset_obj.issuer, "Only asset issuer can update bitasset_data of the asset." );
457 
458  const asset_bitasset_data_object& current_bitasset_data = asset_obj.bitasset_data(d);
459 
460  FC_ASSERT( !current_bitasset_data.has_settlement(), "Cannot update a bitasset after a global settlement has executed" );
461 
462  bool after_hf_core_922_931 = ( d.get_dynamic_global_properties().next_maintenance_time > HARDFORK_CORE_922_931_TIME );
463 
464  // Are we changing the backing asset?
465  if( op.new_options.short_backing_asset != current_bitasset_data.options.short_backing_asset )
466  {
467  FC_ASSERT( asset_obj.dynamic_asset_data_id(d).current_supply == 0,
468  "Cannot update a bitasset if there is already a current supply." );
469 
470  const asset_object& new_backing_asset = op.new_options.short_backing_asset(d); // check if the asset exists
471 
472  if( after_hf_core_922_931 ) // TODO remove this check after hard fork if things in `else` did not occur
473  {
474  FC_ASSERT( op.new_options.short_backing_asset != asset_obj.get_id(),
475  "Cannot update an asset to be backed by itself." );
476 
477  if( current_bitasset_data.is_prediction_market )
478  {
479  FC_ASSERT( asset_obj.precision == new_backing_asset.precision,
480  "The precision of the asset and backing asset must be equal." );
481  }
482 
483  if( asset_obj.issuer == GRAPHENE_COMMITTEE_ACCOUNT )
484  {
485  if( new_backing_asset.is_market_issued() )
486  {
487  FC_ASSERT( new_backing_asset.bitasset_data(d).options.short_backing_asset == asset_id_type(),
488  "May not modify a blockchain-controlled market asset to be backed by an asset which is not "
489  "backed by CORE." );
490 
491  check_children_of_bitasset( d, op, new_backing_asset, after_hf_core_922_931 );
492  }
493  else
494  {
495  FC_ASSERT( new_backing_asset.get_id() == asset_id_type(),
496  "May not modify a blockchain-controlled market asset to be backed by an asset which is not "
497  "market issued asset nor CORE." );
498  }
499  }
500  else
501  {
502  // not a committee issued asset
503 
504  // If we're changing to a backing_asset that is not CORE, we need to look at any
505  // asset ( "CHILD" ) that has this one as a backing asset. If CHILD is committee-owned,
506  // the change is not allowed. If CHILD is user-owned, then this asset's backing
507  // asset must be either CORE or a UIA.
508  if ( new_backing_asset.get_id() != asset_id_type() ) // not backed by CORE
509  {
510  check_children_of_bitasset( d, op, new_backing_asset, after_hf_core_922_931 );
511  }
512 
513  }
514 
515  // Check if the new backing asset is itself backed by something. It must be CORE or a UIA
516  if ( new_backing_asset.is_market_issued() )
517  {
518  asset_id_type backing_backing_asset_id = new_backing_asset.bitasset_data(d).options.short_backing_asset;
519  FC_ASSERT( (backing_backing_asset_id == asset_id_type() || !backing_backing_asset_id(d).is_market_issued()),
520  "A BitAsset cannot be backed by a BitAsset that itself is backed by a BitAsset.");
521  }
522  }
523  else // prior to HF 922 / 931
524  {
525  // code to check if issues occurred before hard fork. TODO cleanup after hard fork
