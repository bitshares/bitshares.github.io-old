55  bool login_api::login(const string& user, const string& password)
56  {
57  optional< api_access_info > acc = _app.get_api_access_info( user );
58  if( !acc.valid() )
59  return false;
60  if( acc->password_hash_b64 != "*" )
61  {
62  std::string password_salt = fc::base64_decode( acc->password_salt_b64 );
63  std::string acc_password_hash = fc::base64_decode( acc->password_hash_b64 );
64 
65  fc::sha256 hash_obj = fc::sha256::hash( password + password_salt );
66  if( hash_obj.data_size() != acc_password_hash.length() )
67  return false;
68  if( memcmp( hash_obj.data(), acc_password_hash.c_str(), hash_obj.data_size() ) != 0 )
69  return false;
70  }
71 
72  for( const std::string& api_name : acc->allowed_apis )
73  enable_api( api_name );
74  return true;
75  }

312  vector<operation_history_object> history_api::get_account_history( account_id_type account,
313  operation_history_id_type stop,
314  unsigned limit,
315  operation_history_id_type start ) const
316  {
317  FC_ASSERT( _app.chain_database() );
318  const auto& db = *_app.chain_database();
319  FC_ASSERT( limit <= 100 );
320  vector<operation_history_object> result;
321  try {
322  const account_transaction_history_object& node = account(db).statistics(db).most_recent_op(db);
323  if(start == operation_history_id_type() || start.instance.value > node.operation_id.instance.value)
324  start = node.operation_id;
325  } catch(...) { return result; }
326 
327  const auto& hist_idx = db.get_index_type<account_transaction_history_index>();
328  const auto& by_op_idx = hist_idx.indices().get<by_op>();
329  auto index_start = by_op_idx.begin();
330  auto itr = by_op_idx.lower_bound(boost::make_tuple(account, start));
331 
332  while(itr != index_start && itr->account == account && itr->operation_id.instance.value > stop.instance.value && result.size() < limit)
333  {
334  if(itr->operation_id.instance.value <= start.instance.value)
335  result.push_back(itr->operation_id(db));
336  --itr;
337  }
338  if(stop.instance.value == 0 && result.size() < limit && itr->account == account) {
339  result.push_back(itr->operation_id(db));
340  }
341 
342  return result;
343  }

345  vector<operation_history_object> history_api::get_account_history_operations( account_id_type account,
346  int operation_id,
347  operation_history_id_type start,
348  operation_history_id_type stop,
349  unsigned limit) const
350  {
351  FC_ASSERT( _app.chain_database() );
352  const auto& db = *_app.chain_database();
353  FC_ASSERT( limit <= 100 );
354  vector<operation_history_object> result;
355  const auto& stats = account(db).statistics(db);
356  if( stats.most_recent_op == account_transaction_history_id_type() ) return result;
357  const account_transaction_history_object* node = &stats.most_recent_op(db);
358  if( start == operation_history_id_type() )
359  start = node->operation_id;
360

369  node = nullptr;
370  else node = &node->next(db);
371  }
372  if( stop.instance.value == 0 && result.size() < limit ) {
373  const account_transaction_history_object head = account_transaction_history_id_type()(db);
374  if( head.account == account && head.operation_id(db).op.which() == operation_id )
