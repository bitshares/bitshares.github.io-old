50  bool login_api::login(const string& user, const string& password)
51  {
52  optional< api_access_info > acc = _app.get_api_access_info( user );
53  if( !acc.valid() )
54  return false;
55  if( acc->password_hash_b64 != "*" )
56  {
57  std::string password_salt = fc::base64_decode( acc->password_salt_b64 );
58  std::string acc_password_hash = fc::base64_decode( acc->password_hash_b64 );
59 
60  fc::sha256 hash_obj = fc::sha256::hash( password + password_salt );
61  if( hash_obj.data_size() != acc_password_hash.length() )
62  return false;
63  if( memcmp( hash_obj.data(), acc_password_hash.c_str(), hash_obj.data_size() ) != 0 )
64  return false;
65  }
66 
67  for( const std::string& api_name : acc->allowed_apis )
68  enable_api( api_name );
69  return true;
70  }

98  void network_broadcast_api::on_applied_block( const signed_block& b )
99  {
100  if( _callbacks.size() )
101  {
103  auto capture_this = shared_from_this();
104  for( uint32_t trx_num = 0; trx_num < b.transactions.size(); ++trx_num )
105  {
106  const auto& trx = b.transactions[trx_num];
107  auto id = trx.id();
108  auto itr = _callbacks.find(id);
109  if( itr != _callbacks.end() )
110  {
111  auto block_num = b.block_num();
112  auto& callback = _callbacks.find(id)->second;
113  fc::async( [capture_this,this,id,block_num,trx_num,trx,callback](){ callback( fc::variant(transaction_confirmation{ id, block_num, trx_num, trx}) ); } );
114  }
115  }

313  vector<order_history_object> history_api::get_fill_order_history( asset_id_type a, asset_id_type b, uint32_t limit )const
314  {
315  FC_ASSERT(_app.chain_database());
316  const auto& db = *_app.chain_database();
317  if( a > b ) std::swap(a,b);
318  const auto& history_idx = db.get_index_type<graphene::market_history::history_index>().indices().get<by_key>();
319  history_key hkey;
320  hkey.base = a;
321  hkey.quote = b;
322  hkey.sequence = std::numeric_limits<int64_t>::min();
323 
324  uint32_t count = 0;
325  auto itr = history_idx.lower_bound( hkey );
326  vector<order_history_object> result;
327  while( itr != history_idx.end() )
328  {
329  if( itr->key.base != a || itr->key.quote != b ) break;
330  result.push_back( *itr );
331  ++itr;
332  ++count;
333  if( count > limit ) break;
334  }
335 
336  return result;
337  }

339  vector<operation_history_object> history_api::get_account_history(account_id_type account, operation_history_id_type stop, unsigned limit, operation_history_id_type start) const
340  {
341  FC_ASSERT(_app.chain_database());
342  const auto& db = *_app.chain_database();
343  FC_ASSERT(limit <= 100);
344  vector<operation_history_object> result;
345  const auto& stats = account(db).statistics(db);
346  if(stats.most_recent_op == account_transaction_history_id_type()) return result;
347  const account_transaction_history_object* node = &stats.most_recent_op(db);
348  if(start == operation_history_id_type())
349  start = node->id;
350  while(node && node->operation_id.instance.value > stop.instance.value && result.size() < limit)
351  {
352  if(node->id.instance() <= start.instance.value)
353  result.push_back(node->operation_id(db));
354  if(node->next == account_transaction_history_id_type())
355  node = nullptr;
356  else node = db.find(node->next);
357  }
358  return result;
359  }
