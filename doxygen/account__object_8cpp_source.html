32 share_type cut_fee(share_type a, uint16_t p)
33 {
34  if( a == 0 || p == 0 )
35  return 0;
36  if( p == GRAPHENE_100_PERCENT )
37  return a;
38 
39  fc::uint128 r(a.value);
40  r *= p;
41  r /= GRAPHENE_100_PERCENT;
42  return r.to_uint64();
43 }

51 void account_statistics_object::process_fees(const account_object& a, database& d) const
52 {
53  if( pending_fees > 0 || pending_vested_fees > 0 )
54  {
55  auto pay_out_fees = [&](const account_object& account, share_type core_fee_total, bool require_vesting)
56  {
57  // Check the referrer -- if he's no longer a member, pay to the lifetime referrer instead.
58  // No need to check the registrar; registrars are required to be lifetime members.
59  if( account.referrer(d).is_basic_account(d.head_block_time()) )
60  d.modify(account, [](account_object& a) {
61  a.referrer = a.lifetime_referrer;
62  });
63 
64  share_type network_cut = cut_fee(core_fee_total, account.network_fee_percentage);
65  assert( network_cut <= core_fee_total );
66 
67 #ifndef NDEBUG
68  const auto& props = d.get_global_properties();
69 
70  share_type reserveed = cut_fee(network_cut, props.parameters.reserve_percent_of_fee);
71  share_type accumulated = network_cut - reserveed;
72  assert( accumulated + reserveed == network_cut );
73 #endif
74  share_type lifetime_cut = cut_fee(core_fee_total, account.lifetime_referrer_fee_percentage);
75  share_type referral = core_fee_total - network_cut - lifetime_cut;
76 
77  d.modify(asset_dynamic_data_id_type()(d), [network_cut](asset_dynamic_data_object& d) {
78  d.accumulated_fees += network_cut;
79  });
80 
81  // Potential optimization: Skip some of this math and object lookups by special casing on the account type.
82  // For example, if the account is a lifetime member, we can skip all this and just deposit the referral to
83  // it directly.
84  share_type referrer_cut = cut_fee(referral, account.referrer_rewards_percentage);
85  share_type registrar_cut = referral - referrer_cut;
86 
87  d.deposit_cashback(d.get(account.lifetime_referrer), lifetime_cut, require_vesting);
88  d.deposit_cashback(d.get(account.referrer), referrer_cut, require_vesting);
89  d.deposit_cashback(d.get(account.registrar), registrar_cut, require_vesting);
90 
91  assert( referrer_cut + registrar_cut + accumulated + reserveed + lifetime_cut == core_fee_total );
92  };
93 
94  pay_out_fees(a, pending_fees, true);
95  pay_out_fees(a, pending_vested_fees, false);
96 
97  d.modify(*this, [&](account_statistics_object& s) {
98  s.lifetime_fees_paid += pending_fees + pending_vested_fees;
99  s.pending_fees = 0;
100  s.pending_vested_fees = 0;
101  });
102  }
103 }
