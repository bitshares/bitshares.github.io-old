43 void database::update_global_dynamic_data( const signed_block& b )
44 {
45  const dynamic_global_property_object& _dgp =
46  dynamic_global_property_id_type(0)(*this);
47 
48  uint32_t missed_blocks = get_slot_at_time( b.timestamp );
49  assert( missed_blocks != 0 );
50  missed_blocks--;
51  for( uint32_t i = 0; i < missed_blocks; ++i ) {
52  const auto& witness_missed = get_scheduled_witness( i+1 )(*this);
53  if( witness_missed.id != b.witness ) {
54  /*
55  const auto& witness_account = witness_missed.witness_account(*this);
56  if( (fc::time_point::now() - b.timestamp) < fc::seconds(30) )
57  wlog( "Witness ${name} missed block ${n} around ${t}", ("name",witness_account.name)("n",b.block_num())("t",b.timestamp) );
58  */
59 
60  modify( witness_missed, [&]( witness_object& w ) {
61  w.total_missed++;
62  });
63  }
64  }
65 
66  // dynamic global properties updating
67  modify( _dgp, [&]( dynamic_global_property_object& dgp ){
68  if( BOOST_UNLIKELY( b.block_num() == 1 ) )
69  dgp.recently_missed_count = 0;
70  else if( _checkpoints.size() && _checkpoints.rbegin()->first >= b.block_num() )
71  dgp.recently_missed_count = 0;
72  else if( missed_blocks )
73  dgp.recently_missed_count += GRAPHENE_RECENTLY_MISSED_COUNT_INCREMENT*missed_blocks;
74  else if( dgp.recently_missed_count > GRAPHENE_RECENTLY_MISSED_COUNT_INCREMENT )
75  dgp.recently_missed_count -= GRAPHENE_RECENTLY_MISSED_COUNT_DECREMENT;
76  else if( dgp.recently_missed_count > 0 )
77  dgp.recently_missed_count--;
78 
79  dgp.head_block_number = b.block_num();
80  dgp.head_block_id = b.id();
81  dgp.time = b.timestamp;
82  dgp.current_witness = b.witness;
83  dgp.recent_slots_filled = (
84  (dgp.recent_slots_filled << 1)
85  + 1) << missed_blocks;
86  dgp.current_aslot += missed_blocks+1;
87  });
88 
89  if( !(get_node_properties().skip_flags & skip_undo_history_check) )
90  {
91  GRAPHENE_ASSERT( _dgp.head_block_number - _dgp.last_irreversible_block_num < GRAPHENE_MAX_UNDO_HISTORY, undo_database_exception,
92  "The database does not have enough undo history to support a blockchain with so many missed blocks. "
93  "Please add a checkpoint if you would like to continue applying blocks beyond this point.",
94  ("last_irreversible_block_num",_dgp.last_irreversible_block_num)("head", _dgp.head_block_number)
95  ("recently_missed",_dgp.recently_missed_count)("max_undo",GRAPHENE_MAX_UNDO_HISTORY) );
96  }

124 void database::update_last_irreversible_block()
125 {
126  const global_property_object& gpo = get_global_properties();
127  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
128 
129  vector< const witness_object* > wit_objs;
130  wit_objs.reserve( gpo.active_witnesses.size() );
131  for( const witness_id_type& wid : gpo.active_witnesses )
132  wit_objs.push_back( &(wid(*this)) );
133 
134  static_assert( GRAPHENE_IRREVERSIBLE_THRESHOLD > 0, "irreversible threshold must be nonzero" );
135 
136  // 1 1 1 2 2 2 2 2 2 2 -> 2 .7*10 = 7
137  // 1 1 1 1 1 1 1 2 2 2 -> 1
138  // 3 3 3 3 3 3 3 3 3 3 -> 3
139 
140  size_t offset = ((GRAPHENE_100_PERCENT - GRAPHENE_IRREVERSIBLE_THRESHOLD) * wit_objs.size() / GRAPHENE_100_PERCENT);
141 
142  std::nth_element( wit_objs.begin(), wit_objs.begin() + offset, wit_objs.end(),
143  []( const witness_object* a, const witness_object* b )
144  {
145  return a->last_confirmed_block_num < b->last_confirmed_block_num;
146  } );
147 
148  uint32_t new_last_irreversible_block_num = wit_objs[offset]->last_confirmed_block_num;
149 
150  if( new_last_irreversible_block_num > dpo.last_irreversible_block_num )
151  {
152  modify( dpo, [&]( dynamic_global_property_object& _dpo )
153  {
154  _dpo.last_irreversible_block_num = new_last_irreversible_block_num;
155  } );
156  }

200 bool database::check_for_blackswan( const asset_object& mia, bool enable_black_swan )
201 {
202  if( !mia.is_market_issued() ) return false;
203 
204  const asset_bitasset_data_object& bitasset = mia.bitasset_data(*this);
205  if( bitasset.has_settlement() ) return true; // already force settled
206  auto settle_price = bitasset.current_feed.settlement_price;
207  if( settle_price.is_null() ) return false; // no feed
208 
209  const call_order_index& call_index = get_index_type<call_order_index>();
210  const auto& call_price_index = call_index.indices().get<by_price>();
211 
212  const limit_order_index& limit_index = get_index_type<limit_order_index>();
213  const auto& limit_price_index = limit_index.indices().get<by_price>();
214 
215  // looking for limit orders selling the most USD for the least CORE
216  auto highest_possible_bid = price::max( mia.id, bitasset.options.short_backing_asset );
217  // stop when limit orders are selling too little USD for too much CORE
218  auto lowest_possible_bid = price::min( mia.id, bitasset.options.short_backing_asset );
219 
220  assert( highest_possible_bid.base.asset_id == lowest_possible_bid.base.asset_id );
221  // NOTE limit_price_index is sorted from greatest to least
222  auto limit_itr = limit_price_index.lower_bound( highest_possible_bid );
223  auto limit_end = limit_price_index.upper_bound( lowest_possible_bid );
224 
225  auto call_min = price::min( bitasset.options.short_backing_asset, mia.id );
226  auto call_max = price::max( bitasset.options.short_backing_asset, mia.id );
227  auto call_itr = call_price_index.lower_bound( call_min );
228  auto call_end = call_price_index.upper_bound( call_max );
229 
230  if( call_itr == call_end ) return false; // no call orders
231 
232  price highest = settle_price;
233  if( limit_itr != limit_end ) {
234  assert( settle_price.base.asset_id == limit_itr->sell_price.base.asset_id );
235  highest = std::max( limit_itr->sell_price, settle_price );
236  }
237 
238  auto least_collateral = call_itr->collateralization();
239  if( ~least_collateral >= highest )
240  {
241  elog( "Black Swan detected: \n"
242  " Least collateralized call: ${lc} ${~lc}\n"
243  // " Highest Bid: ${hb} ${~hb}\n"
244  " Settle Price: ${sp} ${~sp}\n"
245  " Max: ${h} ${~h}\n",
246  ("lc",least_collateral.to_real())("~lc",(~least_collateral).to_real())
247  // ("hb",limit_itr->sell_price.to_real())("~hb",(~limit_itr->sell_price).to_real())
248  ("sp",settle_price.to_real())("~sp",(~settle_price).to_real())
249  ("h",highest.to_real())("~h",(~highest).to_real()) );
250  FC_ASSERT( enable_black_swan, "Black swan was detected during a margin update which is not allowed to trigger a blackswan" );
251  globally_settle_asset(mia, ~least_collateral );
252  return true;
253  }
254  return false;
255 }

257 void database::clear_expired_orders()
258 { try {
259  detail::with_skip_flags( *this,
260  get_node_properties().skip_flags | skip_authority_check, [&](){
261  transaction_evaluation_state cancel_context(this);
262 
263  //Cancel expired limit orders
264  auto& limit_index = get_index_type<limit_order_index>().indices().get<by_expiration>();
265  while( !limit_index.empty() && limit_index.begin()->expiration <= head_block_time() )
266  {
267  limit_order_cancel_operation canceler;
268  const limit_order_object& order = *limit_index.begin();
269  canceler.fee_paying_account = order.seller;
270  canceler.order = order.id;
271  canceler.fee = current_fee_schedule().calculate_fee( canceler );
272  if( canceler.fee.amount > order.deferred_fee )
273  {
274  // Cap auto-cancel fees at deferred_fee; see #549
275  //wlog( "At block ${b}, fee for clearing expired order ${oid} was capped at deferred_fee ${fee}", ("b", head_block_num())("oid", order.id)("fee", order.deferred_fee) );
276  canceler.fee = asset( order.deferred_fee, asset_id_type() );
277  }
278  // we know the fee for this op is set correctly since it is set by the chain.
279  // this allows us to avoid a hung chain:
280  // - if #549 case above triggers
281  // - if the fee is incorrect, which may happen due to #435 (although since cancel is a fixed-fee op, it shouldn't)
282  cancel_context.skip_fee_schedule_check = true;
283  apply_operation(cancel_context, canceler);
284  }
285  });
286 
287  //Process expired force settlement orders
288  auto& settlement_index = get_index_type<force_settlement_index>().indices().get<by_expiration>();
289  if( !settlement_index.empty() )
290  {
291  asset_id_type current_asset = settlement_index.begin()->settlement_asset_id();
292  asset max_settlement_volume;
293  price settlement_fill_price;
294  bool extra_dump = false;
295 
296  auto next_asset = [&current_asset, &settlement_index, &extra_dump] {
297  auto bound = settlement_index.upper_bound(current_asset);
298  if( bound == settlement_index.end() )
299  {
300  if( extra_dump )
301  {
302  ilog( "next_asset() returning false" );
303  }
304  return false;
305  }
306  if( extra_dump )
307  {
308  ilog( "next_asset returning true, bound is ${b}", ("b", *bound) );
309  }
310  current_asset = bound->settlement_asset_id();
311  return true;
312  };
313 
314  uint32_t count = 0;
315 
316  // At each iteration, we either consume the current order and remove it, or we move to the next asset
317  for( auto itr = settlement_index.lower_bound(current_asset);
318  itr != settlement_index.end();
319  itr = settlement_index.lower_bound(current_asset) )
320  {
321  ++count;
322  const force_settlement_object& order = *itr;
323  auto order_id = order.id;
324  current_asset = order.settlement_asset_id();
325  const asset_object& mia_object = get(current_asset);
326  const asset_bitasset_data_object& mia = mia_object.bitasset_data(*this);
327 
328  extra_dump = ((count >= 1000) && (count <= 1020));
329 
330  if( extra_dump )
331  {
332  wlog( "clear_expired_orders() dumping extra data for iteration ${c}", ("c", count) );
333  ilog( "head_block_num is ${hb} current_asset is ${a}", ("hb", head_block_num())("a", current_asset) );
334  }
335 
336  if( mia.has_settlement() )
337  {
338  ilog( "Canceling a force settlement because of black swan" );
339  cancel_order( order );
340  continue;
341  }
342 
343  // Has this order not reached its settlement date?
344  if( order.settlement_date > head_block_time() )
345  {
346  if( next_asset() )
347  {
348  if( extra_dump )
349  {
350  ilog( "next_asset() returned true when order.settlement_date > head_block_time()" );
351  }
352  continue;
353  }
354  break;
355  }
356  // Can we still settle in this asset?
357  if( mia.current_feed.settlement_price.is_null() )
358  {
359  ilog("Canceling a force settlement in ${asset} because settlement price is null",
360  ("asset", mia_object.symbol));
361  cancel_order(order);
362  continue;
363  }
364  if( max_settlement_volume.asset_id != current_asset )
365  max_settlement_volume = mia_object.amount(mia.max_force_settlement_volume(mia_object.dynamic_data(*this).current_supply));
366  if( mia.force_settled_volume >= max_settlement_volume.amount )
367  {
368  /*
369  ilog("Skipping force settlement in ${asset}; settled ${settled_volume} / ${max_volume}",
370  ("asset", mia_object.symbol)("settlement_price_null",mia.current_feed.settlement_price.is_null())
371  ("settled_volume", mia.force_settled_volume)("max_volume", max_settlement_volume));
372  */
373  if( next_asset() )
374  {
375  if( extra_dump )
376  {
377  ilog( "next_asset() returned true when mia.force_settled_volume >= max_settlement_volume.amount" );
378  }
379  continue;
380  }
381  break;
382  }
383 
384  auto& pays = order.balance;
385  auto receives = (order.balance * mia.current_feed.settlement_price);
386  receives.amount = (fc::uint128_t(receives.amount.value) *
387  (GRAPHENE_100_PERCENT - mia.options.force_settlement_offset_percent) / GRAPHENE_100_PERCENT).to_uint64();
388  assert(receives <= order.balance * mia.current_feed.settlement_price);
389 
390  price settlement_price = pays / receives;
391 
392  // Calculate fill_price with a bigger volume to reduce impacts of rounding
393  // TODO replace the calculation with new operator*() and/or operator/()
394  if( settlement_fill_price.base.asset_id != current_asset ) // only calculate once per asset
395  {
396  asset tmp_pays = max_settlement_volume;
397  asset tmp_receives = tmp_pays * mia.current_feed.settlement_price;
398  tmp_receives.amount = (fc::uint128_t(tmp_receives.amount.value) *
399  (GRAPHENE_100_PERCENT - mia.options.force_settlement_offset_percent) / GRAPHENE_100_PERCENT).to_uint64();
400  settlement_fill_price = tmp_pays / tmp_receives;
401  }
402 
403  auto& call_index = get_index_type<call_order_index>().indices().get<by_collateral>();
404  asset settled = mia_object.amount(mia.force_settled_volume);
405  // Match against the least collateralized short until the settlement is finished or we reach max settlements
406  while( settled < max_settlement_volume && find_object(order_id) )
407  {
408  auto itr = call_index.lower_bound(boost::make_tuple(price::min(mia_object.bitasset_data(*this).options.short_backing_asset,
409  mia_object.get_id())));
410  // There should always be a call order, since asset exists!
411  assert(itr != call_index.end() && itr->debt_type() == mia_object.get_id());
412  asset max_settlement = max_settlement_volume - settled;
413 
414  if( order.balance.amount == 0 )
415  {
416  wlog( "0 settlement detected" );
417  cancel_order( order );
418  break;
419  }
420  try {
421  settled += match(*itr, order, settlement_price, max_settlement, settlement_fill_price);
422  }
423  catch ( const black_swan_exception& e ) {
424  wlog( "black swan detected: ${e}", ("e", e.to_detail_string() ) );
425  cancel_order( order );
426  break;
427  }
428  }
429  if( mia.force_settled_volume != settled.amount )
430  {
431  modify(mia, [settled](asset_bitasset_data_object& b) {
432  b.force_settled_volume = settled.amount;
433  });
434  }
435  }
436  }
437 } FC_CAPTURE_AND_RETHROW() }
