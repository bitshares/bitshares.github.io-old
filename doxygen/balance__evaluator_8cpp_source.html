28 void_result balance_claim_evaluator::do_evaluate(const balance_claim_operation& op)
29 {
30  database& d = db();
31  balance = &op.balance_to_claim(d);
32 
33  GRAPHENE_ASSERT(
34  op.balance_owner_key == balance->owner ||
35  pts_address(op.balance_owner_key, false, 56) == balance->owner ||
36  pts_address(op.balance_owner_key, true, 56) == balance->owner ||
37  pts_address(op.balance_owner_key, false, 0) == balance->owner ||
38  pts_address(op.balance_owner_key, true, 0) == balance->owner,
39  balance_claim_owner_mismatch,
40  "Balance owner key was specified as '${op}' but balance's actual owner is '${bal}'",
41  ("op", op.balance_owner_key)
42  ("bal", balance->owner)
43  );
44  if( !(d.get_node_properties().skip_flags & (database::skip_authority_check |
45  database::skip_transaction_signatures)) )
46 
47  FC_ASSERT(op.total_claimed.asset_id == balance->asset_type());
48 
49  if( balance->is_vesting_balance() )
50  {
51  GRAPHENE_ASSERT(
52  balance->vesting_policy->is_withdraw_allowed(
53  { balance->balance,
54  d.head_block_time(),
55  op.total_claimed } ),
56  balance_claim_invalid_claim_amount,
57  "Attempted to claim ${c} from a vesting balance with ${a} available",
58  ("c", op.total_claimed)("a", balance->available(d.head_block_time()))
59  );
60  GRAPHENE_ASSERT(
61  d.head_block_time() - balance->last_claim_date >= fc::days(1),
62  balance_claim_claimed_too_often,
63  "Genesis vesting balances may not be claimed more than once per day."
64  );
65  return {};
66  }
67 
68  FC_ASSERT(op.total_claimed == balance->balance);
69  return {};
70 }

76 void_result balance_claim_evaluator::do_apply(const balance_claim_operation& op)
77 {
78  database& d = db();
79 
80  if( balance->is_vesting_balance() && op.total_claimed < balance->balance )
81  d.modify(*balance, [&](balance_object& b) {
82  b.vesting_policy->on_withdraw({b.balance, d.head_block_time(), op.total_claimed});
83  b.balance -= op.total_claimed;
84  b.last_claim_date = d.head_block_time();
85  });
86  else
87  d.remove(*balance);
88 
89  d.adjust_balance(op.deposit_to_account, op.total_claimed);
90  return {};
91 }
