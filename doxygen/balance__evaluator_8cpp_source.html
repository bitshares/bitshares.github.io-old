25 void_result balance_claim_evaluator::do_evaluate(const balance_claim_operation& op)
26 {
27  database& d = db();
28  balance = &op.balance_to_claim(d);
29 
30  GRAPHENE_ASSERT(
31  op.balance_owner_key == balance->owner ||
32  pts_address(op.balance_owner_key, false, 56) == balance->owner ||
33  pts_address(op.balance_owner_key, true, 56) == balance->owner ||
34  pts_address(op.balance_owner_key, false, 0) == balance->owner ||
35  pts_address(op.balance_owner_key, true, 0) == balance->owner,
36  balance_claim_owner_mismatch,
37  "Balance owner key was specified as '${op}' but balance's actual owner is '${bal}'",
38  ("op", op.balance_owner_key)
39  ("bal", balance->owner)
40  );
41  if( !(d.get_node_properties().skip_flags & (database::skip_authority_check |
42  database::skip_transaction_signatures)) )
43 
44  FC_ASSERT(op.total_claimed.asset_id == balance->asset_type());
45 
46  if( balance->is_vesting_balance() )
47  {
48  GRAPHENE_ASSERT(
49  balance->vesting_policy->is_withdraw_allowed(
50  { balance->balance,
51  d.head_block_time(),
52  op.total_claimed } ),
53  balance_claim_invalid_claim_amount,
54  "Attempted to claim ${c} from a vesting balance with ${a} available",
55  ("c", op.total_claimed)("a", balance->available(d.head_block_time()))
56  );
57  GRAPHENE_ASSERT(
58  d.head_block_time() - balance->last_claim_date >= fc::days(1),
59  balance_claim_claimed_too_often,
60  "Genesis vesting balances may not be claimed more than once per day."
61  );
62  return {};
63  }
64 
65  FC_ASSERT(op.total_claimed == balance->balance);
66  return {};
67 }

73 void_result balance_claim_evaluator::do_apply(const balance_claim_operation& op)
74 {
75  database& d = db();
76 
77  if( balance->is_vesting_balance() && op.total_claimed < balance->balance )
78  d.modify(*balance, [&](balance_object& b) {
79  b.vesting_policy->on_withdraw({b.balance, d.head_block_time(), op.total_claimed});
80  b.balance -= op.total_claimed;
81  b.last_claim_date = d.head_block_time();
82  });
83  else
84  d.remove(*balance);
85 
86  d.adjust_balance(op.deposit_to_account, op.total_claimed);
87  return {};
88 }
