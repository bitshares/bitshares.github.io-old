57 share_type call_order_object::get_max_debt_to_cover( price match_price,
58  price feed_price,
59  const uint16_t maintenance_collateral_ratio )const
60 { try {
61  // be defensive here, make sure feed_price is in collateral / debt format
62  if( feed_price.base.asset_id != call_price.base.asset_id )
63  feed_price = ~feed_price;
64 
65  FC_ASSERT( feed_price.base.asset_id == call_price.base.asset_id
66  && feed_price.quote.asset_id == call_price.quote.asset_id );
67 
68  if( call_price > feed_price ) // feed protected. be defensive here, although this should be guaranteed by caller
69  return 0;
70 
71  if( !target_collateral_ratio.valid() ) // target cr is not set
72  return debt;
73 
74  uint16_t tcr = std::max( *target_collateral_ratio, maintenance_collateral_ratio ); // use mcr if target cr is too small
75 
76  // be defensive here, make sure match_price is in collateral / debt format
77  if( match_price.base.asset_id != call_price.base.asset_id )
78  match_price = ~match_price;
79 
80  FC_ASSERT( match_price.base.asset_id == call_price.base.asset_id
81  && match_price.quote.asset_id == call_price.quote.asset_id );
82 
83  typedef boost::multiprecision::int256_t i256;
84  i256 mp_debt_amt = match_price.quote.amount.value;
85  i256 mp_coll_amt = match_price.base.amount.value;
86  i256 fp_debt_amt = feed_price.quote.amount.value;
87  i256 fp_coll_amt = feed_price.base.amount.value;
88 
89  // firstly we calculate without the fraction (x), the result could be a bit too small
90  i256 numerator = fp_coll_amt * mp_debt_amt * debt.value * tcr
91  - fp_debt_amt * mp_debt_amt * collateral.value * GRAPHENE_COLLATERAL_RATIO_DENOM;
92  if( numerator < 0 ) // feed protected, actually should not be true here, just check to be safe
93  return 0;
94 
95  i256 denominator = fp_coll_amt * mp_debt_amt * tcr - fp_debt_amt * mp_coll_amt * GRAPHENE_COLLATERAL_RATIO_DENOM;
96  if( denominator <= 0 ) // black swan
97  return debt;
98 
99  // note: if add 1 here, will result in 1.5x imperfection rate;
100  // however, due to rounding, the result could still be a bit too big, thus imperfect.
101  i256 to_cover_i256 = ( numerator / denominator );
102  if( to_cover_i256 >= debt.value ) // avoid possible overflow
103  return debt;
104  share_type to_cover_amt = static_cast< int64_t >( to_cover_i256 );
105 
106  // stabilize
107  // note: rounding up-down results in 3x imperfection rate in comparison to down-down-up
108  asset to_pay = asset( to_cover_amt, debt_type() ) * match_price;
109  asset to_cover = to_pay * match_price;
110  to_pay = to_cover.multiply_and_round_up( match_price );
111 
112  if( to_cover.amount >= debt || to_pay.amount >= collateral ) // to be safe
113  return debt;
114  FC_ASSERT( to_pay.amount < collateral && to_cover.amount < debt );
115 
116  // check collateral ratio after filled, if it's OK, we return
117  price new_call_price = price::call_price( get_debt() - to_cover, get_collateral() - to_pay, tcr );
118  if( new_call_price > feed_price )
119  return to_cover.amount;
120 
121  // be here, to_cover is too small due to rounding. deal with the fraction
122  numerator += fp_coll_amt * mp_debt_amt * tcr; // plus the fraction
123  to_cover_i256 = ( numerator / denominator ) + 1;
124  if( to_cover_i256 >= debt.value ) // avoid possible overflow
125  to_cover_i256 = debt.value;
126  to_cover_amt = static_cast< int64_t >( to_cover_i256 );
127 
128  asset max_to_pay = ( ( to_cover_amt == debt.value ) ? get_collateral()
129  : asset( to_cover_amt, debt_type() ).multiply_and_round_up( match_price ) );
130  if( max_to_pay.amount > collateral )
131  max_to_pay.amount = collateral;
132 
133  asset max_to_cover = ( ( max_to_pay.amount == collateral ) ? get_debt() : ( max_to_pay * match_price ) );
134  if( max_to_cover.amount >= debt ) // to be safe
135  {
136  max_to_pay.amount = collateral;
137  max_to_cover.amount = debt;
138  }
139 
140  if( max_to_pay <= to_pay || max_to_cover <= to_cover ) // strange data. should skip binary search and go on, but doesn't help much
141  return debt;
142  FC_ASSERT( max_to_pay > to_pay && max_to_cover > to_cover );
143 
144  asset min_to_pay = to_pay;
145  asset min_to_cover = to_cover;
146 
147  // try with binary search to find a good value
148  // note: actually binary search can not always provide perfect result here,
149  // due to rounding, collateral ratio is not always increasing while to_pay or to_cover is increasing
150  bool max_is_ok = false;
151  while( true )
152  {
153  // get the mean
154  if( match_price.base.amount < match_price.quote.amount ) // step of collateral is smaller
155  {
156  to_pay.amount = ( min_to_pay.amount + max_to_pay.amount + 1 ) / 2; // should not overflow. round up here
157  if( to_pay.amount == max_to_pay.amount )
158  to_cover.amount = max_to_cover.amount;
159  else
160  {
161  to_cover = to_pay * match_price;
162  if( to_cover.amount >= max_to_cover.amount ) // can be true when max_is_ok is false
163  {
164  to_pay.amount = max_to_pay.amount;
165  to_cover.amount = max_to_cover.amount;
166  }
167  else
168  {
169  to_pay = to_cover.multiply_and_round_up( match_price ); // stabilization, no change or become smaller
170  FC_ASSERT( to_pay.amount < max_to_pay.amount );
171  }
172  }
173  }
174  else // step of debt is smaller or equal
175  {
176  to_cover.amount = ( min_to_cover.amount + max_to_cover.amount ) / 2; // should not overflow. round down here
177  if( to_cover.amount == max_to_cover.amount )
178  to_pay.amount = max_to_pay.amount;
179  else
180  {
181  to_pay = to_cover.multiply_and_round_up( match_price );
182  if( to_pay.amount >= max_to_pay.amount ) // can be true when max_is_ok is false
183  {
184  to_pay.amount = max_to_pay.amount;
185  to_cover.amount = max_to_cover.amount;
186  }
187  else
188  {
189  to_cover = to_pay * match_price; // stabilization, to_cover should have increased
190  if( to_cover.amount >= max_to_cover.amount ) // to be safe
191  {
192  to_pay.amount = max_to_pay.amount;
193  to_cover.amount = max_to_cover.amount;
194  }
195  }
196  }
197  }
198 
199  // check again to see if we've moved away from the minimums, if not, use the maximums directly
200  if( to_pay.amount <= min_to_pay.amount || to_cover.amount <= min_to_cover.amount
201  || to_pay.amount > max_to_pay.amount || to_cover.amount > max_to_cover.amount )
202  {
203  to_pay.amount = max_to_pay.amount;
204  to_cover.amount = max_to_cover.amount;
205  }
206 
207  // check the mean
208  if( to_pay.amount == max_to_pay.amount && ( max_is_ok || to_pay.amount == collateral ) )
209  return to_cover.amount;
210  FC_ASSERT( to_pay.amount < collateral && to_cover.amount < debt );
211 
212  new_call_price = price::call_price( get_debt() - to_cover, get_collateral() - to_pay, tcr );
213  if( new_call_price > feed_price ) // good
214  {
215  if( to_pay.amount == max_to_pay.amount )
216  return to_cover.amount;
217  max_to_pay.amount = to_pay.amount;
218  max_to_cover.amount = to_cover.amount;
219  max_is_ok = true;
220  }
221  else // not good
222  {
223  if( to_pay.amount == max_to_pay.amount )
224  break;
225  min_to_pay.amount = to_pay.amount;
226  min_to_cover.amount = to_cover.amount;
227  }
228  }
229 
230  // be here, max_to_cover is too small due to rounding. search forward
231  for( uint64_t d1 = 0, d2 = 1, d3 = 1; ; d1 = d2, d2 = d3, d3 = d1 + d2 ) // 1,1,2,3,5,8,...
232  {
233  if( match_price.base.amount > match_price.quote.amount ) // step of debt is smaller
234  {
235  to_pay.amount += d2;
236  if( to_pay.amount >= collateral )
237  return debt;
238  to_cover = to_pay * match_price;
239  if( to_cover.amount >= debt )
240  return debt;
241  to_pay = to_cover.multiply_and_round_up( match_price ); // stabilization
242  if( to_pay.amount >= collateral )
243  return debt;
244  }
245  else // step of collateral is smaller or equal
246  {
247  to_cover.amount += d2;
248  if( to_cover.amount >= debt )
249  return debt;
250  to_pay = to_cover.multiply_and_round_up( match_price );
251  if( to_pay.amount >= collateral )
252  return debt;
253  to_cover = to_pay * match_price; // stabilization
254  if( to_cover.amount >= debt )
255  return debt;
256  }
257 
258  // check
259  FC_ASSERT( to_pay.amount < collateral && to_cover.amount < debt );
260 
261  new_call_price = price::call_price( get_debt() - to_cover, get_collateral() - to_pay, tcr );
262  if( new_call_price > feed_price ) // good
263  return to_cover.amount;
264  }
