129 bool database::_push_block(const signed_block& new_block)
130 { try {
131  uint32_t skip = get_node_properties().skip_flags;
132  if( !(skip&skip_fork_db) )
133  {
135  // verify that the block signer is in the current set of active witnesses.
136 
137  shared_ptr<fork_item> new_head = _fork_db.push_block(new_block);
138  //If the head block from the longest chain does not build off of the current head, we need to switch forks.
139  if( new_head->data.previous != head_block_id() )
140  {
141  //If the newly pushed block is the same height as head, we get head back in new_head
142  //Only switch forks if new_head is actually higher than head
143  if( new_head->data.block_num() > head_block_num() )
144  {
145  wlog( "Switching to fork: ${id}", ("id",new_head->data.id()) );
146  auto branches = _fork_db.fetch_branch_from(new_head->data.id(), head_block_id());
147 
148  // pop blocks until we hit the forked block
149  while( head_block_id() != branches.second.back()->data.previous )
150  {
151  ilog( "popping block #${n} ${id}", ("n",head_block_num())("id",head_block_id()) );
152  pop_block();
153  }
154 
155  // push all blocks on the new fork
156  for( auto ritr = branches.first.rbegin(); ritr != branches.first.rend(); ++ritr )
157  {
158  ilog( "pushing block from fork #${n} ${id}", ("n",(*ritr)->data.block_num())("id",(*ritr)->id) );
159  optional<fc::exception> except;
160  try {
161  undo_database::session session = _undo_db.start_undo_session();
162  apply_block( (*ritr)->data, skip );
163  _block_id_to_block.store( (*ritr)->id, (*ritr)->data );
164  session.commit();
165  }
166  catch ( const fc::exception& e ) { except = e; }
167  if( except )
168  {
169  wlog( "exception thrown while switching forks ${e}", ("e",except->to_detail_string() ) );
170  // remove the rest of branches.first from the fork_db, those blocks are invalid
171  while( ritr != branches.first.rend() )
172  {
173  ilog( "removing block from fork_db #${n} ${id}", ("n",(*ritr)->data.block_num())("id",(*ritr)->id) );
174  _fork_db.remove( (*ritr)->id );
175  ++ritr;
176  }
177  _fork_db.set_head( branches.second.front() );
178 
179  // pop all blocks from the bad fork
180  while( head_block_id() != branches.second.back()->data.previous )
181  {
182  ilog( "popping block #${n} ${id}", ("n",head_block_num())("id",head_block_id()) );
183  pop_block();
184  }
185 
186  ilog( "Switching back to fork: ${id}", ("id",branches.second.front()->data.id()) );
187  // restore all blocks from the good fork
188  for( auto ritr2 = branches.second.rbegin(); ritr2 != branches.second.rend(); ++ritr2 )
189  {
190  ilog( "pushing block #${n} ${id}", ("n",(*ritr2)->data.block_num())("id",(*ritr2)->id) );
191  auto session = _undo_db.start_undo_session();
192  apply_block( (*ritr2)->data, skip );
193  _block_id_to_block.store( (*ritr2)->id, (*ritr2)->data );
194  session.commit();
195  }
196  throw *except;
197  }
198  }
199  return true;
200  }
201  else return false;
202  }
203  }
204 
205  try {
206  auto session = _undo_db.start_undo_session();
207  apply_block(new_block, skip);
208  _block_id_to_block.store(new_block.id(), new_block);
209  session.commit();
210  } catch ( const fc::exception& e ) {
211  elog("Failed to push new block:\n${e}", ("e", e.to_detail_string()));
212  _fork_db.remove(new_block.id());
213  throw;
214  }
215 
216  return false;
217 } FC_CAPTURE_AND_RETHROW( (new_block) ) }

238 processed_transaction database::_push_transaction( const signed_transaction& trx )
239 {
240  // If this is the first transaction pushed after applying a block, start a new undo session.
241  // This allows us to quickly rewind to the clean state of the head block, in case a new block arrives.
242  if( !_pending_tx_session.valid() )
243  _pending_tx_session = _undo_db.start_undo_session();
244 
245  // Create a temporary undo session as a child of _pending_tx_session.
246  // The temporary session will be discarded by the destructor if
247  // _apply_transaction fails. If we make it to merge(), we
248  // apply the changes.
249 
250  auto temp_session = _undo_db.start_undo_session();
251  auto processed_trx = _apply_transaction( trx );
252  _pending_tx.push_back(processed_trx);
253 
254  // notify_changed_objects();
255  // The transaction applied successfully. Merge its changes into the pending block session.
256  temp_session.merge();
257 
258  // notify anyone listening to pending transactions
259  notify_on_pending_transaction( trx );
260  return processed_trx;
261 }

269 processed_transaction database::push_proposal(const proposal_object& proposal)
270 { try {
271  transaction_evaluation_state eval_state(this);
272  eval_state._is_proposed_trx = true;
273 
274  eval_state.operation_results.reserve(proposal.proposed_transaction.operations.size());
275  processed_transaction ptrx(proposal.proposed_transaction);
276  eval_state._trx = &ptrx;
277  size_t old_applied_ops_size = _applied_ops.size();
278 
279  try {
280  auto session = _undo_db.start_undo_session(true);
281  for( auto& op : proposal.proposed_transaction.operations )
282  eval_state.operation_results.emplace_back(apply_operation(eval_state, op));
283  remove(proposal);
284  session.merge();
285  } catch ( const fc::exception& e ) {
286  if( head_block_time() <= HARDFORK_483_TIME )
287  {
288  for( size_t i=old_applied_ops_size,n=_applied_ops.size(); i<n; i++ )
289  {
290  ilog( "removing failed operation from applied_ops: ${op}", ("op", *(_applied_ops[i])) );
291  _applied_ops[i].reset();
292  }
293  }
294  else
295  {
296  _applied_ops.resize( old_applied_ops_size );
297  }
298  elog( "${e}", ("e",e.to_detail_string() ) );
299  throw;
300  }
301 
302  ptrx.operation_results = std::move(eval_state.operation_results);
303  return ptrx;
304 } FC_CAPTURE_AND_RETHROW( (proposal) ) }

321 signed_block database::_generate_block(
322  fc::time_point_sec when,
323  witness_id_type witness_id,
324  const fc::ecc::private_key& block_signing_private_key
325  )
326 {
327  try {
328  uint32_t skip = get_node_properties().skip_flags;
329  uint32_t slot_num = get_slot_at_time( when );
330  FC_ASSERT( slot_num > 0 );
331  witness_id_type scheduled_witness = get_scheduled_witness( slot_num );
332  FC_ASSERT( scheduled_witness == witness_id );
333 
334  const auto& witness_obj = witness_id(*this);
335 
336  if( !(skip & skip_witness_signature) )
337  FC_ASSERT( witness_obj.signing_key == block_signing_private_key.get_public_key() );
338 
339  static const size_t max_block_header_size = fc::raw::pack_size( signed_block_header() ) + 4;
340  auto maximum_block_size = get_global_properties().parameters.maximum_block_size;
341  size_t total_block_size = max_block_header_size;
342 
343  signed_block pending_block;
344 
345  //
346  // The following code throws away existing pending_tx_session and
347  // rebuilds it by re-applying pending transactions.
348  //
349  // This rebuild is necessary because pending transactions' validity
350  // and semantics may have changed since they were received, because
351  // time-based semantics are evaluated based on the current block
352  // time. These changes can only be reflected in the database when
353  // the value of the "when" variable is known, which means we need to
354  // re-apply pending transactions in this method.
355  //
356  _pending_tx_session.reset();
357  _pending_tx_session = _undo_db.start_undo_session();
358 
359  uint64_t postponed_tx_count = 0;
360  // pop pending state (reset to head block state)
361  for( const processed_transaction& tx : _pending_tx )
362  {
363  size_t new_total_size = total_block_size + fc::raw::pack_size( tx );
364 
365  // postpone transaction if it would make block too big
366  if( new_total_size >= maximum_block_size )
367  {
368  postponed_tx_count++;
369  continue;
370  }
371 
372  try
373  {
374  auto temp_session = _undo_db.start_undo_session();
375  processed_transaction ptx = _apply_transaction( tx );
376  temp_session.merge();
377 
378  // We have to recompute pack_size(ptx) because it may be different
379  // than pack_size(tx) (i.e. if one or more results increased
380  // their size)
381  total_block_size += fc::raw::pack_size( ptx );
382  pending_block.transactions.push_back( ptx );
383  }
384  catch ( const fc::exception& e )
385  {
386  // Do nothing, transaction will not be re-applied
387  wlog( "Transaction was not processed while generating block due to ${e}", ("e", e) );
388  wlog( "The transaction was ${t}", ("t", tx) );
389  }
390  }
391  if( postponed_tx_count > 0 )
392  {
393  wlog( "Postponed ${n} transactions due to block size limit", ("n", postponed_tx_count) );
394  }
395 
396  _pending_tx_session.reset();
397 
398  // We have temporarily broken the invariant that
399  // _pending_tx_session is the result of applying _pending_tx, as
400  // _pending_tx now consists of the set of postponed transactions.
401  // However, the push_block() call below will re-create the
402  // _pending_tx_session.
403 
404  pending_block.previous = head_block_id();
405  pending_block.timestamp = when;
406  pending_block.transaction_merkle_root = pending_block.calculate_merkle_root();
407  pending_block.witness = witness_id;
408 
409  if( !(skip & skip_witness_signature) )
410  pending_block.sign( block_signing_private_key );
411 
412  // TODO: Move this to _push_block() so session is restored.
413  if( !(skip & skip_block_size_check) )
414  {
415  FC_ASSERT( fc::raw::pack_size(pending_block) <= get_global_properties().parameters.maximum_block_size );
416  }
417 
418  push_block( pending_block, skip );
419 
420  return pending_block;
421 } FC_CAPTURE_AND_RETHROW( (witness_id) ) }

496 void database::_apply_block( const signed_block& next_block )
497 { try {
498  uint32_t next_block_num = next_block.block_num();
499  uint32_t skip = get_node_properties().skip_flags;
500  _applied_ops.clear();
501 
502  FC_ASSERT( (skip & skip_merkle_check) || next_block.transaction_merkle_root == next_block.calculate_merkle_root(), "", ("next_block.transaction_merkle_root",next_block.transaction_merkle_root)("calc",next_block.calculate_merkle_root())("next_block",next_block)("id",next_block.id()) );
503 
504  const witness_object& signing_witness = validate_block_header(skip, next_block);
505  const auto& global_props = get_global_properties();
506  const auto& dynamic_global_props = get<dynamic_global_property_object>(dynamic_global_property_id_type());
507  bool maint_needed = (dynamic_global_props.next_maintenance_time <= next_block.timestamp);
508 
509  _current_block_num = next_block_num;
510  _current_trx_in_block = 0;
511 
512  for( const auto& trx : next_block.transactions )
513  {
514  /* We do not need to push the undo state for each transaction
515  * because they either all apply and are valid or the
516  * entire block fails to apply. We only need an "undo" state
517  * for transactions when validating broadcast transactions or
518  * when building a block.
519  */
520  apply_transaction( trx, skip );
521  ++_current_trx_in_block;
522  }
523 
524  update_global_dynamic_data(next_block);
525  update_signing_witness(signing_witness, next_block);
526  update_last_irreversible_block();
527 
528  // Are we at the maintenance interval?
529  if( maint_needed )
530  perform_chain_maintenance(next_block, global_props);
531 
532  create_block_summary(next_block);
533  clear_expired_transactions();
534  clear_expired_proposals();
535  clear_expired_orders();
536  update_expired_feeds();
537  update_withdraw_permissions();
538 
539  // n.b., update_maintenance_flag() happens this late
540  // because get_slot_time() / get_slot_at_time() is needed above
541  // TODO: figure out if we could collapse this function into
542  // update_global_dynamic_data() as perhaps these methods only need
543  // to be called for header validation?
544  update_maintenance_flag( maint_needed );
545  update_witness_schedule();
546  if( !_node_property_object.debug_updates.empty() )
547  apply_debug_updates();
548 
549  // notify observers that the block has been applied
550  notify_applied_block( next_block ); //emit
551  _applied_ops.clear();
552 
553  notify_changed_objects();
554 } FC_CAPTURE_AND_RETHROW( (next_block.block_num()) ) }

568 processed_transaction database::_apply_transaction(const signed_transaction& trx)
569 { try {
570  uint32_t skip = get_node_properties().skip_flags;
571 
572  if( true || !(skip&skip_validate) ) /* issue #505 explains why this skip_flag is disabled */
573  trx.validate();
574 
575  auto& trx_idx = get_mutable_index_type<transaction_index>();
576  const chain_id_type& chain_id = get_chain_id();
577  auto trx_id = trx.id();
578  FC_ASSERT( (skip & skip_transaction_dupe_check) ||
579  trx_idx.indices().get<by_trx_id>().find(trx_id) == trx_idx.indices().get<by_trx_id>().end() );
580  transaction_evaluation_state eval_state(this);
581  const chain_parameters& chain_parameters = get_global_properties().parameters;
582  eval_state._trx = &trx;
583 
584  if( !(skip & (skip_transaction_signatures | skip_authority_check) ) )
585  {
586  auto get_active = [&]( account_id_type id ) { return &id(*this).active; };
587  auto get_owner = [&]( account_id_type id ) { return &id(*this).owner; };
588  trx.verify_authority( chain_id, get_active, get_owner, get_global_properties().parameters.max_authority_depth );
589  }
590 
591  //Skip all manner of expiration and TaPoS checking if we're on block 1; It's impossible that the transaction is
592  //expired, and TaPoS makes no sense as no blocks exist.
593  if( BOOST_LIKELY(head_block_num() > 0) )
594  {
595  if( !(skip & skip_tapos_check) )
596  {
597  const auto& tapos_block_summary = block_summary_id_type( trx.ref_block_num )(*this);
598 
599  //Verify TaPoS block summary has correct ID prefix, and that this block's time is not past the expiration
600  FC_ASSERT( trx.ref_block_prefix == tapos_block_summary.block_id._hash[1] );
601  }
602 
603  fc::time_point_sec now = head_block_time();
604 
605  FC_ASSERT( trx.expiration <= now + chain_parameters.maximum_time_until_expiration, "",

649  FC_ASSERT( eval, "No registered evaluator for operation ${op}", ("op",op) );
