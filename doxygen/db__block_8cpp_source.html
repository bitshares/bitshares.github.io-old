121 bool database::_push_block(const signed_block& new_block)
122 { try {
123  uint32_t skip = get_node_properties().skip_flags;
124  if( !(skip&skip_fork_db) )
125  {
127  // verify that the block signer is in the current set of active witnesses.
128 
129  shared_ptr<fork_item> new_head = _fork_db.push_block(new_block);
130  //If the head block from the longest chain does not build off of the current head, we need to switch forks.
131  if( new_head->data.previous != head_block_id() )
132  {
133  //If the newly pushed block is the same height as head, we get head back in new_head
134  //Only switch forks if new_head is actually higher than head
135  if( new_head->data.block_num() > head_block_num() )
136  {
137  wlog( "Switching to fork: ${id}", ("id",new_head->data.id()) );
138  auto branches = _fork_db.fetch_branch_from(new_head->data.id(), head_block_id());
139 
140  // pop blocks until we hit the forked block
141  while( head_block_id() != branches.second.back()->data.previous )
142  pop_block();
143 
144  // push all blocks on the new fork
145  for( auto ritr = branches.first.rbegin(); ritr != branches.first.rend(); ++ritr )
146  {
147  ilog( "pushing blocks from fork ${n} ${id}", ("n",(*ritr)->data.block_num())("id",(*ritr)->data.id()) );
148  optional<fc::exception> except;
149  try {
150  undo_database::session session = _undo_db.start_undo_session();
151  apply_block( (*ritr)->data, skip );
152  _block_id_to_block.store( (*ritr)->id, (*ritr)->data );
153  session.commit();
154  }
155  catch ( const fc::exception& e ) { except = e; }
156  if( except )
157  {
158  wlog( "exception thrown while switching forks ${e}", ("e",except->to_detail_string() ) );
159  // remove the rest of branches.first from the fork_db, those blocks are invalid
160  while( ritr != branches.first.rend() )
161  {
162  _fork_db.remove( (*ritr)->data.id() );
163  ++ritr;
164  }
165  _fork_db.set_head( branches.second.front() );
166 
167  // pop all blocks from the bad fork
168  while( head_block_id() != branches.second.back()->data.previous )
169  pop_block();
170 
171  // restore all blocks from the good fork
172  for( auto ritr = branches.second.rbegin(); ritr != branches.second.rend(); ++ritr )
173  {
174  auto session = _undo_db.start_undo_session();
175  apply_block( (*ritr)->data, skip );
176  _block_id_to_block.store( new_block.id(), (*ritr)->data );
177  session.commit();
178  }
179  throw *except;
180  }
181  }
182  return true;
183  }
184  else return false;
185  }
186  }
187 
188  try {
189  auto session = _undo_db.start_undo_session();
190  apply_block(new_block, skip);
191  _block_id_to_block.store(new_block.id(), new_block);
192  session.commit();
193  } catch ( const fc::exception& e ) {
194  elog("Failed to push new block:\n${e}", ("e", e.to_detail_string()));
195  _fork_db.remove(new_block.id());
196  throw;
197  }
198 
199  return false;
200 } FC_CAPTURE_AND_RETHROW( (new_block) ) }

221 processed_transaction database::_push_transaction( const signed_transaction& trx )
222 {
223  // If this is the first transaction pushed after applying a block, start a new undo session.
224  // This allows us to quickly rewind to the clean state of the head block, in case a new block arrives.
225  if( !_pending_tx_session.valid() )
226  _pending_tx_session = _undo_db.start_undo_session();
227 
228  // Create a temporary undo session as a child of _pending_tx_session.
229  // The temporary session will be discarded by the destructor if
230  // _apply_transaction fails. If we make it to merge(), we
231  // apply the changes.
232 
233  auto temp_session = _undo_db.start_undo_session();
234  auto processed_trx = _apply_transaction( trx );
235  _pending_tx.push_back(processed_trx);
236 
237  notify_changed_objects();
238  // The transaction applied successfully. Merge its changes into the pending block session.
239  temp_session.merge();
240 
241  // notify anyone listening to pending transactions
242  on_pending_transaction( trx );
243  return processed_trx;
244 }

291 signed_block database::_generate_block(
292  fc::time_point_sec when,
293  witness_id_type witness_id,
294  const fc::ecc::private_key& block_signing_private_key
295  )
296 {
297  try {
298  uint32_t skip = get_node_properties().skip_flags;
299  uint32_t slot_num = get_slot_at_time( when );
300  FC_ASSERT( slot_num > 0 );
301  witness_id_type scheduled_witness = get_scheduled_witness( slot_num );
302  FC_ASSERT( scheduled_witness == witness_id );
303 
304  const auto& witness_obj = witness_id(*this);
305 
306  if( !(skip & skip_witness_signature) )
307  FC_ASSERT( witness_obj.signing_key == block_signing_private_key.get_public_key() );
308 
309  static const size_t max_block_header_size = fc::raw::pack_size( signed_block_header() ) + 4;
310  auto maximum_block_size = get_global_properties().parameters.maximum_block_size;
311  size_t total_block_size = max_block_header_size;
312 
313  signed_block pending_block;
314 
315  //
316  // The following code throws away existing pending_tx_session and
317  // rebuilds it by re-applying pending transactions.
318  //
319  // This rebuild is necessary because pending transactions' validity
320  // and semantics may have changed since they were received, because
321  // time-based semantics are evaluated based on the current block
322  // time. These changes can only be reflected in the database when
323  // the value of the "when" variable is known, which means we need to
324  // re-apply pending transactions in this method.
325  //
326  _pending_tx_session.reset();
327  _pending_tx_session = _undo_db.start_undo_session();
328 
329  uint64_t postponed_tx_count = 0;
330  // pop pending state (reset to head block state)
331  for( const processed_transaction& tx : _pending_tx )
332  {
333  size_t new_total_size = total_block_size + fc::raw::pack_size( tx );
334 
335  // postpone transaction if it would make block too big
336  if( new_total_size >= maximum_block_size )
337  {
338  postponed_tx_count++;
339  continue;
340  }
341 
342  try
343  {
344  auto temp_session = _undo_db.start_undo_session();
345  processed_transaction ptx = _apply_transaction( tx );
346  temp_session.merge();
347 
348  // We have to recompute pack_size(ptx) because it may be different
349  // than pack_size(tx) (i.e. if one or more results increased
350  // their size)
351  total_block_size += fc::raw::pack_size( ptx );
352  pending_block.transactions.push_back( ptx );
353  }
354  catch ( const fc::exception& e )
355  {
356  // Do nothing, transaction will not be re-applied
357  wlog( "Transaction was not processed while generating block due to ${e}", ("e", e) );
358  wlog( "The transaction was ${t}", ("t", tx) );
359  }
360  }
361  if( postponed_tx_count > 0 )
362  {
363  wlog( "Postponed ${n} transactions due to block size limit", ("n", postponed_tx_count) );
364  }
365 
366  _pending_tx_session.reset();
367 
368  // We have temporarily broken the invariant that
369  // _pending_tx_session is the result of applying _pending_tx, as
370  // _pending_tx now consists of the set of postponed transactions.
371  // However, the push_block() call below will re-create the
372  // _pending_tx_session.
373 
374  pending_block.previous = head_block_id();
375  pending_block.timestamp = when;
376  pending_block.transaction_merkle_root = pending_block.calculate_merkle_root();
377  pending_block.witness = witness_id;
378 
379  if( !(skip & skip_witness_signature) )
380  pending_block.sign( block_signing_private_key );
381 
382  // TODO: Move this to _push_block() so session is restored.
383  if( !(skip & skip_block_size_check) )
384  {
385  FC_ASSERT( fc::raw::pack_size(pending_block) <= get_global_properties().parameters.maximum_block_size );
386  }
387 
388  push_block( pending_block, skip );
389 
390  return pending_block;
391 } FC_CAPTURE_AND_RETHROW( (witness_id) ) }

461 void database::_apply_block( const signed_block& next_block )
462 { try {
463  uint32_t next_block_num = next_block.block_num();
464  uint32_t skip = get_node_properties().skip_flags;
465  _applied_ops.clear();
466 
467  FC_ASSERT( (skip & skip_merkle_check) || next_block.transaction_merkle_root == next_block.calculate_merkle_root(), "", ("next_block.transaction_merkle_root",next_block.transaction_merkle_root)("calc",next_block.calculate_merkle_root())("next_block",next_block)("id",next_block.id()) );
468 
469  const witness_object& signing_witness = validate_block_header(skip, next_block);
470  const auto& global_props = get_global_properties();
471  const auto& dynamic_global_props = get<dynamic_global_property_object>(dynamic_global_property_id_type());
472  bool maint_needed = (dynamic_global_props.next_maintenance_time <= next_block.timestamp);
473 
474  _current_block_num = next_block_num;
475  _current_trx_in_block = 0;
476 
477  for( const auto& trx : next_block.transactions )
478  {
479  /* We do not need to push the undo state for each transaction
480  * because they either all apply and are valid or the
481  * entire block fails to apply. We only need an "undo" state
482  * for transactions when validating broadcast transactions or
483  * when building a block.
484  */
485  apply_transaction( trx, skip | skip_transaction_signatures );
486  ++_current_trx_in_block;
487  }
488 
489  update_global_dynamic_data(next_block);
490  update_signing_witness(signing_witness, next_block);
491  update_last_irreversible_block();
492 
493  // Are we at the maintenance interval?
494  if( maint_needed )
495  perform_chain_maintenance(next_block, global_props);
496 
497  create_block_summary(next_block);
498  clear_expired_transactions();
499  clear_expired_proposals();
500  clear_expired_orders();
501  update_expired_feeds();
502  update_withdraw_permissions();
503 
504  // n.b., update_maintenance_flag() happens this late
505  // because get_slot_time() / get_slot_at_time() is needed above
506  // TODO: figure out if we could collapse this function into
507  // update_global_dynamic_data() as perhaps these methods only need
508  // to be called for header validation?
509  update_maintenance_flag( maint_needed );
510  update_witness_schedule();
511 
512  // notify observers that the block has been applied
513  applied_block( next_block ); //emit
514  _applied_ops.clear();
515 
516  notify_changed_objects();
517 } FC_CAPTURE_AND_RETHROW( (next_block.block_num()) ) }

548 processed_transaction database::_apply_transaction(const signed_transaction& trx)
549 { try {
550  uint32_t skip = get_node_properties().skip_flags;
551  trx.validate();
552  auto& trx_idx = get_mutable_index_type<transaction_index>();
553  const chain_id_type& chain_id = get_chain_id();
554  auto trx_id = trx.id();
555  FC_ASSERT( (skip & skip_transaction_dupe_check) ||
556  trx_idx.indices().get<by_trx_id>().find(trx_id) == trx_idx.indices().get<by_trx_id>().end() );
557  transaction_evaluation_state eval_state(this);
558  const chain_parameters& chain_parameters = get_global_properties().parameters;
559  eval_state._trx = &trx;
560 
561  if( !(skip & (skip_transaction_signatures | skip_authority_check) ) )
562  {
563  auto get_active = [&]( account_id_type id ) { return &id(*this).active; };
564  auto get_owner = [&]( account_id_type id ) { return &id(*this).owner; };
565  trx.verify_authority( chain_id, get_active, get_owner, get_global_properties().parameters.max_authority_depth );
566  }
567 
568  //Skip all manner of expiration and TaPoS checking if we're on block 1; It's impossible that the transaction is
569  //expired, and TaPoS makes no sense as no blocks exist.
570  if( BOOST_LIKELY(head_block_num() > 0) )
571  {
572  if( !(skip & skip_tapos_check) )
573  {
574  const auto& tapos_block_summary = block_summary_id_type( trx.ref_block_num )(*this);
575 
576  //Verify TaPoS block summary has correct ID prefix, and that this block's time is not past the expiration
577  FC_ASSERT( trx.ref_block_prefix == tapos_block_summary.block_id._hash[1] );
578  }
579 
580  fc::time_point_sec now = head_block_time();
581 
582  FC_ASSERT( trx.expiration <= now + chain_parameters.maximum_time_until_expiration, "",
583  ("trx.expiration",trx.expiration)("now",now)("max_til_exp",chain_parameters.maximum_time_until_expiration));
584  FC_ASSERT( now <= trx.expiration, "", ("now",now)("trx.exp",trx.expiration) );
585  }
586 
587  //Insert transaction into unique transactions database.
588  if( !(skip & skip_transaction_dupe_check) )
589  {
590  create<transaction_object>([&](transaction_object& transaction) {
591  transaction.trx_id = trx_id;
592  transaction.trx = trx;
593  });
594  }
595 
596  eval_state.operation_results.reserve(trx.operations.size());
597 
598  //Finally process the operations
599  processed_transaction ptrx(trx);
600  _current_op_in_trx = 0;

619  uint64_t u_which = uint64_t( i_which );
