91 bool database::push_block( const signed_block& new_block, uint32_t skip )
92 { try {
93  if( !(skip&skip_fork_db) )
94  {
95  wdump((new_block.id())(new_block.previous));
96  auto new_head = _fork_db.push_block( new_block );
97  //If the head block from the longest chain does not build off of the current head, we need to switch forks.
98  if( new_head->data.previous != head_block_id() )
99  {
100  edump((new_head->data.previous));
101  //If the newly pushed block is the same height as head, we get head back in new_head
102  //Only switch forks if new_head is actually higher than head
103  if( new_head->data.block_num() > head_block_num() )
104  {
105  auto branches = _fork_db.fetch_branch_from( new_head->data.id(), _pending_block.previous );
106  for( auto item : branches.first )
107  {
108  wdump( ("new")(item->id)(item->data.previous) );
109  }
110  for( auto item : branches.second )
111  {
112  wdump( ("old")(item->id)(item->data.previous) );
113  }
114 
115  // pop blocks until we hit the forked block
116  while( head_block_id() != branches.second.back()->data.previous )
117  pop_block();
118 
119  // push all blocks on the new fork
120  for( auto ritr = branches.first.rbegin(); ritr != branches.first.rend(); ++ritr )
121  {
122  optional<fc::exception> except;
123  try {
124  auto session = _undo_db.start_undo_session();
125  apply_block( (*ritr)->data, skip );
126  _block_id_to_block.store( new_block.id(), (*ritr)->data );
127  session.commit();
128  }
129  catch ( const fc::exception& e ) { except = e; }
130  if( except )
131  {
132  elog( "Encountered error when switching to a longer fork at id ${id}. Going back.",
133  ("id", (*ritr)->id) );
134  // remove the rest of branches.first from the fork_db, those blocks are invalid
135  while( ritr != branches.first.rend() )
136  {
137  _fork_db.remove( (*ritr)->data.id() );
138  ++ritr;
139  }
140  _fork_db.set_head( branches.second.front() );
141 
142  // pop all blocks from the bad fork
143  while( head_block_id() != branches.second.back()->data.previous )
144  pop_block();
145 
146  // restore all blocks from the good fork
147  for( auto ritr = branches.second.rbegin(); ritr != branches.second.rend(); ++ritr )
148  {
149  auto session = _undo_db.start_undo_session();
150  apply_block( (*ritr)->data, skip );
151  _block_id_to_block.store( new_block.id(), (*ritr)->data );
152  session.commit();
153  }
154  throw *except;
155  }
156  }
157  return true;
158  }
159  else return false;
160  }
161  }
162 
163  // If there is a pending block session, then the database state is dirty with pending transactions.
164  // Drop the pending session to reset the database to a clean head block state.
165  // TODO: Preserve pending transactions, and re-apply any which weren't included in the new block.
166  clear_pending();
167 
168  try {
169  auto session = _undo_db.start_undo_session();
170  apply_block( new_block, skip );
171  _block_id_to_block.store( new_block.id(), new_block );
172  session.commit();
173  } catch ( const fc::exception& e ) {
174  elog("Failed to push new block:\n${e}", ("e", e.to_detail_string()));
175  _fork_db.remove(new_block.id());
176  throw;
177  }
178 
179  return false;
180 } FC_CAPTURE_AND_RETHROW( (new_block) ) }

191 processed_transaction database::push_transaction( const signed_transaction& trx, uint32_t skip )
192 {
193  //wdump((trx.digest())(trx.id()));
194  // If this is the first transaction pushed after applying a block, start a new undo session.
195  // This allows us to quickly rewind to the clean state of the head block, in case a new block arrives.
196  if( !_pending_block_session ) _pending_block_session = _undo_db.start_undo_session();
197  auto session = _undo_db.start_undo_session();
198  auto processed_trx = apply_transaction( trx, skip );
199  _pending_block.transactions.push_back(processed_trx);
200 
201  FC_ASSERT( (skip & skip_block_size_check) ||
202  fc::raw::pack_size(_pending_block) <= get_global_properties().parameters.maximum_block_size );
203 
204  // The transaction applied successfully. Merge its changes into the pending block session.
205  session.merge();
206  return processed_trx;

209 processed_transaction database::push_proposal(const proposal_object& proposal)
210 {
211  transaction_evaluation_state eval_state(this);
212  eval_state._is_proposed_trx = true;
213 
214  //Inject the approving authorities into the transaction eval state
215  std::transform(proposal.required_active_approvals.begin(),
216  proposal.required_active_approvals.end(),
217  std::inserter(eval_state.approved_by, eval_state.approved_by.begin()),
218  []( account_id_type id ) {
219  return std::make_pair(id, authority::active);
220  });
221  std::transform(proposal.required_owner_approvals.begin(),
222  proposal.required_owner_approvals.end(),
223  std::inserter(eval_state.approved_by, eval_state.approved_by.begin()),
224  []( account_id_type id ) {
225  return std::make_pair(id, authority::owner);
226  });
227 
228  ilog("Attempting to push proposal ${prop}", ("prop", proposal));
229  idump((eval_state.approved_by));
230 
231  eval_state.operation_results.reserve(proposal.proposed_transaction.operations.size());
232  processed_transaction ptrx(proposal.proposed_transaction);
233  eval_state._trx = &ptrx;
234 
235  auto session = _undo_db.start_undo_session();
236  for( auto& op : proposal.proposed_transaction.operations )
237  eval_state.operation_results.emplace_back(apply_operation(eval_state, op));
238  remove(proposal);
239  session.merge();
240 
241  ptrx.operation_results = std::move(eval_state.operation_results);
242  return ptrx;
243 }

332 void database::apply_block( const signed_block& next_block, uint32_t skip )
333 { try {
334  _applied_ops.clear();
335 
336  FC_ASSERT( (skip & skip_merkle_check) || next_block.transaction_merkle_root == next_block.calculate_merkle_root() );
337 
338  const witness_object& signing_witness = validate_block_header(skip, next_block);
339  const auto& global_props = get_global_properties();
340  const auto& dynamic_global_props = get<dynamic_global_property_object>(dynamic_global_property_id_type());
341 
342  _current_block_num = next_block.block_num();
343  _current_trx_in_block = 0;
344 
345  for( const auto& trx : next_block.transactions )
346  {
347  /* We do not need to push the undo state for each transaction
348  * because they either all apply and are valid or the
349  * entire block fails to apply. We only need an "undo" state
350  * for transactions when validating broadcast transactions or
351  * when building a block.
352  */
353  apply_transaction( trx, skip | skip_transaction_signatures );
354  ++_current_trx_in_block;
355  }
356 
357  update_witness_schedule(next_block);
358  update_global_dynamic_data(next_block);
359  update_signing_witness(signing_witness, next_block);
360 
361  auto current_block_interval = global_props.parameters.block_interval;
362 
363  // Are we at the maintenance interval?
364  if( dynamic_global_props.next_maintenance_time <= next_block.timestamp )
365  // This will update _pending_block.timestamp if the block interval has changed
366  perform_chain_maintenance(next_block, global_props);
367 
368  create_block_summary(next_block);
369  clear_expired_transactions();
370  clear_expired_proposals();
371  clear_expired_orders();
372  update_expired_feeds();
373  update_withdraw_permissions();
374 
375  // notify observers that the block has been applied
376  applied_block( next_block ); //emit
377  _applied_ops.clear();
378 
379  const auto& head_undo = _undo_db.head();
380  vector<object_id_type> changed_ids; changed_ids.reserve(head_undo.old_values.size());
381  for( const auto& item : head_undo.old_values ) changed_ids.push_back(item.first);
382  changed_objects(changed_ids);
383 
384 
385  update_pending_block(next_block, current_block_interval);
386 } FC_CAPTURE_AND_RETHROW( (next_block.block_num())(skip) ) }

388 processed_transaction database::apply_transaction( const signed_transaction& trx, uint32_t skip )
389 { try {
390  trx.validate();
391  auto& trx_idx = get_mutable_index_type<transaction_index>();
392  auto trx_id = trx.id();
393  FC_ASSERT( (skip & skip_transaction_dupe_check) ||
394  trx_idx.indices().get<by_trx_id>().find(trx_id) == trx_idx.indices().get<by_trx_id>().end() );
395  transaction_evaluation_state eval_state(this, skip&skip_authority_check );
396  const chain_parameters& chain_parameters = get_global_properties().parameters;
397  eval_state._trx = &trx;
398 
399  //This check is used only if this transaction has an absolute expiration time.
400  if( !(skip & skip_transaction_signatures) && trx.relative_expiration == 0 )
401  {
402  for( const auto& sig : trx.signatures )
403  {
404  FC_ASSERT( sig.first(*this).key_address() == fc::ecc::public_key( sig.second, trx.digest() ), "",
405  ("trx",trx)
406  ("digest",trx.digest())
407  ("sig.first",sig.first)
408  ("key_address",sig.first(*this).key_address())
409  ("addr", address(fc::ecc::public_key( sig.second, trx.digest() ))) );
410  }
411  }
412 
413  //If we're skipping tapos check, but not dupe check, assume all transactions have maximum expiration time.
414  fc::time_point_sec trx_expiration = _pending_block.timestamp + chain_parameters.maximum_time_until_expiration;
415 
416  //Skip all manner of expiration and TaPoS checking if we're on block 1; It's impossible that the transaction is
417  //expired, and TaPoS makes no sense as no blocks exist.
418  if( BOOST_LIKELY(head_block_num() > 0) )
419  {
420  if( !(skip & skip_tapos_check) && trx.relative_expiration != 0 )
421  {
422  //Check the TaPoS reference and expiration time
423  //Remember that the TaPoS block number is abbreviated; it contains only the lower 16 bits.
424  //Lookup TaPoS block summary by block number (remember block summary instances are the block numbers)
425  const block_summary_object& tapos_block_summary
426  = static_cast<const block_summary_object&>(get_index<block_summary_object>()
427  .get(block_summary_id_type((head_block_num() & ~0xffff)
428  + trx.ref_block_num)));
429 
430  //This is the signature check for transactions with relative expiration.
431  if( !(skip & skip_transaction_signatures) )
432  {
433  for( const auto& sig : trx.signatures )
434  {
435  address trx_addr = fc::ecc::public_key(sig.second, trx.digest(tapos_block_summary.block_id));
436  FC_ASSERT(sig.first(*this).key_address() == trx_addr,
437  "",
438  ("sig.first",sig.first)
439  ("key_address",sig.first(*this).key_address())
440  ("addr", trx_addr));
441  }
442  }
443 
444  //Verify TaPoS block summary has correct ID prefix, and that this block's time is not past the expiration
445  FC_ASSERT( trx.ref_block_prefix == tapos_block_summary.block_id._hash[1] );
446  trx_expiration = tapos_block_summary.timestamp + chain_parameters.block_interval*trx.relative_expiration;
447  } else if( trx.relative_expiration == 0 ) {
448  trx_expiration = fc::time_point_sec(trx.ref_block_prefix);
449  FC_ASSERT( trx_expiration <= _pending_block.timestamp + chain_parameters.maximum_time_until_expiration );
450  }
451  FC_ASSERT( _pending_block.timestamp <= trx_expiration );
452  } else if( !(skip & skip_transaction_signatures) ) {
453  FC_ASSERT(trx.relative_expiration == 0, "May not use transactions with a reference block in block 1!");
454  }
455 
456  //Insert transaction into unique transactions database.
457  if( !(skip & skip_transaction_dupe_check) )
458  {
459  create<transaction_object>([&](transaction_object& transaction) {
460  transaction.expiration = trx_expiration;
461  transaction.trx_id = trx_id;
462  transaction.trx = trx;
463  });
464  }
465 
466  eval_state.operation_results.reserve( trx.operations.size() );
467 
468  processed_transaction ptrx(trx);
469  _current_op_in_trx = 0;
470  for( const auto& op : ptrx.operations )
471  {
472  eval_state.operation_results.emplace_back(apply_operation(eval_state, op));
473  ++_current_op_in_trx;
474  }
475  ptrx.operation_results = std::move( eval_state.operation_results );
476 
477  return ptrx;
478 } FC_CAPTURE_AND_RETHROW( (trx) ) }

480 operation_result database::apply_operation(transaction_evaluation_state& eval_state, const operation& op)
481 {
482  int i_which = op.which();
483  uint64_t u_which = uint64_t( i_which );
484  if( i_which < 0 )
485  assert( "Negative operation tag" && false );
486  if( u_which >= _operation_evaluators.size() )
487  assert( "No registered evaluator for this operation" && false );
488  unique_ptr<op_evaluator>& eval = _operation_evaluators[ u_which ];
489  if( !eval )
490  assert( "No registered evaluator for this operation" && false );
491  auto op_id = push_applied_operation( op );
492  auto result = eval->evaluate( eval_state, op, true );
493  set_applied_operation_result( op_id, result );
494  return result;
495 }
