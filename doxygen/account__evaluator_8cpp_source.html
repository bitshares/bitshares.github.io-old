60 void verify_account_votes( const database& db, const account_options& options )
61 {
62  // ensure account's votes satisfy requirements
63  // NB only the part of vote checking that requires chain state is here,
64  // the rest occurs in account_options::validate()
65 
66  const auto& gpo = db.get_global_properties();
67  const auto& chain_params = gpo.parameters;
68 
69  FC_ASSERT( options.num_witness <= chain_params.maximum_witness_count,
70  "Voted for more witnesses than currently allowed (${c})", ("c", chain_params.maximum_witness_count) );
71  FC_ASSERT( options.num_committee <= chain_params.maximum_committee_count,
72  "Voted for more committee members than currently allowed (${c})", ("c", chain_params.maximum_committee_count) );
73 
74  uint32_t max_vote_id = gpo.next_available_vote_id;
75  bool has_worker_votes = false;
76  for( auto id : options.votes )
77  {
78  FC_ASSERT( id < max_vote_id, "Can not vote for ${id} which does not exist.", ("id",id) );
79  has_worker_votes |= (id.type() == vote_id_type::worker);
80  }
81 
82  if( has_worker_votes && (db.head_block_time() >= HARDFORK_607_TIME) )
83  {
84  const auto& against_worker_idx = db.get_index_type<worker_index>().indices().get<by_vote_against>();
85  for( auto id : options.votes )
86  {
87  if( id.type() == vote_id_type::worker )
88  {
89  FC_ASSERT( against_worker_idx.find( id ) == against_worker_idx.end(),
90  "Can no longer vote against a worker." );
91  }
92  }
93  }
94  if ( db.head_block_time() >= HARDFORK_CORE_143_TIME ) {
95  const auto& approve_worker_idx = db.get_index_type<worker_index>().indices().get<by_vote_for>();
96  const auto& committee_idx = db.get_index_type<committee_member_index>().indices().get<by_vote_id>();
97  const auto& witness_idx = db.get_index_type<witness_index>().indices().get<by_vote_id>();
98  for ( auto id : options.votes ) {
99  switch ( id.type() ) {
100  case vote_id_type::committee:
101  FC_ASSERT( committee_idx.find(id) != committee_idx.end(),
102  "Can not vote for ${id} which does not exist.", ("id",id) );
103  break;
104  case vote_id_type::witness:
105  FC_ASSERT( witness_idx.find(id) != witness_idx.end(),
106  "Can not vote for ${id} which does not exist.", ("id",id) );
107  break;
108  case vote_id_type::worker:
109  FC_ASSERT( approve_worker_idx.find( id ) != approve_worker_idx.end(),
110  "Can not vote for ${id} which does not exist.", ("id",id) );
111  break;
112  default:
113  FC_THROW( "Invalid Vote Type: ${id}", ("id", id) );
114  break;
115  }
116  }
117  }

121 void_result account_create_evaluator::do_evaluate( const account_create_operation& op )
122 { try {
123  database& d = db();
124  if( d.head_block_time() < HARDFORK_516_TIME )
125  {
126  FC_ASSERT( !op.extensions.value.owner_special_authority.valid() );
127  FC_ASSERT( !op.extensions.value.active_special_authority.valid() );
128  }
129  if( d.head_block_time() < HARDFORK_599_TIME )
130  {
131  FC_ASSERT( !op.extensions.value.null_ext.valid() );
132  FC_ASSERT( !op.extensions.value.owner_special_authority.valid() );
133  FC_ASSERT( !op.extensions.value.active_special_authority.valid() );
134  FC_ASSERT( !op.extensions.value.buyback_options.valid() );
135  }
136 
137  FC_ASSERT( d.find_object(op.options.voting_account), "Invalid proxy account specified." );
138  FC_ASSERT( fee_paying_account->is_lifetime_member(), "Only Lifetime members may register an account." );
139  FC_ASSERT( op.referrer(d).is_member(d.head_block_time()), "The referrer must be either a lifetime or annual subscriber." );
140 
141  try
142  {
143  verify_authority_accounts( d, op.owner );
144  verify_authority_accounts( d, op.active );
145  }
146  GRAPHENE_RECODE_EXC( internal_verify_auth_max_auth_exceeded, account_create_max_auth_exceeded )
147  GRAPHENE_RECODE_EXC( internal_verify_auth_account_not_found, account_create_auth_account_not_found )
148 
149  if( op.extensions.value.owner_special_authority.valid() )
150  evaluate_special_authority( d, *op.extensions.value.owner_special_authority );
151  if( op.extensions.value.active_special_authority.valid() )
152  evaluate_special_authority( d, *op.extensions.value.active_special_authority );
153  if( op.extensions.value.buyback_options.valid() )
154  evaluate_buyback_account_options( d, *op.extensions.value.buyback_options );
155  verify_account_votes( d, op.options );
156 
157  auto& acnt_indx = d.get_index_type<account_index>();
158  if( op.name.size() )
159  {
160  auto current_account_itr = acnt_indx.indices().get<by_name>().find( op.name );
161  FC_ASSERT( current_account_itr == acnt_indx.indices().get<by_name>().end(),
162  "Account '${a}' already exists.", ("a",op.name) );
163  }
164 
165  return void_result();
166 } FC_CAPTURE_AND_RETHROW( (op) ) }

306 { try {

309  d.modify( *acnt, [&](account_object& a){

322  if( o.extensions.value.owner_special_authority.valid() )

324  a.owner_special_authority = *(o.extensions.value.owner_special_authority);
325  a.top_n_control_flags = 0;
326  }
327  if( o.extensions.value.active_special_authority.valid() )
328  {
329  a.active_special_authority = *(o.extensions.value.active_special_authority);
330  a.top_n_control_flags = 0;
331  }
332  sa_after = a.has_special_authority();
333  });
334 
335  if( sa_before && (!sa_after) )
336  {
337  const auto& sa_idx = d.get_index_type< special_authority_index >().indices().get<by_account>();
338  auto sa_it = sa_idx.find( o.account );
339  assert( sa_it != sa_idx.end() );
340  d.remove( *sa_it );
341  }
342  else if( (!sa_before) && sa_after )
343  {
344  d.create< special_authority_object >( [&]( special_authority_object& sa )
345  {
346  sa.account = o.account;
347  } );
348  }
349 
350  return void_result();
351 } FC_CAPTURE_AND_RETHROW( (o) ) }

407 void_result account_upgrade_evaluator::do_apply(const account_upgrade_evaluator::operation_type& o)
408 { try {
409  database& d = db();
410 
411  d.modify(*account, [&](account_object& a) {
412  if( o.upgrade_to_lifetime_member )
413  {
414  // Upgrade to lifetime member. I don't care what the account was before.
415  a.statistics(d).process_fees(a, d);
416  a.membership_expiration_date = time_point_sec::maximum();
417  a.referrer = a.registrar = a.lifetime_referrer = a.get_id();
418  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT - a.network_fee_percentage;
419  } else if( a.is_annual_member(d.head_block_time()) ) {
420  // Renew an annual subscription that's still in effect.
421  FC_ASSERT( d.head_block_time() <= HARDFORK_613_TIME );
422  FC_ASSERT(a.membership_expiration_date - d.head_block_time() < fc::days(3650),
423  "May not extend annual membership more than a decade into the future.");
424  a.membership_expiration_date += fc::days(365);
425  } else {
426  // Upgrade from basic account.
427  FC_ASSERT( d.head_block_time() <= HARDFORK_613_TIME );
428  a.statistics(d).process_fees(a, d);
429  assert(a.is_basic_account(d.head_block_time()));
430  a.referrer = a.get_id();
431  a.membership_expiration_date = d.head_block_time() + fc::days(365);
432  }
433  });
434 
435  return {};
436 } FC_RETHROW_EXCEPTIONS( error, "Unable to upgrade account '${a}'", ("a",o.account_to_upgrade(db()).name) ) }
