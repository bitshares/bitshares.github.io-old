47 void_result account_create_evaluator::do_evaluate( const account_create_operation& op )
48 { try {
49  database& d = db();
50  FC_ASSERT( d.find_object(op.options.voting_account), "Invalid proxy account specified." );
51  FC_ASSERT( fee_paying_account->is_lifetime_member(), "Only Lifetime members may register an account." );
52  FC_ASSERT( op.referrer(d).is_member(d.head_block_time()), "The referrer must be either a lifetime or annual subscriber." );
53 
54  const auto& global_props = d.get_global_properties();
55  const auto& chain_params = global_props.parameters;
56 
57  try
58  {
59  verify_authority_accounts( d, op.owner );
60  verify_authority_accounts( d, op.active );
61  }
62  GRAPHENE_RECODE_EXC( internal_verify_auth_max_auth_exceeded, account_create_max_auth_exceeded )
63  GRAPHENE_RECODE_EXC( internal_verify_auth_account_not_found, account_create_auth_account_not_found )
64 
65  uint32_t max_vote_id = global_props.next_available_vote_id;
66 
67  FC_ASSERT( op.options.num_witness <= chain_params.maximum_witness_count,
68  "Voted for more witnesses than currently allowed (${c})", ("c", chain_params.maximum_witness_count) );
69 
70  FC_ASSERT( op.options.num_committee <= chain_params.maximum_committee_count,
71  "Voted for more committee members than currently allowed (${c})", ("c", chain_params.maximum_committee_count) );
72 
73  safe<uint32_t> counts[vote_id_type::VOTE_TYPE_COUNT];
74  for( auto id : op.options.votes )
75  {
76  FC_ASSERT( id < max_vote_id );
77  counts[id.type()]++;
78  }
79  FC_ASSERT(counts[vote_id_type::witness] <= op.options.num_witness,
80  "",
81  ("count", counts[vote_id_type::witness])("num", op.options.num_witness));
82  FC_ASSERT(counts[vote_id_type::committee] <= op.options.num_committee,
83  "",
84  ("count", counts[vote_id_type::committee])("num", op.options.num_committee));
85 
86  auto& acnt_indx = d.get_index_type<account_index>();
87  if( op.name.size() )
88  {
89  auto current_account_itr = acnt_indx.indices().get<by_name>().find( op.name );
90  FC_ASSERT( current_account_itr == acnt_indx.indices().get<by_name>().end() );
91  }
92 
93  return void_result();
94 } FC_CAPTURE_AND_RETHROW( (op) ) }

110  obj.active = o.active;

113  });
114 
115  const auto& dynamic_properties = db().get_dynamic_global_properties();
116  db().modify(dynamic_properties, [](dynamic_global_property_object& p) {
117  ++p.accounts_registered_this_interval;
118  });
119 
120  const auto& global_properties = db().get_global_properties();
121  if( dynamic_properties.accounts_registered_this_interval %
122  global_properties.parameters.accounts_per_fee_scale == 0 )
123  db().modify(global_properties, [&dynamic_properties](global_property_object& p) {
124  p.parameters.current_fees->get<account_create_operation>().basic_fee <<= p.parameters.account_fee_scale_bitshifts;
125  });

127  return new_acnt_object.id;

225 void_result account_upgrade_evaluator::do_apply(const account_upgrade_evaluator::operation_type& o)
226 { try {
227  database& d = db();
228 
229  d.modify(*account, [&](account_object& a) {
230  if( o.upgrade_to_lifetime_member )
231  {
232  // Upgrade to lifetime member. I don't care what the account was before.
233  a.statistics(d).process_fees(a, d);
234  a.membership_expiration_date = time_point_sec::maximum();
235  a.referrer = a.registrar = a.lifetime_referrer = a.get_id();
236  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT - a.network_fee_percentage;
237  } else if( a.is_annual_member(d.head_block_time()) ) {
238  // Renew an annual subscription that's still in effect.
239  FC_ASSERT(a.membership_expiration_date - d.head_block_time() < fc::days(3650),
240  "May not extend annual membership more than a decade into the future.");
241  a.membership_expiration_date += fc::days(365);
242  } else {
243  // Upgrade from basic account.
244  a.statistics(d).process_fees(a, d);
245  assert(a.is_basic_account(d.head_block_time()));
246  a.membership_expiration_date = d.head_block_time() + fc::days(365);
247  }
248  });
249 
250  return {};
251 } FC_RETHROW_EXCEPTIONS( error, "Unable to upgrade account '${a}'", ("a",o.account_to_upgrade(db()).name) ) }
