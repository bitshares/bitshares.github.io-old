51 template<class Index>
52 vector<std::reference_wrapper<const typename Index::object_type>> database::sort_votable_objects(size_t count) const
53 {
54  using ObjectType = typename Index::object_type;
55  const auto& all_objects = get_index_type<Index>().indices();
56  count = std::min(count, all_objects.size());
57  vector<std::reference_wrapper<const ObjectType>> refs;
58  refs.reserve(all_objects.size());
59  std::transform(all_objects.begin(), all_objects.end(),
60  std::back_inserter(refs),
61  [](const ObjectType& o) { return std::cref(o); });
62  std::partial_sort(refs.begin(), refs.begin() + count, refs.end(),
63  [this](const ObjectType& a, const ObjectType& b)->bool {
64  share_type oa_vote = _vote_tally_buffer[a.vote_id];
65  share_type ob_vote = _vote_tally_buffer[b.vote_id];
66  if( oa_vote != ob_vote )
67  return oa_vote > ob_vote;
68  return a.vote_id < b.vote_id;
69  });
70 
71  refs.resize(count, refs.front());
72  return refs;
73 }

116 void database::pay_workers( share_type& budget )
117 {
118 // ilog("Processing payroll! Available budget is ${b}", ("b", budget));
119  vector<std::reference_wrapper<const worker_object>> active_workers;
120  get_index_type<worker_index>().inspect_all_objects([this, &active_workers](const object& o) {
121  const worker_object& w = static_cast<const worker_object&>(o);
122  auto now = head_block_time();
123  if( w.is_active(now) && w.approving_stake() > 0 )
124  active_workers.emplace_back(w);
125  });
126 
127  // worker with more votes is preferred
128  // if two workers exactly tie for votes, worker with lower ID is preferred
129  std::sort(active_workers.begin(), active_workers.end(), [](const worker_object& wa, const worker_object& wb) {
130  share_type wa_vote = wa.approving_stake();
131  share_type wb_vote = wb.approving_stake();
132  if( wa_vote != wb_vote )
133  return wa_vote > wb_vote;
134  return wa.id < wb.id;
135  });
136 
137  for( uint32_t i = 0; i < active_workers.size() && budget > 0; ++i )
138  {
139  const worker_object& active_worker = active_workers[i];
140  share_type requested_pay = active_worker.daily_pay;
141  if( head_block_time() - get_dynamic_global_properties().last_budget_time != fc::days(1) )
142  {
143  fc::uint128 pay(requested_pay.value);
144  pay *= (head_block_time() - get_dynamic_global_properties().last_budget_time).count();
145  pay /= fc::days(1).count();
146  requested_pay = pay.to_uint64();
147  }
148 
149  share_type actual_pay = std::min(budget, requested_pay);
150  //ilog(" ==> Paying ${a} to worker ${w}", ("w", active_worker.id)("a", actual_pay));
151  modify(active_worker, [&](worker_object& w) {
152  w.worker.visit(worker_pay_visitor(actual_pay, *this));
153  });
154 
155  budget -= actual_pay;
156  }
157 }

320 void database::initialize_budget_record( fc::time_point_sec now, budget_record& rec )const
321 {
322  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
323  const asset_object& core = asset_id_type(0)(*this);
324  const asset_dynamic_data_object& core_dd = core.dynamic_asset_data_id(*this);
325 
326  rec.from_initial_reserve = core.reserved(*this);
327  rec.from_accumulated_fees = core_dd.accumulated_fees;
328  rec.from_unused_witness_budget = dpo.witness_budget;
329 
330  if( (dpo.last_budget_time == fc::time_point_sec())
331  || (now <= dpo.last_budget_time) )
332  {
333  rec.time_since_last_budget = 0;
334  return;
335  }
336 
337  int64_t dt = (now - dpo.last_budget_time).to_seconds();
338  rec.time_since_last_budget = uint64_t( dt );
339 
340  // We'll consider accumulated_fees to be reserved at the BEGINNING
341  // of the maintenance interval. However, for speed we only
342  // call modify() on the asset_dynamic_data_object once at the
343  // end of the maintenance interval. Thus the accumulated_fees
344  // are available for the budget at this point, but not included
345  // in core.reserved().
346  share_type reserve = rec.from_initial_reserve + core_dd.accumulated_fees;
347  // Similarly, we consider leftover witness_budget to be burned
348  // at the BEGINNING of the maintenance interval.
349  reserve += dpo.witness_budget;
350 
351  fc::uint128_t budget_u128 = reserve.value;
352  budget_u128 *= uint64_t(dt);
353  budget_u128 *= GRAPHENE_CORE_ASSET_CYCLE_RATE;
354  //round up to the nearest satoshi -- this is necessary to ensure
355  // there isn't an "untouchable" reserve, and we will eventually
356  // be able to use the entire reserve
357  budget_u128 += ((uint64_t(1) << GRAPHENE_CORE_ASSET_CYCLE_RATE_BITS) - 1);
358  budget_u128 >>= GRAPHENE_CORE_ASSET_CYCLE_RATE_BITS;
359  share_type budget;
360  if( budget_u128 < reserve.value )
361  rec.total_budget = share_type(budget_u128.to_uint64());
362  else
363  rec.total_budget = reserve;
364 
365  return;
366 }

371 void database::process_budget()
372 {
373  try
374  {
375  const global_property_object& gpo = get_global_properties();
376  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
377  const asset_dynamic_data_object& core =
378  asset_id_type(0)(*this).dynamic_asset_data_id(*this);
379  fc::time_point_sec now = head_block_time();
380 
381  int64_t time_to_maint = (dpo.next_maintenance_time - now).to_seconds();
382  //
383  // The code that generates the next maintenance time should
384  // only produce a result in the future. If this assert
385  // fails, then the next maintenance time algorithm is buggy.
386  //
387  assert( time_to_maint > 0 );
388  //
389  // Code for setting chain parameters should validate
390  // block_interval > 0 (as well as the humans proposing /
391  // voting on changes to block interval).
392  //
393  assert( gpo.parameters.block_interval > 0 );
394  uint64_t blocks_to_maint = (uint64_t(time_to_maint) + gpo.parameters.block_interval - 1) / gpo.parameters.block_interval;
395 
396  // blocks_to_maint > 0 because time_to_maint > 0,
397  // which means numerator is at least equal to block_interval
398 
399  budget_record rec;
400  initialize_budget_record( now, rec );
401  share_type available_funds = rec.total_budget;
402 
403  share_type witness_budget = gpo.parameters.witness_pay_per_block.value * blocks_to_maint;
404  rec.requested_witness_budget = witness_budget;
405  witness_budget = std::min(witness_budget, available_funds);
406  rec.witness_budget = witness_budget;
407  available_funds -= witness_budget;
408 
409  fc::uint128_t worker_budget_u128 = gpo.parameters.worker_budget_per_day.value;
410  worker_budget_u128 *= uint64_t(time_to_maint);
411  worker_budget_u128 /= 60*60*24;
412 
413  share_type worker_budget;
414  if( worker_budget_u128 >= available_funds.value )
415  worker_budget = available_funds;
416  else
417  worker_budget = worker_budget_u128.to_uint64();
418  rec.worker_budget = worker_budget;
419  available_funds -= worker_budget;
420 
421  share_type leftover_worker_funds = worker_budget;
422  pay_workers(leftover_worker_funds);
423  rec.leftover_worker_funds = leftover_worker_funds;
424  available_funds += leftover_worker_funds;
425 
426  rec.supply_delta = rec.witness_budget
427  + rec.worker_budget
428  - rec.leftover_worker_funds
429  - rec.from_accumulated_fees
430  - rec.from_unused_witness_budget;
431 
432  modify(core, [&]( asset_dynamic_data_object& _core )
433  {
434  _core.current_supply = (_core.current_supply + rec.supply_delta );
435 
436  assert( rec.supply_delta ==
437  witness_budget
438  + worker_budget
439  - leftover_worker_funds
440  - _core.accumulated_fees
441  - dpo.witness_budget
442  );
443  _core.accumulated_fees = 0;
444  });
445 
446  modify(dpo, [&]( dynamic_global_property_object& _dpo )
447  {
448  // Since initial witness_budget was rolled into
449  // available_funds, we replace it with witness_budget
450  // instead of adding it.
451  _dpo.witness_budget = witness_budget;
452  _dpo.last_budget_time = now;
453  });
454 
455  create< budget_record_object >( [&]( budget_record_object& _rec )
456  {
457  _rec.time = head_block_time();
458  _rec.record = rec;
459  });
460 
461  // available_funds is money we could spend, but don't want to.
462  // we simply let it evaporate back into the reserve.
463  }
464  FC_CAPTURE_AND_RETHROW()
465 }

612 void create_buyback_orders( database& db )
613 {
614  const auto& bbo_idx = db.get_index_type< buyback_index >().indices().get<by_id>();
615  const auto& bal_idx = db.get_index_type< account_balance_index >().indices().get< by_account_asset >();
616 
617  for( const buyback_object& bbo : bbo_idx )
618  {
619  const asset_object& asset_to_buy = bbo.asset_to_buy(db);
620  assert( asset_to_buy.buyback_account.valid() );
621 
622  const account_object& buyback_account = (*(asset_to_buy.buyback_account))(db);
623  asset_id_type next_asset = asset_id_type();
624 
625  if( !buyback_account.allowed_assets.valid() )
626  {
627  wlog( "skipping buyback account ${b} at block ${n} because allowed_assets does not exist", ("b", buyback_account)("n", db.head_block_num()) );
628  continue;
629  }
630 
631  while( true )
632  {
633  auto it = bal_idx.lower_bound( boost::make_tuple( buyback_account.id, next_asset ) );
634  if( it == bal_idx.end() )
635  break;
636  if( it->owner != buyback_account.id )
637  break;
638  asset_id_type asset_to_sell = it->asset_type;
639  share_type amount_to_sell = it->balance;
640  next_asset = asset_to_sell + 1;
641  if( asset_to_sell == asset_to_buy.id )
642  continue;
643  if( amount_to_sell == 0 )
644  continue;
645  if( buyback_account.allowed_assets->find( asset_to_sell ) == buyback_account.allowed_assets->end() )
646  {
647  wlog( "buyback account ${b} not selling disallowed holdings of asset ${a} at block ${n}", ("b", buyback_account)("a", asset_to_sell)("n", db.head_block_num()) );
648  continue;
649  }
650 
651  try
652  {
653  transaction_evaluation_state buyback_context(&db);
654  buyback_context.skip_fee_schedule_check = true;
655 
656  limit_order_create_operation create_vop;
657  create_vop.fee = asset( 0, asset_id_type() );
658  create_vop.seller = buyback_account.id;
659  create_vop.amount_to_sell = asset( amount_to_sell, asset_to_sell );
660  create_vop.min_to_receive = asset( 1, asset_to_buy.id );
661  create_vop.expiration = time_point_sec::maximum();
662  create_vop.fill_or_kill = false;
663 
664  limit_order_id_type order_id = db.apply_operation( buyback_context, create_vop ).get< object_id_type >();
665 
666  if( db.find( order_id ) != nullptr )
667  {
668  limit_order_cancel_operation cancel_vop;
669  cancel_vop.fee = asset( 0, asset_id_type() );
670  cancel_vop.order = order_id;
671  cancel_vop.fee_paying_account = buyback_account.id;
672 
673  db.apply_operation( buyback_context, cancel_vop );
674  }
675  }
676  catch( const fc::exception& e )
677  {
678  // we can in fact get here, e.g. if asset issuer of buy/sell asset blacklists/whitelists the buyback account
679  wlog( "Skipping buyback processing selling ${as} for ${ab} for buyback account ${b} at block ${n}; exception was ${e}",
680  ("as", asset_to_sell)("ab", asset_to_buy)("b", buyback_account)("n", db.head_block_num())("e", e.to_detail_string()) );
681  continue;
682  }
683  }
684  }
685  return;

719 void database::process_bids( const asset_bitasset_data_object& bad )
720 {
721  if( bad.is_prediction_market ) return;
722  if( bad.current_feed.settlement_price.is_null() ) return;
723 
724  asset_id_type to_revive_id = (asset( 0, bad.options.short_backing_asset ) * bad.settlement_price).asset_id;
725  const asset_object& to_revive = to_revive_id( *this );
726  const asset_dynamic_data_object& bdd = to_revive.dynamic_data( *this );
727 
728  const auto& bid_idx = get_index_type< collateral_bid_index >().indices().get<by_price>();
729  const auto start = bid_idx.lower_bound( boost::make_tuple( to_revive_id, price::max( bad.options.short_backing_asset, to_revive_id ), collateral_bid_id_type() ) );
730 
731  share_type covered = 0;
732  auto itr = start;
733  while( covered < bdd.current_supply && itr != bid_idx.end() && itr->inv_swan_price.quote.asset_id == to_revive_id )
734  {
735  const collateral_bid_object& bid = *itr;
736  asset total_collateral = bid.inv_swan_price.quote * bad.settlement_price;
737  total_collateral += bid.inv_swan_price.base;
738  price call_price = price::call_price( bid.inv_swan_price.quote, total_collateral, bad.current_feed.maintenance_collateral_ratio );
739  if( ~call_price >= bad.current_feed.settlement_price ) break;
740  covered += bid.inv_swan_price.quote.amount;
741  ++itr;
742  }
743  if( covered < bdd.current_supply ) return;
744 
745  const auto end = itr;
746  share_type to_cover = bdd.current_supply;
747  share_type remaining_fund = bad.settlement_fund;
748  for( itr = start; itr != end; )
749  {
750  const collateral_bid_object& bid = *itr;
751  ++itr;
752  share_type debt = bid.inv_swan_price.quote.amount;
753  share_type collateral = (bid.inv_swan_price.quote * bad.settlement_price).amount;
754  if( bid.inv_swan_price.quote.amount >= to_cover )
755  {
756  debt = to_cover;
757  collateral = remaining_fund;
758  }
759  to_cover -= debt;
760  remaining_fund -= collateral;
761  execute_bid( bid, debt, collateral, bad.current_feed );
762  }
763  FC_ASSERT( remaining_fund == 0 );
764  FC_ASSERT( to_cover == 0 );
765 
766  _cancel_bids_and_revive_mpa( to_revive, bad );
767 }
