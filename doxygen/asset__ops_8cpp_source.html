34 bool is_valid_symbol( const string& symbol )
35 {
36  if( symbol.size() < GRAPHENE_MIN_ASSET_SYMBOL_LENGTH )
37  return false;
38 
39  if( symbol.substr(0,3) == "BIT" )
40  return false;
41 
42  if( symbol.size() > GRAPHENE_MAX_ASSET_SYMBOL_LENGTH )
43  return false;
44 
45  if( !isalpha( symbol.front() ) )
46  return false;
47 
48  if( !isalpha( symbol.back() ) )
49  return false;
50 
51  bool dot_already_present = false;
52  for( const auto c : symbol )
53  {
54  if( (isalpha( c ) && isupper( c )) || isdigit(c) )
55  continue;
56 
57  if( c == '.' )
58  {
59  if( dot_already_present )
60  return false;
61 
62  dot_already_present = true;
63  continue;
64  }
65 
66  return false;
67  }
68 
69  return true;
70 }

204 void asset_options::validate()const
205 {
206  FC_ASSERT( max_supply > 0 );
207  FC_ASSERT( max_supply <= GRAPHENE_MAX_SHARE_SUPPLY );
208  FC_ASSERT( market_fee_percent <= GRAPHENE_100_PERCENT );
209  FC_ASSERT( max_market_fee >= 0 && max_market_fee <= GRAPHENE_MAX_SHARE_SUPPLY );
210  // There must be no high bits in permissions whose meaning is not known.
211  FC_ASSERT( !(issuer_permissions & ~ASSET_ISSUER_PERMISSION_MASK) );
212  // The global_settle flag may never be set (this is a permission only)
213  FC_ASSERT( !(flags & global_settle) );
214  // the witness_fed and committee_fed flags cannot be set simultaneously
215  FC_ASSERT( (flags & (witness_fed_asset | committee_fed_asset)) != (witness_fed_asset | committee_fed_asset) );
216  core_exchange_rate.validate();
217  FC_ASSERT( core_exchange_rate.base.asset_id.instance.value == 0 ||
218  core_exchange_rate.quote.asset_id.instance.value == 0 );
219 
220  if(!whitelist_authorities.empty() || !blacklist_authorities.empty())
221  FC_ASSERT( flags & white_list );
222  for( auto item : whitelist_markets )
223  {
224  FC_ASSERT( blacklist_markets.find(item) == blacklist_markets.end() );
225  }
226  for( auto item : blacklist_markets )
227  {
228  FC_ASSERT( whitelist_markets.find(item) == whitelist_markets.end() );
229  }
