33 share_type asset_bitasset_data_object::max_force_settlement_volume(share_type current_supply) const
34 {
35  if( options.maximum_force_settlement_volume == 0 )
36  return 0;
37  if( options.maximum_force_settlement_volume == GRAPHENE_100_PERCENT )
38  return current_supply + force_settled_volume;

40  fc::uint128 volume = current_supply.value + force_settled_volume.value;

46 void graphene::chain::asset_bitasset_data_object::update_median_feeds(time_point_sec current_time)
47 {
48  current_feed_publication_time = current_time;
49  vector<std::reference_wrapper<const price_feed>> current_feeds;
50  for( const pair<account_id_type, pair<time_point_sec,price_feed>>& f : feeds )
51  {
52  if( (current_time - f.second.first).to_seconds() < options.feed_lifetime_sec &&
53  f.second.first != time_point_sec() )
54  {
55  current_feeds.emplace_back(f.second.second);
56  current_feed_publication_time = std::min(current_feed_publication_time, f.second.first);
57  }
58  }
59 
60  // If there are no valid feeds, or the number available is less than the minimum to calculate a median...
61  if( current_feeds.size() < options.minimum_feeds )
62  {
63  //... don't calculate a median, and set a null feed
64  current_feed_publication_time = current_time;
65  current_feed = price_feed();
66  return;
67  }
68  if( current_feeds.size() == 1 )
69  {
70  current_feed = std::move(current_feeds.front());
71  return;
72  }
73 
74  // *** Begin Median Calculations ***
75  price_feed median_feed;
76  const auto median_itr = current_feeds.begin() + current_feeds.size() / 2;
77 #define CALCULATE_MEDIAN_VALUE(r, data, field_name) \
78  std::nth_element( current_feeds.begin(), median_itr, current_feeds.end(), \
79  [](const price_feed& a, const price_feed& b) { \
80  return a.field_name < b.field_name; \
81  }); \
82  median_feed.field_name = median_itr->get().field_name;
83 
84  BOOST_PP_SEQ_FOR_EACH( CALCULATE_MEDIAN_VALUE, ~, GRAPHENE_PRICE_FEED_FIELDS )
85 #undef CALCULATE_MEDIAN_VALUE
86  // *** End Median Calculations ***
87 
88  current_feed = median_feed;
89 }

93 asset asset_object::amount_from_string(string amount_string) const
94 { try {
95  bool negative_found = false;
96  bool decimal_found = false;
97  for( const char c : amount_string )
98  {
99  if( isdigit( c ) )
100  continue;
101 
102  if( c == '-' && !negative_found )
103  {
104  negative_found = true;
105  continue;
106  }
107 
108  if( c == '.' && !decimal_found )
109  {
110  decimal_found = true;
111  continue;
112  }
113 
114  FC_THROW( (amount_string) );
115  }
116 
117  share_type satoshis = 0;
118 
119  share_type scaled_precision = asset::scaled_precision( precision );
120 
121  const auto decimal_pos = amount_string.find( '.' );
122  const string lhs = amount_string.substr( negative_found, decimal_pos );
123  if( !lhs.empty() )
124  satoshis += fc::safe<int64_t>(std::stoll(lhs)) *= scaled_precision;
125 
126  if( decimal_found )
127  {
128  const size_t max_rhs_size = std::to_string( scaled_precision.value ).substr( 1 ).size();
129 
130  string rhs = amount_string.substr( decimal_pos + 1 );
131  FC_ASSERT( rhs.size() <= max_rhs_size );
132 
133  while( rhs.size() < max_rhs_size )
134  rhs += '0';
135 
136  if( !rhs.empty() )
137  satoshis += std::stoll( rhs );
138  }
139 
140  FC_ASSERT( satoshis <= GRAPHENE_MAX_SHARE_SUPPLY );
141 
142  if( negative_found )
143  satoshis *= -1;
144 
145  return amount(satoshis);
146  } FC_CAPTURE_AND_RETHROW( (amount_string) ) }
