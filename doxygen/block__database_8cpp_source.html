74 void block_database::store( const block_id_type& _id, const signed_block& b )
75 {
76  block_id_type id = _id;
77  if( id == block_id_type() )
78  {
79  id = b.id();
80  elog( "id argument of block_database::store() was not initialized for block ${id}", ("id", id) );
81  }
82  auto num = block_header::num_from_id(id);
83  _block_num_to_pos.seekp( sizeof( index_entry ) * num );
84  index_entry e;
85  _blocks.seekp( 0, _blocks.end );
86  auto vec = fc::raw::pack( b );
87  e.block_pos = _blocks.tellp();
88  e.block_size = vec.size();
89  e.block_id = id;
90  _blocks.write( vec.data(), vec.size() );
91  _block_num_to_pos.write( (char*)&e, sizeof(e) );
92 }

94 void block_database::remove( const block_id_type& id )
95 { try {
96  index_entry e;
97  auto index_pos = sizeof(e)*block_header::num_from_id(id);
98  _block_num_to_pos.seekg( 0, _block_num_to_pos.end );
99  if ( _block_num_to_pos.tellg() <= index_pos )
100  FC_THROW_EXCEPTION(fc::key_not_found_exception, "Block ${id} not contained in block database", ("id", id));
101

103  _block_num_to_pos.read( (char*)&e, sizeof(e) );

129 block_id_type block_database::fetch_block_id( uint32_t block_num )const
130 {
131  assert( block_num != 0 );

133  auto index_pos = sizeof(e)*block_num;
134  _block_num_to_pos.seekg( 0, _block_num_to_pos.end );
135  if ( _block_num_to_pos.tellg() <= int64_t(index_pos) )
136  FC_THROW_EXCEPTION(fc::key_not_found_exception, "Block number ${block_num} not contained in block database", ("block_num", block_num));
137 
138  _block_num_to_pos.seekg( index_pos );
139  _block_num_to_pos.read( (char*)&e, sizeof(e) );
140 
141  FC_ASSERT( e.block_id != block_id_type(), "Empty block_id in block_database (maybe corrupt on disk?)" );

145 optional<signed_block> block_database::fetch_optional( const block_id_type& id )const
146 {

150  auto index_pos = sizeof(e)*block_header::num_from_id(id);
151  _block_num_to_pos.seekg( 0, _block_num_to_pos.end );
152  if ( _block_num_to_pos.tellg() <= index_pos )
153  return {};
154 
155  _block_num_to_pos.seekg( index_pos );
156  _block_num_to_pos.read( (char*)&e, sizeof(e) );
157 
158  if( e.block_id != id ) return optional<signed_block>();
159

161  _blocks.seekg( e.block_pos );
162  if (e.block_size)

164  auto result = fc::raw::unpack<signed_block>(data);
165  FC_ASSERT( result.id() == e.block_id );

167  }

212
