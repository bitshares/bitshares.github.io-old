65  checksum_type signed_block::calculate_merkle_root()const
66  {
67  if( transactions.size() == 0 )
68  return checksum_type();
69 
70  vector<digest_type> ids;
71  ids.resize( transactions.size() );
72  for( uint32_t i = 0; i < transactions.size(); ++i )
73  ids[i] = transactions[i].merkle_digest();
74 
75  vector<digest_type>::size_type current_number_of_hashes = ids.size();
76  while( current_number_of_hashes > 1 )
77  {
78  // hash ID's in pairs
79  uint32_t i_max = current_number_of_hashes - (current_number_of_hashes&1);
80  uint32_t k = 0;
81 
82  for( uint32_t i = 0; i < i_max; i += 2 )
83  ids[k++] = digest_type::hash( std::make_pair( ids[i], ids[i+1] ) );
84 
85  if( current_number_of_hashes&1 )
86  ids[k++] = ids[i_max];
87  current_number_of_hashes = k;
88  }
89  return checksum_type::hash( ids[0] );
90  }
