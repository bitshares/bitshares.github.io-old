53 void database::adjust_balance(account_id_type account, asset delta )
54 { try {
55  if( delta.amount == 0 )
56  return;
57 
58  auto& index = get_index_type<account_balance_index>().indices().get<by_account_asset>();
59  auto itr = index.find(boost::make_tuple(account, delta.asset_id));
60  if(itr == index.end())
61  {
62  FC_ASSERT( delta.amount > 0, "Insufficient Balance: ${a}'s balance of ${b} is less than required ${r}",
63  ("a",account(*this).name)
64  ("b",to_pretty_string(asset(0,delta.asset_id)))
65  ("r",to_pretty_string(-delta)));
66  create<account_balance_object>([account,&delta](account_balance_object& b) {
67  b.owner = account;
68  b.asset_type = delta.asset_id;
69  b.balance = delta.amount.value;
70  });
71  } else {
72  if( delta.amount < 0 )
73  FC_ASSERT( itr->get_balance() >= -delta, "Insufficient Balance: ${a}'s balance of ${b} is less than required ${r}", ("a",account(*this).name)("b",to_pretty_string(itr->get_balance()))("r",to_pretty_string(-delta)));
74  modify(*itr, [delta](account_balance_object& b) {
75  b.adjust_balance(delta);
76  });
77  }
78 
79 } FC_CAPTURE_AND_RETHROW( (account)(delta) ) }

81 optional< vesting_balance_id_type > database::deposit_lazy_vesting(
82  const optional< vesting_balance_id_type >& ovbid,
83  share_type amount, uint32_t req_vesting_seconds,
84  account_id_type req_owner,
85  bool require_vesting )
86 {
87  if( amount == 0 )
88  return optional< vesting_balance_id_type >();
89 
90  fc::time_point_sec now = head_block_time();
91 
92  while( true )
93  {
94  if( !ovbid.valid() )
95  break;
96  const vesting_balance_object& vbo = (*ovbid)(*this);
97  if( vbo.owner != req_owner )
98  break;
99  if( vbo.policy.which() != vesting_policy::tag< cdd_vesting_policy >::value )
100  break;
101  if( vbo.policy.get< cdd_vesting_policy >().vesting_seconds != req_vesting_seconds )
102  break;
103  modify( vbo, [&]( vesting_balance_object& _vbo )
104  {
105  if( require_vesting )
106  _vbo.deposit(now, amount);
107  else
108  _vbo.deposit_vested(now, amount);
109  } );
110  return optional< vesting_balance_id_type >();
111  }
112 
113  const vesting_balance_object& vbo = create< vesting_balance_object >( [&]( vesting_balance_object& _vbo )
114  {
115  _vbo.owner = req_owner;
116  _vbo.balance = amount;
117 
118  cdd_vesting_policy policy;
119  policy.vesting_seconds = req_vesting_seconds;
120  policy.coin_seconds_earned = require_vesting ? 0 : amount.value * policy.vesting_seconds;
121  policy.coin_seconds_earned_last_update = now;
122 
123  _vbo.policy = policy;
124  } );
125 
126  return vbo.id;
127 }

129 void database::deposit_cashback(const account_object& acct, share_type amount, bool require_vesting)
130 {
131  // If we don't have a VBO, or if it has the wrong maturity
132  // due to a policy change, cut it loose.
133 
134  if( amount == 0 )
135  return;
136 
137  if( acct.get_id() == GRAPHENE_COMMITTEE_ACCOUNT || acct.get_id() == GRAPHENE_WITNESS_ACCOUNT ||
138  acct.get_id() == GRAPHENE_RELAXED_COMMITTEE_ACCOUNT || acct.get_id() == GRAPHENE_NULL_ACCOUNT ||
139  acct.get_id() == GRAPHENE_TEMP_ACCOUNT )
140  {
141  // The blockchain's accounts do not get cashback; it simply goes to the reserve pool.
142  modify(get(asset_id_type()).dynamic_asset_data_id(*this), [amount](asset_dynamic_data_object& d) {
143  d.current_supply -= amount;
144  });
145  return;
146  }
147 
148  optional< vesting_balance_id_type > new_vbid = deposit_lazy_vesting(
149  acct.cashback_vb,
150  amount,
151  get_global_properties().parameters.cashback_vesting_period_seconds,
152  acct.id,
153  require_vesting );
154 
155  if( new_vbid.valid() )
156  {
157  modify( acct, [&]( account_object& _acct )
158  {
159  _acct.cashback_vb = *new_vbid;
160  } );
161  }
162 
163  return;
164 }
