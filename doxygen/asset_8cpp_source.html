124  price operator * ( const price& p, const ratio_type& r )
125  { try {
126  p.validate();
127 
128  FC_ASSERT( r.numerator() > 0 && r.denominator() > 0 );
129 
130  if( r.numerator() == r.denominator() ) return p;
131 
132  boost::rational<int128_t> p128( p.base.amount.value, p.quote.amount.value );
133  boost::rational<int128_t> r128( r.numerator(), r.denominator() );
134  auto cp = p128 * r128;
135  auto ocp = cp;
136 
137  bool shrinked = false;
138  static const int128_t max( GRAPHENE_MAX_SHARE_SUPPLY );
139  while( cp.numerator() > max || cp.denominator() > max )
140  {
141  if( cp.numerator() == 1 )
142  {
143  cp = boost::rational<int128_t>( 1, max );
144  break;
145  }
146  else if( cp.denominator() == 1 )
147  {
148  cp = boost::rational<int128_t>( max, 1 );
149  break;
150  }
151  else
152  {
153  cp = boost::rational<int128_t>( cp.numerator() >> 1, cp.denominator() >> 1 );
154  shrinked = true;
155  }
156  }
157  if( shrinked ) // maybe not accurate enough due to rounding, do additional checks here
158  {
159  int128_t num = ocp.numerator();
160  int128_t den = ocp.denominator();
161  if( num > den )
162  {
163  num /= den;
164  if( num > max )
165  num = max;
166  den = 1;
167  }
168  else
169  {
170  den /= num;
171  if( den > max )
172  den = max;
173  num = 1;
174  }
175  boost::rational<int128_t> ncp( num, den );
176  if( num == max || den == max ) // it's on the edge, we know it's accurate enough
177  cp = ncp;
178  else
179  {
180  // from the accurate ocp, now we have ncp and cp. use the one which is closer to ocp.
181  // TODO improve performance
182  auto diff1 = abs( ncp - ocp );
183  auto diff2 = abs( cp - ocp );
184  if( diff1 < diff2 ) cp = ncp;
185  }
186  }
187 
188  price np = asset( cp.numerator().convert_to<int64_t>(), p.base.asset_id )
189  / asset( cp.denominator().convert_to<int64_t>(), p.quote.asset_id );
190 
191  if( ( r.numerator() > r.denominator() && np < p )
192  || ( r.numerator() < r.denominator() && np > p ) )
193  // even with an accurate result, if p is out of valid range, return it
194  np = p;
195 
196  np.validate();
197  return np;
198  } FC_CAPTURE_AND_RETHROW( (p)(r.numerator())(r.denominator()) ) }
