113  bool signed_by( const public_key_type& k )
114  {
115  auto itr = provided_signatures.find(k);
116  if( itr == provided_signatures.end() )
117  {
118  auto pk = available_keys.find(k);
119  if( pk != available_keys.end() )
120  return provided_signatures[k] = true;
121  return false;
122  }
123  return itr->second = true;
124  }

126  optional<map<address,public_key_type>> available_address_sigs;
127  optional<map<address,public_key_type>> provided_address_sigs;
128 
129  bool signed_by( const address& a ) {
130  if( !available_address_sigs ) {
131  available_address_sigs = std::map<address,public_key_type>();
132  provided_address_sigs = std::map<address,public_key_type>();
133  for( auto& item : available_keys ) {
134  (*available_address_sigs)[ address(pts_address(item, false, 56) ) ] = item;
135  (*available_address_sigs)[ address(pts_address(item, true, 56) ) ] = item;
136  (*available_address_sigs)[ address(pts_address(item, false, 0) ) ] = item;
137  (*available_address_sigs)[ address(pts_address(item, true, 0) ) ] = item;
138  (*available_address_sigs)[ address(item) ] = item;
139  }
140  for( auto& item : provided_signatures ) {
141  (*provided_address_sigs)[ address(pts_address(item.first, false, 56) ) ] = item.first;
142  (*provided_address_sigs)[ address(pts_address(item.first, true, 56) ) ] = item.first;
143  (*provided_address_sigs)[ address(pts_address(item.first, false, 0) ) ] = item.first;
144  (*provided_address_sigs)[ address(pts_address(item.first, true, 0) ) ] = item.first;
145  (*provided_address_sigs)[ address(item.first) ] = item.first;
146  }
147  }
148  auto itr = provided_address_sigs->find(a);
149  if( itr == provided_address_sigs->end() )
150  {
151  auto aitr = available_address_sigs->find(a);
152  if( aitr != available_address_sigs->end() ) {
153  auto pk = available_keys.find(aitr->second);
154  if( pk != available_keys.end() )
155  return provided_signatures[aitr->second] = true;
156  return false;
157  }
158  }
159  return provided_signatures[itr->second] = true;
160  }

162  bool check_authority( account_id_type id )
163  {
164  if( approved_by.find(id) != approved_by.end() ) return true;
165  return check_authority( get_active(id) );
166  }

172  bool check_authority( const authority* au, uint32_t depth = 0 )
173  {
174  if( au == nullptr ) return false;
175  const authority& auth = *au;
176 
177  uint32_t total_weight = 0;
178  for( const auto& k : auth.key_auths )
179  if( signed_by( k.first ) )
180  {
181  total_weight += k.second;
182  if( total_weight >= auth.weight_threshold )
183  return true;
184  }
185 
186  for( const auto& k : auth.address_auths )
187  if( signed_by( k.first ) )
188  {
189  total_weight += k.second;
190  if( total_weight >= auth.weight_threshold )
191  return true;
192  }
193 
194  for( const auto& a : auth.account_auths )
195  {
196  if( approved_by.find(a.first) == approved_by.end() )
197  {
198  if( depth == max_recursion )
199  continue;
200  if( check_authority( get_active( a.first ), depth+1 ) )
201  {
202  approved_by.insert( a.first );
203  total_weight += a.second;
204  if( total_weight >= auth.weight_threshold )
205  return true;
206  }
207  }
208  else
209  {
210  total_weight += a.second;
211  if( total_weight >= auth.weight_threshold )
212  return true;
213  }
214  }
215  return total_weight >= auth.weight_threshold;
216  }

249 void verify_authority( const vector<operation>& ops, const flat_set<public_key_type>& sigs,
250  const std::function<const authority*(account_id_type)>& get_active,
251  const std::function<const authority*(account_id_type)>& get_owner,
252  uint32_t max_recursion_depth,
253  bool allow_committe,
254  const flat_set<account_id_type>& active_aprovals,
255  const flat_set<account_id_type>& owner_approvals )
256 { try {
257  flat_set<account_id_type> required_active;
258  flat_set<account_id_type> required_owner;
259  vector<authority> other;
260 
261  for( const auto& op : ops )
262  operation_get_required_authorities( op, required_active, required_owner, other );
263 
264  if( !allow_committe )
265  GRAPHENE_ASSERT( required_active.find(GRAPHENE_COMMITTEE_ACCOUNT) == required_active.end(),
266  invalid_committee_approval, "Committee account may only propose transactions" );
267 
268  sign_state s(sigs,get_active);
269  s.max_recursion = max_recursion_depth;
270  for( auto& id : active_aprovals )
271  s.approved_by.insert( id );
272  for( auto& id : owner_approvals )
273  s.approved_by.insert( id );
274 
275  for( const auto& auth : other )
276  {
277  GRAPHENE_ASSERT( s.check_authority(&auth), tx_missing_other_auth, "Missing Authority", ("auth",auth)("sigs",sigs) );
278  }
279 
280  // fetch all of the top level authorities
281  for( auto id : required_active )
282  {
283  GRAPHENE_ASSERT( s.check_authority(id) ||
284  s.check_authority(get_owner(id)),
285  tx_missing_active_auth, "Missing Active Authority ${id}", ("id",id)("auth",*get_active(id))("owner",*get_owner(id)) );
286  }
287 
288  for( auto id : required_owner )
289  {
290  GRAPHENE_ASSERT( owner_approvals.find(id) != owner_approvals.end() ||
291  s.check_authority(get_owner(id)),
292  tx_missing_owner_auth, "Missing Owner Authority ${id}", ("id",id)("auth",*get_owner(id)) );
293  }
294 
295  GRAPHENE_ASSERT(
296  !s.remove_unused_signatures(),
297  tx_irrelevant_sig,
298  "Unnecessary signature(s) detected"
299  );
300 } FC_CAPTURE_AND_RETHROW( (ops)(sigs) ) }

319 set<public_key_type> signed_transaction::get_required_signatures(
320  const chain_id_type& chain_id,
321  const flat_set<public_key_type>& available_keys,
322  const std::function<const authority*(account_id_type)>& get_active,
323  const std::function<const authority*(account_id_type)>& get_owner,
324  uint32_t max_recursion_depth )const
325 {
326  flat_set<account_id_type> required_active;
327  flat_set<account_id_type> required_owner;
328  vector<authority> other;
329  get_required_authorities( required_active, required_owner, other );
330 
331  flat_set<public_key_type> signature_keys = get_signature_keys( chain_id );
332  sign_state s( signature_keys, get_active, available_keys );
333  s.max_recursion = max_recursion_depth;
334 
335  for( const auto& auth : other )
336  s.check_authority(&auth);
337  for( auto& owner : required_owner )
338  s.check_authority( get_owner( owner ) );
339  for( auto& active : required_active )
340  s.check_authority( active ) || s.check_authority( get_owner( active ) );
341 
342  s.remove_unused_signatures();
343 
344  set<public_key_type> result;
345 
346  for( auto& provided_sig : s.provided_signatures )
347  if( available_keys.find( provided_sig.first ) != available_keys.end()
348  && signature_keys.find( provided_sig.first ) == signature_keys.end() )
349  result.insert( provided_sig.first );
350 
351  return result;
352 }

354 set<public_key_type> signed_transaction::minimize_required_signatures(
355  const chain_id_type& chain_id,
356  const flat_set<public_key_type>& available_keys,
357  const std::function<const authority*(account_id_type)>& get_active,
358  const std::function<const authority*(account_id_type)>& get_owner,
359  uint32_t max_recursion
360  ) const
361 {
362  set< public_key_type > s = get_required_signatures( chain_id, available_keys, get_active, get_owner, max_recursion );
363  flat_set< public_key_type > result( s.begin(), s.end() );
364 
365  for( const public_key_type& k : s )
366  {
367  result.erase( k );
368  try
369  {
370  graphene::chain::verify_authority( operations, result, get_active, get_owner, max_recursion );
371  continue; // element stays erased if verify_authority is ok
372  }
373  catch( const tx_missing_owner_auth& e ) {}

377  }

379 }
