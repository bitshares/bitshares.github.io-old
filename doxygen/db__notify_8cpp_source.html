272 static void get_relevant_accounts( const object* obj, flat_set<account_id_type>& accounts )
273 {
274  if( obj->id.space() == protocol_ids )
275  {
276  switch( (object_type)obj->id.type() )
277  {
278  case null_object_type:
279  case base_object_type:
280  case OBJECT_TYPE_COUNT:
281  return;
282  case account_object_type:{
283  accounts.insert( obj->id );
284  break;
285  } case asset_object_type:{
286  const auto& aobj = dynamic_cast<const asset_object*>(obj);
287  FC_ASSERT( aobj != nullptr );
288  accounts.insert( aobj->issuer );
289  break;
290  } case force_settlement_object_type:{
291  const auto& aobj = dynamic_cast<const force_settlement_object*>(obj);
292  FC_ASSERT( aobj != nullptr );
293  accounts.insert( aobj->owner );
294  break;
295  } case committee_member_object_type:{
296  const auto& aobj = dynamic_cast<const committee_member_object*>(obj);
297  FC_ASSERT( aobj != nullptr );
298  accounts.insert( aobj->committee_member_account );
299  break;
300  } case witness_object_type:{
301  const auto& aobj = dynamic_cast<const witness_object*>(obj);
302  FC_ASSERT( aobj != nullptr );
303  accounts.insert( aobj->witness_account );
304  break;
305  } case limit_order_object_type:{
306  const auto& aobj = dynamic_cast<const limit_order_object*>(obj);
307  FC_ASSERT( aobj != nullptr );
308  accounts.insert( aobj->seller );
309  break;
310  } case call_order_object_type:{
311  const auto& aobj = dynamic_cast<const call_order_object*>(obj);
312  FC_ASSERT( aobj != nullptr );
313  accounts.insert( aobj->borrower );
314  break;
315  } case custom_object_type:{
316  break;
317  } case proposal_object_type:{
318  const auto& aobj = dynamic_cast<const proposal_object*>(obj);
319  FC_ASSERT( aobj != nullptr );
320  transaction_get_impacted_accounts( aobj->proposed_transaction, accounts );
321  break;
322  } case operation_history_object_type:{
323  const auto& aobj = dynamic_cast<const operation_history_object*>(obj);
324  FC_ASSERT( aobj != nullptr );
325  operation_get_impacted_accounts( aobj->op, accounts );
326  break;
327  } case withdraw_permission_object_type:{
328  const auto& aobj = dynamic_cast<const withdraw_permission_object*>(obj);
329  FC_ASSERT( aobj != nullptr );
330  accounts.insert( aobj->withdraw_from_account );
331  accounts.insert( aobj->authorized_account );
332  break;
333  } case vesting_balance_object_type:{
334  const auto& aobj = dynamic_cast<const vesting_balance_object*>(obj);
335  FC_ASSERT( aobj != nullptr );
336  accounts.insert( aobj->owner );
337  break;
338  } case worker_object_type:{
339  const auto& aobj = dynamic_cast<const worker_object*>(obj);
340  FC_ASSERT( aobj != nullptr );
341  accounts.insert( aobj->worker_account );
342  break;
343  } case balance_object_type:{
345  break;
346  }
347  }
348  }
349  else if( obj->id.space() == implementation_ids )
350  {
351  switch( (impl_object_type)obj->id.type() )
352  {
353  case impl_global_property_object_type:
354  break;
355  case impl_dynamic_global_property_object_type:
356  break;
357  case impl_reserved0_object_type:
358  break;
359  case impl_asset_dynamic_data_type:
360  break;
361  case impl_asset_bitasset_data_type:
362  break;
363  case impl_account_balance_object_type:{
364  const auto& aobj = dynamic_cast<const account_balance_object*>(obj);
365  FC_ASSERT( aobj != nullptr );
366  accounts.insert( aobj->owner );
367  break;
368  } case impl_account_statistics_object_type:{
369  const auto& aobj = dynamic_cast<const account_statistics_object*>(obj);
370  FC_ASSERT( aobj != nullptr );
371  accounts.insert( aobj->owner );
372  break;
373  } case impl_transaction_object_type:{
374  const auto& aobj = dynamic_cast<const transaction_object*>(obj);
375  FC_ASSERT( aobj != nullptr );
376  transaction_get_impacted_accounts( aobj->trx, accounts );
377  break;
378  } case impl_blinded_balance_object_type:{
379  const auto& aobj = dynamic_cast<const blinded_balance_object*>(obj);
380  FC_ASSERT( aobj != nullptr );
381  for( const auto& a : aobj->owner.account_auths )
382  accounts.insert( a.first );
383  break;
384  } case impl_block_summary_object_type:
385  break;
386  case impl_account_transaction_history_object_type: {
387  const auto& aobj = dynamic_cast<const account_transaction_history_object*>(obj);
388  FC_ASSERT( aobj != nullptr );
389  accounts.insert( aobj->account );
390  break;
391  } case impl_chain_property_object_type:
392  break;
393  case impl_witness_schedule_object_type:
394  break;
395  case impl_budget_record_object_type:
396  break;
397  case impl_special_authority_object_type:
398  break;
399  case impl_buyback_object_type:
400  break;
401  case impl_fba_accumulator_object_type:
402  break;
403  case impl_collateral_bid_object_type:{
404  const auto& aobj = dynamic_cast<const collateral_bid_object*>(obj);
405  FC_ASSERT( aobj != nullptr );
406  accounts.insert( aobj->bidder );
