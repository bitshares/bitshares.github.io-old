216 static void get_relevant_accounts( const object* obj, flat_set<account_id_type>& accounts )
217 {
218  if( obj->id.space() == protocol_ids )
219  {
220  switch( (object_type)obj->id.type() )
221  {
222  case null_object_type:
223  case base_object_type:
224  case OBJECT_TYPE_COUNT:
225  return;
226  case account_object_type:{
227  accounts.insert( obj->id );
228  break;
229  } case asset_object_type:{
230  const auto& aobj = dynamic_cast<const asset_object*>(obj);
231  assert( aobj != nullptr );
232  accounts.insert( aobj->issuer );
233  break;
234  } case force_settlement_object_type:{
235  const auto& aobj = dynamic_cast<const force_settlement_object*>(obj);
236  assert( aobj != nullptr );
237  accounts.insert( aobj->owner );
238  break;
239  } case committee_member_object_type:{
240  const auto& aobj = dynamic_cast<const committee_member_object*>(obj);
241  assert( aobj != nullptr );
242  accounts.insert( aobj->committee_member_account );
243  break;
244  } case witness_object_type:{
245  const auto& aobj = dynamic_cast<const witness_object*>(obj);
246  assert( aobj != nullptr );
247  accounts.insert( aobj->witness_account );
248  break;
249  } case limit_order_object_type:{
250  const auto& aobj = dynamic_cast<const limit_order_object*>(obj);
251  assert( aobj != nullptr );
252  accounts.insert( aobj->seller );
253  break;
254  } case call_order_object_type:{
255  const auto& aobj = dynamic_cast<const call_order_object*>(obj);
256  assert( aobj != nullptr );
257  accounts.insert( aobj->borrower );
258  break;
259  } case custom_object_type:{
260  break;
261  } case proposal_object_type:{
262  const auto& aobj = dynamic_cast<const proposal_object*>(obj);
263  assert( aobj != nullptr );
264  transaction_get_impacted_accounts( aobj->proposed_transaction, accounts );
265  break;
266  } case operation_history_object_type:{
267  const auto& aobj = dynamic_cast<const operation_history_object*>(obj);
268  assert( aobj != nullptr );
269  operation_get_impacted_accounts( aobj->op, accounts );
270  break;
271  } case withdraw_permission_object_type:{
272  const auto& aobj = dynamic_cast<const withdraw_permission_object*>(obj);
273  assert( aobj != nullptr );
274  accounts.insert( aobj->withdraw_from_account );
275  accounts.insert( aobj->authorized_account );
276  break;
277  } case vesting_balance_object_type:{
278  const auto& aobj = dynamic_cast<const vesting_balance_object*>(obj);
279  assert( aobj != nullptr );
280  accounts.insert( aobj->owner );
281  break;
282  } case worker_object_type:{
283  const auto& aobj = dynamic_cast<const worker_object*>(obj);
284  assert( aobj != nullptr );
285  accounts.insert( aobj->worker_account );
286  break;
287  } case balance_object_type:{
289  break;
290  }
291  }
292  }
293  else if( obj->id.space() == implementation_ids )
294  {
295  switch( (impl_object_type)obj->id.type() )
296  {
297  case impl_global_property_object_type:
298  break;
299  case impl_dynamic_global_property_object_type:
300  break;
301  case impl_reserved0_object_type:
302  break;
303  case impl_asset_dynamic_data_type:
304  break;
305  case impl_asset_bitasset_data_type:
306  break;
307  case impl_account_balance_object_type:{
308  const auto& aobj = dynamic_cast<const account_balance_object*>(obj);
309  assert( aobj != nullptr );
310  accounts.insert( aobj->owner );
311  break;
312  } case impl_account_statistics_object_type:{
313  const auto& aobj = dynamic_cast<const account_statistics_object*>(obj);
314  assert( aobj != nullptr );
315  accounts.insert( aobj->owner );
316  break;
317  } case impl_transaction_object_type:{
318  const auto& aobj = dynamic_cast<const transaction_object*>(obj);
319  assert( aobj != nullptr );
320  transaction_get_impacted_accounts( aobj->trx, accounts );
321  break;
322  } case impl_blinded_balance_object_type:{
323  const auto& aobj = dynamic_cast<const blinded_balance_object*>(obj);
324  assert( aobj != nullptr );
325  for( const auto& a : aobj->owner.account_auths )
326  accounts.insert( a.first );
327  break;
328  } case impl_block_summary_object_type:
329  break;
330  case impl_account_transaction_history_object_type:
331  break;
332  case impl_chain_property_object_type:
333  break;
334  case impl_witness_schedule_object_type:
335  break;

346  assert( aobj != nullptr );
347  accounts.insert( aobj->bidder );
