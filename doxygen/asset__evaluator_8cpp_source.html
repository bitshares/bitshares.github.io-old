37 void_result asset_create_evaluator::do_evaluate( const asset_create_operation& op )
38 { try {
39 
40  database& d = db();
41 
42  const auto& chain_parameters = d.get_global_properties().parameters;
43  FC_ASSERT( op.common_options.whitelist_authorities.size() <= chain_parameters.maximum_asset_whitelist_authorities );
44  FC_ASSERT( op.common_options.blacklist_authorities.size() <= chain_parameters.maximum_asset_whitelist_authorities );
45 
46  // Check that all authorities do exist
47  for( auto id : op.common_options.whitelist_authorities )
48  d.get_object(id);
49  for( auto id : op.common_options.blacklist_authorities )
50  d.get_object(id);
51 
52  auto& asset_indx = d.get_index_type<asset_index>().indices().get<by_symbol>();
53  auto asset_symbol_itr = asset_indx.find( op.symbol );
54  FC_ASSERT( asset_symbol_itr == asset_indx.end() );
55 
56  if( d.head_block_time() > HARDFORK_385_TIME )
57  {
58 
59  if( d.head_block_time() <= HARDFORK_409_TIME )
60  {
61  auto dotpos = op.symbol.find( '.' );
62  if( dotpos != std::string::npos )
63  {
64  auto prefix = op.symbol.substr( 0, dotpos );
65  auto asset_symbol_itr = asset_indx.find( op.symbol );
66  FC_ASSERT( asset_symbol_itr != asset_indx.end(), "Asset ${s} may only be created by issuer of ${p}, but ${p} has not been registered",
67  ("s",op.symbol)("p",prefix) );
68  FC_ASSERT( asset_symbol_itr->issuer == op.issuer, "Asset ${s} may only be created by issuer of ${p}, ${i}",
69  ("s",op.symbol)("p",prefix)("i", op.issuer(d).name) );
70  }
71  }
72  else
73  {
74  auto dotpos = op.symbol.rfind( '.' );
75  if( dotpos != std::string::npos )
76  {
77  auto prefix = op.symbol.substr( 0, dotpos );
78  auto asset_symbol_itr = asset_indx.find( prefix );
79  FC_ASSERT( asset_symbol_itr != asset_indx.end(), "Asset ${s} may only be created by issuer of ${p}, but ${p} has not been registered",
80  ("s",op.symbol)("p",prefix) );
81  FC_ASSERT( asset_symbol_itr->issuer == op.issuer, "Asset ${s} may only be created by issuer of ${p}, ${i}",
82  ("s",op.symbol)("p",prefix)("i", op.issuer(d).name) );
83  }
84  }
85 
86  }
87  else
88  {
89  auto dotpos = op.symbol.find( '.' );
90  if( dotpos != std::string::npos )
91  wlog( "Asset ${s} has a name which requires hardfork 385", ("s",op.symbol) );
92  }
93 
94  if( op.bitasset_opts )
95  {
96  const asset_object& backing = op.bitasset_opts->short_backing_asset(d);
97  if( backing.is_market_issued() )
98  {
99  const asset_bitasset_data_object& backing_bitasset_data = backing.bitasset_data(d);
100  const asset_object& backing_backing = backing_bitasset_data.options.short_backing_asset(d);
101  FC_ASSERT( !backing_backing.is_market_issued(),
102  "May not create a bitasset backed by a bitasset backed by a bitasset." );
103  FC_ASSERT( op.issuer != GRAPHENE_COMMITTEE_ACCOUNT || backing_backing.get_id() == asset_id_type(),
104  "May not create a blockchain-controlled market asset which is not backed by CORE.");
105  } else
106  FC_ASSERT( op.issuer != GRAPHENE_COMMITTEE_ACCOUNT || backing.get_id() == asset_id_type(),
107  "May not create a blockchain-controlled market asset which is not backed by CORE.");
108  FC_ASSERT( op.bitasset_opts->feed_lifetime_sec > chain_parameters.block_interval &&
109  op.bitasset_opts->force_settlement_delay_sec > chain_parameters.block_interval );
110  }
111  if( op.is_prediction_market )
112  {
113  FC_ASSERT( op.bitasset_opts );
114  FC_ASSERT( op.precision == op.bitasset_opts->short_backing_asset(d).precision );
115  }
116 
117  return void_result();
118 } FC_CAPTURE_AND_RETHROW( (op) ) }

276  "May not create a blockchain-controlled market asset which is not backed by CORE.");

278  }
279 
280  if( (d.head_block_time() < HARDFORK_572_TIME) || (a.dynamic_asset_data_id(d).current_supply != 0) )
281  {
282  // new issuer_permissions must be subset of old issuer permissions
283  FC_ASSERT(!(o.new_options.issuer_permissions & ~a.options.issuer_permissions),

289  "Flag change is forbidden by issuer permissions");

298  d.get_object(id);
299  FC_ASSERT( o.new_options.blacklist_authorities.size() <= chain_parameters.maximum_asset_whitelist_authorities );

306 void_result asset_update_evaluator::do_apply(const asset_update_operation& o)
307 { try {
308  database& d = db();
309 
310  // If we are now disabling force settlements, cancel all open force settlement orders
311  if( o.new_options.flags & disable_force_settle && asset_to_update->can_force_settle() )
312  {
313  const auto& idx = d.get_index_type<force_settlement_index>().indices().get<by_expiration>();
314  // Funky iteration code because we're removing objects as we go. We have to re-initialize itr every loop instead
315  // of simply incrementing it.
316  for( auto itr = idx.lower_bound(o.asset_to_update);
317  itr != idx.end() && itr->settlement_asset_id() == o.asset_to_update;
318  itr = idx.lower_bound(o.asset_to_update) )
319  d.cancel_order(*itr);
320  }
321 
322  d.modify(*asset_to_update, [&](asset_object& a) {
323  if( o.new_issuer )
324  a.issuer = *o.new_issuer;
325  a.options = o.new_options;
326  });
327 
328  return void_result();
329 } FC_CAPTURE_AND_RETHROW( (o) ) }

403 void_result asset_update_feed_producers_evaluator::do_apply(const asset_update_feed_producers_evaluator::operation_type& o)
404 { try {
405  db().modify(*bitasset_to_update, [&](asset_bitasset_data_object& a) {
406  //This is tricky because I have a set of publishers coming in, but a map of publisher to feed is stored.
407  //I need to update the map such that the keys match the new publishers, but not munge the old price feeds from
408  //publishers who are being kept.
409  //First, remove any old publishers who are no longer publishers
410  for( auto itr = a.feeds.begin(); itr != a.feeds.end(); )
411  {
412  if( !o.new_feed_producers.count(itr->first) )
413  itr = a.feeds.erase(itr);
414  else
415  ++itr;
416  }
417  //Now, add any new publishers
418  for( auto itr = o.new_feed_producers.begin(); itr != o.new_feed_producers.end(); ++itr )
419  if( !a.feeds.count(*itr) )
420  a.feeds[*itr];
421  a.update_median_feeds(db().head_block_time());
422  });
423  db().check_call_orders( o.asset_to_update(db()) );
424 
425  return void_result();

478 operation_result asset_settle_evaluator::do_apply(const asset_settle_evaluator::operation_type& op)
479 { try {
480  database& d = db();
481  d.adjust_balance(op.account, -op.amount);
482 
483  const auto& bitasset = asset_to_settle->bitasset_data(d);
484  if( bitasset.has_settlement() )
485  {
486  const auto& mia_dyn = asset_to_settle->dynamic_asset_data_id(d);
487 
488  auto settled_amount = op.amount * bitasset.settlement_price;
489  if( op.amount.amount == mia_dyn.current_supply )
490  settled_amount.amount = bitasset.settlement_fund; // avoid rounding problems
491  else
492  FC_ASSERT( settled_amount.amount <= bitasset.settlement_fund ); // should be strictly < except for PM with zero outcome
493 
494  d.modify( bitasset, [&]( asset_bitasset_data_object& obj ){
495  obj.settlement_fund -= settled_amount.amount;
496  });
497 
498  d.adjust_balance(op.account, settled_amount);
499 
500  d.modify( mia_dyn, [&]( asset_dynamic_data_object& obj ){
501  obj.current_supply -= op.amount.amount;
502  });
503 
504  return settled_amount;
505  }
