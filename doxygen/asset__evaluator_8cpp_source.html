36  database& d = db();

38  const auto& chain_parameters = d.get_global_properties().parameters;
39  FC_ASSERT( op.common_options.whitelist_authorities.size() <= chain_parameters.maximum_asset_whitelist_authorities );
40  FC_ASSERT( op.common_options.blacklist_authorities.size() <= chain_parameters.maximum_asset_whitelist_authorities );
41 
42  // Check that all authorities do exist

44  d.get_object(id);
45  for( auto id : op.common_options.blacklist_authorities )
46  d.get_object(id);
47 
48  auto& asset_indx = d.get_index_type<asset_index>().indices().get<by_symbol>();
49  auto asset_symbol_itr = asset_indx.find( op.symbol );
50  FC_ASSERT( asset_symbol_itr == asset_indx.end() );
51 
52  if( d.head_block_time() > HARDFORK_385_TIME )
53  {
54 
55  if( d.head_block_time() <= HARDFORK_409_TIME )
56  {
57  auto dotpos = op.symbol.find( '.' );
58  if( dotpos != std::string::npos )
59  {
60  auto prefix = op.symbol.substr( 0, dotpos );
61  auto asset_symbol_itr = asset_indx.find( op.symbol );
62  FC_ASSERT( asset_symbol_itr != asset_indx.end(), "Asset ${s} may only be created by issuer of ${p}, but ${p} has not been registered",
63  ("s",op.symbol)("p",prefix) );
64  FC_ASSERT( asset_symbol_itr->issuer == op.issuer, "Asset ${s} may only be created by issuer of ${p}, ${i}",
65  ("s",op.symbol)("p",prefix)("i", op.issuer(d).name) );
66  }
67  }
68  else
69  {
70  auto dotpos = op.symbol.rfind( '.' );
71  if( dotpos != std::string::npos )
72  {
73  auto prefix = op.symbol.substr( 0, dotpos );
74  auto asset_symbol_itr = asset_indx.find( prefix );
75  FC_ASSERT( asset_symbol_itr != asset_indx.end(), "Asset ${s} may only be created by issuer of ${p}, but ${p} has not been registered",
76  ("s",op.symbol)("p",prefix) );
77  FC_ASSERT( asset_symbol_itr->issuer == op.issuer, "Asset ${s} may only be created by issuer of ${p}, ${i}",
78  ("s",op.symbol)("p",prefix)("i", op.issuer(d).name) );
79  }
80  }
81 
82  }
83  else
84  {
85  auto dotpos = op.symbol.find( '.' );
86  if( dotpos != std::string::npos )
87  wlog( "Asset ${s} has a name which requires hardfork 385", ("s",op.symbol) );
88  }
89 
90  core_fee_paid -= core_fee_paid.value/2;
91 
92  if( op.bitasset_opts )
93  {
94  const asset_object& backing = op.bitasset_opts->short_backing_asset(d);
95  if( backing.is_market_issued() )

113  }
