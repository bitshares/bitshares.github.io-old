144  virtual operation_result evaluate(const operation& o) final override
145  {
146  auto* eval = static_cast<DerivedEvaluator*>(this);
147  const auto& op = o.get<typename DerivedEvaluator::operation_type>();
148 
149  prepare_fee(op.fee_payer(), op.fee);

157  }
