41  {
42  if( acct.allowed_assets->find( asset_obj.id ) == acct.allowed_assets->end() )
43  return false;
44  // must still pass other checks even if it is in allowed_assets
45  }
46

50  return false;
51  }
52

56  return true;

60  {

64
