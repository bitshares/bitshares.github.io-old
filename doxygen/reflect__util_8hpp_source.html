40 std::string clean_name( const std::string& name )
41 {
42  std::string result;
43  const static std::string prefix = "graphene::chain::";
44  const static std::string suffix = "_operation";
45  // graphene::chain::.*_operation
46  if( (name.size() >= prefix.size() + suffix.size())
47  && (name.substr( 0, prefix.size() ) == prefix)
48  && (name.substr( name.size()-suffix.size(), suffix.size() ) == suffix )
49  )
50  return name.substr( prefix.size(), name.size() - prefix.size() - suffix.size() );
51 
52  wlog( "don't know how to clean name: ${name}", ("name", name) );
53  return name;
54 }
