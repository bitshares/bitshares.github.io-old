23 inline bool sum_below_max_shares( const asset& a, const asset& b )
24 {
25  assert( GRAPHENE_MAX_SHARE_SUPPLY + GRAPHENE_MAX_SHARE_SUPPLY > GRAPHENE_MAX_SHARE_SUPPLY );

32 asset linear_vesting_policy::get_allowed_withdraw( const vesting_policy_context& ctx ) const
33 {
34  if( ctx.now <= begin_date )
35  return asset( 0, ctx.balance.asset_id );
36  if( vesting_seconds == 0 )
37  return ctx.balance;
38 
39  int64_t elapsed_seconds = (ctx.now - begin_date).to_seconds();
40 
41  // if elapsed_seconds <= 0, then ctx.now <= begin_date,
42  // and we should have returned above.
43  assert( elapsed_seconds > 0 );
44 
45  fc::uint128_t total_allowed = begin_balance.value;
46  total_allowed *= uint64_t( elapsed_seconds );
47  total_allowed /= vesting_seconds;
48 
49  if( total_allowed <= total_withdrawn.value )
50  return asset( 0, ctx.balance.asset_id );
51  total_allowed -= total_withdrawn.value;
52  FC_ASSERT( total_allowed <= GRAPHENE_MAX_SHARE_SUPPLY );
53  return asset( total_allowed.to_uint64(), ctx.balance.asset_id );
54 }

78 fc::uint128_t cdd_vesting_policy::compute_coin_seconds_earned( const vesting_policy_context& ctx )const
79 {
80  assert( ctx.now >= coin_seconds_earned_last_update );
81  int64_t delta_seconds = (ctx.now - coin_seconds_earned_last_update).to_seconds();
82  assert( delta_seconds >= 0 );
83 
84  fc::uint128_t delta_coin_seconds = ctx.balance.amount.value;
85  delta_coin_seconds *= delta_seconds;
86 
87  fc::uint128_t coin_seconds_earned_cap = ctx.balance.amount.value;
88  coin_seconds_earned_cap *= vesting_seconds;
89 
90  return std::min(
91  coin_seconds_earned + delta_coin_seconds,
92  coin_seconds_earned_cap
93  );
