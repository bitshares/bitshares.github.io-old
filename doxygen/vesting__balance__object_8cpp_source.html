29 inline bool sum_below_max_shares(const asset& a, const asset& b)
30 {
31  assert(GRAPHENE_MAX_SHARE_SUPPLY + GRAPHENE_MAX_SHARE_SUPPLY > GRAPHENE_MAX_SHARE_SUPPLY);
32  return (a.amount <= GRAPHENE_MAX_SHARE_SUPPLY)
33  && ( b.amount <= GRAPHENE_MAX_SHARE_SUPPLY)
34  && ((a.amount + b.amount) <= GRAPHENE_MAX_SHARE_SUPPLY);
35 }

37 asset linear_vesting_policy::get_allowed_withdraw( const vesting_policy_context& ctx )const
38 {
39  share_type allowed_withdraw = 0;
40 
41  if( ctx.now > begin_timestamp )
42  {
43  const auto elapsed_seconds = (ctx.now - begin_timestamp).to_seconds();
44  assert( elapsed_seconds > 0 );
45 
46  if( elapsed_seconds >= vesting_cliff_seconds )
47  {
48  share_type total_vested = 0;
49  if( elapsed_seconds < vesting_duration_seconds )
50  {
51  total_vested = (fc::uint128_t( begin_balance.value ) * elapsed_seconds / vesting_duration_seconds).to_uint64();
52  }
53  else
54  {
55  total_vested = begin_balance;
56  }
57  assert( total_vested >= 0 );
58 
59  const share_type withdrawn_already = begin_balance - ctx.balance.amount;
60  assert( withdrawn_already >= 0 );
61 
62  allowed_withdraw = total_vested - withdrawn_already;
63  assert( allowed_withdraw >= 0 );
64  }
65  }
66 
67  return asset( allowed_withdraw, ctx.balance.asset_id );
68 }

90 fc::uint128_t cdd_vesting_policy::compute_coin_seconds_earned(const vesting_policy_context& ctx)const
91 {
92  assert(ctx.now >= coin_seconds_earned_last_update);
93  int64_t delta_seconds = (ctx.now - coin_seconds_earned_last_update).to_seconds();
94  assert(delta_seconds >= 0);
95 
96  fc::uint128_t delta_coin_seconds = ctx.balance.amount.value;
97  delta_coin_seconds *= delta_seconds;
98 
99  fc::uint128_t coin_seconds_earned_cap = ctx.balance.amount.value;
100  coin_seconds_earned_cap *= std::max(vesting_seconds, 1u);
101 
102  return std::min(coin_seconds_earned + delta_coin_seconds, coin_seconds_earned_cap);
103 }

111 asset cdd_vesting_policy::get_allowed_withdraw(const vesting_policy_context& ctx)const
112 {
113  if(ctx.now <= start_claim)
114  return asset(0, ctx.balance.asset_id);
115  fc::uint128_t cs_earned = compute_coin_seconds_earned(ctx);
116  fc::uint128_t withdraw_available = cs_earned / std::max(vesting_seconds, 1u);

132 void cdd_vesting_policy::on_withdraw(const vesting_policy_context& ctx)
133 {

135  fc::uint128_t coin_seconds_needed = ctx.amount.amount.value;
136  coin_seconds_needed *= std::max(vesting_seconds, 1u);
137  // is_withdraw_allowed should forbid any withdrawal that
138  // would trigger this assert
