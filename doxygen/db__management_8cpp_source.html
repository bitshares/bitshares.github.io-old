46 void database::reindex(fc::path data_dir, const genesis_state_type& initial_allocation)
47 { try {
48  ilog( "reindexing blockchain" );
49  wipe(data_dir, false);
50  open(data_dir, [&initial_allocation]{return initial_allocation;});
51 
52  auto start = fc::time_point::now();
53  auto last_block = _block_id_to_block.last();
54  if( !last_block ) {
55  elog( "!no last block" );
56  edump((last_block));
57  return;
58  }
59 
60  const auto last_block_num = last_block->block_num();
61 
62  ilog( "Replaying blocks..." );
63  _undo_db.disable();
64  for( uint32_t i = 1; i <= last_block_num; ++i )
65  {
66  if( i % 2000 == 0 ) std::cerr << " " << double(i*100)/last_block_num << "% "<<i << " of " <<last_block_num<<" \n";
67  fc::optional< signed_block > block = _block_id_to_block.fetch_by_number(i);
68  if( !block.valid() )
69  {
70  wlog( "Reindexing terminated due to gap: Block ${i} does not exist!", ("i", i) );
71  uint32_t dropped_count = 0;
72  while( true )
73  {
74  fc::optional< block_id_type > last_id = _block_id_to_block.last_id();
75  // this can trigger if we attempt to e.g. read a file that has block #2 but no block #1
76  if( !last_id.valid() )
77  break;
78  // we've caught up to the gap
79  if( block_header::num_from_id( *last_id ) <= i )
80  break;
81  _block_id_to_block.remove( *last_id );
82  dropped_count++;
83  }
84  wlog( "Dropped ${n} blocks from after the gap", ("n", dropped_count) );
85  break;
86  }
87  apply_block(*block, skip_witness_signature |
88  skip_transaction_signatures |
89  skip_transaction_dupe_check |
90  skip_tapos_check |
91  skip_witness_schedule_check |
92  skip_authority_check);
93  }
94  _undo_db.enable();
95  auto end = fc::time_point::now();
96  ilog( "Done reindexing, elapsed time: ${t} sec", ("t",double((end-start).count())/1000000.0 ) );
97 } FC_CAPTURE_AND_RETHROW( (data_dir) ) }

136 void database::close(bool rewind)
137 {
138  // TODO: Save pending tx's on close()
139  clear_pending();
140 
141  // pop all of the blocks that we can given our undo history, this should
142  // throw when there is no more undo history to pop
143  if( rewind )
144  {
145  try
146  {
147  uint32_t cutoff = get_dynamic_global_properties().last_irreversible_block_num;
148 
149  while( head_block_num() > cutoff )
150  {
151  // elog("pop");
152  block_id_type popped_block_id = head_block_id();
153  pop_block();
154  _fork_db.remove(popped_block_id); // doesn't throw on missing
155  try
156  {
157  _block_id_to_block.remove(popped_block_id);
158  }
159  catch (const fc::key_not_found_exception&)
160  {
161  }
162  }
163  }
164  catch (...)
165  {
166  }
167  }
168 
169  // Since pop_block() will move tx's in the popped blocks into pending,
170  // we have to clear_pending() after we're done popping to get a clean
171  // DB state (issue #336).
172  clear_pending();
173 
174  object_database::flush();
175  object_database::close();
176 
177  if( _block_id_to_block.is_open() )
178  _block_id_to_block.close();
179 
180  _fork_db.reset();
181 }
