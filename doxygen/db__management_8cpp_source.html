49 void database::reindex( fc::path data_dir )
50 { try {
51  auto last_block = _block_id_to_block.last();
52  if( !last_block ) {
53  elog( "!no last block" );
54  edump((last_block));
55  return;
56  }
57  if( last_block->block_num() <= head_block_num()) return;
58 
59  ilog( "reindexing blockchain" );
60  auto start = fc::time_point::now();
61  const auto last_block_num = last_block->block_num();
62  uint32_t flush_point = last_block_num < 10000 ? 0 : last_block_num - 10000;
63  uint32_t undo_point = last_block_num < 50 ? 0 : last_block_num - 50;
64 
65  ilog( "Replaying blocks, starting at ${next}...", ("next",head_block_num() + 1) );
66  if( head_block_num() >= undo_point )
67  {
68  if( head_block_num() > 0 )
69  _fork_db.start_block( *fetch_block_by_number( head_block_num() ) );
70  }
71  else
72  _undo_db.disable();
73  for( uint32_t i = head_block_num() + 1; i <= last_block_num; ++i )
74  {
75  if( i % 10000 == 0 ) std::cerr << " " << double(i*100)/last_block_num << "% "<<i << " of " <<last_block_num<<" \n";
76  if( i == flush_point )
77  {
78  ilog( "Writing database to disk at block ${i}", ("i",i) );
79  flush();
80  ilog( "Done" );
81  }
82  fc::optional< signed_block > block = _block_id_to_block.fetch_by_number(i);
83  if( !block.valid() )
84  {
85  wlog( "Reindexing terminated due to gap: Block ${i} does not exist!", ("i", i) );
86  uint32_t dropped_count = 0;
87  while( true )
88  {
89  fc::optional< block_id_type > last_id = _block_id_to_block.last_id();
90  // this can trigger if we attempt to e.g. read a file that has block #2 but no block #1
91  if( !last_id.valid() )
92  break;
93  // we've caught up to the gap
94  if( block_header::num_from_id( *last_id ) <= i )
95  break;
96  _block_id_to_block.remove( *last_id );
97  dropped_count++;
98  }
99  wlog( "Dropped ${n} blocks from after the gap", ("n", dropped_count) );
100  break;
101  }
102  if( i < undo_point )
103  apply_block(*block, skip_witness_signature |
104  skip_transaction_signatures |
105  skip_transaction_dupe_check |
106  skip_tapos_check |
107  skip_witness_schedule_check |
108  skip_authority_check);
109  else
110  {
111  _undo_db.enable();
112  push_block(*block, skip_witness_signature |
113  skip_transaction_signatures |
114  skip_transaction_dupe_check |
115  skip_tapos_check |
116  skip_witness_schedule_check |
117  skip_authority_check);
118  }
119  }
120  _undo_db.enable();
121  auto end = fc::time_point::now();
122  ilog( "Done reindexing, elapsed time: ${t} sec", ("t",double((end-start).count())/1000000.0 ) );
123 } FC_CAPTURE_AND_RETHROW( (data_dir) ) }

136 void database::open(
137  const fc::path& data_dir,
138  std::function<genesis_state_type()> genesis_loader,
139  const std::string& db_version)
140 {
141  try
142  {
143  bool wipe_object_db = false;
144  if( !fc::exists( data_dir / "db_version" ) )
145  wipe_object_db = true;
146  else
147  {
148  std::string version_string;
149  fc::read_file_contents( data_dir / "db_version", version_string );
150  wipe_object_db = ( version_string != db_version );
151  }
152  if( wipe_object_db ) {
153  ilog("Wiping object_database due to missing or wrong version");
154  object_database::wipe( data_dir );
155  std::ofstream version_file( (data_dir / "db_version").generic_string().c_str(),
156  std::ios::out | std::ios::binary | std::ios::trunc );
157  version_file.write( db_version.c_str(), db_version.size() );
158  version_file.close();
159  }
160 
161  object_database::open(data_dir);
162 
163  _block_id_to_block.open(data_dir / "database" / "block_num_to_block");
164 
165  if( !find(global_property_id_type()) )
166  init_genesis(genesis_loader());
167 
168  fc::optional<block_id_type> last_block = _block_id_to_block.last_id();
169  if( last_block.valid() )
170  {
171  FC_ASSERT( *last_block >= head_block_id(),
172  "last block ID does not match current chain state",
173  ("last_block->id", last_block)("head_block_id",head_block_num()) );
174  reindex( data_dir );
175  }
176  _opened = true;
177  }
178  FC_CAPTURE_LOG_AND_RETHROW( (data_dir) )
179 }

181 void database::close(bool rewind)
182 {
183  // TODO: Save pending tx's on close()
184  clear_pending();
185 
186  // pop all of the blocks that we can given our undo history, this should
187  // throw when there is no more undo history to pop
188  if( rewind )
189  {
190  try
191  {
192  uint32_t cutoff = get_dynamic_global_properties().last_irreversible_block_num;
193 
194  ilog( "Rewinding from ${head} to ${cutoff}", ("head",head_block_num())("cutoff",cutoff) );
195  while( head_block_num() > cutoff )
196  {
197  block_id_type popped_block_id = head_block_id();
198  pop_block();
199  _fork_db.remove(popped_block_id); // doesn't throw on missing
200  }
201  }
202  catch ( const fc::exception& e )
203  {
204  wlog( "Database close unexpected exception: ${e}", ("e", e) );
205  }
206  }
207 
208  // Since pop_block() will move tx's in the popped blocks into pending,
209  // we have to clear_pending() after we're done popping to get a clean
210  // DB state (issue #336).
211  clear_pending();
212 
213  object_database::flush();
214  object_database::close();
215 
216  if( _block_id_to_block.is_open() )
217  _block_id_to_block.close();
218 
219  _fork_db.reset();
220 
221  _opened = false;
222 }
