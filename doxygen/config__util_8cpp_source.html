54  const boost::shared_ptr<bpo::option_description> next(const boost::shared_ptr<bpo::option_description>& o)
55  {
56  const std::string name = o->long_name();
57  if( seen.find( name ) != seen.end() )
58  return nullptr;
59  seen.insert(name);
60  return modifier ? modifier(o) : o;
61  }
62 
63 private:
64  boost::container::flat_set<std::string> seen;
65  const boost::shared_ptr<bpo::option_description> (*modifier)(const boost::shared_ptr<bpo::option_description>&);
66 };

116 // logging config is too complicated to be parsed by boost::program_options,
117 // so we do it by hand
118 static fc::optional<fc::logging_config> load_logging_config_from_ini_file(const fc::path& config_ini_filename)
119 {
120  try
121  {
122  fc::logging_config logging_config;
123  bool found_logging_config = false;
124 
125  boost::property_tree::ptree config_ini_tree;
126  boost::property_tree::ini_parser::read_ini(config_ini_filename.preferred_string().c_str(), config_ini_tree);
127  for (const auto& section : config_ini_tree)
128  {
129  const std::string& section_name = section.first;
130  const boost::property_tree::ptree& section_tree = section.second;
131 
132  const std::string console_appender_section_prefix = "log.console_appender.";
133  const std::string file_appender_section_prefix = "log.file_appender.";
134  const std::string logger_section_prefix = "logger.";
135 
136  if (boost::starts_with(section_name, console_appender_section_prefix))
137  {
138  std::string console_appender_name = section_name.substr(console_appender_section_prefix.length());
139  std::string stream_name = section_tree.get<std::string>("stream");
140 
141  // construct a default console appender config here
142  // stdout/stderr will be taken from ini file, everything else hard-coded here
143  fc::console_appender::config console_appender_config;
144  console_appender_config.level_colors.emplace_back(
145  fc::console_appender::level_color(fc::log_level::debug,
146  fc::console_appender::color::green));
147  console_appender_config.level_colors.emplace_back(
148  fc::console_appender::level_color(fc::log_level::warn,
149  fc::console_appender::color::brown));
150  console_appender_config.level_colors.emplace_back(
151  fc::console_appender::level_color(fc::log_level::error,
152  fc::console_appender::color::cyan));
153  console_appender_config.stream = fc::variant(stream_name).as<fc::console_appender::stream::type>(GRAPHENE_MAX_NESTED_OBJECTS);
154  logging_config.appenders.push_back(fc::appender_config(console_appender_name, "console", fc::variant(console_appender_config, GRAPHENE_MAX_NESTED_OBJECTS)));
155  found_logging_config = true;
156  }
157  else if (boost::starts_with(section_name, file_appender_section_prefix))
158  {
159  std::string file_appender_name = section_name.substr(file_appender_section_prefix.length());
160  fc::path file_name = section_tree.get<std::string>("filename");
161  if (file_name.is_relative())
162  file_name = fc::absolute(config_ini_filename).parent_path() / file_name;
163 
164  int interval = section_tree.get_optional<int>("rotation_interval").get_value_or(60);
165  int limit = section_tree.get_optional<int>("rotation_limit").get_value_or(1);
166 
167  // construct a default file appender config here
168  // filename will be taken from ini file, everything else hard-coded here
169  fc::file_appender::config file_appender_config;
170  file_appender_config.filename = file_name;
171  file_appender_config.flush = true;
172  file_appender_config.rotate = true;
173  file_appender_config.rotation_interval = fc::minutes(interval);
174  file_appender_config.rotation_limit = fc::days(limit);
175  logging_config.appenders.push_back(fc::appender_config(file_appender_name, "file", fc::variant(file_appender_config, GRAPHENE_MAX_NESTED_OBJECTS)));
176  found_logging_config = true;
177  }
178  else if (boost::starts_with(section_name, logger_section_prefix))
179  {
180  std::string logger_name = section_name.substr(logger_section_prefix.length());
181  std::string level_string = section_tree.get<std::string>("level");
182  std::string appenders_string = section_tree.get<std::string>("appenders");
183  fc::logger_config logger_config(logger_name);
184  logger_config.level = fc::variant(level_string).as<fc::log_level>(5);
185  boost::split(logger_config.appenders, appenders_string,
186  boost::is_any_of(" ,"),
187  boost::token_compress_on);
188  logging_config.loggers.push_back(logger_config);
189  found_logging_config = true;
190  }
191  }
192  if (found_logging_config)
193  return logging_config;
194  else
195  return fc::optional<fc::logging_config>();
196  }
197  FC_RETHROW_EXCEPTIONS(warn, "")
198 }

208 static void load_config_file(const fc::path& config_ini_path, const bpo::options_description& cfg_options,
209  bpo::variables_map& options )
210 {
211  deduplicator dedup;
212  bpo::options_description unique_options("Graphene Witness Node");
213  for( const boost::shared_ptr<bpo::option_description> opt : cfg_options.options() )
214  {
215  const boost::shared_ptr<bpo::option_description> od = dedup.next(opt);
216  if( !od ) continue;
217  unique_options.add( od );
218  }
219 
220  // get the basic options
221  bpo::store(bpo::parse_config_file<char>(config_ini_path.preferred_string().c_str(),
222  unique_options, true), options);
223 }

225 static bool load_logging_config_file(const fc::path& config_ini_path)
226 {
227  // try to get logging options from the config file.
228  try
229  {
230  fc::optional<fc::logging_config> logging_config = load_logging_config_from_ini_file(config_ini_path);
231  if (logging_config)
232  {
233  fc::configure_logging(*logging_config);
234  return true;
235  }
236  }
237  catch (const fc::exception& ex)
238  {
239  wlog("Error parsing logging config from logging config file ${config}, using default config", ("config", config_ini_path.preferred_string()));
240  }
241  return false;
242 }

244 static void create_new_config_file(const fc::path& config_ini_path, const fc::path& data_dir,
245  const bpo::options_description& cfg_options )
246 {
247  ilog("Writing new config file at ${path}", ("path", config_ini_path));
248  if( !fc::exists(data_dir) )
249  fc::create_directories(data_dir);
250 
251  auto modify_option_defaults = [](const boost::shared_ptr<bpo::option_description>& o) -> const boost::shared_ptr<bpo::option_description> {
252  const std::string& name = o->long_name();
253  if( name == "partial-operations" )
254  return new_option_description(name, bpo::value<bool>()->default_value(true), o->description() );
255  if( name == "max-ops-per-account" )
256  return new_option_description(name, bpo::value<int>()->default_value(100), o->description() );
257  return o;
258  };
259  deduplicator dedup(modify_option_defaults);
260  std::ofstream out_cfg(config_ini_path.preferred_string());
261  for( const boost::shared_ptr<bpo::option_description> opt : cfg_options.options() )
262  {
263  const boost::shared_ptr<bpo::option_description> od = dedup.next(opt);
264  if( !od ) continue;
265 
266  if( !od->description().empty() )
267  out_cfg << "# " << od->description() << "\n";
268  boost::any store;
269  if( !od->semantic()->apply_default(store) )
270  out_cfg << "# " << od->long_name() << " = \n";
271  else {
272  auto example = od->format_parameter();
273  if( example.empty() )
274  // This is a boolean switch
275  out_cfg << od->long_name() << " = " << "false\n";
276  else {
277  // The string is formatted "arg (=<interesting part>)"
278  example.erase(0, 6);
279  example.erase(example.length()-1);
280  out_cfg << od->long_name() << " = " << example << "\n";
281  }
282  }
283  out_cfg << "\n";
284  }
285 
286  out_cfg << "\n"
287  << "# Logging configuration is loaded from logging.ini by default.\n"
288  << "# If logging.ini exists, logging configuration added in this file will be ignored.\n";
289  out_cfg.close();
290 }

307  void load_configuration_options(const fc::path& data_dir, const bpo::options_description& cfg_options, bpo::variables_map& options)
308  {
309  const auto config_ini_path = data_dir / "config.ini";
310  const auto logging_ini_path = data_dir / "logging.ini";
311 
312  if(!exists(config_ini_path) && fc::exists(logging_ini_path))
313  {
314  // this is an uncommon case
315  create_new_config_file(config_ini_path, data_dir, cfg_options);
316  }
317  else if(!exists(config_ini_path))
318  {
319  // create default config.ini and logging.ini
320  create_new_config_file(config_ini_path, data_dir, cfg_options);
321  create_logging_config_file(logging_ini_path, data_dir);
322  }
323 
324  // load witness node configuration
325  load_config_file(config_ini_path, cfg_options, options);
326 
327  // load logging configuration
328  if (fc::exists(logging_ini_path))
329  {
330  load_logging_config_file(logging_ini_path);
331  }
332  else
333  {
334  // this is the legacy config.ini case
335  load_logging_config_file(config_ini_path);
336  }
337  }
