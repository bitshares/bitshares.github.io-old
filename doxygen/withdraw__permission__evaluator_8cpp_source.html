59 void_result withdraw_permission_claim_evaluator::do_evaluate(const withdraw_permission_claim_evaluator::operation_type& op)
60 { try {
61  const database& d = db();
62  time_point_sec head_block_time = d.head_block_time();
63 
64  const withdraw_permission_object& permit = op.withdraw_permission(d);
65  FC_ASSERT(permit.expiration > head_block_time);
66  FC_ASSERT(permit.authorized_account == op.withdraw_to_account);
67  FC_ASSERT(permit.withdraw_from_account == op.withdraw_from_account);
68  if (head_block_time >= HARDFORK_23_TIME) {
69  FC_ASSERT(permit.period_start_time <= head_block_time);
70  }
71  FC_ASSERT(op.amount_to_withdraw <= permit.available_this_period( head_block_time ) );
72  FC_ASSERT(d.get_balance(op.withdraw_from_account, op.amount_to_withdraw.asset_id) >= op.amount_to_withdraw);
73 
74  const asset_object& _asset = op.amount_to_withdraw.asset_id(d);
75  if( _asset.is_transfer_restricted() ) FC_ASSERT( _asset.issuer == permit.authorized_account || _asset.issuer == permit.withdraw_from_account );
76 
77  const account_object& from = op.withdraw_to_account(d);
78  const account_object& to = permit.authorized_account(d);
79  FC_ASSERT( is_authorized_asset( d, to, _asset ) );
80  FC_ASSERT( is_authorized_asset( d, from, _asset ) );
81 
82  return void_result();
83 } FC_CAPTURE_AND_RETHROW( (op) ) }

85 void_result withdraw_permission_claim_evaluator::do_apply(const withdraw_permission_claim_evaluator::operation_type& op)
86 { try {
87  database& d = db();
88 
89  const withdraw_permission_object& permit = d.get(op.withdraw_permission);
90  d.modify(permit, [&](withdraw_permission_object& p) {
91  auto periods = (d.head_block_time() - p.period_start_time).to_seconds() / p.withdrawal_period_sec;
92  p.period_start_time += periods * p.withdrawal_period_sec;
93  if( periods == 0 )
94  p.claimed_this_period += op.amount_to_withdraw.amount;
95  else
96  p.claimed_this_period = op.amount_to_withdraw.amount;
97  });
98 
99  d.adjust_balance(op.withdraw_from_account, -op.amount_to_withdraw);
100  d.adjust_balance(op.withdraw_to_account, op.amount_to_withdraw);
101 
102  return void_result();
103 } FC_CAPTURE_AND_RETHROW( (op) ) }
