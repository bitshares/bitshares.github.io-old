42 fc::time_point_sec database::get_slot_time(uint32_t slot_num)const
43 {
44  if( slot_num == 0 )
45  return fc::time_point_sec();
46 
47  auto interval = block_interval();
48  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
49 
50  if( head_block_num() == 0 )
51  {
52  // n.b. first block is at genesis_time plus one block interval
53  fc::time_point_sec genesis_time = dpo.time;
54  return genesis_time + slot_num * interval;
55  }
56 
57  int64_t head_block_abs_slot = head_block_time().sec_since_epoch() / interval;
58  fc::time_point_sec head_slot_time(head_block_abs_slot * interval);
59 
60  const global_property_object& gpo = get_global_properties();
61 
62  if( dpo.dynamic_flags & dynamic_global_property_object::maintenance_flag )
63  slot_num += gpo.parameters.maintenance_skip_slots;
64 
65  // "slot 0" is head_slot_time
66  // "slot 1" is head_slot_time,
67  // plus maint interval if head block is a maint block
68  // plus block interval if head block is not a maint block
69  return head_slot_time + (slot_num * interval);
70 }

86 void database::update_witness_schedule()
87 {
88  const witness_schedule_object& wso = witness_schedule_id_type()(*this);
89  const global_property_object& gpo = get_global_properties();
90 
91  if( head_block_num() % gpo.active_witnesses.size() == 0 )
92  {
93  modify( wso, [&]( witness_schedule_object& _wso )
94  {
95  _wso.current_shuffled_witnesses.clear();
96  _wso.current_shuffled_witnesses.reserve( gpo.active_witnesses.size() );
97 
98  for( const witness_id_type& w : gpo.active_witnesses )
99  _wso.current_shuffled_witnesses.push_back( w );
100 
101  auto now_hi = uint64_t(head_block_time().sec_since_epoch()) << 32;
102  for( uint32_t i = 0; i < _wso.current_shuffled_witnesses.size(); ++i )
103  {
106  uint64_t k = now_hi + uint64_t(i)*2685821657736338717ULL;
107  k ^= (k >> 12);
108  k ^= (k << 25);
109  k ^= (k >> 27);
110  k *= 2685821657736338717ULL;
111 
112  uint32_t jmax = _wso.current_shuffled_witnesses.size() - i;
113  uint32_t j = i + k%jmax;
114  std::swap( _wso.current_shuffled_witnesses[i],
115  _wso.current_shuffled_witnesses[j] );
116  }
117  });
118  }
119 }
