117  asset fee_schedule::calculate_fee( const operation& op, const price& core_exchange_rate )const
118  {
119  //idump( (op)(core_exchange_rate) );
120  fee_parameters params; params.set_which(op.which());
121  auto itr = parameters.find(params);
122  if( itr != parameters.end() ) params = *itr;
123  auto base_value = op.visit( calc_fee_visitor( params ) );
124  auto scaled = fc::uint128(base_value) * scale;
125  scaled /= GRAPHENE_100_PERCENT;
126  FC_ASSERT( scaled <= GRAPHENE_MAX_SHARE_SUPPLY );
127  //idump( (base_value)(scaled)(core_exchange_rate) );
128  auto result = asset( scaled.to_uint64(), 0 ) * core_exchange_rate;
129  //FC_ASSERT( result * core_exchange_rate >= asset( scaled.to_uint64()) );
130 
131  while( result * core_exchange_rate < asset( scaled.to_uint64()) )
132  result.amount++;
133 
134  FC_ASSERT( result.amount <= GRAPHENE_MAX_SHARE_SUPPLY );
135  return result;
136  }

145  void chain_parameters::validate()const
146  {
147  current_fees->validate();

152 
153  FC_ASSERT( block_interval >= GRAPHENE_MIN_BLOCK_INTERVAL );
154  FC_ASSERT( block_interval <= GRAPHENE_MAX_BLOCK_INTERVAL );
155  FC_ASSERT( block_interval > 0 );
156  FC_ASSERT( maintenance_interval > block_interval,
157  "Maintenance interval must be longer than block interval" );
158  FC_ASSERT( maintenance_interval % block_interval == 0,
159  "Maintenance interval must be a multiple of block interval" );
160  FC_ASSERT( maximum_transaction_size >= GRAPHENE_MIN_TRANSACTION_SIZE_LIMIT,
161  "Transaction size limit is too low" );
162  FC_ASSERT( maximum_block_size >= GRAPHENE_MIN_BLOCK_SIZE_LIMIT,
163  "Block size limit is too low" );
164  FC_ASSERT( maximum_time_until_expiration > block_interval,
165  "Maximum transaction expiration time must be greater than a block interval" );
166  FC_ASSERT( maximum_proposal_lifetime - committee_proposal_review_period > block_interval,
167  "Committee proposal review period must be less than the maximum proposal lifetime" );
