29 void memo_data::set_message(const fc::ecc::private_key& priv, const fc::ecc::public_key& pub,
30  const string& msg, uint64_t custom_nonce)
31 {
32  if( priv != fc::ecc::private_key() && public_key_type(pub) != public_key_type() )
33  {
34  from = priv.get_public_key();
35  to = pub;
36  if( custom_nonce == 0 )
37  {
38  uint64_t entropy = fc::sha224::hash(fc::ecc::private_key::generate())._hash[0];
39  entropy <<= 32;
40  entropy &= 0xff00000000000000;
41  nonce = (fc::time_point::now().time_since_epoch().count() & 0x00ffffffffffffff) | entropy;
42  } else
43  nonce = custom_nonce;
44  auto secret = priv.get_shared_secret(pub);
45  auto nonce_plus_secret = fc::sha512::hash(fc::to_string(nonce) + secret.str());
46  string text = memo_message(digest_type::hash(msg)._hash[0], msg).serialize();
47  message = fc::aes_encrypt( nonce_plus_secret, vector<char>(text.begin(), text.end()) );
48  }
49  else

56 string memo_data::get_message(const fc::ecc::private_key& priv,
57  const fc::ecc::public_key& pub)const
58 {
59  if( from != public_key_type() )
60  {
61  auto secret = priv.get_shared_secret(pub);
62  auto nonce_plus_secret = fc::sha512::hash(fc::to_string(nonce) + secret.str());
63  auto plain_text = fc::aes_decrypt( nonce_plus_secret, message );
64  auto result = memo_message::deserialize(string(plain_text.begin(), plain_text.end()));
65  FC_ASSERT( result.checksum == uint32_t(digest_type::hash(result.text)._hash[0]) );
66  return result.text;
67  }
68  else
69  {
70  return memo_message::deserialize(string(message.begin(), message.end())).text;
71  }
72 }
