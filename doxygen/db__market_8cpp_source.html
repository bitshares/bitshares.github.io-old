43 void database::globally_settle_asset( const asset_object& mia, const price& settlement_price )
44 { try {
45  const asset_bitasset_data_object& bitasset = mia.bitasset_data(*this);
46  FC_ASSERT( !bitasset.has_settlement(), "black swan already occurred, it should not happen again" );
47 
48  const asset_object& backing_asset = bitasset.options.short_backing_asset(*this);
49  asset collateral_gathered = backing_asset.amount(0);
50 
51  const asset_dynamic_data_object& mia_dyn = mia.dynamic_asset_data_id(*this);
52  auto original_mia_supply = mia_dyn.current_supply;
53 
54  const call_order_index& call_index = get_index_type<call_order_index>();
55  const auto& call_price_index = call_index.indices().get<by_price>();
56 
57  auto maint_time = get_dynamic_global_properties().next_maintenance_time;
58  bool before_core_hardfork_342 = ( maint_time <= HARDFORK_CORE_342_TIME ); // better rounding
59 
60  // cancel all call orders and accumulate it into collateral_gathered
61  auto call_itr = call_price_index.lower_bound( price::min( bitasset.options.short_backing_asset, mia.id ) );
62  auto call_end = call_price_index.upper_bound( price::max( bitasset.options.short_backing_asset, mia.id ) );
63  asset pays;
64  while( call_itr != call_end )
65  {
66  if( before_core_hardfork_342 )
67  {
68  pays = call_itr->get_debt() * settlement_price; // round down, in favor of call order
69 
70  // Be here, the call order can be paying nothing
71  if( pays.amount == 0 && !bitasset.is_prediction_market ) // TODO remove this warning after hard fork core-342
72  wlog( "Something for nothing issue (#184, variant E) occurred at block #${block}", ("block",head_block_num()) );
73  }
74  else
75  pays = call_itr->get_debt().multiply_and_round_up( settlement_price ); // round up, in favor of global settlement fund
76 
77  if( pays > call_itr->get_collateral() )
78  pays = call_itr->get_collateral();
79 
80  collateral_gathered += pays;
81  const auto& order = *call_itr;
82  ++call_itr;
83  FC_ASSERT( fill_call_order( order, pays, order.get_debt(), settlement_price, true ) ); // call order is maker
84  }
85 
86  modify( bitasset, [&]( asset_bitasset_data_object& obj ){
87  assert( collateral_gathered.asset_id == settlement_price.quote.asset_id );
88  obj.settlement_price = mia.amount(original_mia_supply) / collateral_gathered; //settlement_price;
89  obj.settlement_fund = collateral_gathered.amount;
90  });
91 
96  modify( mia_dyn, [&]( asset_dynamic_data_object& obj ){
97  obj.current_supply = original_mia_supply;
98  });
99 
100 } FC_CAPTURE_AND_RETHROW( (mia)(settlement_price) ) }

205 void database::cancel_limit_order( const limit_order_object& order, bool create_virtual_op, bool skip_cancel_fee )
206 {
207  // if need to create a virtual op, try deduct a cancellation fee here.
208  // there are two scenarios when order is cancelled and need to create a virtual op:
209  // 1. due to expiration: always deduct a fee if there is any fee deferred
210  // 2. due to cull_small: deduct a fee after hard fork 604, but not before (will set skip_cancel_fee)
211  const account_statistics_object* seller_acc_stats = nullptr;
212  const asset_dynamic_data_object* fee_asset_dyn_data = nullptr;
213  limit_order_cancel_operation vop;
214  share_type deferred_fee = order.deferred_fee;
215  asset deferred_paid_fee = order.deferred_paid_fee;
216  if( create_virtual_op )
217  {
218  vop.order = order.id;
219  vop.fee_paying_account = order.seller;
220  // only deduct fee if not skipping fee, and there is any fee deferred
221  if( !skip_cancel_fee && deferred_fee > 0 )
222  {
223  asset core_cancel_fee = current_fee_schedule().calculate_fee( vop );
224  // cap the fee
225  if( core_cancel_fee.amount > deferred_fee )
226  core_cancel_fee.amount = deferred_fee;
227  // if there is any CORE fee to deduct, redirect it to referral program
228  if( core_cancel_fee.amount > 0 )
229  {
230  seller_acc_stats = &order.seller( *this ).statistics( *this );
231  modify( *seller_acc_stats, [&]( account_statistics_object& obj ) {
232  obj.pay_fee( core_cancel_fee.amount, get_global_properties().parameters.cashback_vesting_threshold );
233  } );
234  deferred_fee -= core_cancel_fee.amount;
235  // handle originally paid fee if any:
236  // to_deduct = round_up( paid_fee * core_cancel_fee / deferred_core_fee_before_deduct )
237  if( deferred_paid_fee.amount == 0 )
238  {
239  vop.fee = core_cancel_fee;
240  }
241  else
242  {
243  fc::uint128 fee128( deferred_paid_fee.amount.value );
244  fee128 *= core_cancel_fee.amount.value;
245  // to round up
246  fee128 += order.deferred_fee.value;
247  fee128 -= 1;
248  fee128 /= order.deferred_fee.value;
249  share_type cancel_fee_amount = fee128.to_uint64();
250  // cancel_fee should be positive, pay it to asset's accumulated_fees
251  fee_asset_dyn_data = &deferred_paid_fee.asset_id(*this).dynamic_asset_data_id(*this);
252  modify( *fee_asset_dyn_data, [&](asset_dynamic_data_object& addo) {
253  addo.accumulated_fees += cancel_fee_amount;
254  });
255  // cancel_fee should be no more than deferred_paid_fee
256  deferred_paid_fee.amount -= cancel_fee_amount;
257  vop.fee = asset( cancel_fee_amount, deferred_paid_fee.asset_id );
258  }
259  }
260  }
261  }
262 
263  // refund funds in order
264  auto refunded = order.amount_for_sale();
265  if( refunded.asset_id == asset_id_type() )
266  {
267  if( seller_acc_stats == nullptr )
268  seller_acc_stats = &order.seller( *this ).statistics( *this );
269  modify( *seller_acc_stats, [&]( account_statistics_object& obj ) {
270  obj.total_core_in_orders -= refunded.amount;
271  });
272  }
273  adjust_balance(order.seller, refunded);
274 
275  // refund fee
276  // could be virtual op or real op here
277  if( order.deferred_paid_fee.amount == 0 )
278  {
279  // be here, order.create_time <= HARDFORK_CORE_604_TIME, or fee paid in CORE, or no fee to refund.
280  // if order was created before hard fork 604 then cancelled no matter before or after hard fork 604,
281  // see it as fee paid in CORE, deferred_fee should be refunded to order owner but not fee pool
282  adjust_balance( order.seller, deferred_fee );
283  }
284  else // need to refund fee in originally paid asset
285  {
286  adjust_balance(order.seller, deferred_paid_fee);
287  // be here, must have: fee_asset != CORE
288  if( fee_asset_dyn_data == nullptr )
289  fee_asset_dyn_data = &deferred_paid_fee.asset_id(*this).dynamic_asset_data_id(*this);
290  modify( *fee_asset_dyn_data, [&](asset_dynamic_data_object& addo) {
291  addo.fee_pool += deferred_fee;
292  });
293  }
294 
295  if( create_virtual_op )
296  push_applied_operation( vop );
297 
298  remove(order);
299 }

327 bool database::apply_order_before_hardfork_625(const limit_order_object& new_order_object, bool allow_black_swan)
328 {
329  auto order_id = new_order_object.id;
330  const asset_object& sell_asset = get(new_order_object.amount_for_sale().asset_id);
331  const asset_object& receive_asset = get(new_order_object.amount_to_receive().asset_id);
332 
333  // Possible optimization: We only need to check calls if both are true:
334  // - The new order is at the front of the book
335  // - The new order is below the call limit price
336  bool called_some = check_call_orders(sell_asset, allow_black_swan, true); // the first time when checking, call order is maker
337  called_some |= check_call_orders(receive_asset, allow_black_swan, true); // the other side, same as above
338  if( called_some && !find_object(order_id) ) // then we were filled by call order
339  return true;
340 
341  const auto& limit_price_idx = get_index_type<limit_order_index>().indices().get<by_price>();
342 
343  // TODO: it should be possible to simply check the NEXT/PREV iterator after new_order_object to
344  // determine whether or not this order has "changed the book" in a way that requires us to
345  // check orders. For now I just lookup the lower bound and check for equality... this is log(n) vs
346  // constant time check. Potential optimization.
347 
348  auto max_price = ~new_order_object.sell_price;
349  auto limit_itr = limit_price_idx.lower_bound(max_price.max());
350  auto limit_end = limit_price_idx.upper_bound(max_price);
351 
352  bool finished = false;
353  while( !finished && limit_itr != limit_end )
354  {
355  auto old_limit_itr = limit_itr;
356  ++limit_itr;
357  // match returns 2 when only the old order was fully filled. In this case, we keep matching; otherwise, we stop.
358  finished = (match(new_order_object, *old_limit_itr, old_limit_itr->sell_price) != 2);
359  }
360 
361  //Possible optimization: only check calls if the new order completely filled some old order
362  //Do I need to check both assets?
363  check_call_orders(sell_asset, allow_black_swan); // after the new limit order filled some orders on the book,
364  // if a call order matches another order, the call order is taker
365  check_call_orders(receive_asset, allow_black_swan); // the other side, same as above
366 
367  const limit_order_object* updated_order_object = find< limit_order_object >( order_id );
368  if( updated_order_object == nullptr )
369  return true;
370  if( head_block_time() <= HARDFORK_555_TIME )
371  return false;
372  // before #555 we would have done maybe_cull_small_order() logic as a result of fill_order() being called by match() above
373  // however after #555 we need to get rid of small orders -- #555 hardfork defers logic that was done too eagerly before, and
374  // this is the point it's deferred to.
375  return maybe_cull_small_order( *this, *updated_order_object );
376 }

378 bool database::apply_order(const limit_order_object& new_order_object, bool allow_black_swan)
379 {
380  auto order_id = new_order_object.id;
381  asset_id_type sell_asset_id = new_order_object.sell_asset_id();
382  asset_id_type recv_asset_id = new_order_object.receive_asset_id();
383 
384  // We only need to check if the new order will match with others if it is at the front of the book
385  const auto& limit_price_idx = get_index_type<limit_order_index>().indices().get<by_price>();
386  auto limit_itr = limit_price_idx.lower_bound( boost::make_tuple( new_order_object.sell_price, order_id ) );
387  if( limit_itr != limit_price_idx.begin() )
388  {
389  --limit_itr;
390  if( limit_itr->sell_asset_id() == sell_asset_id && limit_itr->receive_asset_id() == recv_asset_id )
391  return false;
392  }
393 
394  // this is the opposite side (on the book)
395  auto max_price = ~new_order_object.sell_price;
396  limit_itr = limit_price_idx.lower_bound( max_price.max() );
397  auto limit_end = limit_price_idx.upper_bound( max_price );
398 
399  // Order matching should be in favor of the taker.
400  // When a new limit order is created, e.g. an ask, need to check if it will match the highest bid.
401  // We were checking call orders first. However, due to MSSR (maximum_short_squeeze_ratio),
402  // effective price of call orders may be worse than limit orders, so we should also check limit orders here.
403 
404  // Question: will a new limit order trigger a black swan event?
405  //
406  // 1. as of writing, it's possible due to the call-order-and-limit-order overlapping issue:
407  // https://github.com/bitshares/bitshares-core/issues/606 .
408  // when it happens, a call order can be very big but don't match with the opposite,
409  // even when price feed is too far away, further than swan price,
410  // if the new limit order is in the same direction with the call orders, it can eat up all the opposite,
411  // then the call order will lose support and trigger a black swan event.
412  // 2. after issue 606 is fixed, there will be no limit order on the opposite side "supporting" the call order,
413  // so a new order in the same direction with the call order won't trigger a black swan event.
414  // 3. calling is one direction. if the new limit order is on the opposite direction,
415  // no matter if matches with the call, it won't trigger a black swan event.
416  // (if a match at MSSP caused a black swan event, it means the call order is already undercollateralized,
417  // which should trigger a black swan event earlier.)
418  //
419  // Since it won't trigger a black swan, no need to check here.
420 
421  // currently we don't do cross-market (triangle) matching.
422  // the limit order will only match with a call order if meet all of these:
423  // 1. it's buying collateral, which means sell_asset is the MIA, receive_asset is the backing asset.
424  // 2. sell_asset is not a prediction market
425  // 3. sell_asset is not globally settled
426  // 4. sell_asset has a valid price feed
427  // 5. the call order's collateral ratio is below or equals to MCR
428  // 6. the limit order provided a good price
429 
430  bool to_check_call_orders = false;
431  const asset_object& sell_asset = sell_asset_id( *this );
432  const asset_bitasset_data_object* sell_abd = nullptr;
433  price call_match_price;
434  if( sell_asset.is_market_issued() )
435  {
436  sell_abd = &sell_asset.bitasset_data( *this );
437  if( sell_abd->options.short_backing_asset == recv_asset_id
438  && !sell_abd->is_prediction_market
439  && !sell_abd->has_settlement()
440  && !sell_abd->current_feed.settlement_price.is_null() )
441  {
442  call_match_price = ~sell_abd->current_feed.max_short_squeeze_price();
443  if( ~new_order_object.sell_price <= call_match_price ) // new limit order price is good enough to match a call
444  to_check_call_orders = true;
445  }
446  }
447 
448  bool finished = false; // whether the new order is gone
449  if( to_check_call_orders )
450  {
451  // check limit orders first, match the ones with better price in comparison to call orders
452  while( !finished && limit_itr != limit_end && limit_itr->sell_price > call_match_price )
453  {
454  auto old_limit_itr = limit_itr;
455  ++limit_itr;
456  // match returns 2 when only the old order was fully filled. In this case, we keep matching; otherwise, we stop.
457  finished = ( match( new_order_object, *old_limit_itr, old_limit_itr->sell_price ) != 2 );
458  }
459 
460  if( !finished )
461  {
462  // check if there are margin calls
463  const auto& call_price_idx = get_index_type<call_order_index>().indices().get<by_price>();
464  auto call_min = price::min( recv_asset_id, sell_asset_id );
465  while( !finished )
466  {
467  // assume hard fork core-343 and core-625 will take place at same time, always check call order with least call_price
468  auto call_itr = call_price_idx.lower_bound( call_min );
469  if( call_itr == call_price_idx.end()
470  || call_itr->debt_type() != sell_asset_id
471  // feed protected https://github.com/cryptonomex/graphene/issues/436
472  || call_itr->call_price > ~sell_abd->current_feed.settlement_price )
473  break;
474  // assume hard fork core-338 and core-625 will take place at same time, not checking HARDFORK_CORE_338_TIME here.
475  int match_result = match( new_order_object, *call_itr, call_match_price,
476  sell_abd->current_feed.settlement_price,
477  sell_abd->current_feed.maintenance_collateral_ratio );
478  // match returns 1 or 3 when the new order was fully filled. In this case, we stop matching; otherwise keep matching.
479  // since match can return 0 due to BSIP38 (hard fork core-834), we no longer only check if the result is 2.
480  if( match_result == 1 || match_result == 3 )
481  finished = true;
482  }
483  }
484  }
485 
486  // still need to check limit orders
487  while( !finished && limit_itr != limit_end )
488  {
489  auto old_limit_itr = limit_itr;
490  ++limit_itr;
491  // match returns 2 when only the old order was fully filled. In this case, we keep matching; otherwise, we stop.
492  finished = ( match( new_order_object, *old_limit_itr, old_limit_itr->sell_price ) != 2 );
493  }
494 
495  const limit_order_object* updated_order_object = find< limit_order_object >( order_id );
496  if( updated_order_object == nullptr )
497  return true;
498 
499  // before #555 we would have done maybe_cull_small_order() logic as a result of fill_order() being called by match() above
500  // however after #555 we need to get rid of small orders -- #555 hardfork defers logic that was done too eagerly before, and
501  // this is the point it's deferred to.
502  return maybe_cull_small_order( *this, *updated_order_object );
503 }

515 int database::match( const limit_order_object& usd, const limit_order_object& core, const price& match_price )
516 {
517  FC_ASSERT( usd.sell_price.quote.asset_id == core.sell_price.base.asset_id );
518  FC_ASSERT( usd.sell_price.base.asset_id == core.sell_price.quote.asset_id );
519  FC_ASSERT( usd.for_sale > 0 && core.for_sale > 0 );
520 
521  auto usd_for_sale = usd.amount_for_sale();
522  auto core_for_sale = core.amount_for_sale();
523 
524  asset usd_pays, usd_receives, core_pays, core_receives;
