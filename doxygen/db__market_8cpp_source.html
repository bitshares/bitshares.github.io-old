239 bool database::apply_order(const limit_order_object& new_order_object, bool allow_black_swan)
240 {
241  auto order_id = new_order_object.id;
242  const asset_object& sell_asset = get(new_order_object.amount_for_sale().asset_id);
243  const asset_object& receive_asset = get(new_order_object.amount_to_receive().asset_id);
244 
245  // Possible optimization: We only need to check calls if both are true:
246  // - The new order is at the front of the book
247  // - The new order is below the call limit price
248  bool called_some = check_call_orders(sell_asset, allow_black_swan, true); // the first time when checking, call order is maker
249  called_some |= check_call_orders(receive_asset, allow_black_swan, true); // the other side, same as above
250  if( called_some && !find_object(order_id) ) // then we were filled by call order
251  return true;
252 
253  const auto& limit_price_idx = get_index_type<limit_order_index>().indices().get<by_price>();
254 
255  // TODO: it should be possible to simply check the NEXT/PREV iterator after new_order_object to
256  // determine whether or not this order has "changed the book" in a way that requires us to
257  // check orders. For now I just lookup the lower bound and check for equality... this is log(n) vs
258  // constant time check. Potential optimization.
259 
260  auto max_price = ~new_order_object.sell_price;
261  auto limit_itr = limit_price_idx.lower_bound(max_price.max());
262  auto limit_end = limit_price_idx.upper_bound(max_price);
263 
264  bool finished = false;
265  while( !finished && limit_itr != limit_end )
266  {
267  auto old_limit_itr = limit_itr;
268  ++limit_itr;
269  // match returns 2 when only the old order was fully filled. In this case, we keep matching; otherwise, we stop.
270  finished = (match(new_order_object, *old_limit_itr, old_limit_itr->sell_price) != 2);
271  }
272 
273  //Possible optimization: only check calls if the new order completely filled some old order
274  //Do I need to check both assets?
275  check_call_orders(sell_asset, allow_black_swan); // after the new limit order filled some orders on the book,
276  // if a call order matches another order, the call order is taker
277  check_call_orders(receive_asset, allow_black_swan); // the other side, same as above
278 
279  const limit_order_object* updated_order_object = find< limit_order_object >( order_id );
280  if( updated_order_object == nullptr )
281  return true;
282  if( head_block_time() <= HARDFORK_555_TIME )
283  return false;
284  // before #555 we would have done maybe_cull_small_order() logic as a result of fill_order() being called by match() above
285  // however after #555 we need to get rid of small orders -- #555 hardfork defers logic that was done too eagerly before, and
286  // this is the point it's deferred to.
287  return maybe_cull_small_order( *this, *updated_order_object );
288 }

301 int database::match( const limit_order_object& usd, const OrderType& core, const price& match_price )
302 {
303  assert( usd.sell_price.quote.asset_id == core.sell_price.base.asset_id );
304  assert( usd.sell_price.base.asset_id == core.sell_price.quote.asset_id );
305  assert( usd.for_sale > 0 && core.for_sale > 0 );
306 
307  auto usd_for_sale = usd.amount_for_sale();
308  auto core_for_sale = core.amount_for_sale();
309 
310  asset usd_pays, usd_receives, core_pays, core_receives;
311 
312  if( usd_for_sale <= core_for_sale * match_price )
313  {
314  core_receives = usd_for_sale;
315  usd_receives = usd_for_sale * match_price;
316  }
317  else
318  {
319  //This line once read: assert( core_for_sale < usd_for_sale * match_price );
320  //This assert is not always true -- see trade_amount_equals_zero in operation_tests.cpp
321  //Although usd_for_sale is greater than core_for_sale * match_price, core_for_sale == usd_for_sale * match_price
322  //Removing the assert seems to be safe -- apparently no asset is created or destroyed.
323  usd_receives = core_for_sale;
324  core_receives = core_for_sale * match_price;
325  }
326 
327  core_pays = usd_receives;
328  usd_pays = core_receives;
329 
330  assert( usd_pays == usd.amount_for_sale() ||
331  core_pays == core.amount_for_sale() );
332 
333  int result = 0;
334  result |= fill_order( usd, usd_pays, usd_receives, false, match_price, false ); // although this function is a template,
335  // right now it only matches one limit order
336  // with another limit order,
337  // the first param is a new order, thus taker
338  result |= fill_order( core, core_pays, core_receives, true, match_price, true ) << 1; // the second param is maker
339  assert( result != 0 );
340  return result;
341 }

601

609  }
610 
611  asset call_pays, call_receives, order_pays, order_receives;
612  if( usd_to_buy >= usd_for_sale )
613  { // fill order
614  call_receives = usd_for_sale;
615  order_receives = usd_for_sale * match_price;
616  call_pays = order_receives;
617  order_pays = usd_for_sale;
618 
619  filled_limit = true;
620  filled_call = (usd_to_buy == usd_for_sale);
621  } else { // fill call
622  call_receives = usd_to_buy;
623  order_receives = usd_to_buy * match_price;
624  call_pays = order_receives;
625  order_pays = usd_to_buy;
626 
627  filled_call = true;
628  if( filled_limit )
629  wlog( "Multiple limit match problem (issue 338) occurred at block #${block}", ("block",head_block_num()) );
630  }
631 
632  FC_ASSERT( filled_call || filled_limit );
633 
634  auto old_call_itr = call_itr;
635  if( filled_call ) ++call_itr;
636  // when for_new_limit_order is true, the call order is maker, otherwise the call order is taker
637  fill_order(*old_call_itr, call_pays, call_receives, match_price, for_new_limit_order );
638 
639  auto old_limit_itr = limit_itr;
640  auto next_limit_itr = std::next( limit_itr );
641  if( filled_limit ) ++limit_itr;
642  // when for_new_limit_order is true, the limit order is taker, otherwise the limit order is maker
643  bool really_filled = fill_order(*old_limit_itr, order_pays, order_receives, true, match_price, !for_new_limit_order );
644  if( !filled_limit && really_filled )
645  {
646  wlog( "Cull_small issue occurred at block #${block}", ("block",head_block_num()) );
647  limit_itr = next_limit_itr;
648  }
649 
650  } // whlie call_itr != call_end

652  return margin_called;
