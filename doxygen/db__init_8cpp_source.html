223 void database::init_genesis(const genesis_state_type& genesis_state)
224 { try {
225  FC_ASSERT( genesis_state.initial_timestamp != time_point_sec(), "Must initialize genesis timestamp." );
226  FC_ASSERT( genesis_state.initial_timestamp.sec_since_epoch() % GRAPHENE_DEFAULT_BLOCK_INTERVAL == 0,
227  "Genesis timestamp must be divisible by GRAPHENE_DEFAULT_BLOCK_INTERVAL." );
228  FC_ASSERT(genesis_state.initial_witness_candidates.size() > 0,
229  "Cannot start a chain with zero witnesses.");
230  FC_ASSERT(genesis_state.initial_active_witnesses <= genesis_state.initial_witness_candidates.size(),
231  "initial_active_witnesses is larger than the number of candidate witnesses.");
232 
233  _undo_db.disable();
234  struct auth_inhibitor {
235  auth_inhibitor(database& db) : db(db), old_flags(db.node_properties().skip_flags)
236  { db.node_properties().skip_flags |= skip_authority_check; }
237  ~auth_inhibitor()
238  { db.node_properties().skip_flags = old_flags; }
239  private:
240  database& db;
241  uint32_t old_flags;
242  } inhibitor(*this);
243 
244  transaction_evaluation_state genesis_eval_state(this);
245 
246  // Create blockchain accounts
247  fc::ecc::private_key null_private_key = fc::ecc::private_key::regenerate(fc::sha256::hash(string("null_key")));
248  create<account_balance_object>([](account_balance_object& b) {
249  b.balance = GRAPHENE_MAX_SHARE_SUPPLY;
250  });
251  const account_object& committee_account =
252  create<account_object>( [&](account_object& n) {
253  n.membership_expiration_date = time_point_sec::maximum();
254  n.network_fee_percentage = GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
255  n.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT - GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
256  n.owner.weight_threshold = 1;
257  n.active.weight_threshold = 1;
258  n.name = "committee-account";
259  n.statistics = create<account_statistics_object>( [&](account_statistics_object& s){ s.owner = n.id; }).id;
260  });
261  FC_ASSERT(committee_account.get_id() == GRAPHENE_COMMITTEE_ACCOUNT);
262  FC_ASSERT(create<account_object>([this](account_object& a) {
263  a.name = "witness-account";
264  a.statistics = create<account_statistics_object>([&](account_statistics_object& s){s.owner = a.id;}).id;
265  a.owner.weight_threshold = 1;
266  a.active.weight_threshold = 1;
267  a.registrar = a.lifetime_referrer = a.referrer = GRAPHENE_WITNESS_ACCOUNT;
268  a.membership_expiration_date = time_point_sec::maximum();
269  a.network_fee_percentage = GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
270  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT - GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
271  }).get_id() == GRAPHENE_WITNESS_ACCOUNT);
272  FC_ASSERT(create<account_object>([this](account_object& a) {
273  a.name = "relaxed-committee-account";
274  a.statistics = create<account_statistics_object>([&](account_statistics_object& s){s.owner = a.id;}).id;
275  a.owner.weight_threshold = 1;
276  a.active.weight_threshold = 1;
277  a.registrar = a.lifetime_referrer = a.referrer = GRAPHENE_RELAXED_COMMITTEE_ACCOUNT;
278  a.membership_expiration_date = time_point_sec::maximum();
279  a.network_fee_percentage = GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
280  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT - GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
281  }).get_id() == GRAPHENE_RELAXED_COMMITTEE_ACCOUNT);
282  FC_ASSERT(create<account_object>([this](account_object& a) {
283  a.name = "null-account";
284  a.statistics = create<account_statistics_object>([&](account_statistics_object& s){s.owner = a.id;}).id;
285  a.owner.weight_threshold = 1;
286  a.active.weight_threshold = 1;
287  a.registrar = a.lifetime_referrer = a.referrer = GRAPHENE_NULL_ACCOUNT;
288  a.membership_expiration_date = time_point_sec::maximum();
289  a.network_fee_percentage = 0;
290  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT;
291  }).get_id() == GRAPHENE_NULL_ACCOUNT);
292  FC_ASSERT(create<account_object>([this](account_object& a) {
293  a.name = "temp-account";
294  a.statistics = create<account_statistics_object>([&](account_statistics_object& s){s.owner = a.id;}).id;
295  a.owner.weight_threshold = 0;
296  a.active.weight_threshold = 0;
297  a.registrar = a.lifetime_referrer = a.referrer = GRAPHENE_TEMP_ACCOUNT;
298  a.membership_expiration_date = time_point_sec::maximum();
299  a.network_fee_percentage = GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
300  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT - GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
301  }).get_id() == GRAPHENE_TEMP_ACCOUNT);
302  FC_ASSERT(create<account_object>([this](account_object& a) {
303  a.name = "proxy-to-self";
304  a.statistics = create<account_statistics_object>([&](account_statistics_object& s){s.owner = a.id;}).id;
305  a.owner.weight_threshold = 1;
306  a.active.weight_threshold = 1;
307  a.registrar = a.lifetime_referrer = a.referrer = GRAPHENE_NULL_ACCOUNT;
308  a.membership_expiration_date = time_point_sec::maximum();
309  a.network_fee_percentage = 0;
310  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT;
311  }).get_id() == GRAPHENE_PROXY_TO_SELF_ACCOUNT);
312 
313  // Create more special accounts
314  while( true )
315  {
316  uint64_t id = get_index<account_object>().get_next_id().instance();
317  if( id >= genesis_state.immutable_parameters.num_special_accounts )
318  break;

320  a.name = "special-account-" + std::to_string(id);
321  a.statistics = create<account_statistics_object>([&](account_statistics_object& s){s.owner = a.id;}).id;
322  a.owner.weight_threshold = 1;
323  a.active.weight_threshold = 1;
324  a.registrar = a.lifetime_referrer = a.referrer = account_id_type(id);
325  a.membership_expiration_date = time_point_sec::maximum();
326  a.network_fee_percentage = GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
327  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT - GRAPHENE_DEFAULT_NETWORK_PERCENT_OF_FEE;
328  });
329  FC_ASSERT( acct.get_id() == account_id_type(id) );

331  }

336  a.current_supply = GRAPHENE_MAX_SHARE_SUPPLY;
337  });

339  create<asset_object>( [&]( asset_object& a ) {
340  a.symbol = GRAPHENE_SYMBOL;
341  a.options.max_supply = genesis_state.max_core_supply;
342  a.precision = GRAPHENE_BLOCKCHAIN_PRECISION_DIGITS;

344  a.options.issuer_permissions = 0;
345  a.issuer = GRAPHENE_NULL_ACCOUNT;

368  a.options.flags = 0;
369  a.options.issuer_permissions = 0;

372  a.options.core_exchange_rate.base.asset_id = asset_id_type(0);
373  a.options.core_exchange_rate.quote.amount = 1;
374  a.options.core_exchange_rate.quote.asset_id = asset_id_type(0);
375  a.dynamic_asset_data_id = dyn_asset.id;
376  });

390  });

392  p.time = genesis_state.initial_timestamp;
393  p.dynamic_flags = 0;
394  p.witness_budget = 0;

396  });

398  FC_ASSERT( (genesis_state.immutable_parameters.min_witness_count & 1) == 1, "min_witness_count must be odd" );

409  // Create initial accounts
410  for( const auto& account : genesis_state.initial_accounts )
411  {

424  cop.options.memo_key = account.active_key;
425  }

427

429  {
430  account_upgrade_operation op;
431  op.account_to_upgrade = account_id;
432  op.upgrade_to_lifetime_member = true;
433  apply_operation(genesis_eval_state, op);
434  }
435  }
