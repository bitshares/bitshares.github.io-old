528  address a3( pts_address(key, false, 0) );
529  address a4( pts_address(key, true, 0) );
530  address a5( key );
531 
532  subscribe_to_item( key );
533  subscribe_to_item( a1 );
534  subscribe_to_item( a2 );
535  subscribe_to_item( a3 );
536  subscribe_to_item( a4 );
537  subscribe_to_item( a5 );
538 
539  const auto& idx = _db.get_index_type<account_index>();
540  const auto& aidx = dynamic_cast<const primary_index<account_index>&>(idx);
541  const auto& refs = aidx.get_secondary_index<graphene::chain::account_member_index>();
542  auto itr = refs.account_to_key_memberships.find(key);
543  vector<account_id_type> result;
544 
545  for( auto& a : {a1,a2,a3,a4,a5} )
546  {
547  auto itr = refs.account_to_address_memberships.find(a);
548  if( itr != refs.account_to_address_memberships.end() )
549  {
550  result.reserve( itr->second.size() );
551  for( auto item : itr->second )
552  {

554  }
555  }
556  }

564  }
