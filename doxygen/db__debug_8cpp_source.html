101 void debug_apply_update( database& db, const fc::variant_object& vo )
102 {
103  static const uint8_t
104  db_action_nil = 0,
105  db_action_create = 1,
106  db_action_write = 2,
107  db_action_update = 3,
108  db_action_delete = 4;
109 
110  // "_action" : "create" object must not exist, unspecified fields take defaults
111  // "_action" : "write" object may exist, is replaced entirely, unspecified fields take defaults
112  // "_action" : "update" object must exist, unspecified fields don't change
113  // "_action" : "delete" object must exist, will be deleted
114 
115  // if _action is unspecified:
116  // - delete if object contains only ID field
117  // - otherwise, write
118 
119  object_id_type oid;
120  uint8_t action = db_action_nil;
121  auto it_id = vo.find("id");
122  FC_ASSERT( it_id != vo.end() );
123 
124  from_variant( it_id->value(), oid );
125  action = ( vo.size() == 1 ) ? db_action_delete : db_action_write;
126 
127  from_variant( vo["id"], oid );
128  if( vo.size() == 1 )
129  action = db_action_delete;
130  auto it_action = vo.find("_action" );
131  if( it_action != vo.end() )
132  {
133  const std::string& str_action = it_action->value().get_string();
134  if( str_action == "create" )
135  action = db_action_create;
136  else if( str_action == "write" )
137  action = db_action_write;
138  else if( str_action == "update" )
139  action = db_action_update;
140  else if( str_action == "delete" )
141  action = db_action_delete;
142  }
143 
144  auto& idx = db.get_index( oid );
145 
146  switch( action )
147  {
148  case db_action_create:
149  /*
150  idx.create( [&]( object& obj )
151  {
152  idx.object_from_variant( vo, obj );
153  } );
154  */
155  FC_ASSERT( false );
156  break;
157  case db_action_write:
158  db.modify( db.get_object( oid ), [&]( object& obj )
159  {
160  idx.object_default( obj );
161  idx.object_from_variant( vo, obj );
162  } );
163  break;
164  case db_action_update:
165  db.modify( db.get_object( oid ), [&]( object& obj )
166  {
167  idx.object_from_variant( vo, obj );
168  } );
169  break;
170  case db_action_delete:
171  db.remove( db.get_object( oid ) );
172  break;
173  default:
174  FC_ASSERT( false );
175  }
176 }

178 void database::apply_debug_updates()
179 {
180  block_id_type head_id = head_block_id();

182  if( it == _node_property_object.debug_updates.end() )

189 {
190  block_id_type head_id = head_block_id();
191  auto it = _node_property_object.debug_updates.find( head_id );
192  if( it == _node_property_object.debug_updates.end() )
193  it = _node_property_object.debug_updates.emplace( head_id, std::vector< fc::variant_object >() ).first;
194  it->second.emplace_back( update );
195 
196  optional<signed_block> head_block = fetch_block_by_id( head_id );
197  FC_ASSERT( head_block.valid() );
198 
199  // What the last block does has been changed by adding to node_property_object, so we have to re-apply it
200  pop_block();
201  push_block( *head_block );
