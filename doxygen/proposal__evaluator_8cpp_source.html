31 void_result proposal_create_evaluator::do_evaluate(const proposal_create_operation& o)
32 { try {
33  const database& d = db();
34  const auto& global_parameters = d.get_global_properties().parameters;
35 
36  FC_ASSERT( o.expiration_time > d.head_block_time(), "Proposal has already expired on creation." );
37  FC_ASSERT( o.expiration_time <= d.head_block_time() + global_parameters.maximum_proposal_lifetime,
38  "Proposal expiration time is too far in the future.");
39  FC_ASSERT( !o.review_period_seconds || fc::seconds(*o.review_period_seconds) < (o.expiration_time - d.head_block_time()),
40  "Proposal review period must be less than its overall lifetime." );
41 
42  {
43  // If we're dealing with the committee authority, make sure this transaction has a sufficient review period.
44  flat_set<account_id_type> auths;
45  vector<authority> other;
46  for( auto& op : o.proposed_ops )
47  {
48  operation_get_required_authorities(op.op, auths, auths, other);
49  }
50 
51  FC_ASSERT( other.size() == 0 ); // TODO: what about other???
52 
53  if( auths.find(GRAPHENE_COMMITTEE_ACCOUNT) != auths.end() )
54  {
55  GRAPHENE_ASSERT(
56  o.review_period_seconds.valid(),
57  proposal_create_review_period_required,
58  "Review period not given, but at least ${min} required",
59  ("min", global_parameters.committee_proposal_review_period)
60  );
61  GRAPHENE_ASSERT(
62  *o.review_period_seconds >= global_parameters.committee_proposal_review_period,
63  proposal_create_review_period_insufficient,
64  "Review period of ${t} specified, but at least ${min} required",
65  ("t", *o.review_period_seconds)
66  ("min", global_parameters.committee_proposal_review_period)
67  );
68  }
69  }
70 
71  for( const op_wrapper& op : o.proposed_ops )
72  _proposed_trx.operations.push_back(op.op);
73  _proposed_trx.validate();
74 
75  return void_result();
76 } FC_CAPTURE_AND_RETHROW( (o) ) }

78 object_id_type proposal_create_evaluator::do_apply(const proposal_create_operation& o)
79 { try {
80  database& d = db();
81 
82  const proposal_object& proposal = d.create<proposal_object>([&](proposal_object& proposal) {
83  _proposed_trx.expiration = o.expiration_time;
84  proposal.proposed_transaction = _proposed_trx;
85  proposal.expiration_time = o.expiration_time;
86  if( o.review_period_seconds )
87  proposal.review_period_time = o.expiration_time - *o.review_period_seconds;
88 
89  //Populate the required approval sets
90  flat_set<account_id_type> required_active;
91  vector<authority> other;
92 
93  // TODO: consider caching values from evaluate?
94  for( auto& op : _proposed_trx.operations )
95  operation_get_required_authorities(op, required_active, proposal.required_owner_approvals, other);
96 
97  //All accounts which must provide both owner and active authority should be omitted from the active authority set;
98  //owner authority approval implies active authority approval.
99  std::set_difference(required_active.begin(), required_active.end(),
100  proposal.required_owner_approvals.begin(), proposal.required_owner_approvals.end(),
101  std::inserter(proposal.required_active_approvals, proposal.required_active_approvals.begin()));
102  });
103 
104  return proposal.id;
105 } FC_CAPTURE_AND_RETHROW( (o) ) }

145 void_result proposal_update_evaluator::do_apply(const proposal_update_operation& o)
146 { try {
147  database& d = db();
148 
149  // Potential optimization: if _executed_proposal is true, we can skip the modify step and make push_proposal skip
150  // signature checks. This isn't done now because I just wrote all the proposals code, and I'm not yet 100% sure the
151  // required approvals are sufficient to authorize the transaction.
152  d.modify(*_proposal, [&o, &d](proposal_object& p) {
153  p.available_active_approvals.insert(o.active_approvals_to_add.begin(), o.active_approvals_to_add.end());
154  p.available_owner_approvals.insert(o.owner_approvals_to_add.begin(), o.owner_approvals_to_add.end());
155  for( account_id_type id : o.active_approvals_to_remove )
156  p.available_active_approvals.erase(id);
157  for( account_id_type id : o.owner_approvals_to_remove )
158  p.available_owner_approvals.erase(id);
159  for( const auto& id : o.key_approvals_to_add )
160  p.available_key_approvals.insert(id);
161  for( const auto& id : o.key_approvals_to_remove )
162  p.available_key_approvals.erase(id);
163  });
164 
165  // If the proposal has a review period, don't bother attempting to authorize/execute it.
166  // Proposals with a review period may never be executed except at their expiration.
167  if( _proposal->review_period_time )
168  return void_result();
169 
170  if( _proposal->is_authorized_to_execute(d) )
171  {
172  // All required approvals are satisfied. Execute!
173  _executed_proposal = true;
174  try {
175  _processed_transaction = d.push_proposal(*_proposal);
176  } catch(fc::exception& e) {
177  wlog("Proposed transaction ${id} failed to apply once approved with exception:\n----\n${reason}\n----\nWill try again when it expires.",
178  ("id", o.proposal)("reason", e.to_detail_string()));
179  _proposal_failed = true;
180  }
181  }
182 
183  return void_result();
184 } FC_CAPTURE_AND_RETHROW( (o) ) }
