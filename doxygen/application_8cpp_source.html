212  std::vector<fc::ip::endpoint> resolve_string_to_ip_endpoints(const std::string& endpoint_string)
213  {
214  try
215  {
216  string::size_type colon_pos = endpoint_string.find(':');
217  if (colon_pos == std::string::npos)
218  FC_THROW("Missing required port number in endpoint string \"${endpoint_string}\"",
219  ("endpoint_string", endpoint_string));
220  std::string port_string = endpoint_string.substr(colon_pos + 1);
221  try
222  {
223  uint16_t port = boost::lexical_cast<uint16_t>(port_string);
224 
225  std::string hostname = endpoint_string.substr(0, colon_pos);
226  std::vector<fc::ip::endpoint> endpoints = fc::resolve(hostname, port);
227  if (endpoints.empty())
228  FC_THROW_EXCEPTION(fc::unknown_host_exception, "The host name can not be resolved: ${hostname}", ("hostname", hostname));
229  return endpoints;
230  }
231  catch (const boost::bad_lexical_cast&)
232  {
233  FC_THROW("Bad port: ${port}", ("port", port_string));
234  }
235  }
236  FC_CAPTURE_AND_RETHROW((endpoint_string))
237  }

484  virtual bool handle_block(const graphene::net::block_message& blk_msg, bool sync_mode,
485  std::vector<fc::uint160_t>& contained_transaction_message_ids) override
486  { try {
487 
488  auto latency = fc::time_point::now() - blk_msg.block.timestamp;
489  if (!sync_mode || blk_msg.block.block_num() % 10000 == 0)
490  {
491  const auto& witness = blk_msg.block.witness(*_chain_db);
492  const auto& witness_account = witness.witness_account(*_chain_db);
493  auto last_irr = _chain_db->get_dynamic_global_properties().last_irreversible_block_num;
494  ilog("Got block: #${n} time: ${t} latency: ${l} ms from: ${w} irreversible: ${i} (-${d})",
495  ("t",blk_msg.block.timestamp)
496  ("n", blk_msg.block.block_num())
497  ("l", (latency.count()/1000))
498  ("w",witness_account.name)
499  ("i",last_irr)("d",blk_msg.block.block_num()-last_irr) );
500  }
501  FC_ASSERT( (latency.count()/1000) > -5000, "Rejecting block with timestamp in the future" );
502 
503  try {
504  // TODO: in the case where this block is valid but on a fork that's too old for us to switch to,
505  // you can help the network code out by throwing a block_older_than_undo_history exception.
506  // when the net code sees that, it will stop trying to push blocks from that chain, but
507  // leave that peer connected so that they can get sync blocks from us
508  bool result = _chain_db->push_block(blk_msg.block, (_is_block_producer | _force_validate) ? database::skip_nothing : database::skip_transaction_signatures);
509 
510  // the block was accepted, so we now know all of the transactions contained in the block
511  if (!sync_mode)
512  {
513  // if we're not in sync mode, there's a chance we will be seeing some transactions
514  // included in blocks before we see the free-floating transaction itself. If that
515  // happens, there's no reason to fetch the transactions, so construct a list of the
516  // transaction message ids we no longer need.
517  // during sync, it is unlikely that we'll see any old
518  for (const processed_transaction& transaction : blk_msg.block.transactions)
519  {
520  graphene::net::trx_message transaction_message(transaction);
521  contained_transaction_message_ids.push_back(graphene::net::message(transaction_message).id());
522  }
523  }
524 
525  return result;
526  } catch ( const graphene::chain::unlinkable_block_exception& e ) {
527  // translate to a graphene::net exception
528  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
529  FC_THROW_EXCEPTION(graphene::net::unlinkable_block_exception, "Error when pushing block:\n${e}", ("e", e.to_detail_string()));
530  } catch( const fc::exception& e ) {
531  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
532  throw;
533  }
534 
535  if( !_is_finished_syncing && !sync_mode )
536  {
537  _is_finished_syncing = true;
538  _self->syncing_finished();
539  }
540  } FC_CAPTURE_AND_RETHROW( (blk_msg)(sync_mode) ) }

708  virtual std::vector<item_hash_t> get_blockchain_synopsis(const item_hash_t& reference_point,
709  uint32_t number_of_blocks_after_reference_point) override
710  { try {
711  std::vector<item_hash_t> synopsis;
712  synopsis.reserve(30);
713  uint32_t high_block_num;
714  uint32_t non_fork_high_block_num;
715  uint32_t low_block_num = _chain_db->last_non_undoable_block_num();
716  std::vector<block_id_type> fork_history;
717 
718  if (reference_point != item_hash_t())
719  {
720  // the node is asking for a summary of the block chain up to a specified
721  // block, which may or may not be on a fork
722  // for now, assume it's not on a fork
723  if (is_included_block(reference_point))
724  {
725  // reference_point is a block we know about and is on the main chain
726  uint32_t reference_point_block_num = block_header::num_from_id(reference_point);
727  assert(reference_point_block_num > 0);
728  high_block_num = reference_point_block_num;
729  non_fork_high_block_num = high_block_num;
730 
731  if (reference_point_block_num < low_block_num)
732  {
733  // we're on the same fork (at least as far as reference_point) but we've passed
734  // reference point and could no longer undo that far if we diverged after that
735  // block. This should probably only happen due to a race condition where
736  // the network thread calls this function, and then immediately pushes a bunch of blocks,
737  // then the main thread finally processes this function.
738  // with the current framework, there's not much we can do to tell the network
739  // thread what our current head block is, so we'll just pretend that
740  // our head is actually the reference point.
741  // this *may* enable us to fetch blocks that we're unable to push, but that should
742  // be a rare case (and correctly handled)
743  low_block_num = reference_point_block_num;
744  }
745  }
746  else
747  {
748  // block is a block we know about, but it is on a fork
749  try
750  {
751  fork_history = _chain_db->get_block_ids_on_fork(reference_point);
752  // returns a vector where the last element is the common ancestor with the preferred chain,
753  // and the first element is the reference point you passed in
754  assert(fork_history.size() >= 2);
755 
756  if( fork_history.front() != reference_point )
757  {
758  edump( (fork_history)(reference_point) );
759  assert(fork_history.front() == reference_point);
760  }
761  block_id_type last_non_fork_block = fork_history.back();
762  fork_history.pop_back(); // remove the common ancestor
763  boost::reverse(fork_history);
764 
765  if (last_non_fork_block == block_id_type()) // if the fork goes all the way back to genesis (does graphene's fork db allow this?)
766  non_fork_high_block_num = 0;
767  else
768  non_fork_high_block_num = block_header::num_from_id(last_non_fork_block);
769 
770  high_block_num = non_fork_high_block_num + fork_history.size();
771  assert(high_block_num == block_header::num_from_id(fork_history.back()));
772  }
773  catch (const fc::exception& e)
774  {
775  // unable to get fork history for some reason. maybe not linked?
776  // we can't return a synopsis of its chain
777  elog("Unable to construct a blockchain synopsis for reference hash ${hash}: ${exception}", ("hash", reference_point)("exception", e));
778  throw;
779  }
780  if (non_fork_high_block_num < low_block_num)
781  {
782  wlog("Unable to generate a usable synopsis because the peer we're generating it for forked too long ago "
783  "(our chains diverge after block #${non_fork_high_block_num} but only undoable to block #${low_block_num})",
784  ("low_block_num", low_block_num)
785  ("non_fork_high_block_num", non_fork_high_block_num));
786  FC_THROW_EXCEPTION(graphene::net::block_older_than_undo_history, "Peer is are on a fork I'm unable to switch to");
787  }
788  }
789  }
790  else
791  {
792  // no reference point specified, summarize the whole block chain
793  high_block_num = _chain_db->head_block_num();
794  non_fork_high_block_num = high_block_num;
795  if (high_block_num == 0)
796  return synopsis; // we have no blocks
797  }
798 
799  if( low_block_num == 0)
800  low_block_num = 1;
801 
802  // at this point:
803  // low_block_num is the block before the first block we can undo,
804  // non_fork_high_block_num is the block before the fork (if the peer is on a fork, or otherwise it is the same as high_block_num)
805  // high_block_num is the block number of the reference block, or the end of the chain if no reference provided
806 
807  // true_high_block_num is the ending block number after the network code appends any item ids it
808  // knows about that we don't
809  uint32_t true_high_block_num = high_block_num + number_of_blocks_after_reference_point;
810  do
811  {
812  // for each block in the synopsis, figure out where to pull the block id from.
813  // if it's <= non_fork_high_block_num, we grab it from the main blockchain;
814  // if it's not, we pull it from the fork history
815  if (low_block_num <= non_fork_high_block_num)
816  synopsis.push_back(_chain_db->get_block_id_for_num(low_block_num));
817  else
818  synopsis.push_back(fork_history[low_block_num - non_fork_high_block_num - 1]);
819  low_block_num += (true_high_block_num - low_block_num + 2) / 2;
820  }
821  while (low_block_num <= high_block_num);
822 
823  //idump((synopsis));
824  return synopsis;
825  } FC_CAPTURE_AND_RETHROW() }
