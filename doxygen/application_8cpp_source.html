217 std::vector<fc::ip::endpoint> application_impl::resolve_string_to_ip_endpoints(const std::string& endpoint_string)
218 {
219  try
220  {
221  string::size_type colon_pos = endpoint_string.find(':');
222  if (colon_pos == std::string::npos)
223  FC_THROW("Missing required port number in endpoint string \"${endpoint_string}\"",
224  ("endpoint_string", endpoint_string));
225  std::string port_string = endpoint_string.substr(colon_pos + 1);
226  try
227  {
228  uint16_t port = boost::lexical_cast<uint16_t>(port_string);
229 
230  std::string hostname = endpoint_string.substr(0, colon_pos);
231  std::vector<fc::ip::endpoint> endpoints = fc::resolve(hostname, port);
232  if (endpoints.empty())
233  FC_THROW_EXCEPTION( fc::unknown_host_exception,
234  "The host name can not be resolved: ${hostname}",
235  ("hostname", hostname) );
236  return endpoints;
237  }
238  catch (const boost::bad_lexical_cast&)
239  {
240  FC_THROW("Bad port: ${port}", ("port", port_string));
241  }
242  }
243  FC_CAPTURE_AND_RETHROW((endpoint_string))
244 }

247 void application_impl::new_connection( const fc::http::websocket_connection_ptr& c )
248 {
249  auto wsc = std::make_shared<fc::rpc::websocket_api_connection>(*c, GRAPHENE_NET_MAX_NESTED_OBJECTS);
250  auto login = std::make_shared<graphene::app::login_api>( std::ref(*_self) );
251  login->enable_api("database_api");
252 
253  wsc->register_api(login->database());
254  wsc->register_api(fc::api<graphene::app::login_api>(login));
255  c->set_session_data( wsc );
256 
257  std::string username = "*";
258  std::string password = "*";
259 
260  // Try to extract login information from "Authorization" header if present
261  std::string auth = c->get_request_header("Authorization");
262  if( boost::starts_with(auth, "Basic ") ) {
263 
264  FC_ASSERT( auth.size() > 6 );
265  auto user_pass = fc::base64_decode(auth.substr(6));
266 
267  std::vector<std::string> parts;
268  boost::split( parts, user_pass, boost::is_any_of(":") );
269 
270  FC_ASSERT(parts.size() == 2);
271 
272  username = parts[0];
273  password = parts[1];
274  }
275 
276  login->login(username, password);
277 }

515 bool application_impl::handle_block(const graphene::net::block_message& blk_msg, bool sync_mode,
516  std::vector<fc::uint160_t>& contained_transaction_message_ids)
517 { try {
518 
519  auto latency = fc::time_point::now() - blk_msg.block.timestamp;
520  if (!sync_mode || blk_msg.block.block_num() % 10000 == 0)
521  {
522  const auto& witness = blk_msg.block.witness(*_chain_db);
523  const auto& witness_account = witness.witness_account(*_chain_db);
524  auto last_irr = _chain_db->get_dynamic_global_properties().last_irreversible_block_num;
525  ilog("Got block: #${n} ${bid} time: ${t} transaction(s): ${x} latency: ${l} ms from: ${w} irreversible: ${i} (-${d})",
526  ("t",blk_msg.block.timestamp)
527  ("n", blk_msg.block.block_num())
528  ("bid", blk_msg.block.id())
529  ("x", blk_msg.block.transactions.size())
530  ("l", (latency.count()/1000))
531  ("w",witness_account.name)
532  ("i",last_irr)("d",blk_msg.block.block_num()-last_irr) );
533  }
534  FC_ASSERT( (latency.count()/1000) > -5000, "Rejecting block with timestamp in the future" );
535 
536  try {
537  const uint32_t skip = (_is_block_producer | _force_validate) ?
538  database::skip_nothing : database::skip_transaction_signatures;
539  bool result = valve.do_serial( [this,&blk_msg,skip] () {
540  _chain_db->precompute_parallel( blk_msg.block, skip ).wait();
541  }, [this,&blk_msg,skip] () {
542  // TODO: in the case where this block is valid but on a fork that's too old for us to switch to,
543  // you can help the network code out by throwing a block_older_than_undo_history exception.
544  // when the net code sees that, it will stop trying to push blocks from that chain, but
545  // leave that peer connected so that they can get sync blocks from us
546  return _chain_db->push_block( blk_msg.block, skip );
547  });
548 
549  // the block was accepted, so we now know all of the transactions contained in the block
550  if (!sync_mode)
551  {
552  // if we're not in sync mode, there's a chance we will be seeing some transactions
553  // included in blocks before we see the free-floating transaction itself. If that
554  // happens, there's no reason to fetch the transactions, so construct a list of the
555  // transaction message ids we no longer need.
556  // during sync, it is unlikely that we'll see any old
557  contained_transaction_message_ids.reserve( contained_transaction_message_ids.size()
558  + blk_msg.block.transactions.size() );
559  for (const processed_transaction& transaction : blk_msg.block.transactions)
560  {
561  graphene::net::trx_message transaction_message(transaction);
562  contained_transaction_message_ids.emplace_back(graphene::net::message(transaction_message).id());
563  }
564  }
565 
566  return result;
567  } catch ( const graphene::chain::unlinkable_block_exception& e ) {
568  // translate to a graphene::net exception
569  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
570  FC_THROW_EXCEPTION( graphene::net::unlinkable_block_exception,
571  "Error when pushing block:\n${e}",
572  ("e", e.to_detail_string()) );
573  } catch( const fc::exception& e ) {
574  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
575  throw;
576  }
577 
578  if( !_is_finished_syncing && !sync_mode )
579  {
580  _is_finished_syncing = true;
581  _self->syncing_finished();
582  }
583 } FC_CAPTURE_AND_RETHROW( (blk_msg)(sync_mode) ) return false; }

623 std::vector<item_hash_t> application_impl::get_block_ids(const std::vector<item_hash_t>& blockchain_synopsis,
624  uint32_t& remaining_item_count,
625  uint32_t limit)
626 { try {
627  vector<block_id_type> result;
628  remaining_item_count = 0;
629  if( _chain_db->head_block_num() == 0 )
630  return result;
631 
632  result.reserve(limit);
633  block_id_type last_known_block_id;
634 
635  if (blockchain_synopsis.empty() ||
636  (blockchain_synopsis.size() == 1 && blockchain_synopsis[0] == block_id_type()))
637  {
638  // peer has sent us an empty synopsis meaning they have no blocks.
639  // A bug in old versions would cause them to send a synopsis containing block 000000000
640  // when they had an empty blockchain, so pretend they sent the right thing here.
641 
642  // do nothing, leave last_known_block_id set to zero
643  }
644  else
645  {
646  bool found_a_block_in_synopsis = false;
647  for (const item_hash_t& block_id_in_synopsis : boost::adaptors::reverse(blockchain_synopsis))
648  if (block_id_in_synopsis == block_id_type() ||
649  (_chain_db->is_known_block(block_id_in_synopsis) && is_included_block(block_id_in_synopsis)))
650  {
651  last_known_block_id = block_id_in_synopsis;
652  found_a_block_in_synopsis = true;
653  break;
654  }
655  if (!found_a_block_in_synopsis)
656  FC_THROW_EXCEPTION( graphene::net::peer_is_on_an_unreachable_fork,
657  "Unable to provide a list of blocks starting at any of the blocks in peer's synopsis" );
658  }
659  for( uint32_t num = block_header::num_from_id(last_known_block_id);
660  num <= _chain_db->head_block_num() && result.size() < limit;
661  ++num )
662  if( num > 0 )
663  result.push_back(_chain_db->get_block_id_for_num(num));
664 
665  if( !result.empty() && block_header::num_from_id(result.back()) < _chain_db->head_block_num() )
666  remaining_item_count = _chain_db->head_block_num() - block_header::num_from_id(result.back());
667 
668  return result;
669 } FC_CAPTURE_AND_RETHROW( (blockchain_synopsis)(remaining_item_count)(limit) ) }

753 std::vector<item_hash_t> application_impl::get_blockchain_synopsis(const item_hash_t& reference_point,
754  uint32_t number_of_blocks_after_reference_point)
755 { try {
756  std::vector<item_hash_t> synopsis;
757  synopsis.reserve(30);
758  uint32_t high_block_num;
759  uint32_t non_fork_high_block_num;
760  uint32_t low_block_num = _chain_db->last_non_undoable_block_num();
761  std::vector<block_id_type> fork_history;
762 
763  if (reference_point != item_hash_t())
764  {
765  // the node is asking for a summary of the block chain up to a specified
766  // block, which may or may not be on a fork
767  // for now, assume it's not on a fork
768  if (is_included_block(reference_point))
769  {
770  // reference_point is a block we know about and is on the main chain
771  uint32_t reference_point_block_num = block_header::num_from_id(reference_point);
772  assert(reference_point_block_num > 0);
773  high_block_num = reference_point_block_num;
774  non_fork_high_block_num = high_block_num;
775 
776  if (reference_point_block_num < low_block_num)
777  {
778  // we're on the same fork (at least as far as reference_point) but we've passed
779  // reference point and could no longer undo that far if we diverged after that
780  // block. This should probably only happen due to a race condition where
781  // the network thread calls this function, and then immediately pushes a bunch of blocks,
782  // then the main thread finally processes this function.
783  // with the current framework, there's not much we can do to tell the network
784  // thread what our current head block is, so we'll just pretend that
785  // our head is actually the reference point.
786  // this *may* enable us to fetch blocks that we're unable to push, but that should
787  // be a rare case (and correctly handled)
788  low_block_num = reference_point_block_num;
789  }
790  }
791  else
792  {
793  // block is a block we know about, but it is on a fork
794  try
795  {
796  fork_history = _chain_db->get_block_ids_on_fork(reference_point);
797  // returns a vector where the last element is the common ancestor with the preferred chain,
798  // and the first element is the reference point you passed in
799  assert(fork_history.size() >= 2);
800 
801  if( fork_history.front() != reference_point )
802  {
803  edump( (fork_history)(reference_point) );
804  assert(fork_history.front() == reference_point);
805  }
806  block_id_type last_non_fork_block = fork_history.back();
807  fork_history.pop_back(); // remove the common ancestor
808  boost::reverse(fork_history);
809 
810  if (last_non_fork_block == block_id_type()) // if the fork goes all the way back to genesis (does graphene's fork db allow this?)
811  non_fork_high_block_num = 0;
812  else
813  non_fork_high_block_num = block_header::num_from_id(last_non_fork_block);
814 
815  high_block_num = non_fork_high_block_num + fork_history.size();
816  assert(high_block_num == block_header::num_from_id(fork_history.back()));
817  }
818  catch (const fc::exception& e)
819  {
820  // unable to get fork history for some reason. maybe not linked?
821  // we can't return a synopsis of its chain
822  elog( "Unable to construct a blockchain synopsis for reference hash ${hash}: ${exception}",
823  ("hash", reference_point)("exception", e) );
824  throw;
825  }
826  if (non_fork_high_block_num < low_block_num)
827  {
828  wlog("Unable to generate a usable synopsis because the peer we're generating it for forked too long ago "
829  "(our chains diverge after block #${non_fork_high_block_num} but only undoable to block #${low_block_num})",
830  ("low_block_num", low_block_num)
831  ("non_fork_high_block_num", non_fork_high_block_num));
832  FC_THROW_EXCEPTION(graphene::net::block_older_than_undo_history, "Peer is are on a fork I'm unable to switch to");
833  }
834  }
835  }
836  else
837  {
838  // no reference point specified, summarize the whole block chain
839  high_block_num = _chain_db->head_block_num();
840  non_fork_high_block_num = high_block_num;
841  if (high_block_num == 0)
842  return synopsis; // we have no blocks
843  }
844 
845  if( low_block_num == 0)
846  low_block_num = 1;
847 
848  // at this point:
849  // low_block_num is the block before the first block we can undo,
850  // non_fork_high_block_num is the block before the fork (if the peer is on a fork, or otherwise it is the same as high_block_num)
851  // high_block_num is the block number of the reference block, or the end of the chain if no reference provided
852 
853  // true_high_block_num is the ending block number after the network code appends any item ids it
854  // knows about that we don't
855  uint32_t true_high_block_num = high_block_num + number_of_blocks_after_reference_point;
856  do
857  {
858  // for each block in the synopsis, figure out where to pull the block id from.
859  // if it's <= non_fork_high_block_num, we grab it from the main blockchain;
860  // if it's not, we pull it from the fork history
861  if (low_block_num <= non_fork_high_block_num)
862  synopsis.push_back(_chain_db->get_block_id_for_num(low_block_num));
863  else
864  synopsis.push_back(fork_history[low_block_num - non_fork_high_block_num - 1]);
865  low_block_num += (true_high_block_num - low_block_num + 2) / 2;
866  }
867  while (low_block_num <= high_block_num);
868 
869  //idump((synopsis));
870  return synopsis;
871 } FC_CAPTURE_AND_RETHROW() }
