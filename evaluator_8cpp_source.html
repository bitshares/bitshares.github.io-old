53  void generic_evaluator::prepare_fee(account_id_type account_id, asset fee)
54  {
55  const database& d = db();
56  fee_from_account = fee;
57  FC_ASSERT( fee.amount >= 0 );
58  fee_paying_account = &account_id(d);
59  fee_paying_account_statistics = &fee_paying_account->statistics(d);
60 
61  fee_asset = &fee.asset_id(d);
62  fee_asset_dyn_data = &fee_asset->dynamic_asset_data_id(d);
63 
64  if( d.head_block_time() > HARDFORK_419_TIME )
65  {
66  FC_ASSERT( is_authorized_asset( d, *fee_paying_account, *fee_asset ), "Account ${acct} '${name}' attempted to pay fee by using asset ${a} '${sym}', which is unauthorized due to whitelist / blacklist",
67  ("acct", fee_paying_account->id)("name", fee_paying_account->name)("a", fee_asset->id)("sym", fee_asset->symbol) );
68  }
69 
70  if( fee_from_account.asset_id == asset_id_type() )
71  core_fee_paid = fee_from_account.amount;
72  else
73  {
74  asset fee_from_pool = fee_from_account * fee_asset->options.core_exchange_rate;
75  FC_ASSERT( fee_from_pool.asset_id == asset_id_type() );
76  core_fee_paid = fee_from_pool.amount;
77  FC_ASSERT( core_fee_paid <= fee_asset_dyn_data->fee_pool, "Fee pool balance of '${b}' is less than the ${r} required to convert ${c}",
78  ("r", db().to_pretty_string( fee_from_pool))("b",db().to_pretty_string(fee_asset_dyn_data->fee_pool))("c",db().to_pretty_string(fee)) );
79  }
80  }
