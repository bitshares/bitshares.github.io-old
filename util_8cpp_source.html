38 fc::uint128 to_capped128( const u256& t )
39 {
40  static u256 max128 = to256( fc::uint128::max_value() );
41  if( t >= max128 )
42  return fc::uint128::max_value();
43  fc::uint128 result;
44  u256 hi(t);
45  hi >>= 64;
46  result.hi = static_cast< uint64_t >( hi );
47  u256 lo(t);
48  hi <<= 64;
49  lo -= hi;
50  result.lo = static_cast< uint64_t >( lo );
51  return result;
52 }

54 string uint128_amount_to_string( const fc::uint128& amount, const uint8_t precision )
55 { try {
56  string s = string( amount );
57  if( precision == 0 || amount == fc::uint128() )
58  return s;
59 
60  std::stringstream ss;
61  uint8_t pos = s.find_last_not_of( '0' ); // should be >= 0
62  uint8_t len = s.size();
63  if( len > precision )
64  {
65  uint8_t left_len = len - precision;
66  ss << s.substr( 0, left_len );
67  if( pos >= left_len )
68  ss << '.' << s.substr( left_len, pos - left_len + 1 );
69  }
70  else
71  {
72  ss << "0.";
73  for( uint8_t i = precision - len; i > 0; --i )
74  ss << '0';
75  ss << s.substr( 0, pos + 1 );
76  }
77  return ss.str();
78 } FC_CAPTURE_AND_RETHROW( (amount)(precision) ) }

80 string price_to_string( const price& _price, const uint8_t base_precision, const uint8_t quote_precision )
81 { try {
82  if( _price.base.amount == 0 )
83  return "0";
84  FC_ASSERT( _price.base.amount >= 0 );
85  FC_ASSERT( _price.quote.amount >= 0 );
86  FC_ASSERT( base_precision <= 19 );
87  FC_ASSERT( quote_precision <= 19 );
88  price new_price = _price;
89  if( new_price.quote.amount == 0 )
90  {
91  new_price.base.amount = std::numeric_limits<int64_t>::max();
92  new_price.quote.amount = 1;
93  }
94 
95  // times (10**19) so won't overflow but have good accuracy
96  fc::uint128 price128 = fc::uint128( new_price.base.amount.value ) * uint64_t(10000000000000000000ULL)
97  / new_price.quote.amount.value;
98 
99  return uint128_amount_to_string( price128, 19 + base_precision - quote_precision );
100 } FC_CAPTURE_AND_RETHROW( (_price)(base_precision)(quote_precision) ) }

102 string price_diff_percent_string( const price& old_price, const price& new_price )
103 { try {
104  FC_ASSERT( old_price.base.asset_id == new_price.base.asset_id );
105  FC_ASSERT( old_price.quote.asset_id == new_price.quote.asset_id );
106  FC_ASSERT( old_price.base.amount >= 0 );
107  FC_ASSERT( old_price.quote.amount >= 0 );
108  FC_ASSERT( new_price.base.amount >= 0 );
109  FC_ASSERT( new_price.quote.amount >= 0 );
110  price old_price1 = old_price;
111  if( old_price.base.amount == 0 )
112  {
113  old_price1.base.amount = 1;
114  old_price1.quote.amount = std::numeric_limits<int64_t>::max();
115  }
116  else if( old_price.quote.amount == 0 )
117  {
118  old_price1.base.amount = std::numeric_limits<int64_t>::max();
119  old_price1.quote.amount = 1;
120  }
121  price new_price1 = new_price;
122  if( new_price.base.amount == 0 )
123  {
124  new_price1.base.amount = 1;
125  new_price1.quote.amount = std::numeric_limits<int64_t>::max();
126  }
127  else if( new_price.quote.amount == 0 )
128  {
129  new_price1.base.amount = std::numeric_limits<int64_t>::max();
130  new_price1.quote.amount = 1;
131  }
132 
133  // change = new/old - 1 = (new_base/new_quote)/(old_base/old_quote) - 1
134  // = (new_base * old_quote) / (new_quote * old_base) - 1
135  // = (new_base * old_quote - new_quote * old_base) / (new_quote * old_base)
136  fc::uint128 new128 = fc::uint128( new_price1.base.amount.value ) * old_price1.quote.amount.value;
137  fc::uint128 old128 = fc::uint128( old_price1.base.amount.value ) * new_price1.quote.amount.value;
138  bool non_negative = (new128 >= old128);
139  fc::uint128 diff128;
140  if( non_negative )
141  diff128 = new128 - old128;
142  else
143  diff128 = old128 - new128;
144  static fc::uint128 max = fc::uint128::max_value() / 10000;
145  if( diff128 <= max )
146  diff128 = diff128 * 10000 / old128;
147  else
148  {
149  u256 diff256 = to256( diff128 );
150  diff256 *= 10000;
151  diff256 /= to256( old128 );
152  diff128 = to_capped128( diff256 );
153  }
154  string diff_str = uint128_amount_to_string( diff128, 2 ); // at most 2 decimal digits
155  if( non_negative || diff_str == "0" )
156  return diff_str;
157  else
158  return "-" + diff_str;
159 } FC_CAPTURE_AND_RETHROW( (old_price)(new_price) ) }
