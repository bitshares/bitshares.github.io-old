54  bool login_api::login(const string& user, const string& password)
55  {
56  optional< api_access_info > acc = _app.get_api_access_info( user );
57  if( !acc.valid() )
58  return false;
59  if( acc->password_hash_b64 != "*" )
60  {
61  std::string password_salt = fc::base64_decode( acc->password_salt_b64 );
62  std::string acc_password_hash = fc::base64_decode( acc->password_hash_b64 );
63 
64  fc::sha256 hash_obj = fc::sha256::hash( password + password_salt );
65  if( hash_obj.data_size() != acc_password_hash.length() )
66  return false;
67  if( memcmp( hash_obj.data(), acc_password_hash.c_str(), hash_obj.data_size() ) != 0 )
68  return false;
69  }
70 
71  for( const std::string& api_name : acc->allowed_apis )
72  enable_api( api_name );
73  return true;
74  }

138  void network_broadcast_api::on_applied_block( const signed_block& b )
139  {
140  if( _callbacks.size() )
141  {
143  auto capture_this = shared_from_this();
144  for( uint32_t trx_num = 0; trx_num < b.transactions.size(); ++trx_num )
145  {
146  const auto& trx = b.transactions[trx_num];
147  auto id = trx.id();
148  auto itr = _callbacks.find(id);
149  if( itr != _callbacks.end() )
150  {
151  auto block_num = b.block_num();
152  auto& callback = _callbacks.find(id)->second;
153  auto v = fc::variant( transaction_confirmation{ id, block_num, trx_num, trx }, GRAPHENE_MAX_NESTED_OBJECTS );
154  fc::async( [capture_this,v,callback]() {
155  callback(v);
156  } );
157  }
158  }
159  }

286  vector<order_history_object> history_api::get_fill_order_history( asset_id_type a, asset_id_type b, uint32_t limit )const
287  {
288  FC_ASSERT(_app.chain_database());
289  const auto& db = *_app.chain_database();
290  if( a > b ) std::swap(a,b);
291  const auto& history_idx = db.get_index_type<graphene::market_history::history_index>().indices().get<by_key>();
292  history_key hkey;

294  hkey.quote = b;
295  hkey.sequence = std::numeric_limits<int64_t>::min();
296 
297  uint32_t count = 0;
298  auto itr = history_idx.lower_bound( hkey );
299  vector<order_history_object> result;
300  while( itr != history_idx.end() && count < limit)

302  if( itr->key.base != a || itr->key.quote != b ) break;
303  result.push_back( *itr );
304  ++itr;
305  ++count;
306  }
307 
308  return result;
309  }

311  vector<operation_history_object> history_api::get_account_history( const std::string account_id_or_name,
312  operation_history_id_type stop,
313  unsigned limit,
314  operation_history_id_type start ) const
315  {
316  FC_ASSERT( _app.chain_database() );
317  const auto& db = *_app.chain_database();
318  FC_ASSERT( limit <= 100 );
319  vector<operation_history_object> result;
320  account_id_type account;
321  try {
322  account = database_api.get_account_id_from_string(account_id_or_name);
323  const account_transaction_history_object& node = account(db).statistics(db).most_recent_op(db);
324  if(start == operation_history_id_type() || start.instance.value > node.operation_id.instance.value)
325  start = node.operation_id;
326  } catch(...) { return result; }
327 
328  const auto& hist_idx = db.get_index_type<account_transaction_history_index>();
329  const auto& by_op_idx = hist_idx.indices().get<by_op>();
330  auto index_start = by_op_idx.begin();
331  auto itr = by_op_idx.lower_bound(boost::make_tuple(account, start));
332 
333  while(itr != index_start && itr->account == account && itr->operation_id.instance.value > stop.instance.value && result.size() < limit)
334  {
335  if(itr->operation_id.instance.value <= start.instance.value)
336  result.push_back(itr->operation_id(db));
337  --itr;
338  }
339  if(stop.instance.value == 0 && result.size() < limit && itr->account == account) {
340  result.push_back(itr->operation_id(db));
341  }
342 
343  return result;
344  }

346  vector<operation_history_object> history_api::get_account_history_operations( const std::string account_id_or_name,
347  int operation_id,
348  operation_history_id_type start,
349  operation_history_id_type stop,
350  unsigned limit) const
351  {
352  FC_ASSERT( _app.chain_database() );
353  const auto& db = *_app.chain_database();
354  FC_ASSERT( limit <= 100 );
355  vector<operation_history_object> result;
356  account_id_type account;
357  try {
358  account = database_api.get_account_id_from_string(account_id_or_name);
359  } catch(...) { return result; }
360  const auto& stats = account(db).statistics(db);
361  if( stats.most_recent_op == account_transaction_history_id_type() ) return result;
362  const account_transaction_history_object* node = &stats.most_recent_op(db);
363  if( start == operation_history_id_type() )
364  start = node->operation_id;
365 
366  while(node && node->operation_id.instance.value > stop.instance.value && result.size() < limit)
367  {
368  if( node->operation_id.instance.value <= start.instance.value ) {
369 
370  if(node->operation_id(db).op.which() == operation_id)
371  result.push_back( node->operation_id(db) );
372  }
373  if( node->next == account_transaction_history_id_type() )
374  node = nullptr;
375  else node = &node->next(db);
376  }
377  if( stop.instance.value == 0 && result.size() < limit ) {
378  const account_transaction_history_object head = account_transaction_history_id_type()(db);
379  if( head.account == account && head.operation_id(db).op.which() == operation_id )
380  result.push_back(head.operation_id(db));
381  }
382  return result;
383  }

386  vector<operation_history_object> history_api::get_relative_account_history( const std::string account_id_or_name,
387  uint32_t stop,
388  unsigned limit,
389  uint32_t start) const
390  {
391  FC_ASSERT( _app.chain_database() );
392  const auto& db = *_app.chain_database();
393  FC_ASSERT(limit <= 100);
394  vector<operation_history_object> result;
395  account_id_type account;
396  try {
397  account = database_api.get_account_id_from_string(account_id_or_name);
398  } catch(...) { return result; }
399  const auto& stats = account(db).statistics(db);
400  if( start == 0 )
401  start = stats.total_ops;
402  else
403  start = min( stats.total_ops, start );
404 
405  if( start >= stop && start > stats.removed_ops && limit > 0 )
406  {
407  const auto& hist_idx = db.get_index_type<account_transaction_history_index>();
408  const auto& by_seq_idx = hist_idx.indices().get<by_seq>();
409 
410  auto itr = by_seq_idx.upper_bound( boost::make_tuple( account, start ) );
411  auto itr_stop = by_seq_idx.lower_bound( boost::make_tuple( account, stop ) );
412 
413  do
414  {
415  --itr;
416  result.push_back( itr->operation_id(db) );
417  }
418  while ( itr != itr_stop && result.size() < limit );
419  }
420  return result;
421  }
