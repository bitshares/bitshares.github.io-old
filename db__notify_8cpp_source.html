273 void get_relevant_accounts( const object* obj, flat_set<account_id_type>& accounts )
274 {
275  if( obj->id.space() == protocol_ids )
276  {
277  switch( (object_type)obj->id.type() )
278  {
279  case null_object_type:
280  case base_object_type:
281  case OBJECT_TYPE_COUNT:
282  return;
283  case account_object_type:{
284  accounts.insert( obj->id );
285  break;
286  } case asset_object_type:{
287  const auto& aobj = dynamic_cast<const asset_object*>(obj);
288  FC_ASSERT( aobj != nullptr );
289  accounts.insert( aobj->issuer );
290  break;
291  } case force_settlement_object_type:{
292  const auto& aobj = dynamic_cast<const force_settlement_object*>(obj);
293  FC_ASSERT( aobj != nullptr );
294  accounts.insert( aobj->owner );
295  break;
296  } case committee_member_object_type:{
297  const auto& aobj = dynamic_cast<const committee_member_object*>(obj);
298  FC_ASSERT( aobj != nullptr );
299  accounts.insert( aobj->committee_member_account );
300  break;
301  } case witness_object_type:{
302  const auto& aobj = dynamic_cast<const witness_object*>(obj);
303  FC_ASSERT( aobj != nullptr );
304  accounts.insert( aobj->witness_account );
305  break;
306  } case limit_order_object_type:{
307  const auto& aobj = dynamic_cast<const limit_order_object*>(obj);
308  FC_ASSERT( aobj != nullptr );
309  accounts.insert( aobj->seller );
310  break;
311  } case call_order_object_type:{
312  const auto& aobj = dynamic_cast<const call_order_object*>(obj);
313  FC_ASSERT( aobj != nullptr );
314  accounts.insert( aobj->borrower );
315  break;
316  } case custom_object_type:{
317  break;
318  } case proposal_object_type:{
319  const auto& aobj = dynamic_cast<const proposal_object*>(obj);
320  FC_ASSERT( aobj != nullptr );
321  transaction_get_impacted_accounts( aobj->proposed_transaction, accounts );
322  break;
323  } case operation_history_object_type:{

326  operation_get_impacted_accounts( aobj->op, accounts );
327  break;
328  } case withdraw_permission_object_type:{
329  const auto& aobj = dynamic_cast<const withdraw_permission_object*>(obj);
330  FC_ASSERT( aobj != nullptr );
331  accounts.insert( aobj->withdraw_from_account );
332  accounts.insert( aobj->authorized_account );
333  break;
334  } case vesting_balance_object_type:{
335  const auto& aobj = dynamic_cast<const vesting_balance_object*>(obj);
336  FC_ASSERT( aobj != nullptr );
337  accounts.insert( aobj->owner );
338  break;
339  } case worker_object_type:{
340  const auto& aobj = dynamic_cast<const worker_object*>(obj);
341  FC_ASSERT( aobj != nullptr );
342  accounts.insert( aobj->worker_account );
343  break;
344  } case balance_object_type:{
346  break;
347  }
348  }
349  }
350  else if( obj->id.space() == implementation_ids )
351  {
352  switch( (impl_object_type)obj->id.type() )
353  {
354  case impl_global_property_object_type:
355  break;
356  case impl_dynamic_global_property_object_type:
357  break;
358  case impl_reserved0_object_type:
359  break;
360  case impl_asset_dynamic_data_type:
361  break;
362  case impl_asset_bitasset_data_type:
363  break;
364  case impl_account_balance_object_type:{
365  const auto& aobj = dynamic_cast<const account_balance_object*>(obj);
366  FC_ASSERT( aobj != nullptr );
367  accounts.insert( aobj->owner );
368  break;
369  } case impl_account_statistics_object_type:{
370  const auto& aobj = dynamic_cast<const account_statistics_object*>(obj);
371  FC_ASSERT( aobj != nullptr );
372  accounts.insert( aobj->owner );
373  break;
374  } case impl_transaction_object_type:{
