52 void database::reindex( fc::path data_dir )
53 { try {
54  auto last_block = _block_id_to_block.last();
55  if( !last_block ) {
56  elog( "!no last block" );
57  edump((last_block));
58  return;
59  }
60  if( last_block->block_num() <= head_block_num()) return;
61 
62  ilog( "reindexing blockchain" );
63  auto start = fc::time_point::now();
64  const auto last_block_num = last_block->block_num();
65  uint32_t flush_point = last_block_num < 10000 ? 0 : last_block_num - 10000;
66  uint32_t undo_point = last_block_num < 50 ? 0 : last_block_num - 50;
67 
68  ilog( "Replaying blocks, starting at ${next}...", ("next",head_block_num() + 1) );
69  if( head_block_num() >= undo_point )
70  {
71  if( head_block_num() > 0 )
72  _fork_db.start_block( *fetch_block_by_number( head_block_num() ) );
73  }
74  else
75  _undo_db.disable();
76  for( uint32_t i = head_block_num() + 1; i <= last_block_num; ++i )
77  {
78  if( i % 10000 == 0 ) std::cerr << " " << double(i*100)/last_block_num << "% "<<i << " of " <<last_block_num<<" \n";
79  if( i == flush_point )
80  {
81  ilog( "Writing database to disk at block ${i}", ("i",i) );
82  flush();
83  ilog( "Done" );
84  }
85  fc::optional< signed_block > block = _block_id_to_block.fetch_by_number(i);
86  if( !block.valid() )
87  {
88  wlog( "Reindexing terminated due to gap: Block ${i} does not exist!", ("i", i) );
89  uint32_t dropped_count = 0;
90  while( true )
91  {
92  fc::optional< block_id_type > last_id = _block_id_to_block.last_id();
93  // this can trigger if we attempt to e.g. read a file that has block #2 but no block #1
94  if( !last_id.valid() )
95  break;
96  // we've caught up to the gap
97  if( block_header::num_from_id( *last_id ) <= i )
98  break;
99  _block_id_to_block.remove( *last_id );
100  dropped_count++;
101  }
102  wlog( "Dropped ${n} blocks from after the gap", ("n", dropped_count) );
103  break;
104  }
105  if( i < undo_point )
106  apply_block(*block, skip_witness_signature |
107  skip_transaction_signatures |
108  skip_transaction_dupe_check |
109  skip_tapos_check |
110  skip_witness_schedule_check |
111  skip_authority_check);
112  else
113  {
114  _undo_db.enable();
115  push_block(*block, skip_witness_signature |
116  skip_transaction_signatures |
117  skip_transaction_dupe_check |
118  skip_tapos_check |
119  skip_witness_schedule_check |
120  skip_authority_check);
121  }
122  }
123  _undo_db.enable();
124  auto end = fc::time_point::now();
125  ilog( "Done reindexing, elapsed time: ${t} sec", ("t",double((end-start).count())/1000000.0 ) );
126 } FC_CAPTURE_AND_RETHROW( (data_dir) ) }

139 void database::open(
140  const fc::path& data_dir,
141  std::function<genesis_state_type()> genesis_loader,
142  const std::string& db_version)
143 {
144  try
145  {
146  bool wipe_object_db = false;
147  if( !fc::exists( data_dir / "db_version" ) )
148  wipe_object_db = true;
149  else
150  {
151  std::string version_string;
152  fc::read_file_contents( data_dir / "db_version", version_string );
153  wipe_object_db = ( version_string != db_version );
154  }
155  if( wipe_object_db ) {
156  ilog("Wiping object_database due to missing or wrong version");
157  object_database::wipe( data_dir );
158  std::ofstream version_file( (data_dir / "db_version").generic_string().c_str(),
159  std::ios::out | std::ios::binary | std::ios::trunc );
160  version_file.write( db_version.c_str(), db_version.size() );
161  version_file.close();
162  }
163 
164  object_database::open(data_dir);
165 
166  _block_id_to_block.open(data_dir / "database" / "block_num_to_block");
167 
168  if( !find(global_property_id_type()) )
169  init_genesis(genesis_loader());
170  else
171  {
172  _p_core_asset_obj = &get( asset_id_type() );
173  _p_core_dynamic_data_obj = &get( asset_dynamic_data_id_type() );
174  _p_global_prop_obj = &get( global_property_id_type() );
175  _p_chain_property_obj = &get( chain_property_id_type() );
176  _p_dyn_global_prop_obj = &get( dynamic_global_property_id_type() );
177  _p_witness_schedule_obj = &get( witness_schedule_id_type() );
178  }
179 
180  fc::optional<block_id_type> last_block = _block_id_to_block.last_id();
181  if( last_block.valid() )
182  {
183  FC_ASSERT( *last_block >= head_block_id(),
184  "last block ID does not match current chain state",
185  ("last_block->id", last_block)("head_block_id",head_block_num()) );
186  reindex( data_dir );
187  }
188  _opened = true;
189  }
190  FC_CAPTURE_LOG_AND_RETHROW( (data_dir) )
191 }

193 void database::close(bool rewind)
194 {
195  // TODO: Save pending tx's on close()
196  clear_pending();
197 
198  // pop all of the blocks that we can given our undo history, this should
199  // throw when there is no more undo history to pop
200  if( rewind )
201  {
202  try
203  {
204  uint32_t cutoff = get_dynamic_global_properties().last_irreversible_block_num;
205 
206  ilog( "Rewinding from ${head} to ${cutoff}", ("head",head_block_num())("cutoff",cutoff) );
207  while( head_block_num() > cutoff )
208  {
209  block_id_type popped_block_id = head_block_id();
210  pop_block();
211  _fork_db.remove(popped_block_id); // doesn't throw on missing
212  }
213  }
214  catch ( const fc::exception& e )
215  {
216  wlog( "Database close unexpected exception: ${e}", ("e", e) );
217  }
218  }
219 
220  // Since pop_block() will move tx's in the popped blocks into pending,
221  // we have to clear_pending() after we're done popping to get a clean
222  // DB state (issue #336).
223  clear_pending();
224 
225  object_database::flush();
226  object_database::close();
227 
228  if( _block_id_to_block.is_open() )
229  _block_id_to_block.close();
230 
231  _fork_db.reset();
232 
233  _opened = false;
234 }
