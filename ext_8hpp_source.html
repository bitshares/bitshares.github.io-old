87  : stream(s), value(v), max_depth(_max_depth - 1)

92  count_left = c.value;

96  void maybe_read_next_which()const
97  {
98  if( count_left > 0 )
99  {
100  unsigned_int w;
101  fc::raw::unpack( stream, w, max_depth );
102  next_which = w.value;
103  }
104  }

106  template< typename Member, class Class, Member (Class::*member)>
107  void operator()( const char* name )const
108  {
109  if( (count_left > 0) && (which == next_which) )
110  {
111  typename Member::value_type temp;
112  fc::raw::unpack( stream, temp, max_depth );
113  (value.*member) = temp;
114  --count_left;
115  maybe_read_next_which();
116  }
117  else
118  (value.*member).reset();

120  }

163 template< typename T >
164 void from_variant( const fc::variant& var, graphene::chain::extension<T>& value, uint32_t max_depth )
165 {
166  value = graphene::chain::extension<T>();
167  if( var.is_null() )
168  return;
169  if( var.is_array() )
170  {
171  FC_ASSERT( var.size() == 0 );
172  return;
173  }
174 
175  graphene_extension_from_variant_visitor<T> vtor( var.get_object(), value.value, max_depth );
176  fc::reflector<T>::visit( vtor );
177  FC_ASSERT( vtor.count_left == 0 ); // unrecognized extension throws here
178 }
