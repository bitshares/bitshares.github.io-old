42 fc::time_point_sec database::get_slot_time(uint32_t slot_num)const
43 {
44  if( slot_num == 0 )
45  return fc::time_point_sec();
46 
47  auto interval = block_interval();
48  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
49 
50  if( head_block_num() == 0 )
51  {
52  // n.b. first block is at genesis_time plus one block interval
53  fc::time_point_sec genesis_time = dpo.time;
54  return genesis_time + slot_num * interval;
55  }
56 
57  int64_t head_block_abs_slot = head_block_time().sec_since_epoch() / interval;
58  fc::time_point_sec head_slot_time(head_block_abs_slot * interval);
59 
60  const global_property_object& gpo = get_global_properties();
61 
62  if( dpo.dynamic_flags & dynamic_global_property_object::maintenance_flag )
63  slot_num += gpo.parameters.maintenance_skip_slots;
64 
65  // "slot 0" is head_slot_time
66  // "slot 1" is head_slot_time,
67  // plus maint interval if head block is a maint block
68  // plus block interval if head block is not a maint block
69  return head_slot_time + (slot_num * interval);
70 }

80 uint32_t database::update_witness_missed_blocks( const signed_block& b )
81 {
82  uint32_t missed_blocks = get_slot_at_time( b.timestamp );
83  FC_ASSERT( missed_blocks != 0, "Trying to push double-produced block onto current block?!" );
84  missed_blocks--;
85  const auto& witnesses = witness_schedule_id_type()(*this).current_shuffled_witnesses;
86  if( missed_blocks < witnesses.size() )
87  for( uint32_t i = 0; i < missed_blocks; ++i ) {
88  const auto& witness_missed = get_scheduled_witness( i+1 )(*this);
89  modify( witness_missed, []( witness_object& w ) {
90  w.total_missed++;
91  });
92  }
93  return missed_blocks;
94 }

102 void database::update_witness_schedule()
103 {
104  const witness_schedule_object& wso = get_witness_schedule_object();
105  const global_property_object& gpo = get_global_properties();
106 
107  if( head_block_num() % gpo.active_witnesses.size() == 0 )
108  {
109  modify( wso, [&]( witness_schedule_object& _wso )
110  {
111  _wso.current_shuffled_witnesses.clear();
112  _wso.current_shuffled_witnesses.reserve( gpo.active_witnesses.size() );
113 
114  for( const witness_id_type& w : gpo.active_witnesses )
115  _wso.current_shuffled_witnesses.push_back( w );
116 
117  auto now_hi = uint64_t(head_block_time().sec_since_epoch()) << 32;
118  for( uint32_t i = 0; i < _wso.current_shuffled_witnesses.size(); ++i )
119  {
122  uint64_t k = now_hi + uint64_t(i)*2685821657736338717ULL;
123  k ^= (k >> 12);
124  k ^= (k << 25);
125  k ^= (k >> 27);
126  k *= 2685821657736338717ULL;
127 
128  uint32_t jmax = _wso.current_shuffled_witnesses.size() - i;
129  uint32_t j = i + k%jmax;
130  std::swap( _wso.current_shuffled_witnesses[i],
131  _wso.current_shuffled_witnesses[j] );
132  }
133  });
134  }
135 }
