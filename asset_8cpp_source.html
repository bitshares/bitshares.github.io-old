104  price operator * ( const price& p, const ratio_type& r )
105  { try {
106  p.validate();
107 
108  FC_ASSERT( r.numerator() > 0 && r.denominator() > 0 );
109 
110  if( r.numerator() == r.denominator() ) return p;
111 
112  boost::rational<int128_t> p128( p.base.amount.value, p.quote.amount.value );
113  boost::rational<int128_t> r128( r.numerator(), r.denominator() );
114  auto cp = p128 * r128;
115  auto ocp = cp;
116 
117  bool shrinked = false;
118  static const int128_t max( GRAPHENE_MAX_SHARE_SUPPLY );
119  while( cp.numerator() > max || cp.denominator() > max )
120  {
121  if( cp.numerator() == 1 )
122  {
123  cp = boost::rational<int128_t>( 1, max );
124  break;
125  }
126  else if( cp.denominator() == 1 )
127  {
128  cp = boost::rational<int128_t>( max, 1 );
129  break;
130  }
131  else
132  {
133  cp = boost::rational<int128_t>( cp.numerator() >> 1, cp.denominator() >> 1 );
134  shrinked = true;
135  }
136  }
137  if( shrinked ) // maybe not accurate enough due to rounding, do additional checks here
138  {
139  int128_t num = ocp.numerator();
140  int128_t den = ocp.denominator();
141  if( num > den )
142  {
143  num /= den;
144  if( num > max )
145  num = max;
146  den = 1;
147  }
148  else
149  {
150  den /= num;
151  if( den > max )
152  den = max;
153  num = 1;
154  }
155  boost::rational<int128_t> ncp( num, den );
156  if( num == max || den == max ) // it's on the edge, we know it's accurate enough
157  cp = ncp;
158  else
159  {
160  // from the accurate ocp, now we have ncp and cp. use the one which is closer to ocp.
161  // TODO improve performance
162  auto diff1 = abs( ncp - ocp );
163  auto diff2 = abs( cp - ocp );
164  if( diff1 < diff2 ) cp = ncp;
165  }
166  }
167 
168  price np = asset( cp.numerator().convert_to<int64_t>(), p.base.asset_id )
169  / asset( cp.denominator().convert_to<int64_t>(), p.quote.asset_id );
170 
171  if( ( r.numerator() > r.denominator() && np < p )
172  || ( r.numerator() < r.denominator() && np > p ) )
173  // even with an accurate result, if p is out of valid range, return it
174  np = p;
175 
176  np.validate();
177  return np;
178  } FC_CAPTURE_AND_RETHROW( (p)(r.numerator())(r.denominator()) ) }
