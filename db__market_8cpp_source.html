207 void database::cancel_limit_order( const limit_order_object& order, bool create_virtual_op, bool skip_cancel_fee )
208 {
209  // if need to create a virtual op, try deduct a cancellation fee here.
210  // there are two scenarios when order is cancelled and need to create a virtual op:
211  // 1. due to expiration: always deduct a fee if there is any fee deferred
212  // 2. due to cull_small: deduct a fee after hard fork 604, but not before (will set skip_cancel_fee)
213  const account_statistics_object* seller_acc_stats = nullptr;
214  const asset_dynamic_data_object* fee_asset_dyn_data = nullptr;
215  limit_order_cancel_operation vop;
216  share_type deferred_fee = order.deferred_fee;
217  asset deferred_paid_fee = order.deferred_paid_fee;
218  if( create_virtual_op )
219  {
220  vop.order = order.id;
221  vop.fee_paying_account = order.seller;
222  // only deduct fee if not skipping fee, and there is any fee deferred
223  if( !skip_cancel_fee && deferred_fee > 0 )
224  {
225  asset core_cancel_fee = current_fee_schedule().calculate_fee( vop );
226  // cap the fee
227  if( core_cancel_fee.amount > deferred_fee )
228  core_cancel_fee.amount = deferred_fee;
229  // if there is any CORE fee to deduct, redirect it to referral program
230  if( core_cancel_fee.amount > 0 )
231  {
232  seller_acc_stats = &order.seller( *this ).statistics( *this );
233  modify( *seller_acc_stats, [&]( account_statistics_object& obj ) {
234  obj.pay_fee( core_cancel_fee.amount, get_global_properties().parameters.cashback_vesting_threshold );
235  } );
236  deferred_fee -= core_cancel_fee.amount;
237  // handle originally paid fee if any:
238  // to_deduct = round_up( paid_fee * core_cancel_fee / deferred_core_fee_before_deduct )
239  if( deferred_paid_fee.amount == 0 )
240  {
241  vop.fee = core_cancel_fee;
242  }
243  else
244  {
245  fc::uint128 fee128( deferred_paid_fee.amount.value );
246  fee128 *= core_cancel_fee.amount.value;
247  // to round up
248  fee128 += order.deferred_fee.value;
249  fee128 -= 1;
250  fee128 /= order.deferred_fee.value;
251  share_type cancel_fee_amount = fee128.to_uint64();
252  // cancel_fee should be positive, pay it to asset's accumulated_fees
253  fee_asset_dyn_data = &deferred_paid_fee.asset_id(*this).dynamic_asset_data_id(*this);
254  modify( *fee_asset_dyn_data, [&](asset_dynamic_data_object& addo) {
255  addo.accumulated_fees += cancel_fee_amount;
256  });
257  // cancel_fee should be no more than deferred_paid_fee
258  deferred_paid_fee.amount -= cancel_fee_amount;
259  vop.fee = asset( cancel_fee_amount, deferred_paid_fee.asset_id );
260  }
261  }
262  }
263  }
264 
265  // refund funds in order
266  auto refunded = order.amount_for_sale();
267  if( refunded.asset_id == asset_id_type() )
268  {
269  if( seller_acc_stats == nullptr )
270  seller_acc_stats = &order.seller( *this ).statistics( *this );
271  modify( *seller_acc_stats, [&]( account_statistics_object& obj ) {
272  obj.total_core_in_orders -= refunded.amount;
273  });
274  }
275  adjust_balance(order.seller, refunded);
276 
277  // refund fee
278  // could be virtual op or real op here
279  if( order.deferred_paid_fee.amount == 0 )
280  {
281  // be here, order.create_time <= HARDFORK_CORE_604_TIME, or fee paid in CORE, or no fee to refund.
282  // if order was created before hard fork 604 then cancelled no matter before or after hard fork 604,
283  // see it as fee paid in CORE, deferred_fee should be refunded to order owner but not fee pool
284  adjust_balance( order.seller, deferred_fee );
285  }
286  else // need to refund fee in originally paid asset
287  {
288  adjust_balance(order.seller, deferred_paid_fee);
289  // be here, must have: fee_asset != CORE
290  if( fee_asset_dyn_data == nullptr )
291  fee_asset_dyn_data = &deferred_paid_fee.asset_id(*this).dynamic_asset_data_id(*this);
292  modify( *fee_asset_dyn_data, [&](asset_dynamic_data_object& addo) {
293  addo.fee_pool += deferred_fee;
294  });
295  }
296 
297  if( create_virtual_op )
298  push_applied_operation( vop );
299 
300  remove(order);
301 }

329 bool database::apply_order_before_hardfork_625(const limit_order_object& new_order_object, bool allow_black_swan)
330 {
331  auto order_id = new_order_object.id;
332  const asset_object& sell_asset = get(new_order_object.amount_for_sale().asset_id);
333  const asset_object& receive_asset = get(new_order_object.amount_to_receive().asset_id);
334 
335  // Possible optimization: We only need to check calls if both are true:
336  // - The new order is at the front of the book
337  // - The new order is below the call limit price
338  bool called_some = check_call_orders(sell_asset, allow_black_swan, true); // the first time when checking, call order is maker
339  called_some |= check_call_orders(receive_asset, allow_black_swan, true); // the other side, same as above
340  if( called_some && !find_object(order_id) ) // then we were filled by call order
341  return true;
342 
343  const auto& limit_price_idx = get_index_type<limit_order_index>().indices().get<by_price>();
344 
345  // TODO: it should be possible to simply check the NEXT/PREV iterator after new_order_object to
346  // determine whether or not this order has "changed the book" in a way that requires us to
347  // check orders. For now I just lookup the lower bound and check for equality... this is log(n) vs
348  // constant time check. Potential optimization.
349 
350  auto max_price = ~new_order_object.sell_price;
351  auto limit_itr = limit_price_idx.lower_bound(max_price.max());
352  auto limit_end = limit_price_idx.upper_bound(max_price);
353 
354  bool finished = false;
355  while( !finished && limit_itr != limit_end )
356  {
357  auto old_limit_itr = limit_itr;
358  ++limit_itr;
359  // match returns 2 when only the old order was fully filled. In this case, we keep matching; otherwise, we stop.
360  finished = (match(new_order_object, *old_limit_itr, old_limit_itr->sell_price) != 2);
361  }
362 
363  //Possible optimization: only check calls if the new order completely filled some old order
364  //Do I need to check both assets?
365  check_call_orders(sell_asset, allow_black_swan); // after the new limit order filled some orders on the book,
366  // if a call order matches another order, the call order is taker
367  check_call_orders(receive_asset, allow_black_swan); // the other side, same as above
368 
369  const limit_order_object* updated_order_object = find< limit_order_object >( order_id );
370  if( updated_order_object == nullptr )
371  return true;
372  if( head_block_time() <= HARDFORK_555_TIME )
373  return false;
374  // before #555 we would have done maybe_cull_small_order() logic as a result of fill_order() being called by match() above
375  // however after #555 we need to get rid of small orders -- #555 hardfork defers logic that was done too eagerly before, and
376  // this is the point it's deferred to.
377  return maybe_cull_small_order( *this, *updated_order_object );
378 }

380 bool database::apply_order(const limit_order_object& new_order_object, bool allow_black_swan)
381 {
382  auto order_id = new_order_object.id;
383  asset_id_type sell_asset_id = new_order_object.sell_asset_id();
384  asset_id_type recv_asset_id = new_order_object.receive_asset_id();
385 
386  // We only need to check if the new order will match with others if it is at the front of the book
387  const auto& limit_price_idx = get_index_type<limit_order_index>().indices().get<by_price>();
388  auto limit_itr = limit_price_idx.lower_bound( boost::make_tuple( new_order_object.sell_price, order_id ) );
389  if( limit_itr != limit_price_idx.begin() )
390  {
391  --limit_itr;
392  if( limit_itr->sell_asset_id() == sell_asset_id && limit_itr->receive_asset_id() == recv_asset_id )
393  return false;
394  }
395 
396  // this is the opposite side (on the book)
397  auto max_price = ~new_order_object.sell_price;
398  limit_itr = limit_price_idx.lower_bound( max_price.max() );
399  auto limit_end = limit_price_idx.upper_bound( max_price );
400 
401  // Order matching should be in favor of the taker.
402  // When a new limit order is created, e.g. an ask, need to check if it will match the highest bid.
403  // We were checking call orders first. However, due to MSSR (maximum_short_squeeze_ratio),
404  // effective price of call orders may be worse than limit orders, so we should also check limit orders here.
405 
406  // Question: will a new limit order trigger a black swan event?
407  //
408  // 1. as of writing, it's possible due to the call-order-and-limit-order overlapping issue:
409  // https://github.com/bitshares/bitshares-core/issues/606 .
410  // when it happens, a call order can be very big but don't match with the opposite,
411  // even when price feed is too far away, further than swan price,
412  // if the new limit order is in the same direction with the call orders, it can eat up all the opposite,
413  // then the call order will lose support and trigger a black swan event.
414  // 2. after issue 606 is fixed, there will be no limit order on the opposite side "supporting" the call order,
415  // so a new order in the same direction with the call order won't trigger a black swan event.
416  // 3. calling is one direction. if the new limit order is on the opposite direction,
417  // no matter if matches with the call, it won't trigger a black swan event.
418  // (if a match at MSSP caused a black swan event, it means the call order is already undercollateralized,
419  // which should trigger a black swan event earlier.)
420  //
421  // Since it won't trigger a black swan, no need to check here.
422 
423  // currently we don't do cross-market (triangle) matching.
424  // the limit order will only match with a call order if meet all of these:
425  // 1. it's buying collateral, which means sell_asset is the MIA, receive_asset is the backing asset.
426  // 2. sell_asset is not a prediction market
427  // 3. sell_asset is not globally settled
428  // 4. sell_asset has a valid price feed
429  // 5. the call order's collateral ratio is below or equals to MCR
430  // 6. the limit order provided a good price
431 
432  bool to_check_call_orders = false;
433  const asset_object& sell_asset = sell_asset_id( *this );
434  const asset_bitasset_data_object* sell_abd = nullptr;
435  price call_match_price;
436  if( sell_asset.is_market_issued() )
437  {
438  sell_abd = &sell_asset.bitasset_data( *this );
439  if( sell_abd->options.short_backing_asset == recv_asset_id
440  && !sell_abd->is_prediction_market
441  && !sell_abd->has_settlement()
442  && !sell_abd->current_feed.settlement_price.is_null() )
443  {
444  call_match_price = ~sell_abd->current_feed.max_short_squeeze_price();
445  if( ~new_order_object.sell_price <= call_match_price ) // new limit order price is good enough to match a call
446  to_check_call_orders = true;
447  }
448  }
449 
450  bool finished = false; // whether the new order is gone
451  if( to_check_call_orders )
452  {
453  // check limit orders first, match the ones with better price in comparison to call orders
454  while( !finished && limit_itr != limit_end && limit_itr->sell_price > call_match_price )
455  {
456  auto old_limit_itr = limit_itr;
457  ++limit_itr;
458  // match returns 2 when only the old order was fully filled. In this case, we keep matching; otherwise, we stop.
459  finished = ( match( new_order_object, *old_limit_itr, old_limit_itr->sell_price ) != 2 );
460  }
461 
462  if( !finished )
463  {
464  // check if there are margin calls
465  const auto& call_price_idx = get_index_type<call_order_index>().indices().get<by_price>();
466  auto call_min = price::min( recv_asset_id, sell_asset_id );
467  while( !finished )
468  {
469  // assume hard fork core-343 and core-625 will take place at same time, always check call order with least call_price
470  auto call_itr = call_price_idx.lower_bound( call_min );
471  if( call_itr == call_price_idx.end()
472  || call_itr->debt_type() != sell_asset_id
473  // feed protected https://github.com/cryptonomex/graphene/issues/436
474  || call_itr->call_price > ~sell_abd->current_feed.settlement_price )
475  break;
476  // assume hard fork core-338 and core-625 will take place at same time, not checking HARDFORK_CORE_338_TIME here.
477  int match_result = match( new_order_object, *call_itr, call_match_price,
478  sell_abd->current_feed.settlement_price,
479  sell_abd->current_feed.maintenance_collateral_ratio );
480  // match returns 1 or 3 when the new order was fully filled. In this case, we stop matching; otherwise keep matching.
481  // since match can return 0 due to BSIP38 (hard fork core-834), we no longer only check if the result is 2.
482  if( match_result == 1 || match_result == 3 )
483  finished = true;
484  }
485  }
486  }
487 
488  // still need to check limit orders
489  while( !finished && limit_itr != limit_end )
490  {
491  auto old_limit_itr = limit_itr;
492  ++limit_itr;
493  // match returns 2 when only the old order was fully filled. In this case, we keep matching; otherwise, we stop.
494  finished = ( match( new_order_object, *old_limit_itr, old_limit_itr->sell_price ) != 2 );
495  }
496 
497  const limit_order_object* updated_order_object = find< limit_order_object >( order_id );
498  if( updated_order_object == nullptr )
499  return true;
500 
501  // before #555 we would have done maybe_cull_small_order() logic as a result of fill_order() being called by match() above
502  // however after #555 we need to get rid of small orders -- #555 hardfork defers logic that was done too eagerly before, and
503  // this is the point it's deferred to.
504  return maybe_cull_small_order( *this, *updated_order_object );
505 }

517 int database::match( const limit_order_object& usd, const limit_order_object& core, const price& match_price )
518 {
519  FC_ASSERT( usd.sell_price.quote.asset_id == core.sell_price.base.asset_id );
520  FC_ASSERT( usd.sell_price.base.asset_id == core.sell_price.quote.asset_id );
521  FC_ASSERT( usd.for_sale > 0 && core.for_sale > 0 );
522 
523  auto usd_for_sale = usd.amount_for_sale();
524  auto core_for_sale = core.amount_for_sale();
525 
526  asset usd_pays, usd_receives, core_pays, core_receives;
