129 bool database::_push_block(const signed_block& new_block)
130 { try {
131  uint32_t skip = get_node_properties().skip_flags;
132  if( !(skip&skip_fork_db) )
133  {
135  // verify that the block signer is in the current set of active witnesses.
136 
137  shared_ptr<fork_item> new_head = _fork_db.push_block(new_block);
138  //If the head block from the longest chain does not build off of the current head, we need to switch forks.
139  if( new_head->data.previous != head_block_id() )
140  {
141  //If the newly pushed block is the same height as head, we get head back in new_head
142  //Only switch forks if new_head is actually higher than head
143  if( new_head->data.block_num() > head_block_num() )
144  {
145  wlog( "Switching to fork: ${id}", ("id",new_head->data.id()) );
146  auto branches = _fork_db.fetch_branch_from(new_head->data.id(), head_block_id());
147 
148  // pop blocks until we hit the forked block
149  while( head_block_id() != branches.second.back()->data.previous )
150  {
151  ilog( "popping block #${n} ${id}", ("n",head_block_num())("id",head_block_id()) );
152  pop_block();
153  }
154 
155  // push all blocks on the new fork
156  for( auto ritr = branches.first.rbegin(); ritr != branches.first.rend(); ++ritr )
157  {
158  ilog( "pushing block from fork #${n} ${id}", ("n",(*ritr)->data.block_num())("id",(*ritr)->id) );
159  optional<fc::exception> except;
160  try {
161  undo_database::session session = _undo_db.start_undo_session();
162  apply_block( (*ritr)->data, skip );
163  _block_id_to_block.store( (*ritr)->id, (*ritr)->data );
164  session.commit();
165  }
166  catch ( const fc::exception& e ) { except = e; }
167  if( except )
168  {
169  wlog( "exception thrown while switching forks ${e}", ("e",except->to_detail_string() ) );
170  // remove the rest of branches.first from the fork_db, those blocks are invalid
171  while( ritr != branches.first.rend() )
172  {
173  ilog( "removing block from fork_db #${n} ${id}", ("n",(*ritr)->data.block_num())("id",(*ritr)->id) );
174  _fork_db.remove( (*ritr)->id );
175  ++ritr;
176  }
177  _fork_db.set_head( branches.second.front() );
178 
179  // pop all blocks from the bad fork
180  while( head_block_id() != branches.second.back()->data.previous )
181  {
182  ilog( "popping block #${n} ${id}", ("n",head_block_num())("id",head_block_id()) );
183  pop_block();
184  }
185 
186  ilog( "Switching back to fork: ${id}", ("id",branches.second.front()->data.id()) );
187  // restore all blocks from the good fork
188  for( auto ritr2 = branches.second.rbegin(); ritr2 != branches.second.rend(); ++ritr2 )
189  {
190  ilog( "pushing block #${n} ${id}", ("n",(*ritr2)->data.block_num())("id",(*ritr2)->id) );
191  auto session = _undo_db.start_undo_session();
192  apply_block( (*ritr2)->data, skip );
193  _block_id_to_block.store( (*ritr2)->id, (*ritr2)->data );
194  session.commit();
195  }
196  throw *except;
197  }
198  }
199  return true;
200  }
201  else return false;
202  }
203  }
204 
205  try {
206  auto session = _undo_db.start_undo_session();
207  apply_block(new_block, skip);
208  _block_id_to_block.store(new_block.id(), new_block);
209  session.commit();
210  } catch ( const fc::exception& e ) {
211  elog("Failed to push new block:\n${e}", ("e", e.to_detail_string()));
212  _fork_db.remove(new_block.id());
213  throw;
214  }
215 
216  return false;
217 } FC_CAPTURE_AND_RETHROW( (new_block) ) }

238 processed_transaction database::_push_transaction( const signed_transaction& trx )
239 {
240  // If this is the first transaction pushed after applying a block, start a new undo session.
241  // This allows us to quickly rewind to the clean state of the head block, in case a new block arrives.
242  if( !_pending_tx_session.valid() )
243  _pending_tx_session = _undo_db.start_undo_session();
244 
245  // Create a temporary undo session as a child of _pending_tx_session.
246  // The temporary session will be discarded by the destructor if
247  // _apply_transaction fails. If we make it to merge(), we
248  // apply the changes.
249 
250  auto temp_session = _undo_db.start_undo_session();
251  auto processed_trx = _apply_transaction( trx );
252  _pending_tx.push_back(processed_trx);
253 
254  // notify_changed_objects();
255  // The transaction applied successfully. Merge its changes into the pending block session.
256  temp_session.merge();
257 
258  // notify anyone listening to pending transactions
259  notify_on_pending_transaction( trx );
260  return processed_trx;
261 }

287 processed_transaction database::push_proposal(const proposal_object& proposal)
288 { try {
289  transaction_evaluation_state eval_state(this);
290  eval_state._is_proposed_trx = true;
291 
292  eval_state.operation_results.reserve(proposal.proposed_transaction.operations.size());
293  processed_transaction ptrx(proposal.proposed_transaction);
294  eval_state._trx = &ptrx;
295  size_t old_applied_ops_size = _applied_ops.size();
296 
297  try {
298  push_proposal_nesting_guard guard( _push_proposal_nesting_depth, *this );
299  if( _undo_db.size() >= _undo_db.max_size() )
300  _undo_db.set_max_size( _undo_db.size() + 1 );
301  auto session = _undo_db.start_undo_session(true);
302  for( auto& op : proposal.proposed_transaction.operations )
303  eval_state.operation_results.emplace_back(apply_operation(eval_state, op));
304  remove(proposal);
305  session.merge();
306  } catch ( const fc::exception& e ) {
307  if( head_block_time() <= HARDFORK_483_TIME )
308  {
309  for( size_t i=old_applied_ops_size,n=_applied_ops.size(); i<n; i++ )
310  {
311  ilog( "removing failed operation from applied_ops: ${op}", ("op", *(_applied_ops[i])) );
312  _applied_ops[i].reset();
313  }
314  }
315  else
316  {
317  _applied_ops.resize( old_applied_ops_size );
318  }
319  wlog( "${e}", ("e",e.to_detail_string() ) );
320  throw;
321  }
322 
323  ptrx.operation_results = std::move(eval_state.operation_results);
324  return ptrx;
325 } FC_CAPTURE_AND_RETHROW( (proposal) ) }

342 signed_block database::_generate_block(
343  fc::time_point_sec when,
344  witness_id_type witness_id,
345  const fc::ecc::private_key& block_signing_private_key
346  )
347 {
348  try {
349  uint32_t skip = get_node_properties().skip_flags;
350  uint32_t slot_num = get_slot_at_time( when );
351  FC_ASSERT( slot_num > 0 );
352  witness_id_type scheduled_witness = get_scheduled_witness( slot_num );
353  FC_ASSERT( scheduled_witness == witness_id );
354 
355  const auto& witness_obj = witness_id(*this);
356 
357  if( !(skip & skip_witness_signature) )
358  FC_ASSERT( witness_obj.signing_key == block_signing_private_key.get_public_key() );
359 
360  static const size_t max_block_header_size = fc::raw::pack_size( signed_block_header() ) + 4;
361  auto maximum_block_size = get_global_properties().parameters.maximum_block_size;
362  size_t total_block_size = max_block_header_size;
363 
364  signed_block pending_block;
365 
366  //
367  // The following code throws away existing pending_tx_session and
368  // rebuilds it by re-applying pending transactions.
369  //
370  // This rebuild is necessary because pending transactions' validity
371  // and semantics may have changed since they were received, because
372  // time-based semantics are evaluated based on the current block
373  // time. These changes can only be reflected in the database when
374  // the value of the "when" variable is known, which means we need to
375  // re-apply pending transactions in this method.
376  //
377  _pending_tx_session.reset();
378  _pending_tx_session = _undo_db.start_undo_session();
379 
380  uint64_t postponed_tx_count = 0;
381  // pop pending state (reset to head block state)
382  for( const processed_transaction& tx : _pending_tx )
383  {
384  size_t new_total_size = total_block_size + fc::raw::pack_size( tx );
385 
386  // postpone transaction if it would make block too big
387  if( new_total_size >= maximum_block_size )
388  {
389  postponed_tx_count++;
390  continue;
391  }
392 
393  try
394  {
395  auto temp_session = _undo_db.start_undo_session();
396  processed_transaction ptx = _apply_transaction( tx );
397  temp_session.merge();
398 
399  // We have to recompute pack_size(ptx) because it may be different
400  // than pack_size(tx) (i.e. if one or more results increased
401  // their size)
402  total_block_size += fc::raw::pack_size( ptx );
403  pending_block.transactions.push_back( ptx );
404  }
405  catch ( const fc::exception& e )
406  {
407  // Do nothing, transaction will not be re-applied
408  wlog( "Transaction was not processed while generating block due to ${e}", ("e", e) );
409  wlog( "The transaction was ${t}", ("t", tx) );
410  }
411  }
412  if( postponed_tx_count > 0 )
413  {
414  wlog( "Postponed ${n} transactions due to block size limit", ("n", postponed_tx_count) );
415  }
416 
417  _pending_tx_session.reset();
418 
419  // We have temporarily broken the invariant that
420  // _pending_tx_session is the result of applying _pending_tx, as
421  // _pending_tx now consists of the set of postponed transactions.
422  // However, the push_block() call below will re-create the
423  // _pending_tx_session.
424 
425  pending_block.previous = head_block_id();
426  pending_block.timestamp = when;
427  pending_block.transaction_merkle_root = pending_block.calculate_merkle_root();
428  pending_block.witness = witness_id;
429 
430  if( !(skip & skip_witness_signature) )
431  pending_block.sign( block_signing_private_key );
432 
433  // TODO: Move this to _push_block() so session is restored.
434  if( !(skip & skip_block_size_check) )
435  {
436  FC_ASSERT( fc::raw::pack_size(pending_block) <= get_global_properties().parameters.maximum_block_size );
437  }
438 
439  push_block( pending_block, skip );
440 
441  return pending_block;
442 } FC_CAPTURE_AND_RETHROW( (witness_id) ) }

517 void database::_apply_block( const signed_block& next_block )
518 { try {
519  uint32_t next_block_num = next_block.block_num();
520  uint32_t skip = get_node_properties().skip_flags;
521  _applied_ops.clear();
522 
523  FC_ASSERT( (skip & skip_merkle_check) || next_block.transaction_merkle_root == next_block.calculate_merkle_root(), "", ("next_block.transaction_merkle_root",next_block.transaction_merkle_root)("calc",next_block.calculate_merkle_root())("next_block",next_block)("id",next_block.id()) );
524 
525  const witness_object& signing_witness = validate_block_header(skip, next_block);
526  const auto& global_props = get_global_properties();
527  const auto& dynamic_global_props = get_dynamic_global_properties();
528  bool maint_needed = (dynamic_global_props.next_maintenance_time <= next_block.timestamp);
529 
530  _current_block_num = next_block_num;
531  _current_trx_in_block = 0;
532 
533  _issue_453_affected_assets.clear();
534 
535  for( const auto& trx : next_block.transactions )
536  {
537  /* We do not need to push the undo state for each transaction
538  * because they either all apply and are valid or the
539  * entire block fails to apply. We only need an "undo" state
540  * for transactions when validating broadcast transactions or
541  * when building a block.
542  */
543  apply_transaction( trx, skip );
544  ++_current_trx_in_block;
545  }
546 
547  const uint32_t missed = update_witness_missed_blocks( next_block );
548  update_global_dynamic_data( next_block, missed );
549  update_signing_witness(signing_witness, next_block);
550  update_last_irreversible_block();
551 
552  // Are we at the maintenance interval?
553  if( maint_needed )
554  perform_chain_maintenance(next_block, global_props);
555 
556  create_block_summary(next_block);
557  clear_expired_transactions();
558  clear_expired_proposals();
559  clear_expired_orders();
560  update_expired_feeds(); // this will update expired feeds and some core exchange rates
561  update_core_exchange_rates(); // this will update remaining core exchange rates
562  update_withdraw_permissions();
563 
564  // n.b., update_maintenance_flag() happens this late
565  // because get_slot_time() / get_slot_at_time() is needed above
566  // TODO: figure out if we could collapse this function into
567  // update_global_dynamic_data() as perhaps these methods only need
568  // to be called for header validation?
569  update_maintenance_flag( maint_needed );
570  update_witness_schedule();
571  if( !_node_property_object.debug_updates.empty() )
572  apply_debug_updates();
573 
574  // notify observers that the block has been applied
575  notify_applied_block( next_block ); //emit
576  _applied_ops.clear();
577 
578  notify_changed_objects();
579 } FC_CAPTURE_AND_RETHROW( (next_block.block_num()) ) }

593 processed_transaction database::_apply_transaction(const signed_transaction& trx)
594 { try {
595  uint32_t skip = get_node_properties().skip_flags;
596 
597  if( true || !(skip&skip_validate) ) /* issue #505 explains why this skip_flag is disabled */
598  trx.validate();
599 
600  auto& trx_idx = get_mutable_index_type<transaction_index>();
601  const chain_id_type& chain_id = get_chain_id();
602  auto trx_id = trx.id();
603  FC_ASSERT( (skip & skip_transaction_dupe_check) ||
604  trx_idx.indices().get<by_trx_id>().find(trx_id) == trx_idx.indices().get<by_trx_id>().end() );
605  transaction_evaluation_state eval_state(this);
606  const chain_parameters& chain_parameters = get_global_properties().parameters;
607  eval_state._trx = &trx;
608 
609  if( !(skip & (skip_transaction_signatures | skip_authority_check) ) )
610  {
611  auto get_active = [&]( account_id_type id ) { return &id(*this).active; };
612  auto get_owner = [&]( account_id_type id ) { return &id(*this).owner; };
613  trx.verify_authority( chain_id, get_active, get_owner, get_global_properties().parameters.max_authority_depth );
614  }
615 
616  //Skip all manner of expiration and TaPoS checking if we're on block 1; It's impossible that the transaction is
617  //expired, and TaPoS makes no sense as no blocks exist.
618  if( BOOST_LIKELY(head_block_num() > 0) )
619  {
620  if( !(skip & skip_tapos_check) )
621  {
622  const auto& tapos_block_summary = block_summary_id_type( trx.ref_block_num )(*this);
623 
624  //Verify TaPoS block summary has correct ID prefix, and that this block's time is not past the expiration
625  FC_ASSERT( trx.ref_block_prefix == tapos_block_summary.block_id._hash[1] );
626  }
627 
628  fc::time_point_sec now = head_block_time();
629 
630  FC_ASSERT( trx.expiration <= now + chain_parameters.maximum_time_until_expiration, "",
631  ("trx.expiration",trx.expiration)("now",now)("max_til_exp",chain_parameters.maximum_time_until_expiration));
632  FC_ASSERT( now <= trx.expiration, "", ("now",now)("trx.exp",trx.expiration) );
633  }
634 
635  //Insert transaction into unique transactions database.
636  if( !(skip & skip_transaction_dupe_check) )
637  {
638  create<transaction_object>([&](transaction_object& transaction) {
639  transaction.trx_id = trx_id;
640  transaction.trx = trx;
641  });
642  }
643 
644  eval_state.operation_results.reserve(trx.operations.size());
645 
646  //Finally process the operations
647  processed_transaction ptrx(trx);
648  _current_op_in_trx = 0;
649  for( const auto& op : ptrx.operations )
650  {
651  eval_state.operation_results.emplace_back(apply_operation(eval_state, op));
652  ++_current_op_in_trx;
653  }
654  ptrx.operation_results = std::move(eval_state.operation_results);
655 
656  if( head_block_time() < HARDFORK_CORE_1040_TIME ) // TODO totally remove this code block after hard fork
657  {
658  //Make sure the temp account has no non-zero balances
659  const auto& index = get_index_type<account_balance_index>().indices().get<by_account_asset>();
660  auto range = index.equal_range( boost::make_tuple( GRAPHENE_TEMP_ACCOUNT ) );
661  std::for_each(range.first, range.second, [](const account_balance_object& b) { FC_ASSERT(b.balance == 0); });
662  }
663 
664  return ptrx;
665 } FC_CAPTURE_AND_RETHROW( (trx) ) }

668 { try {
669  int i_which = op.which();
670  uint64_t u_which = uint64_t( i_which );
671  FC_ASSERT( i_which >= 0, "Negative operation tag in operation ${op}", ("op",op) );
672  FC_ASSERT( u_which < _operation_evaluators.size(), "No registered evaluator for operation ${op}", ("op",op) );
673  unique_ptr<op_evaluator>& eval = _operation_evaluators[ u_which ];
674  FC_ASSERT( eval, "No registered evaluator for operation ${op}", ("op",op) );
675  auto op_id = push_applied_operation( op );
676  auto result = eval->evaluate( eval_state, op, true );
677  set_applied_operation_result( op_id, result );
678  return result;
