325 fc::variants database_api_impl::get_objects(const vector<object_id_type>& ids)const
326 {
327  if( _subscribe_callback ) {
328  for( auto id : ids )
329  {
330  if( id.type() == operation_history_object_type && id.space() == protocol_ids ) continue;
331  if( id.type() == impl_account_transaction_history_object_type && id.space() == implementation_ids ) continue;
332 
333  this->subscribe_to_item( id );
334  }
335  }
336 
337  fc::variants result;
338  result.reserve(ids.size());
339 
340  std::transform(ids.begin(), ids.end(), std::back_inserter(result),
341  [this](object_id_type id) -> fc::variant {
342  if(auto obj = _db.find_object(id))
343  return obj->to_variant();
344  return {};
345  });
346 
347  return result;
348 }

682  return results;

686  FC_ASSERT( assets[1], "Invalid quote asset symbol: ${s}", ("s",quote) );
687 
688  auto base_id = assets[0]->id;
689  auto quote_id = assets[1]->id;
690 
691  if (ostart_price.valid()) {
692  FC_ASSERT(ostart_price->base.asset_id == base_id, "Base asset inconsistent with start price");
693  FC_ASSERT(ostart_price->quote.asset_id == quote_id, "Quote asset inconsistent with start price");
694  }

697  limit_order_multi_index_type::index<by_account>::type::const_iterator lower_itr;
698  limit_order_multi_index_type::index<by_account>::type::const_iterator upper_itr;
699 
700  // if both order_id and price are invalid, query the first page
701  if ( !ostart_id.valid() && !ostart_price.valid() )
702  {
703  lower_itr = index_by_account.lower_bound(std::make_tuple(account->id, price::max(base_id, quote_id)));
704  }
705  else if ( ostart_id.valid() )
706  {
707  // in case of the order been deleted during page querying
708  const limit_order_object *p_loo = _db.find(*ostart_id);
709 
710  if ( !p_loo )
711  {
712  if ( ostart_price.valid() )
713  {
714  lower_itr = index_by_account.lower_bound(std::make_tuple(account->id, *ostart_price, *ostart_id));
715  }
716  else
717  {
718  // start order id been deleted, yet not provided price either
719  FC_THROW("Order id invalid (maybe just been canceled?), and start price not provided");
720  }
721  }
722  else
723  {
724  const limit_order_object &loo = *p_loo;
725 
726  // in case of the order not belongs to specified account or market
727  FC_ASSERT(loo.sell_price.base.asset_id == base_id, "Order base asset inconsistent");
728  FC_ASSERT(loo.sell_price.quote.asset_id == quote_id, "Order quote asset inconsistent with order");
729  FC_ASSERT(loo.seller == account->get_id(), "Order not owned by specified account");
730 
731  lower_itr = index_by_account.lower_bound(std::make_tuple(account->id, loo.sell_price, *ostart_id));
732  }
733  }
734  else
735  {
736  // if reach here start_price must be valid
737  lower_itr = index_by_account.lower_bound(std::make_tuple(account->id, *ostart_price));
738  }
739 
740  upper_itr = index_by_account.upper_bound(std::make_tuple(account->id, price::min(base_id, quote_id)));
741 
742  // Add the account's orders
743  for ( ; lower_itr != upper_itr && count < limit; ++lower_itr, ++count)
744  {
745  const limit_order_object &order = *lower_itr;
746  results.emplace_back(order);
747  }
748 
749  return results;
