36 bool is_valid_symbol( const string& symbol )
37 {
38  static const std::locale& loc = std::locale::classic();
39  if( symbol.size() < GRAPHENE_MIN_ASSET_SYMBOL_LENGTH )
40  return false;
41 
42  if( symbol.substr(0,3) == "BIT" )
43  return false;
44 
45  if( symbol.size() > GRAPHENE_MAX_ASSET_SYMBOL_LENGTH )
46  return false;
47 
48  if( !isalpha( symbol.front(), loc ) )
49  return false;
50 
51  if( !isalnum( symbol.back(), loc ) )
52  return false;
53 
54  bool dot_already_present = false;
55  for( const auto c : symbol )
56  {
57  if( (isalpha( c, loc ) && isupper( c, loc )) || isdigit( c, loc ) )
58  continue;
59 
60  if( c == '.' )
61  {
62  if( dot_already_present )
63  return false;
64 
65  dot_already_present = true;
66  continue;
67  }
68 
69  return false;
70  }
71 
72  return true;
73 }

81 {

213 void asset_options::validate()const
214 {
215  FC_ASSERT( max_supply > 0 );
216  FC_ASSERT( max_supply <= GRAPHENE_MAX_SHARE_SUPPLY );
217  FC_ASSERT( market_fee_percent <= GRAPHENE_100_PERCENT );
218  FC_ASSERT( max_market_fee >= 0 && max_market_fee <= GRAPHENE_MAX_SHARE_SUPPLY );
219  // There must be no high bits in permissions whose meaning is not known.
220  FC_ASSERT( !(issuer_permissions & ~ASSET_ISSUER_PERMISSION_MASK) );
221  // The global_settle flag may never be set (this is a permission only)
222  FC_ASSERT( !(flags & global_settle) );
223  // the witness_fed and committee_fed flags cannot be set simultaneously
224  FC_ASSERT( (flags & (witness_fed_asset | committee_fed_asset)) != (witness_fed_asset | committee_fed_asset) );
225  core_exchange_rate.validate();
226  FC_ASSERT( core_exchange_rate.base.asset_id.instance.value == 0 ||
227  core_exchange_rate.quote.asset_id.instance.value == 0 );
228 
229  if(!whitelist_authorities.empty() || !blacklist_authorities.empty())
230  FC_ASSERT( flags & white_list );
231  for( auto item : whitelist_markets )
232  {
233  FC_ASSERT( blacklist_markets.find(item) == blacklist_markets.end() );
234  }
235  for( auto item : blacklist_markets )
236  {
237  FC_ASSERT( whitelist_markets.find(item) == whitelist_markets.end() );
238  }
239 }
