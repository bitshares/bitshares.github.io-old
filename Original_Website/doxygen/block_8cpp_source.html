62  checksum_type signed_block::calculate_merkle_root()const
63  {
64  if( transactions.size() == 0 )
65  return checksum_type();
66 
67  vector<digest_type> ids;
68  ids.resize( transactions.size() );
69  for( uint32_t i = 0; i < transactions.size(); ++i )
70  ids[i] = transactions[i].merkle_digest();
71 
72  vector<digest_type>::size_type current_number_of_hashes = ids.size();
73  while( current_number_of_hashes > 1 )
74  {
75  // hash ID's in pairs
76  uint32_t i_max = current_number_of_hashes - (current_number_of_hashes&1);
77  uint32_t k = 0;
78 
79  for( uint32_t i = 0; i < i_max; i += 2 )
80  ids[k++] = digest_type::hash( std::make_pair( ids[i], ids[i+1] ) );
81 
82  if( current_number_of_hashes&1 )
83  ids[k++] = ids[i_max];
84  current_number_of_hashes = k;
85  }
86  return checksum_type::hash( ids[0] );
87  }
