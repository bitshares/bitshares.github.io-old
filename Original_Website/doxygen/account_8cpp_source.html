121 bool is_cheap_name( const string& n )
122 {
123  bool v = false;
124  for( auto c : n )
125  {
126  if( c >= '0' && c <= '9' ) return true;
127  if( c == '.' || c == '-' || c == '/' ) return true;
128  switch( c )
129  {
130  case 'a':
131  case 'e':
132  case 'i':
133  case 'o':
134  case 'u':
135  case 'y':
136  v = true;
137  }
138  }
139  if( !v )
140  return true;
141  return false;
