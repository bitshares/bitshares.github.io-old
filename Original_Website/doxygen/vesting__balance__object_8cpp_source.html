26 inline bool sum_below_max_shares(const asset& a, const asset& b)
27 {
28  assert(GRAPHENE_MAX_SHARE_SUPPLY + GRAPHENE_MAX_SHARE_SUPPLY > GRAPHENE_MAX_SHARE_SUPPLY);

34 asset linear_vesting_policy::get_allowed_withdraw( const vesting_policy_context& ctx )const
35 {
36  share_type allowed_withdraw = 0;
37 
38  if( ctx.now > begin_timestamp )
39  {
40  const auto elapsed_seconds = (ctx.now - begin_timestamp).to_seconds();
41  assert( elapsed_seconds > 0 );
42 
43  if( elapsed_seconds >= vesting_cliff_seconds )
44  {
45  share_type total_vested = 0;
46  if( elapsed_seconds < vesting_duration_seconds )
47  {
48  total_vested = (fc::uint128_t( begin_balance.value ) * elapsed_seconds / vesting_duration_seconds).to_uint64();
49  }
50  else
51  {
52  total_vested = begin_balance;
53  }
54  assert( total_vested >= 0 );
55 
56  const share_type withdrawn_already = begin_balance - ctx.balance.amount;
57  assert( withdrawn_already >= 0 );
58 
59  allowed_withdraw = total_vested - withdrawn_already;
60  assert( allowed_withdraw >= 0 );
61  }
62  }
63 
64  return asset( allowed_withdraw, ctx.amount.asset_id );
65 }

87 fc::uint128_t cdd_vesting_policy::compute_coin_seconds_earned(const vesting_policy_context& ctx)const
88 {
89  assert(ctx.now >= coin_seconds_earned_last_update);
90  int64_t delta_seconds = (ctx.now - coin_seconds_earned_last_update).to_seconds();
91  assert(delta_seconds >= 0);
92 
93  fc::uint128_t delta_coin_seconds = ctx.balance.amount.value;
94  delta_coin_seconds *= delta_seconds;
95 
96  fc::uint128_t coin_seconds_earned_cap = ctx.balance.amount.value;
97  coin_seconds_earned_cap *= std::max(vesting_seconds, 1u);
98 
99  return std::min(coin_seconds_earned + delta_coin_seconds, coin_seconds_earned_cap);
100 }

108 asset cdd_vesting_policy::get_allowed_withdraw(const vesting_policy_context& ctx)const
109 {
110  if(ctx.now <= start_claim)
111  return asset(0, ctx.balance.asset_id);
112  fc::uint128_t cs_earned = compute_coin_seconds_earned(ctx);
113  fc::uint128_t withdraw_available = cs_earned / std::max(vesting_seconds, 1u);

129 void cdd_vesting_policy::on_withdraw(const vesting_policy_context& ctx)
130 {
