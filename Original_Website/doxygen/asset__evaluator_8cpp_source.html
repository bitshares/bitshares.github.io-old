36  database& d = db();

38  const auto& chain_parameters = d.get_global_properties().parameters;
39  FC_ASSERT( op.common_options.whitelist_authorities.size() <= chain_parameters.maximum_asset_whitelist_authorities );
40  FC_ASSERT( op.common_options.blacklist_authorities.size() <= chain_parameters.maximum_asset_whitelist_authorities );
41 
42  // Check that all authorities do exist

45  for( auto id : op.common_options.blacklist_authorities )

50  FC_ASSERT( asset_symbol_itr == asset_indx.end() );
51 
52  if( d.head_block_time() <= HARDFORK_409_TIME )
53  {
54  auto dotpos = op.symbol.find( '.' );
55  if( dotpos != std::string::npos )
56  {
57  auto prefix = op.symbol.substr( 0, dotpos );
58  auto asset_symbol_itr = asset_indx.find( op.symbol );
59  FC_ASSERT( asset_symbol_itr != asset_indx.end(), "Asset ${s} may only be created by issuer of ${p}, but ${p} has not been registered",
60  ("s",op.symbol)("p",prefix) );
61  FC_ASSERT( asset_symbol_itr->issuer == op.issuer, "Asset ${s} may only be created by issuer of ${p}, ${i}",
62  ("s",op.symbol)("p",prefix)("i", op.issuer(d).name) );
63  }
64  }
65  else
66  {
67  auto dotpos = op.symbol.rfind( '.' );
68  if( dotpos != std::string::npos )
69  {
70  auto prefix = op.symbol.substr( 0, dotpos );
71  auto asset_symbol_itr = asset_indx.find( prefix );
72  FC_ASSERT( asset_symbol_itr != asset_indx.end(), "Asset ${s} may only be created by issuer of ${p}, but ${p} has not been registered",
73  ("s",op.symbol)("p",prefix) );
74  FC_ASSERT( asset_symbol_itr->issuer == op.issuer, "Asset ${s} may only be created by issuer of ${p}, ${i}",
75  ("s",op.symbol)("p",prefix)("i", op.issuer(d).name) );
76  }
77  }
78 
79  core_fee_paid -= core_fee_paid.value/2;
80 
81  if( op.bitasset_opts )
82  {
83  const asset_object& backing = op.bitasset_opts->short_backing_asset(d);
84  if( backing.is_market_issued() )
85  {
86  const asset_bitasset_data_object& backing_bitasset_data = backing.bitasset_data(d);
87  const asset_object& backing_backing = backing_bitasset_data.options.short_backing_asset(d);
88  FC_ASSERT( !backing_backing.is_market_issued(),
89  "May not create a bitasset backed by a bitasset backed by a bitasset." );

91  "May not create a blockchain-controlled market asset which is not backed by CORE.");
92  } else
93  FC_ASSERT( op.issuer != GRAPHENE_COMMITTEE_ACCOUNT || backing.get_id() == asset_id_type(),
94  "May not create a blockchain-controlled market asset which is not backed by CORE.");
95  FC_ASSERT( op.bitasset_opts->feed_lifetime_sec > chain_parameters.block_interval &&

101  FC_ASSERT( op.precision == op.bitasset_opts->short_backing_asset(d).precision );
102  }

286  // If we are now disabling force settlements, cancel all open force settlement orders
