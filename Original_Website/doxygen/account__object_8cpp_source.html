29 share_type cut_fee(share_type a, uint16_t p)
30 {
31  if( a == 0 || p == 0 )
32  return 0;
33  if( p == GRAPHENE_100_PERCENT )
34  return a;
35 
36  fc::uint128 r(a.value);
37  r *= p;
38  r /= GRAPHENE_100_PERCENT;
39  return r.to_uint64();
40 }

42 bool account_object::is_authorized_asset(const asset_object& asset_obj, const database& d) const
43 {
44  if( d.head_block_time() > HARDFORK_416_TIME )
45  {
46  if( !(asset_obj.options.flags & white_list) )
47  return true;
48  }
49 
50  for( const auto id : blacklisting_accounts )
51  {
52  if( asset_obj.options.blacklist_authorities.find(id) != asset_obj.options.blacklist_authorities.end() )
53  return false;
54  }
55 
56  if( d.head_block_time() > HARDFORK_415_TIME )
57  {
58  if( asset_obj.options.whitelist_authorities.size() == 0 )
59  return true;
60  }
61 
62  for( const auto id : whitelisting_accounts )
63  {
64  if( asset_obj.options.whitelist_authorities.find(id) != asset_obj.options.whitelist_authorities.end() )
65  return true;
66  }
67 
68  return false;
69 }

77 void account_statistics_object::process_fees(const account_object& a, database& d) const
78 {
79  if( pending_fees > 0 || pending_vested_fees > 0 )
80  {
81  auto pay_out_fees = [&](const account_object& account, share_type core_fee_total, bool require_vesting)
82  {
83  // Check the referrer -- if he's no longer a member, pay to the lifetime referrer instead.
84  // No need to check the registrar; registrars are required to be lifetime members.
85  if( account.referrer(d).is_basic_account(d.head_block_time()) )
86  d.modify(account, [](account_object& a) {
87  a.referrer = a.lifetime_referrer;
88  });
89 
90  share_type network_cut = cut_fee(core_fee_total, account.network_fee_percentage);
91  assert( network_cut <= core_fee_total );
92 
93 #ifndef NDEBUG
94  const auto& props = d.get_global_properties();
95 
96  share_type reserveed = cut_fee(network_cut, props.parameters.reserve_percent_of_fee);
97  share_type accumulated = network_cut - reserveed;
98  assert( accumulated + reserveed == network_cut );
99 #endif
100  share_type lifetime_cut = cut_fee(core_fee_total, account.lifetime_referrer_fee_percentage);
101  share_type referral = core_fee_total - network_cut - lifetime_cut;
102 
103  d.modify(asset_dynamic_data_id_type()(d), [network_cut](asset_dynamic_data_object& d) {
104  d.accumulated_fees += network_cut;
105  });
106 
107  // Potential optimization: Skip some of this math and object lookups by special casing on the account type.
108  // For example, if the account is a lifetime member, we can skip all this and just deposit the referral to
109  // it directly.
110  share_type referrer_cut = cut_fee(referral, account.referrer_rewards_percentage);
111  share_type registrar_cut = referral - referrer_cut;
112 
113  d.deposit_cashback(d.get(account.lifetime_referrer), lifetime_cut, require_vesting);
114  d.deposit_cashback(d.get(account.referrer), referrer_cut, require_vesting);
115  d.deposit_cashback(d.get(account.registrar), registrar_cut, require_vesting);
116 
117  assert( referrer_cut + registrar_cut + accumulated + reserveed + lifetime_cut == core_fee_total );
118  };
119 
120  pay_out_fees(a, pending_fees, true);
121  pay_out_fees(a, pending_vested_fees, false);
122 
123  d.modify(*this, [&](account_statistics_object& s) {
124  s.lifetime_fees_paid += pending_fees + pending_vested_fees;
125  s.pending_fees = 0;
126  s.pending_vested_fees = 0;
127  });
128  }
129 }

139 void account_object::options_type::validate() const
140 {
141  auto needed_witnesses = num_witness;
142  auto needed_committee = num_committee;
143 
144  for( vote_id_type id : votes )
145  if( id.type() == vote_id_type::witness && needed_witnesses )
146  --needed_witnesses;
147  else if ( id.type() == vote_id_type::committee && needed_committee )
148  --needed_committee;
149 
150  FC_ASSERT( needed_witnesses == 0 && needed_committee == 0,
151  "May not specify fewer witnesses or committee members than the number voted for.");
152 }
