119  asset fee_schedule::calculate_fee( const operation& op, const price& core_exchange_rate )const
120  {
121  //idump( (op)(core_exchange_rate) );
122  fee_parameters params; params.set_which(op.which());
123  auto itr = parameters.find(params);
124  if( itr != parameters.end() ) params = *itr;
125  auto base_value = op.visit( calc_fee_visitor( params ) );
126  auto scaled = fc::uint128(base_value) * scale;
127  scaled /= GRAPHENE_100_PERCENT;
128  FC_ASSERT( scaled <= GRAPHENE_MAX_SHARE_SUPPLY );
129  //idump( (base_value)(scaled)(core_exchange_rate) );
130  auto result = asset( scaled.to_uint64(), 0 ) * core_exchange_rate;
131  //FC_ASSERT( result * core_exchange_rate >= asset( scaled.to_uint64()) );
132 
133  while( result * core_exchange_rate < asset( scaled.to_uint64()) )
134  result.amount++;
135 
136  FC_ASSERT( result.amount <= GRAPHENE_MAX_SHARE_SUPPLY );
137  return result;
138  }

140  asset fee_schedule::set_fee( operation& op, const price& core_exchange_rate )const
141  {
142  auto f = calculate_fee( op, core_exchange_rate );
143  auto f_max = f;
144  for( int i=0; i<MAX_FEE_STABILIZATION_ITERATION; i++ )
145  {
146  op.visit( set_fee_visitor( f_max ) );
147  auto f2 = calculate_fee( op, core_exchange_rate );
148  if( f == f2 )
149  break;
150  f_max = std::max( f_max, f2 );
151  f = f2;
152  if( i == 0 )
153  {
154  // no need for warnings on later iterations
155  wlog( "set_fee requires multiple iterations to stabilize with core_exchange_rate ${p} on operation ${op}",
156  ("p", core_exchange_rate) ("op", op) );
157  }
158  }
159  return f_max;
160  }

182  "Maximum transaction expiration time must be greater than a block interval" );
