51 void_result assert_evaluator::do_evaluate( const assert_operation& o )
52 { try {
53  const database& _db = db();
54  uint32_t skip = _db.get_node_properties().skip_flags;
55  auto max_predicate_opcode = _db.get_global_properties().parameters.max_predicate_opcode;
56 
57  if( skip & database::skip_assert_evaluation )
58  return void_result();
59 
60  for( const auto& p : o.predicates )
61  {
62  FC_ASSERT( p.which() >= 0 );
63  FC_ASSERT( unsigned(p.which()) < max_predicate_opcode );
