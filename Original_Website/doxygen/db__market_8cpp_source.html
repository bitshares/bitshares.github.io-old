132 bool database::apply_order(const limit_order_object& new_order_object, bool allow_black_swan)
133 {
134  auto order_id = new_order_object.id;
135  const asset_object& sell_asset = get(new_order_object.amount_for_sale().asset_id);
136  const asset_object& receive_asset = get(new_order_object.amount_to_receive().asset_id);
137 
138  // Possible optimization: We only need to check calls if both are true:
139  // - The new order is at the front of the book
140  // - The new order is below the call limit price
141  bool called_some = check_call_orders(sell_asset, allow_black_swan);
142  called_some |= check_call_orders(receive_asset, allow_black_swan);
143  if( called_some && !find_object(order_id) ) // then we were filled by call order
144  return true;
145 
146  const auto& limit_price_idx = get_index_type<limit_order_index>().indices().get<by_price>();
147 
148  // TODO: it should be possible to simply check the NEXT/PREV iterator after new_order_object to
149  // determine whether or not this order has "changed the book" in a way that requires us to
150  // check orders. For now I just lookup the lower bound and check for equality... this is log(n) vs
151  // constant time check. Potential optimization.
152 
153  auto max_price = ~new_order_object.sell_price;
154  auto limit_itr = limit_price_idx.lower_bound(max_price.max());
155  auto limit_end = limit_price_idx.upper_bound(max_price);
156 
157  bool finished = false;
158  while( !finished && limit_itr != limit_end )
159  {
160  auto old_limit_itr = limit_itr;
161  ++limit_itr;
162  // match returns 2 when only the old order was fully filled. In this case, we keep matching; otherwise, we stop.
163  finished = (match(new_order_object, *old_limit_itr, old_limit_itr->sell_price) != 2);
164  }
165 
166  //Possible optimization: only check calls if the new order completely filled some old order
167  //Do I need to check both assets?
168  check_call_orders(sell_asset, allow_black_swan);
169  check_call_orders(receive_asset, allow_black_swan);
170 
171  return find_object(order_id) == nullptr;
172 }

407 
408  if( check_for_blackswan( mia, enable_black_swan ) )
409  return false;
410 
411  const asset_bitasset_data_object& bitasset = mia.bitasset_data(*this);
412  if( bitasset.is_prediction_market ) return false;
413  if( bitasset.current_feed.settlement_price.is_null() ) return false;
414 
415  const call_order_index& call_index = get_index_type<call_order_index>();
416  const auto& call_price_index = call_index.indices().get<by_price>();
417 
418  const limit_order_index& limit_index = get_index_type<limit_order_index>();
419  const auto& limit_price_index = limit_index.indices().get<by_price>();
420 
421  // looking for limit orders selling the most USD for the least CORE
422  auto max_price = price::max( mia.id, bitasset.options.short_backing_asset );
423  // stop when limit orders are selling too little USD for too much CORE
424  auto min_price = bitasset.current_feed.max_short_squeeze_price();
425 
426  assert( max_price.base.asset_id == min_price.base.asset_id );
427  // NOTE limit_price_index is sorted from greatest to least
428  auto limit_itr = limit_price_index.lower_bound( max_price );
429  auto limit_end = limit_price_index.upper_bound( min_price );
430 
431  if( limit_itr == limit_end )
432  return false;
433 
434  auto call_min = price::min( bitasset.options.short_backing_asset, mia.id );
435  auto call_max = price::max( bitasset.options.short_backing_asset, mia.id );
436  auto call_itr = call_price_index.lower_bound( call_min );
437  auto call_end = call_price_index.upper_bound( call_max );
438 
439  bool filled_limit = false;
440  bool margin_called = false;
441 
442  while( !check_for_blackswan( mia, enable_black_swan ) && call_itr != call_end )
443  {
444  bool filled_call = false;
445  price match_price;
446  asset usd_for_sale;
447  if( limit_itr != limit_end )
448  {
449  assert( limit_itr != limit_price_index.end() );
450  match_price = limit_itr->sell_price;
451  usd_for_sale = limit_itr->amount_for_sale();
452  }
453  else return margin_called;
454

456 
457  // would be margin called, but there is no matching order #436
458  bool feed_protected = ( bitasset.current_feed.settlement_price > ~call_itr->call_price );
459  if( feed_protected && (head_block_time() > HARDFORK_436_TIME) )
460  return margin_called;
461 
462  // would be margin called, but there is no matching order
463  if( match_price > ~call_itr->call_price )
464  return margin_called;
465 
466  if( feed_protected )
467  {
468  ilog( "Feed protected margin call executing (HARDFORK_436_TIME not here yet)" );
469  idump( (*call_itr) );
470  idump( (*limit_itr) );
471  }
472 
473  // idump((*call_itr));
474  // idump((*limit_itr));
475 
476  // ilog( "match_price <= ~call_itr->call_price performing a margin call" );
477 
478  margin_called = true;
479 
480  auto usd_to_buy = call_itr->get_debt();
481 
482  if( usd_to_buy * match_price > call_itr->get_collateral() )
483  {
484  elog( "black swan detected" );

486  FC_ASSERT( enable_black_swan );

492  if( usd_to_buy >= usd_for_sale )

494  call_receives = usd_for_sale;
