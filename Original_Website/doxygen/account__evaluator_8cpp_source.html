50 void_result account_create_evaluator::do_evaluate( const account_create_operation& op )
51 { try {
52  database& d = db();
53  FC_ASSERT( d.find_object(op.options.voting_account), "Invalid proxy account specified." );
54  FC_ASSERT( fee_paying_account->is_lifetime_member(), "Only Lifetime members may register an account." );
55  FC_ASSERT( op.referrer(d).is_member(d.head_block_time()), "The referrer must be either a lifetime or annual subscriber." );
56 
57  const auto& global_props = d.get_global_properties();
58  const auto& chain_params = global_props.parameters;
59 
60  try
61  {
62  verify_authority_accounts( d, op.owner );
63  verify_authority_accounts( d, op.active );
64  }
65  GRAPHENE_RECODE_EXC( internal_verify_auth_max_auth_exceeded, account_create_max_auth_exceeded )
66  GRAPHENE_RECODE_EXC( internal_verify_auth_account_not_found, account_create_auth_account_not_found )
67 
68  uint32_t max_vote_id = global_props.next_available_vote_id;
69 
70  FC_ASSERT( op.options.num_witness <= chain_params.maximum_witness_count,
71  "Voted for more witnesses than currently allowed (${c})", ("c", chain_params.maximum_witness_count) );
72 
73  FC_ASSERT( op.options.num_committee <= chain_params.maximum_committee_count,
74  "Voted for more committee members than currently allowed (${c})", ("c", chain_params.maximum_committee_count) );
75 
76  safe<uint32_t> counts[vote_id_type::VOTE_TYPE_COUNT];
77  for( auto id : op.options.votes )
78  {
79  FC_ASSERT( id < max_vote_id );
80  counts[id.type()]++;
81  }
82  FC_ASSERT(counts[vote_id_type::witness] <= op.options.num_witness,
83  "",
84  ("count", counts[vote_id_type::witness])("num", op.options.num_witness));
85  FC_ASSERT(counts[vote_id_type::committee] <= op.options.num_committee,
86  "",
87  ("count", counts[vote_id_type::committee])("num", op.options.num_committee));
88 
89  auto& acnt_indx = d.get_index_type<account_index>();
90  if( op.name.size() )
91  {
92  auto current_account_itr = acnt_indx.indices().get<by_name>().find( op.name );
93  FC_ASSERT( current_account_itr == acnt_indx.indices().get<by_name>().end() );
94  }
95 
96  return void_result();
97 } FC_CAPTURE_AND_RETHROW( (op) ) }

99 object_id_type account_create_evaluator::do_apply( const account_create_operation& o )
100 { try {
101 
102  uint16_t referrer_percent = o.referrer_percent;
103  bool has_small_percent = (
104  (db().head_block_time() <= HARDFORK_453_TIME)
105  && (o.referrer != o.registrar )
106  && (o.referrer_percent != 0 )
107  && (o.referrer_percent <= 0x100)
108  );
109 
110  if( has_small_percent )
111  {
112  if( referrer_percent >= 100 )
113  {
114  wlog( "between 100% and 0x100%: ${o}", ("o", o) );
115  }
116  referrer_percent = referrer_percent*100;
117  if( referrer_percent > GRAPHENE_100_PERCENT )
118  referrer_percent = GRAPHENE_100_PERCENT;
119  }
120 
121  const auto& new_acnt_object = db().create<account_object>( [&]( account_object& obj ){
122  obj.registrar = o.registrar;
123  obj.referrer = o.referrer;
124  obj.lifetime_referrer = o.referrer(db()).lifetime_referrer;
125 
126  auto& params = db().get_global_properties().parameters;
127  obj.network_fee_percentage = params.network_percent_of_fee;
128  obj.lifetime_referrer_fee_percentage = params.lifetime_referrer_percent_of_fee;
129  obj.referrer_rewards_percentage = referrer_percent;
130 
131  obj.name = o.name;
132  obj.owner = o.owner;
133  obj.active = o.active;
134  obj.options = o.options;
135  obj.statistics = db().create<account_statistics_object>([&](account_statistics_object& s){s.owner = obj.id;}).id;
136  });
137 
138  if( has_small_percent )
139  {
140  wlog( "Account affected by #453 registered in block ${n}: ${na} reg=${reg} ref=${ref}:${refp} ltr=${ltr}:${ltrp}",
141  ("n", db().head_block_num()) ("na", new_acnt_object.id)
142  ("reg", o.registrar) ("ref", o.referrer) ("ltr", new_acnt_object.lifetime_referrer)
143  ("refp", new_acnt_object.referrer_rewards_percentage) ("ltrp", new_acnt_object.lifetime_referrer_fee_percentage) );
144  wlog( "Affected account object is ${o}", ("o", new_acnt_object) );
145  }
146 
147  const auto& dynamic_properties = db().get_dynamic_global_properties();
148  db().modify(dynamic_properties, [](dynamic_global_property_object& p) {
149  ++p.accounts_registered_this_interval;
150  });
151 
152  const auto& global_properties = db().get_global_properties();
153  if( dynamic_properties.accounts_registered_this_interval %
154  global_properties.parameters.accounts_per_fee_scale == 0 )
155  db().modify(global_properties, [&dynamic_properties](global_property_object& p) {
156  p.parameters.current_fees->get<account_create_operation>().basic_fee <<= p.parameters.account_fee_scale_bitshifts;
157  });
158 
159  return new_acnt_object.id;
160 } FC_CAPTURE_AND_RETHROW((o)) }

257 void_result account_upgrade_evaluator::do_apply(const account_upgrade_evaluator::operation_type& o)
258 { try {
259  database& d = db();
260 
261  d.modify(*account, [&](account_object& a) {
262  if( o.upgrade_to_lifetime_member )
263  {
264  // Upgrade to lifetime member. I don't care what the account was before.
265  a.statistics(d).process_fees(a, d);
266  a.membership_expiration_date = time_point_sec::maximum();
267  a.referrer = a.registrar = a.lifetime_referrer = a.get_id();
268  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT - a.network_fee_percentage;
269  } else if( a.is_annual_member(d.head_block_time()) ) {
270  // Renew an annual subscription that's still in effect.
271  FC_ASSERT(a.membership_expiration_date - d.head_block_time() < fc::days(3650),
272  "May not extend annual membership more than a decade into the future.");
273  a.membership_expiration_date += fc::days(365);
274  } else {
275  // Upgrade from basic account.
276  a.statistics(d).process_fees(a, d);
277  assert(a.is_basic_account(d.head_block_time()));
278  a.referrer = a.get_id();
279  a.membership_expiration_date = d.head_block_time() + fc::days(365);
280  }
281  });
282 
283  return {};
284 } FC_RETHROW_EXCEPTIONS( error, "Unable to upgrade account '${a}'", ("a",o.account_to_upgrade(db()).name) ) }
