27  const string& msg, uint64_t custom_nonce)
28 {
29  if( priv != fc::ecc::private_key() && public_key_type(pub) != public_key_type() )
30  {
31  from = priv.get_public_key();
32  to = pub;
33  if( custom_nonce == 0 )
34  {
35  uint64_t entropy = fc::sha224::hash(fc::ecc::private_key::generate())._hash[0];
36  entropy <<= 32;
37  entropy &= 0xff00000000000000;
38  nonce = (fc::time_point::now().time_since_epoch().count() & 0x00ffffffffffffff) | entropy;
39  } else

53 string memo_data::get_message(const fc::ecc::private_key& priv,
54  const fc::ecc::public_key& pub)const
55 {
56  if( from != public_key_type() )
57  {
58  auto secret = priv.get_shared_secret(pub);
59  auto nonce_plus_secret = fc::sha512::hash(fc::to_string(nonce) + secret.str());
60  auto plain_text = fc::aes_decrypt( nonce_plus_secret, message );
61  auto result = memo_message::deserialize(string(plain_text.begin(), plain_text.end()));
62  FC_ASSERT( result.checksum == uint32_t(digest_type::hash(result.text)._hash[0]) );
63  return result.text;
64  }
65  else
66  {
67  return memo_message::deserialize(string(message.begin(), message.end())).text;
68  }
