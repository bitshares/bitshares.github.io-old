148  std::vector<fc::ip::endpoint> resolve_string_to_ip_endpoints(const std::string& endpoint_string)
149  {
150  try
151  {
152  string::size_type colon_pos = endpoint_string.find(':');
153  if (colon_pos == std::string::npos)
154  FC_THROW("Missing required port number in endpoint string \"${endpoint_string}\"",
155  ("endpoint_string", endpoint_string));
156  std::string port_string = endpoint_string.substr(colon_pos + 1);
157  try
158  {
159  uint16_t port = boost::lexical_cast<uint16_t>(port_string);
160 
161  std::string hostname = endpoint_string.substr(0, colon_pos);
162  std::vector<fc::ip::endpoint> endpoints = fc::resolve(hostname, port);
163  if (endpoints.empty())
164  FC_THROW_EXCEPTION(fc::unknown_host_exception, "The host name can not be resolved: ${hostname}", ("hostname", hostname));
165  return endpoints;
166  }
167  catch (const boost::bad_lexical_cast&)
168  {
169  FC_THROW("Bad port: ${port}", ("port", port_string));
170  }
171  }
172  FC_CAPTURE_AND_RETHROW((endpoint_string))
173  }

438  virtual bool handle_block(const graphene::net::block_message& blk_msg, bool sync_mode,
439  std::vector<fc::uint160_t>& contained_transaction_message_ids) override
440  { try {
441 
442  auto latency = graphene::time::now() - blk_msg.block.timestamp;
443  if (!sync_mode || blk_msg.block.block_num() % 10000 == 0)
444  {
445  const auto& witness = blk_msg.block.witness(*_chain_db);
446  const auto& witness_account = witness.witness_account(*_chain_db);
447  auto last_irr = _chain_db->get_dynamic_global_properties().last_irreversible_block_num;
448  ilog("Got block: #${n} time: ${t} latency: ${l} ms from: ${w} irreversible: ${i} (-${d})",
449  ("t",blk_msg.block.timestamp)
450  ("n", blk_msg.block.block_num())
451  ("l", (latency.count()/1000))
452  ("w",witness_account.name)
453  ("i",last_irr)("d",blk_msg.block.block_num()-last_irr) );
454  }
455 
456  try {
457  // TODO: in the case where this block is valid but on a fork that's too old for us to switch to,
458  // you can help the network code out by throwing a block_older_than_undo_history exception.
459  // when the net code sees that, it will stop trying to push blocks from that chain, but
460  // leave that peer connected so that they can get sync blocks from us
461  bool result = _chain_db->push_block(blk_msg.block, (_is_block_producer | _force_validate) ? database::skip_nothing : database::skip_transaction_signatures);
462 
463  // the block was accepted, so we now know all of the transactions contained in the block
464  if (!sync_mode)
465  {
466  // if we're not in sync mode, there's a chance we will be seeing some transactions
467  // included in blocks before we see the free-floating transaction itself. If that
468  // happens, there's no reason to fetch the transactions, so construct a list of the
469  // transaction message ids we no longer need.
470  // during sync, it is unlikely that we'll see any old
471  for (const processed_transaction& transaction : blk_msg.block.transactions)
472  {
473  graphene::net::trx_message transaction_message(transaction);
474  contained_transaction_message_ids.push_back(graphene::net::message(transaction_message).id());
475  }
476  }
477 
478  return result;
479  } catch ( const graphene::chain::unlinkable_block_exception& e ) {
480  // translate to a graphene::net exception
481  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
482  FC_THROW_EXCEPTION(graphene::net::unlinkable_block_exception, "Error when pushing block:\n${e}", ("e", e.to_detail_string()));
483  } catch( const fc::exception& e ) {
484  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
485  throw;
486  }
487 
488  if( !_is_finished_syncing && !sync_mode )
489  {
490  _is_finished_syncing = true;
491  _self->syncing_finished();
492  }
493  } FC_CAPTURE_AND_RETHROW( (blk_msg)(sync_mode) ) }

558  (_chain_db->is_known_block(block_id_in_synopsis) && is_included_block(block_id_in_synopsis)))

561  found_a_block_in_synopsis = true;
562  break;
563  }
564  if (!found_a_block_in_synopsis)
565  FC_THROW_EXCEPTION(graphene::net::peer_is_on_an_unreachable_fork, "Unable to provide a list of blocks starting at any of the blocks in peer's synopsis");
566  }
567  for( uint32_t num = block_header::num_from_id(last_known_block_id);
568  num <= _chain_db->head_block_num() && result.size() < limit;
569  ++num )
570  if( num > 0 )
571  result.push_back(_chain_db->get_block_id_for_num(num));
572 
573  if( !result.empty() && block_header::num_from_id(result.back()) < _chain_db->head_block_num() )
574  remaining_item_count = _chain_db->head_block_num() - block_header::num_from_id(result.back());
575 
576  return result;
577  } FC_CAPTURE_AND_RETHROW( (blockchain_synopsis)(remaining_item_count)(limit) ) }

661  virtual std::vector<item_hash_t> get_blockchain_synopsis(const item_hash_t& reference_point,
662  uint32_t number_of_blocks_after_reference_point) override
663  { try {
664  std::vector<item_hash_t> synopsis;
665  synopsis.reserve(30);
666  uint32_t high_block_num;
667  uint32_t non_fork_high_block_num;
668  uint32_t low_block_num = _chain_db->last_non_undoable_block_num();
669  std::vector<block_id_type> fork_history;
670 
671  if (reference_point != item_hash_t())
672  {
673  // the node is asking for a summary of the block chain up to a specified
674  // block, which may or may not be on a fork
675  // for now, assume it's not on a fork
676  if (is_included_block(reference_point))
677  {
678  // reference_point is a block we know about and is on the main chain
679  uint32_t reference_point_block_num = block_header::num_from_id(reference_point);
680  assert(reference_point_block_num > 0);
681  high_block_num = reference_point_block_num;
682  non_fork_high_block_num = high_block_num;
683 
684  if (reference_point_block_num < low_block_num)
685  {
686  // we're on the same fork (at least as far as reference_point) but we've passed
687  // reference point and could no longer undo that far if we diverged after that
688  // block. This should probably only happen due to a race condition where
689  // the network thread calls this function, and then immediately pushes a bunch of blocks,
690  // then the main thread finally processes this function.
691  // with the current framework, there's not much we can do to tell the network
692  // thread what our current head block is, so we'll just pretend that
693  // our head is actually the reference point.
694  // this *may* enable us to fetch blocks that we're unable to push, but that should
695  // be a rare case (and correctly handled)
696  low_block_num = reference_point_block_num;
697  }
698  }
699  else
700  {
701  // block is a block we know about, but it is on a fork
702  try
703  {
704  fork_history = _chain_db->get_block_ids_on_fork(reference_point);
705  // returns a vector where the last element is the common ancestor with the preferred chain,
706  // and the first element is the reference point you passed in
707  assert(fork_history.size() >= 2);
708 
709  if( fork_history.front() != reference_point )
710  {
711  edump( (fork_history)(reference_point) );
712  assert(fork_history.front() == reference_point);
713  }
714  block_id_type last_non_fork_block = fork_history.back();
715  fork_history.pop_back(); // remove the common ancestor
716  boost::reverse(fork_history);
717 
718  if (last_non_fork_block == block_id_type()) // if the fork goes all the way back to genesis (does graphene's fork db allow this?)
719  non_fork_high_block_num = 0;
720  else
721  non_fork_high_block_num = block_header::num_from_id(last_non_fork_block);
722 
723  high_block_num = non_fork_high_block_num + fork_history.size();
724  assert(high_block_num == block_header::num_from_id(fork_history.back()));
725  }
726  catch (const fc::exception& e)
727  {
728  // unable to get fork history for some reason. maybe not linked?
729  // we can't return a synopsis of its chain
730  elog("Unable to construct a blockchain synopsis for reference hash ${hash}: ${exception}", ("hash", reference_point)("exception", e));
731  throw;
732  }
733  if (non_fork_high_block_num < low_block_num)
734  {
735  wlog("Unable to generate a usable synopsis because the peer we're generating it for forked too long ago "
736  "(our chains diverge after block #${non_fork_high_block_num} but only undoable to block #${low_block_num})",
737  ("low_block_num", low_block_num)
738  ("non_fork_high_block_num", non_fork_high_block_num));
739  FC_THROW_EXCEPTION(graphene::net::block_older_than_undo_history, "Peer is are on a fork I'm unable to switch to");
740  }
741  }
742  }
743  else
744  {
745  // no reference point specified, summarize the whole block chain
746  high_block_num = _chain_db->head_block_num();
747  non_fork_high_block_num = high_block_num;
748  if (high_block_num == 0)
749  return synopsis; // we have no blocks
750  }
751 
752  // at this point:
753  // low_block_num is the block before the first block we can undo,
754  // non_fork_high_block_num is the block before the fork (if the peer is on a fork, or otherwise it is the same as high_block_num)
755  // high_block_num is the block number of the reference block, or the end of the chain if no reference provided
756 
757  // true_high_block_num is the ending block number after the network code appends any item ids it
758  // knows about that we don't
759  uint32_t true_high_block_num = high_block_num + number_of_blocks_after_reference_point;
760  do
761  {
762  // for each block in the synopsis, figure out where to pull the block id from.
763  // if it's <= non_fork_high_block_num, we grab it from the main blockchain;
764  // if it's not, we pull it from the fork history
765  if (low_block_num <= non_fork_high_block_num)
766  synopsis.push_back(_chain_db->get_block_id_for_num(low_block_num));
767  else
768  synopsis.push_back(fork_history[low_block_num - non_fork_high_block_num - 1]);
769  low_block_num += (true_high_block_num - low_block_num + 2) / 2;
770  }
771  while (low_block_num <= high_block_num);
772 
773  idump((synopsis));
774  return synopsis;
775  } FC_CAPTURE_AND_RETHROW() }
