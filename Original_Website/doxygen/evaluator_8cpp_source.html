48  void generic_evaluator::prepare_fee(account_id_type account_id, asset fee)
49  {
50  const database& d = db();
51  fee_from_account = fee;
52  FC_ASSERT( fee.amount >= 0 );
53  fee_paying_account = &account_id(d);
54  fee_paying_account_statistics = &fee_paying_account->statistics(d);
55 
56  fee_asset = &fee.asset_id(d);
57  fee_asset_dyn_data = &fee_asset->dynamic_asset_data_id(d);
58 
59  if( d.head_block_time() > HARDFORK_419_TIME )
60  {
61  FC_ASSERT( fee_paying_account->is_authorized_asset( *fee_asset, d ), "Account ${acct} '${name}' attempted to pay fee by using asset ${a} '${sym}', which is unauthorized due to whitelist / blacklist",
62  ("acct", fee_paying_account->id)("name", fee_paying_account->name)("a", fee_asset->id)("sym", fee_asset->symbol) );
63  }
64 
65  if( fee_from_account.asset_id == asset_id_type() )
66  core_fee_paid = fee_from_account.amount;
67  else
68  {
69  asset fee_from_pool = fee_from_account * fee_asset->options.core_exchange_rate;
70  FC_ASSERT( fee_from_pool.asset_id == asset_id_type() );
71  core_fee_paid = fee_from_pool.amount;
72  FC_ASSERT( core_fee_paid <= fee_asset_dyn_data->fee_pool, "Fee pool balance of '${b}' is less than the ${r} required to convert ${c}",
73  ("r", db().to_pretty_string( fee_from_pool))("b",db().to_pretty_string(fee_asset_dyn_data->fee_pool))("c",db().to_pretty_string(fee)) );
74  }
