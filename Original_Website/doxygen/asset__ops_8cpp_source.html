31 bool is_valid_symbol( const string& symbol )
32 {
33  if( symbol.size() < GRAPHENE_MIN_ASSET_SYMBOL_LENGTH )
34  return false;
35 
36  if( symbol.size() > GRAPHENE_MAX_ASSET_SYMBOL_LENGTH )
37  return false;
38 
39  if( !isalpha( symbol.front() ) )
40  return false;
41 
42  if( !isalpha( symbol.back() ) )
43  return false;
44 
45  bool dot_already_present = false;
46  for( const auto c : symbol )
47  {
48  if( (isalpha( c ) && isupper( c )) || isdigit(c) )
49  continue;
50 
51  if( c == '.' )
52  {
53  if( dot_already_present )
54  return false;
55 
56  dot_already_present = true;
57  continue;
58  }
59 
60  return false;
61  }
62 
63  return true;
64 }

199 void asset_options::validate()const
200 {
201  FC_ASSERT( max_supply > 0 );
202  FC_ASSERT( max_supply <= GRAPHENE_MAX_SHARE_SUPPLY );
203  FC_ASSERT( market_fee_percent <= GRAPHENE_100_PERCENT );
204  FC_ASSERT( max_market_fee >= 0 && max_market_fee <= GRAPHENE_MAX_SHARE_SUPPLY );
205  // There must be no high bits in permissions whose meaning is not known.
206  FC_ASSERT( !(issuer_permissions & ~ASSET_ISSUER_PERMISSION_MASK) );
207  // The global_settle flag may never be set (this is a permission only)
208  FC_ASSERT( !(flags & global_settle) );
209  // the witness_fed and committee_fed flags cannot be set simultaneously
210  FC_ASSERT( (flags & (witness_fed_asset | committee_fed_asset)) != (witness_fed_asset | committee_fed_asset) );
211  core_exchange_rate.validate();
212  FC_ASSERT( core_exchange_rate.base.asset_id.instance.value == 0 ||
213  core_exchange_rate.quote.asset_id.instance.value == 0 );
214 
215  if(!whitelist_authorities.empty() || !blacklist_authorities.empty())
216  FC_ASSERT( flags & white_list );
217  for( auto item : whitelist_markets )
218  {
219  FC_ASSERT( blacklist_markets.find(item) == blacklist_markets.end() );
220  }
221  for( auto item : blacklist_markets )
222  {
223  FC_ASSERT( whitelist_markets.find(item) == whitelist_markets.end() );
224  }
225 }
