40  bool address::is_valid( const std::string& base58str, const std::string& prefix )
41  {
42  const size_t prefix_len = prefix.size();
43  if( base58str.size() <= prefix_len )
44  return false;
45  if( base58str.substr( 0, prefix_len ) != prefix )
46  return false;
47  std::vector<char> v;
48  try
49  {
50  v = fc::from_base58( base58str.substr( prefix_len ) );
51  }
52  catch( const fc::parse_error_exception& e )
53  {
54  return false;
55  }
56  if( v.size() != sizeof( fc::ripemd160 ) + 4 )
57  return false;
58  const fc::ripemd160 checksum = fc::ripemd160::hash( v.data(), v.size() - 4 );
59  if( memcmp( v.data() + 20, (char*)checksum._hash, 4 ) != 0 )
60  return false;
61  return true;
62  }
