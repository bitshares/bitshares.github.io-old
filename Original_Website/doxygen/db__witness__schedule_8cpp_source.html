40 fc::time_point_sec database::get_slot_time(uint32_t slot_num)const
41 {
42  if( slot_num == 0 )
43  return fc::time_point_sec();
44 
45  auto interval = block_interval();
46  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
47 
48  if( head_block_num() == 0 )
49  {
50  // n.b. first block is at genesis_time plus one block interval
51  fc::time_point_sec genesis_time = dpo.time;
52  return genesis_time + slot_num * interval;
53  }
54 
55  int64_t head_block_abs_slot = head_block_time().sec_since_epoch() / interval;
56  fc::time_point_sec head_slot_time(head_block_abs_slot * interval);
57 
58  const global_property_object& gpo = get_global_properties();
59 
60  if( dpo.dynamic_flags & dynamic_global_property_object::maintenance_flag )
61  slot_num += gpo.parameters.maintenance_skip_slots;
62 
63  // "slot 0" is head_slot_time
64  // "slot 1" is head_slot_time,
65  // plus maint interval if head block is a maint block
66  // plus block interval if head block is not a maint block
67  return head_slot_time + (slot_num * interval);
68 }

84 void database::update_witness_schedule()
85 {
86  const witness_schedule_object& wso = witness_schedule_id_type()(*this);
87  const global_property_object& gpo = get_global_properties();
88 
89  if( head_block_num() % gpo.active_witnesses.size() == 0 )
90  {
91  modify( wso, [&]( witness_schedule_object& _wso )
92  {
93  _wso.current_shuffled_witnesses.clear();
94  _wso.current_shuffled_witnesses.reserve( gpo.active_witnesses.size() );
95 
96  for( const witness_id_type& w : gpo.active_witnesses )
97  _wso.current_shuffled_witnesses.push_back( w );
98 
99  auto now_hi = uint64_t(head_block_time().sec_since_epoch()) << 32;
100  for( uint32_t i = 0; i < _wso.current_shuffled_witnesses.size(); ++i )
101  {
104  uint64_t k = now_hi + uint64_t(i)*2685821657736338717ULL;
105  k ^= (k >> 12);
106  k ^= (k << 25);
107  k ^= (k >> 27);
108  k *= 2685821657736338717ULL;
109 
110  uint32_t jmax = _wso.current_shuffled_witnesses.size() - i;
111  uint32_t j = i + k%jmax;
112  std::swap( _wso.current_shuffled_witnesses[i],
113  _wso.current_shuffled_witnesses[j] );
114  }
115  });
116  }
117 }
