155  template<typename T>
156  class op_evaluator_impl : public op_evaluator
157  {
158  public:
159  virtual operation_result evaluate(transaction_evaluation_state& eval_state, const operation& op, bool apply = true) override
160  {
161  // fc::exception from observers are suppressed.
162  // fc::exception from evaluation is deferred (re-thrown
163  // after all observers receive evaluation_failed)
164 
165  T eval;
166  shared_ptr<fc::exception> evaluation_exception;
167  size_t observer_count = 0;
168  operation_result result;
169 
170  for( const auto& obs : eval_observers )
171  {
172  try
173  {
174  obs->pre_evaluate(eval_state, op, apply, &eval);
175  }
176  catch( const fc::exception& e )
177  {
178  elog( "suppressed exception in observer pre method:\n${e}", ( "e", e.to_detail_string() ) );
179  }
180  observer_count++;
181  }
182 
183  try
184  {
185  result = eval.start_evaluate(eval_state, op, apply);
186  }
187  catch( const fc::exception& e )
188  {
189  evaluation_exception = e.dynamic_copy_exception();
190  }
191 
192  while( observer_count > 0 )
193  {
194  --observer_count;
195  const auto& obs = eval_observers[observer_count];
196  try
197  {
198  if( evaluation_exception )
199  obs->post_evaluate(eval_state, op, apply, &eval, result);
200  else
201  obs->evaluation_failed(eval_state, op, apply, &eval, result);
202  }
203  catch( const fc::exception& e )
204  {
205  elog( "suppressed exception in observer post method:\n${e}", ( "e", e.to_detail_string() ) );
206  }
207  }
208 
209  if( evaluation_exception )
210  evaluation_exception->dynamic_rethrow_exception();
211  return result;
212  }
213  };
