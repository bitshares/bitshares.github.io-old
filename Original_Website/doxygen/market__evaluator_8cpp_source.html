32 void_result limit_order_create_evaluator::do_evaluate(const limit_order_create_operation& op)
33 { try {
34  const database& d = db();
35 
36  FC_ASSERT( op.expiration >= d.head_block_time() );
37 
38  _seller = this->fee_paying_account;
39  _sell_asset = &op.amount_to_sell.asset_id(d);
40  _receive_asset = &op.min_to_receive.asset_id(d);
41 
42  if( _sell_asset->options.whitelist_markets.size() )
43  FC_ASSERT( _sell_asset->options.whitelist_markets.find(_receive_asset->id) != _sell_asset->options.whitelist_markets.end() );
44  if( _sell_asset->options.blacklist_markets.size() )
45  FC_ASSERT( _sell_asset->options.blacklist_markets.find(_receive_asset->id) == _sell_asset->options.blacklist_markets.end() );
46 
47  if( d.head_block_time() <= HARDFORK_416_TIME )
48  {
49  if( _sell_asset->options.flags & white_list ) FC_ASSERT( _seller->is_authorized_asset( *_sell_asset, d ) );
50  if( _receive_asset->options.flags & white_list ) FC_ASSERT( _seller->is_authorized_asset( *_receive_asset, d ) );
51  }
52  else
53  {
54  FC_ASSERT( _seller->is_authorized_asset( *_sell_asset, d ) );
55  FC_ASSERT( _seller->is_authorized_asset( *_receive_asset, d ) );
56  }
57 
58  FC_ASSERT( d.get_balance( *_seller, *_sell_asset ) >= op.amount_to_sell, "insufficient balance",
59  ("balance",d.get_balance(*_seller,*_sell_asset))("amount_to_sell",op.amount_to_sell) );
60 
61  return void_result();
62 } FC_CAPTURE_AND_RETHROW( (op) ) }

127 void_result call_order_update_evaluator::do_evaluate(const call_order_update_operation& o)
128 { try {
129  database& d = db();
130 
131  _paying_account = &o.funding_account(d);
132  _debt_asset = &o.delta_debt.asset_id(d);
133  FC_ASSERT( _debt_asset->is_market_issued(), "Unable to cover ${sym} as it is not a collateralized asset.",
134  ("sym", _debt_asset->symbol) );
135 
136  _bitasset_data = &_debt_asset->bitasset_data(d);
137 
140  FC_ASSERT( !_bitasset_data->has_settlement() );
141 
142  FC_ASSERT( o.delta_collateral.asset_id == _bitasset_data->options.short_backing_asset );
143 
144  if( _bitasset_data->is_prediction_market )
145  FC_ASSERT( o.delta_collateral.amount == o.delta_debt.amount );
146  else if( _bitasset_data->current_feed.settlement_price.is_null() )
147  FC_THROW_EXCEPTION(insufficient_feeds, "Cannot borrow asset with no price feed.");
148 
149  if( o.delta_debt.amount < 0 )
150  {
151  FC_ASSERT( d.get_balance(*_paying_account, *_debt_asset) >= o.delta_debt,
152  "Cannot cover by ${c} when payer only has ${b}",
153  ("c", o.delta_debt.amount)("b", d.get_balance(*_paying_account, *_debt_asset).amount) );
154  }
155 
156  if( o.delta_collateral.amount > 0 )
157  {
158  FC_ASSERT( d.get_balance(*_paying_account, _bitasset_data->options.short_backing_asset(d)) >= o.delta_collateral,
159  "Cannot increase collateral by ${c} when payer only has ${b}", ("c", o.delta_collateral.amount)
160  ("b", d.get_balance(*_paying_account, o.delta_collateral.asset_id(d)).amount) );
161  }
162 
163  return void_result();
164 } FC_CAPTURE_AND_RETHROW( (o) ) }

167 void_result call_order_update_evaluator::do_apply(const call_order_update_operation& o)
168 { try {
169  database& d = db();
170 
171  if( o.delta_debt.amount != 0 )
172  {
173  d.adjust_balance( o.funding_account, o.delta_debt );
174 
175  // Deduct the debt paid from the total supply of the debt asset.
176  d.modify(_debt_asset->dynamic_asset_data_id(d), [&](asset_dynamic_data_object& dynamic_asset) {
177  dynamic_asset.current_supply += o.delta_debt.amount;
178  assert(dynamic_asset.current_supply >= 0);
179  });
180  }
181 
182  if( o.delta_collateral.amount != 0 )
183  {
184  d.adjust_balance( o.funding_account, -o.delta_collateral );
185 
186  // Adjust the total core in orders accodingly
187  if( o.delta_collateral.asset_id == asset_id_type() )
188  {
189  d.modify(_paying_account->statistics(d), [&](account_statistics_object& stats) {
190  stats.total_core_in_orders += o.delta_collateral.amount;
191  });
192  }
193  }
194 
195 
196  auto& call_idx = d.get_index_type<call_order_index>().indices().get<by_account>();
197  auto itr = call_idx.find( boost::make_tuple(o.funding_account, o.delta_debt.asset_id) );
198  const call_order_object* call_obj = nullptr;
199 
200  if( itr == call_idx.end() )
201  {
202  FC_ASSERT( o.delta_collateral.amount > 0 );
203  FC_ASSERT( o.delta_debt.amount > 0 );
204 
205  call_obj = &d.create<call_order_object>( [&](call_order_object& call ){
206  call.borrower = o.funding_account;
207  call.collateral = o.delta_collateral.amount;
208  call.debt = o.delta_debt.amount;
209  call.call_price = price::call_price(o.delta_debt, o.delta_collateral,
210  _bitasset_data->current_feed.maintenance_collateral_ratio);
211 
212  });
213  }
214  else
215  {
216  call_obj = &*itr;
217 
218  d.modify( *call_obj, [&]( call_order_object& call ){
219  call.collateral += o.delta_collateral.amount;
220  call.debt += o.delta_debt.amount;
221  if( call.debt > 0 )
222  {
223  call.call_price = price::call_price(call.get_debt(), call.get_collateral(),
224  _bitasset_data->current_feed.maintenance_collateral_ratio);
225  }
226  });
227  }
228 
229  auto debt = call_obj->get_debt();
230  if( debt.amount == 0 )
231  {
232  FC_ASSERT( call_obj->collateral == 0 );
233  d.remove( *call_obj );
234  return void_result();
235  }
236 
237  FC_ASSERT(call_obj->collateral > 0 && call_obj->debt > 0);
238 
239  // then we must check for margin calls and other issues
240  if( !_bitasset_data->is_prediction_market )
241  {
242  call_order_id_type call_order_id = call_obj->id;
243 
244  // check to see if the order needs to be margin called now, but don't allow black swans and require there to be
245  // limit orders available that could be used to fill the order.
246  if( d.check_call_orders( *_debt_asset, false ) )
247  {
248  const auto call_obj = d.find(call_order_id);
249  // if we filled at least one call order, we are OK if we totally filled.
250  GRAPHENE_ASSERT(
251  !call_obj,
252  call_order_update_unfilled_margin_call,
253  "Updating call order would trigger a margin call that cannot be fully filled",
254  ("a", ~call_obj->call_price )("b", _bitasset_data->current_feed.settlement_price)
255  );
256  }
257  else
258  {
259  const auto call_obj = d.find(call_order_id);
260  FC_ASSERT( call_obj, "no margin call was executed and yet the call object was deleted" );
261  //edump( (~call_obj->call_price) ("<")( _bitasset_data->current_feed.settlement_price) );
262  // We didn't fill any call orders. This may be because we
263  // aren't in margin call territory, or it may be because there
264  // were no matching orders. In the latter case, we throw.
265  GRAPHENE_ASSERT(
266  ~call_obj->call_price < _bitasset_data->current_feed.settlement_price,
267  call_order_update_unfilled_margin_call,
268  "Updating call order would trigger a margin call that cannot be fully filled",
269  ("a", ~call_obj->call_price )("b", _bitasset_data->current_feed.settlement_price)
270  );
271  }
272  }
273 
274  return void_result();
275 } FC_CAPTURE_AND_RETHROW( (o) ) }
