119 void database::update_last_irreversible_block()
120 {
121  const global_property_object& gpo = get_global_properties();
122  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
123 
124  vector< const witness_object* > wit_objs;
125  wit_objs.reserve( gpo.active_witnesses.size() );
126  for( const witness_id_type& wid : gpo.active_witnesses )
127  wit_objs.push_back( &(wid(*this)) );
128 
129  static_assert( GRAPHENE_IRREVERSIBLE_THRESHOLD > 0, "irreversible threshold must be nonzero" );
130 
131  // 1 1 1 2 2 2 2 2 2 2 -> 2 .7*10 = 7
132  // 1 1 1 1 1 1 1 2 2 2 -> 1
133  // 3 3 3 3 3 3 3 3 3 3 -> 3
134 
135  size_t offset = ((GRAPHENE_100_PERCENT - GRAPHENE_IRREVERSIBLE_THRESHOLD) * wit_objs.size() / GRAPHENE_100_PERCENT);
136 
137  std::nth_element( wit_objs.begin(), wit_objs.begin() + offset, wit_objs.end(),
138  []( const witness_object* a, const witness_object* b )
139  {
140  return a->last_confirmed_block_num < b->last_confirmed_block_num;
141  } );
142 
143  uint32_t new_last_irreversible_block_num = wit_objs[offset]->last_confirmed_block_num;
144 
145  if( new_last_irreversible_block_num > dpo.last_irreversible_block_num )
146  {
147  modify( dpo, [&]( dynamic_global_property_object& _dpo )
148  {
149  _dpo.last_irreversible_block_num = new_last_irreversible_block_num;
150  } );
151  }

195 bool database::check_for_blackswan( const asset_object& mia, bool enable_black_swan )
196 {
197  if( !mia.is_market_issued() ) return false;
198 
199  const asset_bitasset_data_object& bitasset = mia.bitasset_data(*this);
200  if( bitasset.has_settlement() ) return true; // already force settled
201  auto settle_price = bitasset.current_feed.settlement_price;
202  if( settle_price.is_null() ) return false; // no feed
203 
204  const call_order_index& call_index = get_index_type<call_order_index>();
205  const auto& call_price_index = call_index.indices().get<by_price>();
206 
207  const limit_order_index& limit_index = get_index_type<limit_order_index>();
208  const auto& limit_price_index = limit_index.indices().get<by_price>();
209 
210  // looking for limit orders selling the most USD for the least CORE
211  auto highest_possible_bid = price::max( mia.id, bitasset.options.short_backing_asset );
212  // stop when limit orders are selling too little USD for too much CORE
213  auto lowest_possible_bid = price::min( mia.id, bitasset.options.short_backing_asset );
214 
215  assert( highest_possible_bid.base.asset_id == lowest_possible_bid.base.asset_id );
216  // NOTE limit_price_index is sorted from greatest to least
217  auto limit_itr = limit_price_index.lower_bound( highest_possible_bid );
218  auto limit_end = limit_price_index.upper_bound( lowest_possible_bid );
219 
220  auto call_min = price::min( bitasset.options.short_backing_asset, mia.id );
221  auto call_max = price::max( bitasset.options.short_backing_asset, mia.id );
222  auto call_itr = call_price_index.lower_bound( call_min );
223  auto call_end = call_price_index.upper_bound( call_max );
224 
225  if( call_itr == call_end ) return false; // no call orders
226 
227  price highest = settle_price;
228  if( limit_itr != limit_end ) {
229  assert( settle_price.base.asset_id == limit_itr->sell_price.base.asset_id );
230  highest = std::max( limit_itr->sell_price, settle_price );
231  }
232 
233  auto least_collateral = call_itr->collateralization();
234  if( ~least_collateral >= highest )
235  {
236  elog( "Black Swan detected: \n"
237  " Least collateralized call: ${lc} ${~lc}\n"
238  // " Highest Bid: ${hb} ${~hb}\n"
239  " Settle Price: ${sp} ${~sp}\n"
240  " Max: ${h} ${~h}\n",
241  ("lc",least_collateral.to_real())("~lc",(~least_collateral).to_real())
242  // ("hb",limit_itr->sell_price.to_real())("~hb",(~limit_itr->sell_price).to_real())
243  ("sp",settle_price.to_real())("~sp",(~settle_price).to_real())
244  ("h",highest.to_real())("~h",(~highest).to_real()) );
245  FC_ASSERT( enable_black_swan, "Black swan was detected during a margin update which is not allowed to trigger a blackswan" );
246  globally_settle_asset(mia, ~least_collateral );
247  return true;
248  }
249  return false;
250 }

252 void database::clear_expired_orders()
253 {
254  detail::with_skip_flags( *this,
255  get_node_properties().skip_flags | skip_authority_check, [&](){
256  transaction_evaluation_state cancel_context(this);
257 
258  //Cancel expired limit orders
259  auto& limit_index = get_index_type<limit_order_index>().indices().get<by_expiration>();
260  while( !limit_index.empty() && limit_index.begin()->expiration <= head_block_time() )
261  {
262  limit_order_cancel_operation canceler;
263  const limit_order_object& order = *limit_index.begin();
264  canceler.fee_paying_account = order.seller;
265  canceler.order = order.id;
266  apply_operation(cancel_context, canceler);
267  }
268  });
269 
270  //Process expired force settlement orders
271  auto& settlement_index = get_index_type<force_settlement_index>().indices().get<by_expiration>();
272  if( !settlement_index.empty() )
273  {
274  asset_id_type current_asset = settlement_index.begin()->settlement_asset_id();
275  asset max_settlement_volume;
276 
277  auto next_asset = [&current_asset, &settlement_index] {
278  auto bound = settlement_index.upper_bound(current_asset);
279  if( bound == settlement_index.end() )
280  return false;
281  current_asset = bound->settlement_asset_id();
282  return true;
283  };
284 
285  // At each iteration, we either consume the current order and remove it, or we move to the next asset
286  for( auto itr = settlement_index.lower_bound(current_asset);
287  itr != settlement_index.end();
288  itr = settlement_index.lower_bound(current_asset) )
289  {
290  const force_settlement_object& order = *itr;
291  auto order_id = order.id;
292  current_asset = order.settlement_asset_id();
293  const asset_object& mia_object = get(current_asset);
294  const asset_bitasset_data_object mia = mia_object.bitasset_data(*this);
295 
296  if( mia.has_settlement() )
297  {
298  ilog( "Canceling a force settlement because of black swan" );
299  cancel_order( order );
300  continue;
301  }
302 
303  // Has this order not reached its settlement date?
304  if( order.settlement_date > head_block_time() )
305  {
306  if( next_asset() )
307  continue;
308  break;
309  }
310  // Can we still settle in this asset?
311  if( mia.current_feed.settlement_price.is_null() )
312  {
313  ilog("Canceling a force settlement in ${asset} because settlement price is null",
314  ("asset", mia_object.symbol));
315  cancel_order(order);
316  continue;
317  }
318  if( max_settlement_volume.asset_id != current_asset )
319  max_settlement_volume = mia_object.amount(mia.max_force_settlement_volume(mia_object.dynamic_data(*this).current_supply));
320  if( mia.force_settled_volume >= max_settlement_volume.amount )
321  {
322  /*
323  ilog("Skipping force settlement in ${asset}; settled ${settled_volume} / ${max_volume}",
324  ("asset", mia_object.symbol)("settlement_price_null",mia.current_feed.settlement_price.is_null())
325  ("settled_volume", mia.force_settled_volume)("max_volume", max_settlement_volume));
326  */
327  if( next_asset() )
328  continue;
329  break;
330  }
331 
332  auto& pays = order.balance;
333  auto receives = (order.balance * mia.current_feed.settlement_price);
334  receives.amount = (fc::uint128_t(receives.amount.value) *
335  (GRAPHENE_100_PERCENT - mia.options.force_settlement_offset_percent) / GRAPHENE_100_PERCENT).to_uint64();
336  assert(receives <= order.balance * mia.current_feed.settlement_price);
337 
338  price settlement_price = pays / receives;
339 
340  auto& call_index = get_index_type<call_order_index>().indices().get<by_collateral>();
341  asset settled = mia_object.amount(mia.force_settled_volume);
342  // Match against the least collateralized short until the settlement is finished or we reach max settlements
343  while( settled < max_settlement_volume && find_object(order_id) )
344  {
345  auto itr = call_index.lower_bound(boost::make_tuple(price::min(mia_object.bitasset_data(*this).options.short_backing_asset,
346  mia_object.get_id())));
347  // There should always be a call order, since asset exists!
348  assert(itr != call_index.end() && itr->debt_type() == mia_object.get_id());
349  asset max_settlement = max_settlement_volume - settled;
350 
351  try {
352  settled += match(*itr, order, settlement_price, max_settlement);
353  }
354  catch ( const black_swan_exception& e ) {
355  wlog( "black swan detected: ${e}", ("e", e.to_detail_string() ) );
356  cancel_order( order );
357  break;
358  }
359  }
360  modify(mia, [settled](asset_bitasset_data_object& b) {
361  b.force_settled_volume = settled.amount;
362  });
363  }
364  }
365 }
