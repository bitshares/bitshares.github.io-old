39 template<class Index>
40 vector<std::reference_wrapper<const typename Index::object_type>> database::sort_votable_objects(size_t count) const
41 {
42  using ObjectType = typename Index::object_type;
43  const auto& all_objects = get_index_type<Index>().indices();
44  count = std::min(count, all_objects.size());
45  vector<std::reference_wrapper<const ObjectType>> refs;
46  refs.reserve(all_objects.size());
47  std::transform(all_objects.begin(), all_objects.end(),
48  std::back_inserter(refs),
49  [](const ObjectType& o) { return std::cref(o); });
50  std::partial_sort(refs.begin(), refs.begin() + count, refs.end(),
51  [this](const ObjectType& a, const ObjectType& b)->bool {
52  share_type oa_vote = _vote_tally_buffer[a.vote_id];
53  share_type ob_vote = _vote_tally_buffer[b.vote_id];
54  if( oa_vote != ob_vote )
55  return oa_vote > ob_vote;
56  return a.vote_id < b.vote_id;
57  });
58 
59  refs.resize(count, refs.front());
60  return refs;
61 }

103 void database::pay_workers( share_type& budget )
104 {
105 // ilog("Processing payroll! Available budget is ${b}", ("b", budget));
106  vector<std::reference_wrapper<const worker_object>> active_workers;
107  get_index_type<worker_index>().inspect_all_objects([this, &active_workers](const object& o) {
108  const worker_object& w = static_cast<const worker_object&>(o);
109  auto now = head_block_time();
110  if( w.is_active(now) && w.approving_stake(_vote_tally_buffer) > 0 )
111  active_workers.emplace_back(w);
112  });
113 
114  // worker with more votes is preferred
115  // if two workers exactly tie for votes, worker with lower ID is preferred
116  std::sort(active_workers.begin(), active_workers.end(), [this](const worker_object& wa, const worker_object& wb) {
117  share_type wa_vote = wa.approving_stake(_vote_tally_buffer);
118  share_type wb_vote = wb.approving_stake(_vote_tally_buffer);
119  if( wa_vote != wb_vote )
120  return wa_vote > wb_vote;
121  return wa.id < wb.id;
122  });
123 
124  for( uint32_t i = 0; i < active_workers.size() && budget > 0; ++i )
125  {
126  const worker_object& active_worker = active_workers[i];
127  share_type requested_pay = active_worker.daily_pay;
128  if( head_block_time() - get_dynamic_global_properties().last_budget_time != fc::days(1) )
129  {
130  fc::uint128 pay(requested_pay.value);
131  pay *= (head_block_time() - get_dynamic_global_properties().last_budget_time).count();
132  pay /= fc::days(1).count();
133  requested_pay = pay.to_uint64();
134  }
135 
136  share_type actual_pay = std::min(budget, requested_pay);
137  //ilog(" ==> Paying ${a} to worker ${w}", ("w", active_worker.id)("a", actual_pay));
138  modify(active_worker, [&](worker_object& w) {
139  w.worker.visit(worker_pay_visitor(actual_pay, *this));
140  });
141 
142  budget -= actual_pay;
143  }
144 }

292 void database::initialize_budget_record( fc::time_point_sec now, budget_record& rec )const
293 {
294  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
295  const asset_object& core = asset_id_type(0)(*this);
296  const asset_dynamic_data_object& core_dd = core.dynamic_asset_data_id(*this);
297 
298  rec.from_initial_reserve = core.reserved(*this);
299  rec.from_accumulated_fees = core_dd.accumulated_fees;
300  rec.from_unused_witness_budget = dpo.witness_budget;
301 
302  if( (dpo.last_budget_time == fc::time_point_sec())
303  || (now <= dpo.last_budget_time) )
304  {
305  rec.time_since_last_budget = 0;
306  return;
307  }
308 
309  int64_t dt = (now - dpo.last_budget_time).to_seconds();
310  rec.time_since_last_budget = uint64_t( dt );
311 
312  // We'll consider accumulated_fees to be reserved at the BEGINNING
313  // of the maintenance interval. However, for speed we only
314  // call modify() on the asset_dynamic_data_object once at the
315  // end of the maintenance interval. Thus the accumulated_fees
316  // are available for the budget at this point, but not included
317  // in core.reserved().
318  share_type reserve = rec.from_initial_reserve + core_dd.accumulated_fees;
319  // Similarly, we consider leftover witness_budget to be burned
320  // at the BEGINNING of the maintenance interval.
321  reserve += dpo.witness_budget;
322 
323  fc::uint128_t budget_u128 = reserve.value;
324  budget_u128 *= uint64_t(dt);
325  budget_u128 *= GRAPHENE_CORE_ASSET_CYCLE_RATE;
326  //round up to the nearest satoshi -- this is necessary to ensure
327  // there isn't an "untouchable" reserve, and we will eventually
328  // be able to use the entire reserve
329  budget_u128 += ((uint64_t(1) << GRAPHENE_CORE_ASSET_CYCLE_RATE_BITS) - 1);
330  budget_u128 >>= GRAPHENE_CORE_ASSET_CYCLE_RATE_BITS;
331  share_type budget;
332  if( budget_u128 < reserve.value )
333  rec.total_budget = share_type(budget_u128.to_uint64());
334  else
335  rec.total_budget = reserve;
336 
337  return;
338 }

343 void database::process_budget()
344 {
345  try
346  {
347  const global_property_object& gpo = get_global_properties();
348  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
349  const asset_dynamic_data_object& core =
350  asset_id_type(0)(*this).dynamic_asset_data_id(*this);
351  fc::time_point_sec now = head_block_time();
352 
353  int64_t time_to_maint = (dpo.next_maintenance_time - now).to_seconds();
354  //
355  // The code that generates the next maintenance time should
356  // only produce a result in the future. If this assert
357  // fails, then the next maintenance time algorithm is buggy.
358  //
359  assert( time_to_maint > 0 );
360  //
361  // Code for setting chain parameters should validate
362  // block_interval > 0 (as well as the humans proposing /
363  // voting on changes to block interval).
364  //
365  assert( gpo.parameters.block_interval > 0 );
366  uint64_t blocks_to_maint = (uint64_t(time_to_maint) + gpo.parameters.block_interval - 1) / gpo.parameters.block_interval;
367 
368  // blocks_to_maint > 0 because time_to_maint > 0,
369  // which means numerator is at least equal to block_interval
370 
371  budget_record rec;
372  initialize_budget_record( now, rec );
373  share_type available_funds = rec.total_budget;
374 
375  share_type witness_budget = gpo.parameters.witness_pay_per_block.value * blocks_to_maint;
376  rec.requested_witness_budget = witness_budget;
377  witness_budget = std::min(witness_budget, available_funds);
378  rec.witness_budget = witness_budget;
379  available_funds -= witness_budget;
380 
381  fc::uint128_t worker_budget_u128 = gpo.parameters.worker_budget_per_day.value;
382  worker_budget_u128 *= uint64_t(time_to_maint);
383  worker_budget_u128 /= 60*60*24;
384 
385  share_type worker_budget;
386  if( worker_budget_u128 >= available_funds.value )
387  worker_budget = available_funds;
388  else
389  worker_budget = worker_budget_u128.to_uint64();
390  rec.worker_budget = worker_budget;
391  available_funds -= worker_budget;
392 
393  share_type leftover_worker_funds = worker_budget;
394  pay_workers(leftover_worker_funds);
395  rec.leftover_worker_funds = leftover_worker_funds;
396  available_funds += leftover_worker_funds;
397 
398  rec.supply_delta = rec.witness_budget
399  + rec.worker_budget
400  - rec.leftover_worker_funds
401  - rec.from_accumulated_fees
402  - rec.from_unused_witness_budget;
403 
404  modify(core, [&]( asset_dynamic_data_object& _core )
405  {
406  _core.current_supply = (_core.current_supply + rec.supply_delta );
407 
408  assert( rec.supply_delta ==
409  witness_budget
410  + worker_budget
411  - leftover_worker_funds
412  - _core.accumulated_fees
413  - dpo.witness_budget
414  );
415  _core.accumulated_fees = 0;
416  });
417 
418  modify(dpo, [&]( dynamic_global_property_object& _dpo )
419  {
420  // Since initial witness_budget was rolled into
421  // available_funds, we replace it with witness_budget
422  // instead of adding it.
423  _dpo.witness_budget = witness_budget;
424  _dpo.last_budget_time = now;
425  });
426 
427  create< budget_record_object >( [&]( budget_record_object& _rec )
428  {
429  _rec.time = head_block_time();
430  _rec.record = rec;
431  });
432 
433  // available_funds is money we could spend, but don't want to.
434  // we simply let it evaporate back into the reserve.
435  }
436  FC_CAPTURE_AND_RETHROW()
437 }

461  // specifying the opinions.

477  d._vote_tally_buffer[offset] += voting_stake;
