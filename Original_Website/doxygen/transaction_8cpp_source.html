107  bool signed_by( const public_key_type& k )
108  {
109  auto itr = provided_signatures.find(k);
110  if( itr == provided_signatures.end() )
111  {
112  auto pk = available_keys.find(k);
113  if( pk != available_keys.end() )
114  return provided_signatures[k] = true;
115  return false;
116  }
117  return itr->second = true;
118  }

123  bool signed_by( const address& a ) {
124  if( !available_address_sigs ) {
125  available_address_sigs = std::map<address,public_key_type>();
126  provided_address_sigs = std::map<address,public_key_type>();
127  for( auto& item : available_keys ) {
128  (*available_address_sigs)[ address(pts_address(item, false, 56) ) ] = item;
129  (*available_address_sigs)[ address(pts_address(item, true, 56) ) ] = item;
130  (*available_address_sigs)[ address(pts_address(item, false, 0) ) ] = item;
131  (*available_address_sigs)[ address(pts_address(item, true, 0) ) ] = item;
132  (*available_address_sigs)[ address(item) ] = item;
133  }
134  for( auto& item : provided_signatures ) {
135  (*provided_address_sigs)[ address(pts_address(item.first, false, 56) ) ] = item.first;
136  (*provided_address_sigs)[ address(pts_address(item.first, true, 56) ) ] = item.first;
137  (*provided_address_sigs)[ address(pts_address(item.first, false, 0) ) ] = item.first;
138  (*provided_address_sigs)[ address(pts_address(item.first, true, 0) ) ] = item.first;
139  (*provided_address_sigs)[ address(item.first) ] = item.first;
140  }
141  }
142  auto itr = provided_address_sigs->find(a);
143  if( itr == provided_address_sigs->end() )
144  {
145  auto aitr = available_address_sigs->find(a);
146  if( aitr != available_address_sigs->end() ) {
147  auto pk = available_keys.find(aitr->second);
148  if( pk != available_keys.end() )
149  return provided_signatures[aitr->second] = true;
150  return false;
151  }
152  }
153  return provided_signatures[itr->second] = true;
154  }

156  bool check_authority( account_id_type id )
157  {
158  if( approved_by.find(id) != approved_by.end() ) return true;
159  return check_authority( get_active(id) );
160  }

166  bool check_authority( const authority* au, uint32_t depth = 0 )
167  {
168  if( au == nullptr ) return false;
169  const authority& auth = *au;
170 
171  uint32_t total_weight = 0;
172  for( const auto& k : auth.key_auths )
173  if( signed_by( k.first ) )
174  {
175  total_weight += k.second;
176  if( total_weight >= auth.weight_threshold )
177  return true;
178  }
179 
180  for( const auto& k : auth.address_auths )
181  if( signed_by( k.first ) )
182  {
183  total_weight += k.second;
184  if( total_weight >= auth.weight_threshold )
185  return true;
186  }
187 
188  for( const auto& a : auth.account_auths )
189  {
190  if( approved_by.find(a.first) == approved_by.end() )
191  {
192  if( depth == max_recursion )
193  return false;
194  if( check_authority( get_active( a.first ), depth+1 ) )
195  {
196  approved_by.insert( a.first );
197  total_weight += a.second;
198  if( total_weight >= auth.weight_threshold )
199  return true;
200  }
201  }
202  else
203  {
204  total_weight += a.second;
205  if( total_weight >= auth.weight_threshold )
206  return true;
207  }
208  }
209  return total_weight >= auth.weight_threshold;
210  }

243 void verify_authority( const vector<operation>& ops, const flat_set<public_key_type>& sigs,
244  const std::function<const authority*(account_id_type)>& get_active,
245  const std::function<const authority*(account_id_type)>& get_owner,
246  uint32_t max_recursion_depth,
247  bool allow_committe,
248  const flat_set<account_id_type>& active_aprovals,
249  const flat_set<account_id_type>& owner_approvals )
250 { try {
251  flat_set<account_id_type> required_active;
252  flat_set<account_id_type> required_owner;
253  vector<authority> other;
254 
255  for( const auto& op : ops )
256  operation_get_required_authorities( op, required_active, required_owner, other );
257 
258  if( !allow_committe )
259  GRAPHENE_ASSERT( required_active.find(GRAPHENE_COMMITTEE_ACCOUNT) == required_active.end(),
260  invalid_committee_approval, "Committee account may only propose transactions" );
261 
262  sign_state s(sigs,get_active);
263  s.max_recursion = max_recursion_depth;
264  for( auto& id : active_aprovals )
265  s.approved_by.insert( id );
266  for( auto& id : owner_approvals )
267  s.approved_by.insert( id );
268 
269  for( const auto& auth : other )
270  {
271  GRAPHENE_ASSERT( s.check_authority(&auth), tx_missing_other_auth, "Missing Authority", ("auth",auth)("sigs",sigs) );
272  }
273 
274  // fetch all of the top level authorities
275  for( auto id : required_active )
276  {
277  GRAPHENE_ASSERT( s.check_authority(id) ||
278  s.check_authority(get_owner(id)),
279  tx_missing_active_auth, "Missing Active Authority ${id}", ("id",id)("auth",*get_active(id))("owner",*get_owner(id)) );
280  }
281 
282  for( auto id : required_owner )
283  {
284  GRAPHENE_ASSERT( owner_approvals.find(id) != owner_approvals.end() ||
285  s.check_authority(get_owner(id)),
286  tx_missing_owner_auth, "Missing Owner Authority ${id}", ("id",id)("auth",*get_owner(id)) );
287  }
288 
289  GRAPHENE_ASSERT(
290  !s.remove_unused_signatures(),
291  tx_irrelevant_sig,
292  "Unnecessary signature(s) detected"
293  );
294 } FC_CAPTURE_AND_RETHROW( (ops)(sigs) ) }

314  const chain_id_type& chain_id,
315  const flat_set<public_key_type>& available_keys,
316  const std::function<const authority*(account_id_type)>& get_active,
317  const std::function<const authority*(account_id_type)>& get_owner,

324 
325 
326  sign_state s(get_signature_keys( chain_id ),get_active,available_keys);
327  s.max_recursion = max_recursion_depth;
328 
329  for( const auto& auth : other )
330  s.check_authority(&auth);
331  for( auto& owner : required_owner )
332  s.check_authority( get_owner( owner ) );
333  for( auto& active : required_active )
334  s.check_authority( active );
335 
336  s.remove_unused_signatures();

339 
340  for( auto& provided_sig : s.provided_signatures )
341  if( available_keys.find( provided_sig.first ) != available_keys.end() )
342  result.insert( provided_sig.first );
343 
344  return result;

347 set<public_key_type> signed_transaction::minimize_required_signatures(
348  const chain_id_type& chain_id,
349  const flat_set<public_key_type>& available_keys,
350  const std::function<const authority*(account_id_type)>& get_active,
