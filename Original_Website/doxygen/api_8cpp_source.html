53  bool login_api::login(const string& user, const string& password)
54  {
55  optional< api_access_info > acc = _app.get_api_access_info( user );
56  if( !acc.valid() )
57  return false;
58  if( acc->password_hash_b64 != "*" )
59  {
60  std::string password_salt = fc::base64_decode( acc->password_salt_b64 );
61  std::string acc_password_hash = fc::base64_decode( acc->password_hash_b64 );
62 
63  fc::sha256 hash_obj = fc::sha256::hash( password + password_salt );
64  if( hash_obj.data_size() != acc_password_hash.length() )
65  return false;
66  if( memcmp( hash_obj.data(), acc_password_hash.c_str(), hash_obj.data_size() ) != 0 )
67  return false;
68  }
69 
70  for( const std::string& api_name : acc->allowed_apis )
71  enable_api( api_name );
72  return true;
73  }

101  void network_broadcast_api::on_applied_block( const signed_block& b )
102  {
103  if( _callbacks.size() )
104  {
106  auto capture_this = shared_from_this();
107  for( uint32_t trx_num = 0; trx_num < b.transactions.size(); ++trx_num )
108  {
109  const auto& trx = b.transactions[trx_num];
110  auto id = trx.id();
111  auto itr = _callbacks.find(id);
112  if( itr != _callbacks.end() )
113  {
114  auto block_num = b.block_num();
115  auto& callback = _callbacks.find(id)->second;
116  fc::async( [capture_this,this,id,block_num,trx_num,trx,callback](){ callback( fc::variant(transaction_confirmation{ id, block_num, trx_num, trx}) ); } );
117  }
118  }
119  }

203  vector<account_id_type> get_relevant_accounts( const object* obj )
204  {
205  vector<account_id_type> result;
206  if( obj->id.space() == protocol_ids )
207  {
208  switch( (object_type)obj->id.type() )

211  case base_object_type:
212  case OBJECT_TYPE_COUNT:

214  case account_object_type:{

217  } case asset_object_type:{

219  assert( aobj != nullptr );
220  result.push_back( aobj->issuer );
221  break;
222  } case force_settlement_object_type:{
223  const auto& aobj = dynamic_cast<const force_settlement_object*>(obj);

229  assert( aobj != nullptr );

231  break;

233  const auto& aobj = dynamic_cast<const witness_object*>(obj);
234  assert( aobj != nullptr );
235  result.push_back( aobj->witness_account );
236  break;
237  } case limit_order_object_type:{
238  const auto& aobj = dynamic_cast<const limit_order_object*>(obj);
239  assert( aobj != nullptr );
240  result.push_back( aobj->seller );
241  break;
242  } case call_order_object_type:{
243  const auto& aobj = dynamic_cast<const call_order_object*>(obj);
244  assert( aobj != nullptr );
245  result.push_back( aobj->borrower );
246  break;
247  } case custom_object_type:{
248  } case proposal_object_type:{
249  const auto& aobj = dynamic_cast<const proposal_object*>(obj);
250  assert( aobj != nullptr );
251  flat_set<account_id_type> impacted;
252  transaction_get_impacted_accounts( aobj->proposed_transaction, impacted );

255  break;

258  assert( aobj != nullptr );

262  for( auto& item : impacted ) result.emplace_back(item);
263  break;

270  } case vesting_balance_object_type:{
271  const auto& aobj = dynamic_cast<const vesting_balance_object*>(obj);

275  } case worker_object_type:{

278  result.push_back( aobj->worker_account );
279  break;
280  } case balance_object_type:{
282  }
283  }
284  }
285  else if( obj->id.space() == implementation_ids )
286  {
287  switch( (impl_object_type)obj->id.type() )
288  {
289  case impl_global_property_object_type:{
290  } case impl_dynamic_global_property_object_type:{
291  } case impl_reserved0_object_type:{
292  } case impl_asset_dynamic_data_type:{

295  } case impl_account_balance_object_type:{

331  vector<order_history_object> history_api::get_fill_order_history( asset_id_type a, asset_id_type b, uint32_t limit )const
332  {
333  FC_ASSERT(_app.chain_database());
334  const auto& db = *_app.chain_database();
335  if( a > b ) std::swap(a,b);
336  const auto& history_idx = db.get_index_type<graphene::market_history::history_index>().indices().get<by_key>();
337  history_key hkey;
338  hkey.base = a;
339  hkey.quote = b;
340  hkey.sequence = std::numeric_limits<int64_t>::min();
341 
342  uint32_t count = 0;
343  auto itr = history_idx.lower_bound( hkey );
344  vector<order_history_object> result;
345  while( itr != history_idx.end() )
346  {
347  if( itr->key.base != a || itr->key.quote != b ) break;
348  result.push_back( *itr );
349  ++itr;
350  ++count;
351  if( count > limit ) break;
352  }
353 
354  return result;
355  }

357  vector<operation_history_object> history_api::get_account_history(account_id_type account, operation_history_id_type stop, unsigned limit, operation_history_id_type start) const
358  {
359  FC_ASSERT(_app.chain_database());
360  const auto& db = *_app.chain_database();
361  FC_ASSERT(limit <= 100);
362  vector<operation_history_object> result;
363  const auto& stats = account(db).statistics(db);
364  if(stats.most_recent_op == account_transaction_history_id_type()) return result;
365  const account_transaction_history_object* node = &stats.most_recent_op(db);
366  if(start == operation_history_id_type())
367  start = node->id;
368  while(node && node->operation_id.instance.value > stop.instance.value && result.size() < limit)
369  {
370  if(node->id.instance() <= start.instance.value)
371  result.push_back(node->operation_id(db));
372  if(node->next == account_transaction_history_id_type())
373  node = nullptr;
374  else node = db.find(node->next);
375  }
376  return result;
377  }
