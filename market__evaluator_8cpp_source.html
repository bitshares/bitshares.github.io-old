41 void_result limit_order_create_evaluator::do_evaluate(const limit_order_create_operation& op)
42 { try {
43  const database& d = db();
44 
45  FC_ASSERT( op.expiration >= d.head_block_time() );
46 
47  _seller = this->fee_paying_account;
48  _sell_asset = &op.amount_to_sell.asset_id(d);
49  _receive_asset = &op.min_to_receive.asset_id(d);
50 
51  if( _sell_asset->options.whitelist_markets.size() )
52  FC_ASSERT( _sell_asset->options.whitelist_markets.find(_receive_asset->id) != _sell_asset->options.whitelist_markets.end() );
53  if( _sell_asset->options.blacklist_markets.size() )
54  FC_ASSERT( _sell_asset->options.blacklist_markets.find(_receive_asset->id) == _sell_asset->options.blacklist_markets.end() );
55 
56  FC_ASSERT( is_authorized_asset( d, *_seller, *_sell_asset ) );
57  FC_ASSERT( is_authorized_asset( d, *_seller, *_receive_asset ) );
58 
59  FC_ASSERT( d.get_balance( *_seller, *_sell_asset ) >= op.amount_to_sell, "insufficient balance",
60  ("balance",d.get_balance(*_seller,*_sell_asset))("amount_to_sell",op.amount_to_sell) );
61 
62  return void_result();
63 } FC_CAPTURE_AND_RETHROW( (op) ) }

104

106  obj.seller = _seller->id;
107  obj.for_sale = op.amount_to_sell.amount;
108  obj.sell_price = op.get_price();
109  obj.expiration = op.expiration;

194 void_result call_order_update_evaluator::do_apply(const call_order_update_operation& o)
195 { try {
196  database& d = db();
197 
198  if( o.delta_debt.amount != 0 )
199  {
200  d.adjust_balance( o.funding_account, o.delta_debt );
201 
202  // Deduct the debt paid from the total supply of the debt asset.
203  d.modify(_debt_asset->dynamic_asset_data_id(d), [&](asset_dynamic_data_object& dynamic_asset) {
204  dynamic_asset.current_supply += o.delta_debt.amount;
205  FC_ASSERT(dynamic_asset.current_supply >= 0);
206  });
207  }
208 
209  if( o.delta_collateral.amount != 0 )
210  {
211  d.adjust_balance( o.funding_account, -o.delta_collateral );
212 
213  // Adjust the total core in orders accodingly
214  if( o.delta_collateral.asset_id == asset_id_type() )
215  {
216  d.modify(_paying_account->statistics(d), [&](account_statistics_object& stats) {
217  stats.total_core_in_orders += o.delta_collateral.amount;
218  });
219  }
220  }
221 
222 
223  auto& call_idx = d.get_index_type<call_order_index>().indices().get<by_account>();
224  auto itr = call_idx.find( boost::make_tuple(o.funding_account, o.delta_debt.asset_id) );
225  const call_order_object* call_obj = nullptr;
226 
227  optional<price> old_collateralization;
228  optional<share_type> old_debt;
229 
230  optional<uint16_t> new_target_cr = o.extensions.value.target_collateral_ratio;
231 
232  if( itr == call_idx.end() )
233  {
234  FC_ASSERT( o.delta_collateral.amount > 0 );
235  FC_ASSERT( o.delta_debt.amount > 0 );
236 
237  call_obj = &d.create<call_order_object>( [&](call_order_object& call ){
238  call.borrower = o.funding_account;
239  call.collateral = o.delta_collateral.amount;
240  call.debt = o.delta_debt.amount;
241  call.call_price = price::call_price(o.delta_debt, o.delta_collateral,
242  _bitasset_data->current_feed.maintenance_collateral_ratio);
243  call.target_collateral_ratio = new_target_cr;
244  });
245  }
246  else
247  {
248  call_obj = &*itr;
249  old_collateralization = call_obj->collateralization();
250  old_debt = call_obj->debt;
251  auto new_collateral = call_obj->collateral + o.delta_collateral.amount;
252  auto new_debt = *old_debt + o.delta_debt.amount;
253 
254  // Forbid zero collateral with nonzero debt (avoids divide by zero when calculating call price below)
255  FC_ASSERT(!(new_collateral == 0 && new_debt != 0), "Cannot have zero collateral and nonzero debt");
256 
257  d.modify( *call_obj, [&]( call_order_object& call ){
258  call.collateral += o.delta_collateral.amount;
259  call.debt += o.delta_debt.amount;
260  if( call.debt > 0 )
261  {
262  call.call_price = price::call_price(call.get_debt(), call.get_collateral(),
263  _bitasset_data->current_feed.maintenance_collateral_ratio);
264  }
265  call.target_collateral_ratio = new_target_cr;
266  });
267  }
268 
269  if( call_obj->debt == 0 )
270  {
271  FC_ASSERT( call_obj->collateral == 0 );
272  d.remove( *call_obj );
273  return void_result();
274  }
275 
276  FC_ASSERT(call_obj->collateral > 0 && call_obj->debt > 0);
277 
278  // then we must check for margin calls and other issues
279  if( !_bitasset_data->is_prediction_market )
280  {
281  call_order_id_type call_order_id = call_obj->id;
282 
283  // check to see if the order needs to be margin called now, but don't allow black swans and require there to be
284  // limit orders available that could be used to fill the order.
285  // Note: due to https://github.com/bitshares/bitshares-core/issues/649,
286  // the first call order may be unable to be updated if the second one is undercollateralized.
287  if( d.check_call_orders( *_debt_asset, false ) )
288  {
289  const auto call_obj = d.find(call_order_id);
290  // before hard fork core-583: if we filled at least one call order, we are OK if we totally filled.
291  // after hard fork core-583: we want to allow increasing collateral
292  // Note: increasing collateral won't get the call order itself matched (instantly margin called)
293  // if there is at least a call order get matched but didn't cause a black swan event,
294  // current order must have got matched. in this case, it's OK if it's totally filled.
295  GRAPHENE_ASSERT(
296  !call_obj,
297  call_order_update_unfilled_margin_call,
298  "Updating call order would trigger a margin call that cannot be fully filled",
299  ("a", ~call_obj->call_price )("b", _bitasset_data->current_feed.settlement_price)
300  );
301  }
302  else
303  {
304  const auto call_obj = d.find(call_order_id);
305  // we know no black swan event has occurred
306  FC_ASSERT( call_obj, "no margin call was executed and yet the call object was deleted" );
307  if( d.head_block_time() <= HARDFORK_CORE_583_TIME ) // TODO remove after hard fork core-583
308  {
309  // We didn't fill any call orders. This may be because we
310  // aren't in margin call territory, or it may be because there
311  // were no matching orders. In the latter case, we throw.
312  GRAPHENE_ASSERT(
313  ~call_obj->call_price < _bitasset_data->current_feed.settlement_price,
314  call_order_update_unfilled_margin_call,
315  "Updating call order would trigger a margin call that cannot be fully filled",
316  ("a", ~call_obj->call_price )("b", _bitasset_data->current_feed.settlement_price)
317  );
318  }
319  else // after hard fork, always allow call order to be updated if collateral ratio is increased and debt is not increased
320  {
321  // We didn't fill any call orders. This may be because we
322  // aren't in margin call territory, or it may be because there
323  // were no matching orders. In the latter case,
324  // if collateral ratio is not increased or debt is increased, we throw.
325  // be here, we know no margin call was executed,
326  // so call_obj's collateral ratio should be set only by op
327  FC_ASSERT( ( old_collateralization.valid() && call_obj->debt <= *old_debt
328  && call_obj->collateralization() > *old_collateralization )
329  || ~call_obj->call_price < _bitasset_data->current_feed.settlement_price,
330  "Can only increase collateral ratio without increasing debt if would trigger a margin call that cannot be fully filled",
331  ("new_call_price", ~call_obj->call_price )
332  ("settlement_price", _bitasset_data->current_feed.settlement_price)
333  ("old_debt", old_debt)
334  ("new_debt", call_obj->debt)
335  ("old_collateralization", old_collateralization)
336  ("new_collateralization", call_obj->collateralization() )
337  );
338  }
339  }
340  }
341 
342  return void_result();
343 } FC_CAPTURE_AND_RETHROW( (o) ) }

346 { try {
347  database& d = db();
348 
349  FC_ASSERT( d.head_block_time() > HARDFORK_CORE_216_TIME, "Not yet!" );
350

352  _debt_asset = &o.debt_covered.asset_id(d);

355

357 
358  FC_ASSERT( _bitasset_data->has_settlement() );

361 
362  FC_ASSERT( !_bitasset_data->is_prediction_market, "Cannot bid on a prediction market!" );
363 
364  if( o.additional_collateral.amount > 0 )
365  {
366  FC_ASSERT( d.get_balance(*_paying_account, _bitasset_data->options.short_backing_asset(d)) >= o.additional_collateral,
367  "Cannot bid ${c} collateral when payer only has ${b}", ("c", o.additional_collateral.amount)
368  ("b", d.get_balance(*_paying_account, o.additional_collateral.asset_id(d)).amount) );
369  }
370 
371  const collateral_bid_index& bids = d.get_index_type<collateral_bid_index>();
372  const auto& index = bids.indices().get<by_account>();
373  const auto& bid = index.find( boost::make_tuple( o.debt_covered.asset_id, o.bidder ) );
374  if( bid != index.end() )
375  _bid = &(*bid);
376  else
377  FC_ASSERT( o.debt_covered.amount > 0, "Can't find bid to cancel?!");
378 
379  return void_result();
380 } FC_CAPTURE_AND_RETHROW( (o) ) }
