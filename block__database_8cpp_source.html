78 void block_database::store( const block_id_type& _id, const signed_block& b )
79 {
80  block_id_type id = _id;
81  if( id == block_id_type() )
82  {
83  id = b.id();
84  elog( "id argument of block_database::store() was not initialized for block ${id}", ("id", id) );
85  }
86  _block_num_to_pos.seekp( sizeof( index_entry ) * int64_t(block_header::num_from_id(id)) );
87  index_entry e;
88  _blocks.seekp( 0, _blocks.end );
89  auto vec = fc::raw::pack( b );
90  e.block_pos = _blocks.tellp();
91  e.block_size = vec.size();
92  e.block_id = id;
93  _blocks.write( vec.data(), vec.size() );
94  _block_num_to_pos.write( (char*)&e, sizeof(e) );
95 }

116 bool block_database::contains( const block_id_type& id )const
117 {
118  if( id == block_id_type() )
119  return false;
120 
121  index_entry e;
122  int64_t index_pos = sizeof(e) * int64_t(block_header::num_from_id(id));
123  _block_num_to_pos.seekg( 0, _block_num_to_pos.end );
124  if ( _block_num_to_pos.tellg() < int64_t(index_pos + sizeof(e)) )
125  return false;
126  _block_num_to_pos.seekg( index_pos );
127  _block_num_to_pos.read( (char*)&e, sizeof(e) );
128 
129  return e.block_id == id && e.block_size > 0;
130 }

209 optional<index_entry> block_database::last_index_entry()const {
210  try
211  {
212  index_entry e;
213 
214  _block_num_to_pos.seekg( 0, _block_num_to_pos.end );
215  std::streampos pos = _block_num_to_pos.tellg();
216  if( pos < long(sizeof(index_entry)) )
217  return optional<index_entry>();
218 
219  pos -= pos % sizeof(index_entry);
220 
221  _blocks.seekg( 0, _block_num_to_pos.end );
222  const std::streampos blocks_size = _blocks.tellg();
223  while( pos > 0 )
224  {
225  pos -= sizeof(index_entry);
226  _block_num_to_pos.seekg( pos );
227  _block_num_to_pos.read( (char*)&e, sizeof(e) );
228  if( _block_num_to_pos.gcount() == sizeof(e) && e.block_size > 0
229  && int64_t(e.block_pos + e.block_size) <= blocks_size )
230  try
231  {
232  vector<char> data( e.block_size );
233  _blocks.seekg( e.block_pos );
234  _blocks.read( data.data(), e.block_size );
235  if( _blocks.gcount() == long(e.block_size) )
236  {
237  const signed_block block = fc::raw::unpack<signed_block>(data);
238  if( block.id() == e.block_id )
239  return e;
240  }
241  }
242  catch (const fc::exception&)
243  {
244  }
245  catch (const std::exception&)
246  {
247  }
248  fc::resize_file( _index_filename, pos );
249  }
250  }
251  catch (const fc::exception&)
252  {
253  }
254  catch (const std::exception&)
255  {
256  }
257  return optional<index_entry>();
258 }
