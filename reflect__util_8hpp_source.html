40 std::string clean_name( const std::string& name )
41 {
42  const static std::string prefix = "graphene::chain::";
43  const static std::string suffix = "_operation";
44  // graphene::chain::.*_operation
45  if( (name.size() >= prefix.size() + suffix.size())
46  && (name.substr( 0, prefix.size() ) == prefix)
47  && (name.substr( name.size()-suffix.size(), suffix.size() ) == suffix )
48  )
49  return name.substr( prefix.size(), name.size() - prefix.size() - suffix.size() );
50 
51  wlog( "don't know how to clean name: ${name}", ("name", name) );
52  return name;
53 }

95 template< typename T >
96 T from_which_variant( int which, const variant& v, uint32_t max_depth )
97 {
98  // Parse a variant for a known which()
99  T dummy;
100  dummy.set_which( which );
101  impl::from_which_visitor< T > vtor(v, max_depth);
102  return dummy.visit( vtor );
103 }

105 template<typename T>
106 static_variant_map create_static_variant_map()
107 {
108  T dummy;
109  int n = dummy.count();
110  FC_ASSERT( n <= std::numeric_limits<uint16_t>::max(), "Too many items in this static_variant" );
111  impl::static_variant_map_visitor vtor;
112  for( int i=0; i<n; i++ )
113  {
114  dummy.set_which(i);
115  vtor.which = i;
116  dummy.visit( vtor );
117  }
118  return vtor.m;
119 }
