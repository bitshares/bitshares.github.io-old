43 void database::update_global_dynamic_data( const signed_block& b, const uint32_t missed_blocks )
44 {
45  const dynamic_global_property_object& _dgp = get_dynamic_global_properties();
46 
47  // dynamic global properties updating
48  modify( _dgp, [&b,this,missed_blocks]( dynamic_global_property_object& dgp ){
49  const uint32_t block_num = b.block_num();
50  if( BOOST_UNLIKELY( block_num == 1 ) )
51  dgp.recently_missed_count = 0;
52  else if( _checkpoints.size() && _checkpoints.rbegin()->first >= block_num )
53  dgp.recently_missed_count = 0;
54  else if( missed_blocks )
55  dgp.recently_missed_count += GRAPHENE_RECENTLY_MISSED_COUNT_INCREMENT*missed_blocks;
56  else if( dgp.recently_missed_count > GRAPHENE_RECENTLY_MISSED_COUNT_INCREMENT )
57  dgp.recently_missed_count -= GRAPHENE_RECENTLY_MISSED_COUNT_DECREMENT;
58  else if( dgp.recently_missed_count > 0 )
59  dgp.recently_missed_count--;
60 
61  dgp.head_block_number = block_num;
62  dgp.head_block_id = b.id();
63  dgp.time = b.timestamp;
64  dgp.current_witness = b.witness;
65  dgp.recent_slots_filled = (
66  (dgp.recent_slots_filled << 1)
67  + 1) << missed_blocks;
68  dgp.current_aslot += missed_blocks+1;
69  });
70 
71  if( !(get_node_properties().skip_flags & skip_undo_history_check) )
72  {
73  GRAPHENE_ASSERT( _dgp.head_block_number - _dgp.last_irreversible_block_num < GRAPHENE_MAX_UNDO_HISTORY, undo_database_exception,
74  "The database does not have enough undo history to support a blockchain with so many missed blocks. "
75  "Please add a checkpoint if you would like to continue applying blocks beyond this point.",
76  ("last_irreversible_block_num",_dgp.last_irreversible_block_num)("head", _dgp.head_block_number)
77  ("recently_missed",_dgp.recently_missed_count)("max_undo",GRAPHENE_MAX_UNDO_HISTORY) );
78  }
79 
80  _undo_db.set_max_size( _dgp.head_block_number - _dgp.last_irreversible_block_num + 1 );
81  _fork_db.set_max_size( _dgp.head_block_number - _dgp.last_irreversible_block_num + 1 );
82 }

106 void database::update_last_irreversible_block()
107 {
108  const global_property_object& gpo = get_global_properties();
109  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
110 
111  // TODO for better performance, move this to db_maint, because only need to do it once per maintenance interval
112  vector< const witness_object* > wit_objs;
113  wit_objs.reserve( gpo.active_witnesses.size() );
114  for( const witness_id_type& wid : gpo.active_witnesses )
115  wit_objs.push_back( &(wid(*this)) );
116 
117  static_assert( GRAPHENE_IRREVERSIBLE_THRESHOLD > 0, "irreversible threshold must be nonzero" );
118 
119  // 1 1 1 2 2 2 2 2 2 2 -> 2 .7*10 = 7
120  // 1 1 1 1 1 1 1 2 2 2 -> 1
121  // 3 3 3 3 3 3 3 3 3 3 -> 3
122 
123  size_t offset = ((GRAPHENE_100_PERCENT - GRAPHENE_IRREVERSIBLE_THRESHOLD) * wit_objs.size() / GRAPHENE_100_PERCENT);
124 
125  std::nth_element( wit_objs.begin(), wit_objs.begin() + offset, wit_objs.end(),
126  []( const witness_object* a, const witness_object* b )
127  {
128  return a->last_confirmed_block_num < b->last_confirmed_block_num;
129  } );
130 
131  uint32_t new_last_irreversible_block_num = wit_objs[offset]->last_confirmed_block_num;
132 
133  if( new_last_irreversible_block_num > dpo.last_irreversible_block_num )
134  {
135  modify( dpo, [&]( dynamic_global_property_object& _dpo )
136  {
137  _dpo.last_irreversible_block_num = new_last_irreversible_block_num;
138  } );
139  }

183 bool database::check_for_blackswan( const asset_object& mia, bool enable_black_swan,
184  const asset_bitasset_data_object* bitasset_ptr )
185 {
186  if( !mia.is_market_issued() ) return false;
187 
188  const asset_bitasset_data_object& bitasset = ( bitasset_ptr ? *bitasset_ptr : mia.bitasset_data(*this) );
189  if( bitasset.has_settlement() ) return true; // already force settled
190  auto settle_price = bitasset.current_feed.settlement_price;
191  if( settle_price.is_null() ) return false; // no feed
192 
193  const call_order_index& call_index = get_index_type<call_order_index>();
194  const auto& call_price_index = call_index.indices().get<by_price>();
195 
196  auto call_min = price::min( bitasset.options.short_backing_asset, mia.id );
197  auto call_max = price::max( bitasset.options.short_backing_asset, mia.id );
198  auto call_itr = call_price_index.lower_bound( call_min );
199  auto call_end = call_price_index.upper_bound( call_max );
200 
201  if( call_itr == call_end ) return false; // no call orders
202 
203  price highest = settle_price;
204 
205  const auto& dyn_prop = get_dynamic_global_properties();
206  auto maint_time = dyn_prop.next_maintenance_time;
207  if( maint_time > HARDFORK_CORE_338_TIME )
208  // due to #338, we won't check for black swan on incoming limit order, so need to check with MSSP here
209  highest = bitasset.current_feed.max_short_squeeze_price();
210 
211  const limit_order_index& limit_index = get_index_type<limit_order_index>();
212  const auto& limit_price_index = limit_index.indices().get<by_price>();
213 
214  // looking for limit orders selling the most USD for the least CORE
215  auto highest_possible_bid = price::max( mia.id, bitasset.options.short_backing_asset );
216  // stop when limit orders are selling too little USD for too much CORE
217  auto lowest_possible_bid = price::min( mia.id, bitasset.options.short_backing_asset );
218 
219  FC_ASSERT( highest_possible_bid.base.asset_id == lowest_possible_bid.base.asset_id );
220  // NOTE limit_price_index is sorted from greatest to least
221  auto limit_itr = limit_price_index.lower_bound( highest_possible_bid );
222  auto limit_end = limit_price_index.upper_bound( lowest_possible_bid );
223 
224  if( limit_itr != limit_end ) {
225  FC_ASSERT( highest.base.asset_id == limit_itr->sell_price.base.asset_id );
226  highest = std::max( limit_itr->sell_price, highest );
227  }
228 
229  auto least_collateral = call_itr->collateralization();
230  if( ~least_collateral >= highest )
231  {
232  wdump( (*call_itr) );
233  elog( "Black Swan detected on asset ${symbol} (${id}) at block ${b}: \n"
234  " Least collateralized call: ${lc} ${~lc}\n"
235  // " Highest Bid: ${hb} ${~hb}\n"
236  " Settle Price: ${~sp} ${sp}\n"
237  " Max: ${~h} ${h}\n",
238  ("id",mia.id)("symbol",mia.symbol)("b",head_block_num())
239  ("lc",least_collateral.to_real())("~lc",(~least_collateral).to_real())
240  // ("hb",limit_itr->sell_price.to_real())("~hb",(~limit_itr->sell_price).to_real())
241  ("sp",settle_price.to_real())("~sp",(~settle_price).to_real())
242  ("h",highest.to_real())("~h",(~highest).to_real()) );
243  edump((enable_black_swan));
244  FC_ASSERT( enable_black_swan, "Black swan was detected during a margin update which is not allowed to trigger a blackswan" );
245  if( maint_time > HARDFORK_CORE_338_TIME && ~least_collateral <= settle_price )
246  // global settle at feed price if possible
247  globally_settle_asset(mia, settle_price );
248  else
249  globally_settle_asset(mia, ~least_collateral );
250  return true;
251  }
252  return false;
253 }

255 void database::clear_expired_orders()
256 { try {
257  //Cancel expired limit orders
258  auto head_time = head_block_time();
259  auto maint_time = get_dynamic_global_properties().next_maintenance_time;
260 
261  bool before_core_hardfork_184 = ( maint_time <= HARDFORK_CORE_184_TIME ); // something-for-nothing
262  bool before_core_hardfork_342 = ( maint_time <= HARDFORK_CORE_342_TIME ); // better rounding
263  bool before_core_hardfork_606 = ( maint_time <= HARDFORK_CORE_606_TIME ); // feed always trigger call
264 
265  auto& limit_index = get_index_type<limit_order_index>().indices().get<by_expiration>();
266  while( !limit_index.empty() && limit_index.begin()->expiration <= head_time )
267  {
268  const limit_order_object& order = *limit_index.begin();
269  auto base_asset = order.sell_price.base.asset_id;
270  auto quote_asset = order.sell_price.quote.asset_id;
271  cancel_limit_order( order );
272  if( before_core_hardfork_606 )
273  {
274  // check call orders
275  // Comments below are copied from limit_order_cancel_evaluator::do_apply(...)
276  // Possible optimization: order can be called by cancelling a limit order
277  // if the canceled order was at the top of the book.
278  // Do I need to check calls in both assets?
279  check_call_orders( base_asset( *this ) );
280  check_call_orders( quote_asset( *this ) );
281  }
282  }
283 
284  //Process expired force settlement orders
285  auto& settlement_index = get_index_type<force_settlement_index>().indices().get<by_expiration>();
286  if( !settlement_index.empty() )
287  {
288  asset_id_type current_asset = settlement_index.begin()->settlement_asset_id();
289  asset max_settlement_volume;
290  price settlement_fill_price;
291  price settlement_price;
292  bool current_asset_finished = false;
293  bool extra_dump = false;
294 
295  auto next_asset = [&current_asset, &current_asset_finished, &settlement_index, &extra_dump] {
296  auto bound = settlement_index.upper_bound(current_asset);
297  if( bound == settlement_index.end() )
298  {
299  if( extra_dump )
300  {
301  ilog( "next_asset() returning false" );
302  }
303  return false;
304  }
305  if( extra_dump )
306  {
307  ilog( "next_asset returning true, bound is ${b}", ("b", *bound) );
308  }
309  current_asset = bound->settlement_asset_id();
310  current_asset_finished = false;
311  return true;
312  };
313 
314  uint32_t count = 0;
315 
316  // At each iteration, we either consume the current order and remove it, or we move to the next asset
317  for( auto itr = settlement_index.lower_bound(current_asset);
318  itr != settlement_index.end();
319  itr = settlement_index.lower_bound(current_asset) )
320  {
321  ++count;
322  const force_settlement_object& order = *itr;
323  auto order_id = order.id;
324  current_asset = order.settlement_asset_id();
325  const asset_object& mia_object = get(current_asset);
326  const asset_bitasset_data_object& mia = mia_object.bitasset_data(*this);
327 
328  extra_dump = ((count >= 1000) && (count <= 1020));
329 
330  if( extra_dump )
331  {
332  wlog( "clear_expired_orders() dumping extra data for iteration ${c}", ("c", count) );
333  ilog( "head_block_num is ${hb} current_asset is ${a}", ("hb", head_block_num())("a", current_asset) );
334  }
335 
336  if( mia.has_settlement() )
337  {
338  ilog( "Canceling a force settlement because of black swan" );
339  cancel_settle_order( order );
340  continue;
341  }
342 
343  // Has this order not reached its settlement date?
344  if( order.settlement_date > head_time )
345  {
346  if( next_asset() )
347  {
348  if( extra_dump )
349  {
350  ilog( "next_asset() returned true when order.settlement_date > head_block_time()" );
351  }
352  continue;
353  }
354  break;
355  }
356  // Can we still settle in this asset?
357  if( mia.current_feed.settlement_price.is_null() )
358  {
359  ilog("Canceling a force settlement in ${asset} because settlement price is null",
360  ("asset", mia_object.symbol));
361  cancel_settle_order(order);
362  continue;
363  }
364  if( max_settlement_volume.asset_id != current_asset )
365  max_settlement_volume = mia_object.amount(mia.max_force_settlement_volume(mia_object.dynamic_data(*this).current_supply));
366  // When current_asset_finished is true, this would be the 2nd time processing the same order.
367  // In this case, we move to the next asset.
368  if( mia.force_settled_volume >= max_settlement_volume.amount || current_asset_finished )
369  {
370  /*
371  ilog("Skipping force settlement in ${asset}; settled ${settled_volume} / ${max_volume}",
372  ("asset", mia_object.symbol)("settlement_price_null",mia.current_feed.settlement_price.is_null())
373  ("settled_volume", mia.force_settled_volume)("max_volume", max_settlement_volume));
374  */
375  if( next_asset() )
376  {
377  if( extra_dump )
378  {
379  ilog( "next_asset() returned true when mia.force_settled_volume >= max_settlement_volume.amount" );
380  }
381  continue;
382  }
383  break;
384  }
385 
386  if( settlement_fill_price.base.asset_id != current_asset ) // only calculate once per asset
387  settlement_fill_price = mia.current_feed.settlement_price
388  / ratio_type( GRAPHENE_100_PERCENT - mia.options.force_settlement_offset_percent,
389  GRAPHENE_100_PERCENT );
390 
391  if( before_core_hardfork_342 )
392  {
393  auto& pays = order.balance;
394  auto receives = (order.balance * mia.current_feed.settlement_price);
395  receives.amount = ( fc::uint128_t(receives.amount.value) *
396  (GRAPHENE_100_PERCENT - mia.options.force_settlement_offset_percent) /
397  GRAPHENE_100_PERCENT ).to_uint64();
398  assert(receives <= order.balance * mia.current_feed.settlement_price);
399  settlement_price = pays / receives;
400  }
401  else if( settlement_price.base.asset_id != current_asset ) // only calculate once per asset
402  settlement_price = settlement_fill_price;
403 
404  auto& call_index = get_index_type<call_order_index>().indices().get<by_collateral>();
405  asset settled = mia_object.amount(mia.force_settled_volume);
406  // Match against the least collateralized short until the settlement is finished or we reach max settlements
407  while( settled < max_settlement_volume && find_object(order_id) )
408  {
409  auto itr = call_index.lower_bound(boost::make_tuple(price::min(mia_object.bitasset_data(*this).options.short_backing_asset,
410  mia_object.get_id())));
411  // There should always be a call order, since asset exists!
412  assert(itr != call_index.end() && itr->debt_type() == mia_object.get_id());
413  asset max_settlement = max_settlement_volume - settled;
414 
415  if( order.balance.amount == 0 )
416  {
417  wlog( "0 settlement detected" );
418  cancel_settle_order( order );
419  break;
420  }
421  try {
422  asset new_settled = match(*itr, order, settlement_price, max_settlement, settlement_fill_price);
423  if( !before_core_hardfork_184 && new_settled.amount == 0 ) // unable to fill this settle order
424  {
425  if( find_object( order_id ) ) // the settle order hasn't been cancelled
426  current_asset_finished = true;
427  break;
428  }
429  settled += new_settled;
430  // before hard fork core-342, `new_settled > 0` is always true, we'll have:
431  // * call order is completely filled (thus itr will change in next loop), or
432  // * settle order is completely filled (thus find_object(order_id) will be false so will break out), or
433  // * reached max_settlement_volume limit (thus new_settled == max_settlement so will break out).
434  //
435  // after hard fork core-342, if new_settled > 0, we'll have:
436  // * call order is completely filled (thus itr will change in next loop), or
437  // * settle order is completely filled (thus find_object(order_id) will be false so will break out), or
438  // * reached max_settlement_volume limit, but it's possible that new_settled < max_settlement,
439  // in this case, new_settled will be zero in next iteration of the loop, so no need to check here.
440  }
441  catch ( const black_swan_exception& e ) {
442  wlog( "Cancelling a settle_order since it may trigger a black swan: ${o}, ${e}",
443  ("o", order)("e", e.to_detail_string()) );
444  cancel_settle_order( order );
445  break;
446  }
447  }
448  if( mia.force_settled_volume != settled.amount )
449  {
450  modify(mia, [settled](asset_bitasset_data_object& b) {
451  b.force_settled_volume = settled.amount;
452  });
453  }
454  }
455  }
456 } FC_CAPTURE_AND_RETHROW() }

458 void database::update_expired_feeds()
459 {

462

468  ++itr; // not always process begin() because old code skipped updating some assets before hf 615

470  const asset_object* asset_ptr = nullptr;

486  {

516 void database::update_core_exchange_rates()
517 {
518  const auto& idx = get_index_type<asset_bitasset_data_index>().indices().get<by_cer_update>();
519  if( idx.begin() != idx.end() )
520  {
521  for( auto itr = idx.rbegin(); itr->need_to_update_cer(); itr = idx.rbegin() )
522  {
523  const asset_bitasset_data_object& b = *itr;
524  const asset_object& a = b.asset_id( *this );
525  if( a.options.core_exchange_rate != b.current_feed.core_exchange_rate )
526  {
