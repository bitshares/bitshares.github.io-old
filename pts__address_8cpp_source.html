51  pts_address::pts_address( const fc::ecc::public_key& pub, bool compressed, uint8_t version )
52  {
53  fc::sha256 sha2;
54  if( compressed )
55  {
56  auto dat = pub.serialize();
57  sha2 = fc::sha256::hash(dat.data, sizeof(dat) );
58  }
59  else
60  {
61  auto dat = pub.serialize_ecc_point();
62  sha2 = fc::sha256::hash(dat.data, sizeof(dat) );
63  }
64  auto rep = fc::ripemd160::hash((char*)&sha2,sizeof(sha2));
65  addr.data[0] = version;
66  memcpy( addr.data+1, (char*)&rep, sizeof(rep) );
67  auto check = fc::sha256::hash( addr.data, sizeof(rep)+1 );
68  check = fc::sha256::hash(check);
69  memcpy( addr.data+1+sizeof(rep), (char*)&check, 4 );
70  }

76  bool pts_address::is_valid()const
77  {
78  auto check = fc::sha256::hash( addr.data, sizeof(fc::ripemd160)+1 );
79  check = fc::sha256::hash(check);
80  return memcmp( addr.data+1+sizeof(fc::ripemd160), (char*)&check, 4 ) == 0;
81  }
