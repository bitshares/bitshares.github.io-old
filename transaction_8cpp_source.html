109 struct sign_state
110 {
114  bool signed_by( const public_key_type& k )
115  {
116  auto itr = provided_signatures.find(k);
117  if( itr == provided_signatures.end() )
118  {
119  auto pk = available_keys.find(k);
120  if( pk != available_keys.end() )
121  return provided_signatures[k] = true;
122  return false;
123  }
124  return itr->second = true;
125  }

127  optional<map<address,public_key_type>> available_address_sigs;
128  optional<map<address,public_key_type>> provided_address_sigs;
129 
130  bool signed_by( const address& a ) {
131  if( !available_address_sigs ) {
132  available_address_sigs = std::map<address,public_key_type>();
133  provided_address_sigs = std::map<address,public_key_type>();
134  for( auto& item : available_keys ) {
135  (*available_address_sigs)[ address(pts_address(item, false, 56) ) ] = item;
136  (*available_address_sigs)[ address(pts_address(item, true, 56) ) ] = item;
137  (*available_address_sigs)[ address(pts_address(item, false, 0) ) ] = item;
138  (*available_address_sigs)[ address(pts_address(item, true, 0) ) ] = item;
139  (*available_address_sigs)[ address(item) ] = item;
140  }
141  for( auto& item : provided_signatures ) {
142  (*provided_address_sigs)[ address(pts_address(item.first, false, 56) ) ] = item.first;
143  (*provided_address_sigs)[ address(pts_address(item.first, true, 56) ) ] = item.first;
144  (*provided_address_sigs)[ address(pts_address(item.first, false, 0) ) ] = item.first;
145  (*provided_address_sigs)[ address(pts_address(item.first, true, 0) ) ] = item.first;
146  (*provided_address_sigs)[ address(item.first) ] = item.first;
147  }
148  }
149  auto itr = provided_address_sigs->find(a);
150  if( itr == provided_address_sigs->end() )
151  {
152  auto aitr = available_address_sigs->find(a);
153  if( aitr != available_address_sigs->end() ) {
154  auto pk = available_keys.find(aitr->second);
155  if( pk != available_keys.end() )
156  return provided_signatures[aitr->second] = true;
157  return false;
158  }
159  }
160  return provided_signatures[itr->second] = true;
161  }
162 
163  bool check_authority( account_id_type id )
164  {
165  if( approved_by.find(id) != approved_by.end() ) return true;
166  return check_authority( get_active(id) );
167  }

173  bool check_authority( const authority* au, uint32_t depth = 0 )
174  {
175  if( au == nullptr ) return false;
176  const authority& auth = *au;
177 
178  uint32_t total_weight = 0;
179  for( const auto& k : auth.key_auths )
180  if( signed_by( k.first ) )
181  {
182  total_weight += k.second;
183  if( total_weight >= auth.weight_threshold )
184  return true;
185  }
186 
187  for( const auto& k : auth.address_auths )
188  if( signed_by( k.first ) )
189  {
190  total_weight += k.second;
191  if( total_weight >= auth.weight_threshold )
192  return true;
193  }
194 
195  for( const auto& a : auth.account_auths )
196  {
197  if( approved_by.find(a.first) == approved_by.end() )
198  {
199  if( depth == max_recursion )
200  continue;
201  if( check_authority( get_active( a.first ), depth+1 ) )
202  {
203  approved_by.insert( a.first );
204  total_weight += a.second;
205  if( total_weight >= auth.weight_threshold )
206  return true;
207  }
208  }
209  else
210  {
211  total_weight += a.second;
212  if( total_weight >= auth.weight_threshold )
213  return true;
214  }
215  }
216  return total_weight >= auth.weight_threshold;
217  }

220  {
221  vector<public_key_type> remove_sigs;

250 void verify_authority( const vector<operation>& ops, const flat_set<public_key_type>& sigs,
251  const std::function<const authority*(account_id_type)>& get_active,
252  const std::function<const authority*(account_id_type)>& get_owner,
253  uint32_t max_recursion_depth,
254  bool allow_committe,
255  const flat_set<account_id_type>& active_aprovals,
256  const flat_set<account_id_type>& owner_approvals )
257 { try {
258  flat_set<account_id_type> required_active;
259  flat_set<account_id_type> required_owner;
260  vector<authority> other;
261 
262  for( const auto& op : ops )
263  operation_get_required_authorities( op, required_active, required_owner, other );
264 
265  if( !allow_committe )
266  GRAPHENE_ASSERT( required_active.find(GRAPHENE_COMMITTEE_ACCOUNT) == required_active.end(),
267  invalid_committee_approval, "Committee account may only propose transactions" );
268 
269  sign_state s(sigs,get_active);
270  s.max_recursion = max_recursion_depth;
271  for( auto& id : active_aprovals )
272  s.approved_by.insert( id );
273  for( auto& id : owner_approvals )
274  s.approved_by.insert( id );
275 
276  for( const auto& auth : other )
277  {
278  GRAPHENE_ASSERT( s.check_authority(&auth), tx_missing_other_auth, "Missing Authority", ("auth",auth)("sigs",sigs) );
279  }
280 
281  // fetch all of the top level authorities
282  for( auto id : required_active )
283  {
284  GRAPHENE_ASSERT( s.check_authority(id) ||
285  s.check_authority(get_owner(id)),
286  tx_missing_active_auth, "Missing Active Authority ${id}", ("id",id)("auth",*get_active(id))("owner",*get_owner(id)) );
287  }
288 
289  for( auto id : required_owner )
290  {
291  GRAPHENE_ASSERT( owner_approvals.find(id) != owner_approvals.end() ||
292  s.check_authority(get_owner(id)),
293  tx_missing_owner_auth, "Missing Owner Authority ${id}", ("id",id)("auth",*get_owner(id)) );
294  }
295 
296  GRAPHENE_ASSERT(
297  !s.remove_unused_signatures(),
298  tx_irrelevant_sig,
299  "Unnecessary signature(s) detected"
300  );
301 } FC_CAPTURE_AND_RETHROW( (ops)(sigs) ) }

320 set<public_key_type> signed_transaction::get_required_signatures(
321  const chain_id_type& chain_id,
322  const flat_set<public_key_type>& available_keys,
323  const std::function<const authority*(account_id_type)>& get_active,
324  const std::function<const authority*(account_id_type)>& get_owner,
325  uint32_t max_recursion_depth )const
326 {
327  flat_set<account_id_type> required_active;
328  flat_set<account_id_type> required_owner;
329  vector<authority> other;
330  get_required_authorities( required_active, required_owner, other );
331 
332  flat_set<public_key_type> signature_keys = get_signature_keys( chain_id );
333  sign_state s( signature_keys, get_active, available_keys );
334  s.max_recursion = max_recursion_depth;
335 
336  for( const auto& auth : other )
337  s.check_authority(&auth);
338  for( auto& owner : required_owner )
339  s.check_authority( get_owner( owner ) );
340  for( auto& active : required_active )
341  s.check_authority( active ) || s.check_authority( get_owner( active ) );
342 
343  s.remove_unused_signatures();
344 
345  set<public_key_type> result;
346 
347  for( auto& provided_sig : s.provided_signatures )
348  if( available_keys.find( provided_sig.first ) != available_keys.end()
349  && signature_keys.find( provided_sig.first ) == signature_keys.end() )
350  result.insert( provided_sig.first );
351 
352  return result;
353 }

355 set<public_key_type> signed_transaction::minimize_required_signatures(
356  const chain_id_type& chain_id,
357  const flat_set<public_key_type>& available_keys,
358  const std::function<const authority*(account_id_type)>& get_active,
359  const std::function<const authority*(account_id_type)>& get_owner,
360  uint32_t max_recursion
361  ) const
362 {

364  flat_set< public_key_type > result( s.begin(), s.end() );
365 
366  for( const public_key_type& k : s )
367  {
368  result.erase( k );
369  try
370  {
371  graphene::chain::verify_authority( operations, result, get_active, get_owner, max_recursion );
372  continue; // element stays erased if verify_authority is ok
