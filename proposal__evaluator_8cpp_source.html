34 struct proposal_operation_hardfork_visitor
35 {
36  typedef void result_type;
37  const fc::time_point_sec block_time;
38  const fc::time_point_sec next_maintenance_time;
39 
40  proposal_operation_hardfork_visitor( const fc::time_point_sec bt, const fc::time_point_sec nmt )
41  : block_time(bt), next_maintenance_time(nmt) {}
42 
43  template<typename T>
44  void operator()(const T &v) const {}
45 
46  // TODO review and cleanup code below after hard fork
47  // hf_834
48  void operator()(const graphene::chain::call_order_update_operation &v) const {
49  if (next_maintenance_time <= HARDFORK_CORE_834_TIME) {
50  FC_ASSERT( !v.extensions.value.target_collateral_ratio.valid(),
51  "Can not set target_collateral_ratio in call_order_update_operation before hardfork 834." );
52  }
53  }
54  // hf_620
55  void operator()(const graphene::chain::asset_create_operation &v) const {
56  if (block_time < HARDFORK_CORE_620_TIME) {
57  static const std::locale &loc = std::locale::classic();
58  FC_ASSERT(isalpha(v.symbol.back(), loc), "Asset ${s} must end with alpha character before hardfork 620", ("s", v.symbol));
59  }
60  }
61  // hf_199
62  void operator()(const graphene::chain::asset_update_issuer_operation &v) const {
63  if (block_time < HARDFORK_CORE_199_TIME) {
64  FC_ASSERT(false, "Not allowed until hardfork 199");
65  }
66  }
67  // hf_188
68  void operator()(const graphene::chain::asset_claim_pool_operation &v) const {
69  if (block_time < HARDFORK_CORE_188_TIME) {
70  FC_ASSERT(false, "Not allowed until hardfork 188");
71  }
72  }
73  // hf_588
74  // issue #588
75  //
76  // As a virtual operation which has no evaluator `asset_settle_cancel_operation`
77  // originally won't be packed into blocks, yet its loose `validate()` method
78  // make it able to slip into blocks.
79  //
80  // We need to forbid this operation being packed into blocks via proposal but
81  // this will lead to a hardfork (this operation in proposal will denied by new
82  // node while accept by old node), so a hardfork guard code needed and a
83  // consensus upgrade over all nodes needed in future. And because the
84  // `validate()` method not suitable to check database status, so we put the
85  // code here.
86  //
87  // After the hardfork, all nodes will deny packing this operation into a block,
88  // and then we will check whether exists a proposal containing this kind of
89  // operation, if not exists, we can harden the `validate()` method to deny
90  // it in a earlier stage.
91  //
92  void operator()(const graphene::chain::asset_settle_cancel_operation &v) const {
93  if (block_time > HARDFORK_CORE_588_TIME) {
94  FC_ASSERT(!"Virtual operation");
95  }
96  }
97  // loop and self visit in proposals
98  void operator()(const graphene::chain::proposal_create_operation &v) const {
99  for (const op_wrapper &op : v.proposed_ops)
100  op.op.visit(*this);
101  }

116 void_result proposal_create_evaluator::do_evaluate(const proposal_create_operation& o)
117 { try {
118  const database& d = db();
119 
120  // Calling the proposal hardfork visitor
121  const fc::time_point_sec block_time = d.head_block_time();
122  const fc::time_point_sec next_maint_time = d.get_dynamic_global_properties().next_maintenance_time;
123  proposal_operation_hardfork_visitor vtor( block_time, next_maint_time );
124  vtor( o );
125  if( block_time < HARDFORK_CORE_214_TIME )
126  { // cannot be removed after hf, unfortunately
127  hardfork_visitor_214 hf214;
128  for (const op_wrapper &op : o.proposed_ops)
129  op.op.visit( hf214 );
130  }
131 
132  const auto& global_parameters = d.get_global_properties().parameters;
133 
134  FC_ASSERT( o.expiration_time > block_time, "Proposal has already expired on creation." );
135  FC_ASSERT( o.expiration_time <= block_time + global_parameters.maximum_proposal_lifetime,
136  "Proposal expiration time is too far in the future.");
137  FC_ASSERT( !o.review_period_seconds || fc::seconds(*o.review_period_seconds) < (o.expiration_time - block_time),
138  "Proposal review period must be less than its overall lifetime." );
139 
140  {
141  // If we're dealing with the committee authority, make sure this transaction has a sufficient review period.
142  flat_set<account_id_type> auths;
143  vector<authority> other;
144  for( auto& op : o.proposed_ops )
145  {
146  operation_get_required_authorities(op.op, auths, auths, other);
147  }
148 
149  FC_ASSERT( other.size() == 0 ); // TODO: what about other???
150 
151  if( auths.find(GRAPHENE_COMMITTEE_ACCOUNT) != auths.end() )
152  {
153  GRAPHENE_ASSERT(
154  o.review_period_seconds.valid(),
155  proposal_create_review_period_required,
156  "Review period not given, but at least ${min} required",
157  ("min", global_parameters.committee_proposal_review_period)
158  );
159  GRAPHENE_ASSERT(
160  *o.review_period_seconds >= global_parameters.committee_proposal_review_period,
161  proposal_create_review_period_insufficient,
162  "Review period of ${t} specified, but at least ${min} required",
163  ("t", *o.review_period_seconds)
164  ("min", global_parameters.committee_proposal_review_period)
165  );
166  }
167  }
168 
169  for( const op_wrapper& op : o.proposed_ops )
170  _proposed_trx.operations.push_back(op.op);
171 
172  _proposed_trx.validate();
173 
174  return void_result();
175 } FC_CAPTURE_AND_RETHROW( (o) ) }

177 object_id_type proposal_create_evaluator::do_apply(const proposal_create_operation& o)
178 { try {
179  database& d = db();
180 
181  const proposal_object& proposal = d.create<proposal_object>([&](proposal_object& proposal) {
182  _proposed_trx.expiration = o.expiration_time;
183  proposal.proposed_transaction = _proposed_trx;
184  proposal.expiration_time = o.expiration_time;
185  proposal.proposer = o.fee_paying_account;
186  if( o.review_period_seconds )
187  proposal.review_period_time = o.expiration_time - *o.review_period_seconds;
188 
189  //Populate the required approval sets
190  flat_set<account_id_type> required_active;
191  vector<authority> other;
192 
193  // TODO: consider caching values from evaluate?
194  for( auto& op : _proposed_trx.operations )
195  operation_get_required_authorities(op, required_active, proposal.required_owner_approvals, other);
196 
197  //All accounts which must provide both owner and active authority should be omitted from the active authority set;
198  //owner authority approval implies active authority approval.
199  std::set_difference(required_active.begin(), required_active.end(),
200  proposal.required_owner_approvals.begin(), proposal.required_owner_approvals.end(),
201  std::inserter(proposal.required_active_approvals, proposal.required_active_approvals.begin()));
202  });
203 
204  return proposal.id;
205 } FC_CAPTURE_AND_RETHROW( (o) ) }

207 void_result proposal_update_evaluator::do_evaluate(const proposal_update_operation& o)
208 { try {
209  database& d = db();
210 
211  _proposal = &o.proposal(d);
212 
213  if( _proposal->review_period_time && d.head_block_time() >= *_proposal->review_period_time )
214  FC_ASSERT( o.active_approvals_to_add.empty() && o.owner_approvals_to_add.empty(),
215  "This proposal is in its review period. No new approvals may be added." );
216 
217  for( account_id_type id : o.active_approvals_to_remove )
218  {
219  FC_ASSERT( _proposal->available_active_approvals.find(id) != _proposal->available_active_approvals.end(),
220  "", ("id", id)("available", _proposal->available_active_approvals) );
221  }
222  for( account_id_type id : o.owner_approvals_to_remove )
223  {
224  FC_ASSERT( _proposal->available_owner_approvals.find(id) != _proposal->available_owner_approvals.end(),
225  "", ("id", id)("available", _proposal->available_owner_approvals) );
226  }
227 
228  return void_result();

231 void_result proposal_update_evaluator::do_apply(const proposal_update_operation& o)
232 { try {
233  database& d = db();
234 
235  // Potential optimization: if _executed_proposal is true, we can skip the modify step and make push_proposal skip
236  // signature checks. This isn't done now because I just wrote all the proposals code, and I'm not yet 100% sure the
237  // required approvals are sufficient to authorize the transaction.
238  d.modify(*_proposal, [&o](proposal_object& p) {
239  p.available_active_approvals.insert(o.active_approvals_to_add.begin(), o.active_approvals_to_add.end());
240  p.available_owner_approvals.insert(o.owner_approvals_to_add.begin(), o.owner_approvals_to_add.end());
241  for( account_id_type id : o.active_approvals_to_remove )
242  p.available_active_approvals.erase(id);
243  for( account_id_type id : o.owner_approvals_to_remove )
244  p.available_owner_approvals.erase(id);
245  for( const auto& id : o.key_approvals_to_add )
246  p.available_key_approvals.insert(id);
247  for( const auto& id : o.key_approvals_to_remove )
248  p.available_key_approvals.erase(id);
249  });
250 
251  // If the proposal has a review period, don't bother attempting to authorize/execute it.
252  // Proposals with a review period may never be executed except at their expiration.
253  if( _proposal->review_period_time )
254  return void_result();
255 
256  if( _proposal->is_authorized_to_execute(d) )
257  {
258  // All required approvals are satisfied. Execute!
259  _executed_proposal = true;
260  try {
261  _processed_transaction = d.push_proposal(*_proposal);
262  } catch(fc::exception& e) {
263  d.modify(*_proposal, [&e](proposal_object& p) {
264  p.fail_reason = e.to_string(fc::log_level(fc::log_level::all));
265  });
266  wlog("Proposed transaction ${id} failed to apply once approved with exception:\n----\n${reason}\n----\nWill try again when it expires.",
267  ("id", o.proposal)("reason", e.to_detail_string()));
268  _proposal_failed = true;
269  }
270  }
271 
272  return void_result();
273 } FC_CAPTURE_AND_RETHROW( (o) ) }
