50 void proposal_update_operation::validate() const
51 {
52  FC_ASSERT(fee.amount >= 0);
53  FC_ASSERT(!(active_approvals_to_add.empty() && active_approvals_to_remove.empty() &&
54  owner_approvals_to_add.empty() && owner_approvals_to_remove.empty() &&
55  key_approvals_to_add.empty() && key_approvals_to_remove.empty()));
56  for( auto a : active_approvals_to_add )
57  {
58  FC_ASSERT(active_approvals_to_remove.find(a) == active_approvals_to_remove.end(),
59  "Cannot add and remove approval at the same time.");
60  }
61  for( auto a : owner_approvals_to_add )
62  {
63  FC_ASSERT(owner_approvals_to_remove.find(a) == owner_approvals_to_remove.end(),
64  "Cannot add and remove approval at the same time.");
65  }
66  for( auto a : key_approvals_to_add )
67  {
68  FC_ASSERT(key_approvals_to_remove.find(a) == key_approvals_to_remove.end(),
69  "Cannot add and remove approval at the same time.");

71 }

84 {
85  authority auth;

87  auth.key_auths[k] = 1;
88  for( const auto& k : key_approvals_to_remove )
