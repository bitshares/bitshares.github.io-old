83 { try {
84  database& d = db();
85  vote_id_type for_id, against_id;
86  d.modify(d.get_global_properties(), [&for_id, &against_id](global_property_object& p) {
87  for_id = get_next_vote_id(p, vote_id_type::worker);
88  against_id = get_next_vote_id(p, vote_id_type::worker);
89  });
90 
91  return d.create<worker_object>([&](worker_object& w) {
92  w.worker_account = o.owner;
93  w.daily_pay = o.daily_pay;
94  w.work_begin_date = o.work_begin_date;
95  w.work_end_date = o.work_end_date;
96  w.name = o.name;
97  w.url = o.url;
98  w.vote_for = for_id;
99  w.vote_against = against_id;
100 
101  w.worker.set_which(o.initializer.which());
