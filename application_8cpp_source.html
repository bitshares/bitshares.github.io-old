217 std::vector<fc::ip::endpoint> application_impl::resolve_string_to_ip_endpoints(const std::string& endpoint_string)
218 {
219  try
220  {
221  string::size_type colon_pos = endpoint_string.find(':');
222  if (colon_pos == std::string::npos)
223  FC_THROW("Missing required port number in endpoint string \"${endpoint_string}\"",
224  ("endpoint_string", endpoint_string));
225  std::string port_string = endpoint_string.substr(colon_pos + 1);
226  try
227  {
228  uint16_t port = boost::lexical_cast<uint16_t>(port_string);
229 
230  std::string hostname = endpoint_string.substr(0, colon_pos);
231  std::vector<fc::ip::endpoint> endpoints = fc::resolve(hostname, port);
232  if (endpoints.empty())
233  FC_THROW_EXCEPTION( fc::unknown_host_exception,
234  "The host name can not be resolved: ${hostname}",
235  ("hostname", hostname) );
236  return endpoints;
237  }
238  catch (const boost::bad_lexical_cast&)
239  {
240  FC_THROW("Bad port: ${port}", ("port", port_string));
241  }
242  }
243  FC_CAPTURE_AND_RETHROW((endpoint_string))
244 }

247 void application_impl::new_connection( const fc::http::websocket_connection_ptr& c )
248 {
249  auto wsc = std::make_shared<fc::rpc::websocket_api_connection>(*c, GRAPHENE_NET_MAX_NESTED_OBJECTS);
250  auto login = std::make_shared<graphene::app::login_api>( std::ref(*_self) );
251  login->enable_api("database_api");
252 
253  wsc->register_api(login->database());
254  wsc->register_api(fc::api<graphene::app::login_api>(login));
255  c->set_session_data( wsc );
256 
257  std::string username = "*";
258  std::string password = "*";
259 
260  // Try to extract login information from "Authorization" header if present
261  std::string auth = c->get_request_header("Authorization");
262  if( boost::starts_with(auth, "Basic ") ) {
263 
264  FC_ASSERT( auth.size() > 6 );
265  auto user_pass = fc::base64_decode(auth.substr(6));
266 
267  std::vector<std::string> parts;
268  boost::split( parts, user_pass, boost::is_any_of(":") );
269 
270  FC_ASSERT(parts.size() == 2);
271 
272  username = parts[0];
273  password = parts[1];
274  }
275 
276  login->login(username, password);
277 }

495 bool application_impl::handle_block(const graphene::net::block_message& blk_msg, bool sync_mode,
496  std::vector<fc::uint160_t>& contained_transaction_message_ids)
497 { try {
498 
499  auto latency = fc::time_point::now() - blk_msg.block.timestamp;
500  if (!sync_mode || blk_msg.block.block_num() % 10000 == 0)
501  {
502  const auto& witness = blk_msg.block.witness(*_chain_db);
503  const auto& witness_account = witness.witness_account(*_chain_db);
504  auto last_irr = _chain_db->get_dynamic_global_properties().last_irreversible_block_num;
505  ilog("Got block: #${n} ${bid} time: ${t} latency: ${l} ms from: ${w} irreversible: ${i} (-${d})",
506  ("t",blk_msg.block.timestamp)
507  ("n", blk_msg.block.block_num())
508  ("bid", blk_msg.block.id())
509  ("l", (latency.count()/1000))
510  ("w",witness_account.name)
511  ("i",last_irr)("d",blk_msg.block.block_num()-last_irr) );
512  }
513  FC_ASSERT( (latency.count()/1000) > -5000, "Rejecting block with timestamp in the future" );
514 
515  try {
516  // TODO: in the case where this block is valid but on a fork that's too old for us to switch to,
517  // you can help the network code out by throwing a block_older_than_undo_history exception.
518  // when the net code sees that, it will stop trying to push blocks from that chain, but
519  // leave that peer connected so that they can get sync blocks from us
520  bool result = _chain_db->push_block( blk_msg.block,
521  (_is_block_producer | _force_validate) ?
522  database::skip_nothing : database::skip_transaction_signatures );
523 
524  // the block was accepted, so we now know all of the transactions contained in the block
525  if (!sync_mode)
526  {
527  // if we're not in sync mode, there's a chance we will be seeing some transactions
528  // included in blocks before we see the free-floating transaction itself. If that
529  // happens, there's no reason to fetch the transactions, so construct a list of the
530  // transaction message ids we no longer need.
531  // during sync, it is unlikely that we'll see any old
532  for (const processed_transaction& transaction : blk_msg.block.transactions)
533  {
534  graphene::net::trx_message transaction_message(transaction);
535  contained_transaction_message_ids.push_back(graphene::net::message(transaction_message).id());
536  }
537  }
538 
539  return result;
540  } catch ( const graphene::chain::unlinkable_block_exception& e ) {
541  // translate to a graphene::net exception
542  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
543  FC_THROW_EXCEPTION( graphene::net::unlinkable_block_exception,
544  "Error when pushing block:\n${e}",
545  ("e", e.to_detail_string()) );
546  } catch( const fc::exception& e ) {
547  elog("Error when pushing block:\n${e}", ("e", e.to_detail_string()));
548  throw;
549  }
550 
551  if( !_is_finished_syncing && !sync_mode )
552  {
553  _is_finished_syncing = true;
554  _self->syncing_finished();
555  }

595 std::vector<item_hash_t> application_impl::get_block_ids(const std::vector<item_hash_t>& blockchain_synopsis,
596  uint32_t& remaining_item_count,
597  uint32_t limit)
598 { try {
599  vector<block_id_type> result;
600  remaining_item_count = 0;
601  if( _chain_db->head_block_num() == 0 )
602  return result;
603 
604  result.reserve(limit);
605  block_id_type last_known_block_id;
606 
607  if (blockchain_synopsis.empty() ||
608  (blockchain_synopsis.size() == 1 && blockchain_synopsis[0] == block_id_type()))
609  {
610  // peer has sent us an empty synopsis meaning they have no blocks.
611  // A bug in old versions would cause them to send a synopsis containing block 000000000
612  // when they had an empty blockchain, so pretend they sent the right thing here.
613 
614  // do nothing, leave last_known_block_id set to zero
615  }
616  else
617  {
618  bool found_a_block_in_synopsis = false;
619  for (const item_hash_t& block_id_in_synopsis : boost::adaptors::reverse(blockchain_synopsis))
620  if (block_id_in_synopsis == block_id_type() ||
621  (_chain_db->is_known_block(block_id_in_synopsis) && is_included_block(block_id_in_synopsis)))
622  {
623  last_known_block_id = block_id_in_synopsis;
624  found_a_block_in_synopsis = true;
625  break;
626  }
627  if (!found_a_block_in_synopsis)
628  FC_THROW_EXCEPTION( graphene::net::peer_is_on_an_unreachable_fork,
629  "Unable to provide a list of blocks starting at any of the blocks in peer's synopsis" );
630  }
631  for( uint32_t num = block_header::num_from_id(last_known_block_id);
632  num <= _chain_db->head_block_num() && result.size() < limit;
633  ++num )
634  if( num > 0 )
635  result.push_back(_chain_db->get_block_id_for_num(num));
636 
637  if( !result.empty() && block_header::num_from_id(result.back()) < _chain_db->head_block_num() )
638  remaining_item_count = _chain_db->head_block_num() - block_header::num_from_id(result.back());
639 
640  return result;
641 } FC_CAPTURE_AND_RETHROW( (blockchain_synopsis)(remaining_item_count)(limit) ) }

725 std::vector<item_hash_t> application_impl::get_blockchain_synopsis(const item_hash_t& reference_point,
726  uint32_t number_of_blocks_after_reference_point)
727 { try {
728  std::vector<item_hash_t> synopsis;
729  synopsis.reserve(30);
730  uint32_t high_block_num;
731  uint32_t non_fork_high_block_num;
732  uint32_t low_block_num = _chain_db->last_non_undoable_block_num();
733  std::vector<block_id_type> fork_history;
734 
735  if (reference_point != item_hash_t())
736  {
737  // the node is asking for a summary of the block chain up to a specified
738  // block, which may or may not be on a fork
739  // for now, assume it's not on a fork
740  if (is_included_block(reference_point))
741  {
742  // reference_point is a block we know about and is on the main chain
743  uint32_t reference_point_block_num = block_header::num_from_id(reference_point);
744  assert(reference_point_block_num > 0);
745  high_block_num = reference_point_block_num;
746  non_fork_high_block_num = high_block_num;
747 
748  if (reference_point_block_num < low_block_num)
749  {
750  // we're on the same fork (at least as far as reference_point) but we've passed
751  // reference point and could no longer undo that far if we diverged after that
752  // block. This should probably only happen due to a race condition where
753  // the network thread calls this function, and then immediately pushes a bunch of blocks,
754  // then the main thread finally processes this function.
755  // with the current framework, there's not much we can do to tell the network
756  // thread what our current head block is, so we'll just pretend that
757  // our head is actually the reference point.
758  // this *may* enable us to fetch blocks that we're unable to push, but that should
759  // be a rare case (and correctly handled)
760  low_block_num = reference_point_block_num;
761  }
762  }
763  else
764  {
765  // block is a block we know about, but it is on a fork
766  try
767  {
768  fork_history = _chain_db->get_block_ids_on_fork(reference_point);
769  // returns a vector where the last element is the common ancestor with the preferred chain,
770  // and the first element is the reference point you passed in
771  assert(fork_history.size() >= 2);
772 
773  if( fork_history.front() != reference_point )
774  {
775  edump( (fork_history)(reference_point) );
776  assert(fork_history.front() == reference_point);
777  }
778  block_id_type last_non_fork_block = fork_history.back();
779  fork_history.pop_back(); // remove the common ancestor
780  boost::reverse(fork_history);
781 
782  if (last_non_fork_block == block_id_type()) // if the fork goes all the way back to genesis (does graphene's fork db allow this?)
783  non_fork_high_block_num = 0;
784  else
785  non_fork_high_block_num = block_header::num_from_id(last_non_fork_block);
786 
787  high_block_num = non_fork_high_block_num + fork_history.size();
788  assert(high_block_num == block_header::num_from_id(fork_history.back()));
789  }
790  catch (const fc::exception& e)
791  {
792  // unable to get fork history for some reason. maybe not linked?
793  // we can't return a synopsis of its chain
794  elog( "Unable to construct a blockchain synopsis for reference hash ${hash}: ${exception}",
795  ("hash", reference_point)("exception", e) );
796  throw;
797  }
798  if (non_fork_high_block_num < low_block_num)
799  {
800  wlog("Unable to generate a usable synopsis because the peer we're generating it for forked too long ago "
801  "(our chains diverge after block #${non_fork_high_block_num} but only undoable to block #${low_block_num})",
802  ("low_block_num", low_block_num)
803  ("non_fork_high_block_num", non_fork_high_block_num));
804  FC_THROW_EXCEPTION(graphene::net::block_older_than_undo_history, "Peer is are on a fork I'm unable to switch to");
805  }
806  }
807  }
808  else
809  {
810  // no reference point specified, summarize the whole block chain
811  high_block_num = _chain_db->head_block_num();
812  non_fork_high_block_num = high_block_num;
813  if (high_block_num == 0)
814  return synopsis; // we have no blocks
815  }
816 
817  if( low_block_num == 0)
818  low_block_num = 1;
819 
820  // at this point:
821  // low_block_num is the block before the first block we can undo,
822  // non_fork_high_block_num is the block before the fork (if the peer is on a fork, or otherwise it is the same as high_block_num)
823  // high_block_num is the block number of the reference block, or the end of the chain if no reference provided
824 
825  // true_high_block_num is the ending block number after the network code appends any item ids it
826  // knows about that we don't
827  uint32_t true_high_block_num = high_block_num + number_of_blocks_after_reference_point;
828  do
829  {
830  // for each block in the synopsis, figure out where to pull the block id from.
831  // if it's <= non_fork_high_block_num, we grab it from the main blockchain;
832  // if it's not, we pull it from the fork history
833  if (low_block_num <= non_fork_high_block_num)
834  synopsis.push_back(_chain_db->get_block_id_for_num(low_block_num));
835  else
836  synopsis.push_back(fork_history[low_block_num - non_fork_high_block_num - 1]);
837  low_block_num += (true_high_block_num - low_block_num + 2) / 2;
838  }
839  while (low_block_num <= high_block_num);
840 
841  //idump((synopsis));
842  return synopsis;
843 } FC_CAPTURE_AND_RETHROW() }
