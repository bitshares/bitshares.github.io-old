51 template<class Index>
52 vector<std::reference_wrapper<const typename Index::object_type>> database::sort_votable_objects(size_t count) const
53 {
54  using ObjectType = typename Index::object_type;
55  const auto& all_objects = get_index_type<Index>().indices();
56  count = std::min(count, all_objects.size());
57  vector<std::reference_wrapper<const ObjectType>> refs;
58  refs.reserve(all_objects.size());
59  std::transform(all_objects.begin(), all_objects.end(),
60  std::back_inserter(refs),
61  [](const ObjectType& o) { return std::cref(o); });
62  std::partial_sort(refs.begin(), refs.begin() + count, refs.end(),
63  [this](const ObjectType& a, const ObjectType& b)->bool {
64  share_type oa_vote = _vote_tally_buffer[a.vote_id];
65  share_type ob_vote = _vote_tally_buffer[b.vote_id];
66  if( oa_vote != ob_vote )
67  return oa_vote > ob_vote;
68  return a.vote_id < b.vote_id;
69  });
70 
71  refs.resize(count, refs.front());
72  return refs;
73 }

152 void database::pay_workers( share_type& budget )
153 {
154  const auto head_time = head_block_time();
155 // ilog("Processing payroll! Available budget is ${b}", ("b", budget));
156  vector<std::reference_wrapper<const worker_object>> active_workers;
157  // TODO optimization: add by_expiration index to avoid iterating through all objects
158  get_index_type<worker_index>().inspect_all_objects([head_time, &active_workers](const object& o) {
159  const worker_object& w = static_cast<const worker_object&>(o);
160  if( w.is_active(head_time) && w.approving_stake() > 0 )
161  active_workers.emplace_back(w);
162  });
163 
164  // worker with more votes is preferred
165  // if two workers exactly tie for votes, worker with lower ID is preferred
166  std::sort(active_workers.begin(), active_workers.end(), [](const worker_object& wa, const worker_object& wb) {
167  share_type wa_vote = wa.approving_stake();
168  share_type wb_vote = wb.approving_stake();
169  if( wa_vote != wb_vote )
170  return wa_vote > wb_vote;
171  return wa.id < wb.id;
172  });
173 
174  const auto last_budget_time = get_dynamic_global_properties().last_budget_time;
175  const auto passed_time_ms = head_time - last_budget_time;
176  const auto passed_time_count = passed_time_ms.count();
177  const auto day_count = fc::days(1).count();
178  for( uint32_t i = 0; i < active_workers.size() && budget > 0; ++i )
179  {
180  const worker_object& active_worker = active_workers[i];
181  share_type requested_pay = active_worker.daily_pay;
182 
183  // Note: if there is a good chance that passed_time_count == day_count,
184  // for better performance, can avoid the 128 bit calculation by adding a check.
185  // Since it's not the case on BitShares mainnet, we're not using a check here.
186  fc::uint128 pay(requested_pay.value);
187  pay *= passed_time_count;
188  pay /= day_count;
189  requested_pay = pay.to_uint64();
190 
191  share_type actual_pay = std::min(budget, requested_pay);
192  //ilog(" ==> Paying ${a} to worker ${w}", ("w", active_worker.id)("a", actual_pay));
193  modify(active_worker, [&](worker_object& w) {
194  w.worker.visit(worker_pay_visitor(actual_pay, *this));
195  });
196 
197  budget -= actual_pay;
198  }

404 void database::initialize_budget_record( fc::time_point_sec now, budget_record& rec )const
405 {
406  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
407  const asset_object& core = get_core_asset();
408  const asset_dynamic_data_object& core_dd = get_core_dynamic_data();
409 
410  rec.from_initial_reserve = core.reserved(*this);
411  rec.from_accumulated_fees = core_dd.accumulated_fees;
412  rec.from_unused_witness_budget = dpo.witness_budget;
413 
414  if( (dpo.last_budget_time == fc::time_point_sec())
415  || (now <= dpo.last_budget_time) )
416  {
417  rec.time_since_last_budget = 0;
418  return;
419  }
420 
421  int64_t dt = (now - dpo.last_budget_time).to_seconds();
422  rec.time_since_last_budget = uint64_t( dt );
423 
424  // We'll consider accumulated_fees to be reserved at the BEGINNING
425  // of the maintenance interval. However, for speed we only
426  // call modify() on the asset_dynamic_data_object once at the
427  // end of the maintenance interval. Thus the accumulated_fees
428  // are available for the budget at this point, but not included
429  // in core.reserved().
430  share_type reserve = rec.from_initial_reserve + core_dd.accumulated_fees;
431  // Similarly, we consider leftover witness_budget to be burned
432  // at the BEGINNING of the maintenance interval.
433  reserve += dpo.witness_budget;
434 
435  fc::uint128_t budget_u128 = reserve.value;
436  budget_u128 *= uint64_t(dt);
437  budget_u128 *= GRAPHENE_CORE_ASSET_CYCLE_RATE;
438  //round up to the nearest satoshi -- this is necessary to ensure
439  // there isn't an "untouchable" reserve, and we will eventually
440  // be able to use the entire reserve
441  budget_u128 += ((uint64_t(1) << GRAPHENE_CORE_ASSET_CYCLE_RATE_BITS) - 1);
442  budget_u128 >>= GRAPHENE_CORE_ASSET_CYCLE_RATE_BITS;
443  if( budget_u128 < reserve.value )
444  rec.total_budget = share_type(budget_u128.to_uint64());
445  else
446  rec.total_budget = reserve;
447 
448  return;
449 }

454 void database::process_budget()
455 {
456  try
457  {
458  const global_property_object& gpo = get_global_properties();
459  const dynamic_global_property_object& dpo = get_dynamic_global_properties();
460  const asset_dynamic_data_object& core = get_core_dynamic_data();
461  fc::time_point_sec now = head_block_time();
462 
463  int64_t time_to_maint = (dpo.next_maintenance_time - now).to_seconds();
464  //
465  // The code that generates the next maintenance time should
466  // only produce a result in the future. If this assert
467  // fails, then the next maintenance time algorithm is buggy.
468  //
469  assert( time_to_maint > 0 );
470  //
471  // Code for setting chain parameters should validate
472  // block_interval > 0 (as well as the humans proposing /
473  // voting on changes to block interval).
474  //
475  assert( gpo.parameters.block_interval > 0 );
476  uint64_t blocks_to_maint = (uint64_t(time_to_maint) + gpo.parameters.block_interval - 1) / gpo.parameters.block_interval;
477 
478  // blocks_to_maint > 0 because time_to_maint > 0,
479  // which means numerator is at least equal to block_interval
480 
481  budget_record rec;
482  initialize_budget_record( now, rec );
483  share_type available_funds = rec.total_budget;
484 
485  share_type witness_budget = gpo.parameters.witness_pay_per_block.value * blocks_to_maint;
486  rec.requested_witness_budget = witness_budget;
487  witness_budget = std::min(witness_budget, available_funds);
488  rec.witness_budget = witness_budget;
489  available_funds -= witness_budget;
490 
491  fc::uint128_t worker_budget_u128 = gpo.parameters.worker_budget_per_day.value;
492  worker_budget_u128 *= uint64_t(time_to_maint);
493  worker_budget_u128 /= 60*60*24;
494 
495  share_type worker_budget;
496  if( worker_budget_u128 >= available_funds.value )
497  worker_budget = available_funds;
498  else
499  worker_budget = worker_budget_u128.to_uint64();
500  rec.worker_budget = worker_budget;
501  available_funds -= worker_budget;
502 
503  share_type leftover_worker_funds = worker_budget;
504  pay_workers(leftover_worker_funds);
505  rec.leftover_worker_funds = leftover_worker_funds;
506  available_funds += leftover_worker_funds;
507 
508  rec.supply_delta = rec.witness_budget
509  + rec.worker_budget
510  - rec.leftover_worker_funds
511  - rec.from_accumulated_fees
512  - rec.from_unused_witness_budget;
513 
514  modify(core, [&]( asset_dynamic_data_object& _core )
515  {
516  _core.current_supply = (_core.current_supply + rec.supply_delta );
517 
518  assert( rec.supply_delta ==
519  witness_budget
520  + worker_budget
521  - leftover_worker_funds
522  - _core.accumulated_fees
523  - dpo.witness_budget
524  );
525  _core.accumulated_fees = 0;
526  });
527 
528  modify(dpo, [&]( dynamic_global_property_object& _dpo )
529  {
530  // Since initial witness_budget was rolled into
531  // available_funds, we replace it with witness_budget
532  // instead of adding it.
533  _dpo.witness_budget = witness_budget;
534  _dpo.last_budget_time = now;
535  });
536 
537  create< budget_record_object >( [&]( budget_record_object& _rec )
538  {
539  _rec.time = head_block_time();
540  _rec.record = rec;
541  });
542 
543  // available_funds is money we could spend, but don't want to.
544  // we simply let it evaporate back into the reserve.
545  }
546  FC_CAPTURE_AND_RETHROW()
547 }

607 void split_fba_balance(
608  database& db,
609  uint64_t fba_id,
610  uint16_t network_pct,
611  uint16_t designated_asset_buyback_pct,
612  uint16_t designated_asset_issuer_pct
613 )
614 {
615  FC_ASSERT( uint32_t(network_pct) + uint32_t(designated_asset_buyback_pct) + uint32_t(designated_asset_issuer_pct) == GRAPHENE_100_PERCENT );
616  const fba_accumulator_object& fba = fba_accumulator_id_type( fba_id )(db);
617  if( fba.accumulated_fba_fees == 0 )
618  return;
619 
620  const asset_dynamic_data_object& core_dd = db.get_core_dynamic_data();
621 
622  if( !fba.is_configured(db) )
623  {
624  ilog( "${n} core given to network at block ${b} due to non-configured FBA", ("n", fba.accumulated_fba_fees)("b", db.head_block_time()) );
625  db.modify( core_dd, [&]( asset_dynamic_data_object& _core_dd )
626  {
627  _core_dd.current_supply -= fba.accumulated_fba_fees;
628  } );
629  db.modify( fba, [&]( fba_accumulator_object& _fba )
630  {
631  _fba.accumulated_fba_fees = 0;
632  } );
633  return;
634  }
635 
636  fc::uint128_t buyback_amount_128 = fba.accumulated_fba_fees.value;
637  buyback_amount_128 *= designated_asset_buyback_pct;
638  buyback_amount_128 /= GRAPHENE_100_PERCENT;
639  share_type buyback_amount = buyback_amount_128.to_uint64();
640 
641  fc::uint128_t issuer_amount_128 = fba.accumulated_fba_fees.value;
642  issuer_amount_128 *= designated_asset_issuer_pct;
643  issuer_amount_128 /= GRAPHENE_100_PERCENT;
644  share_type issuer_amount = issuer_amount_128.to_uint64();
645 
646  // this assert should never fail
647  FC_ASSERT( buyback_amount + issuer_amount <= fba.accumulated_fba_fees );
648 
649  share_type network_amount = fba.accumulated_fba_fees - (buyback_amount + issuer_amount);
650 
651  const asset_object& designated_asset = (*fba.designated_asset)(db);
652 
653  if( network_amount != 0 )
654  {
655  db.modify( core_dd, [&]( asset_dynamic_data_object& _core_dd )
656  {
657  _core_dd.current_supply -= network_amount;
658  } );
659  }

663  vop.fba_id = fba.id;

674  if( vop.amount != 0 )
675  {
676  db.adjust_balance( designated_asset.issuer, asset(issuer_amount) );
677  db.push_applied_operation(vop);
678  }
679

693 void create_buyback_orders( database& db )
694 {
695  const auto& bbo_idx = db.get_index_type< buyback_index >().indices().get<by_id>();
696  const auto& bal_idx = db.get_index_type< account_balance_index >().indices().get< by_account_asset >();
697 
698  for( const buyback_object& bbo : bbo_idx )
699  {
700  const asset_object& asset_to_buy = bbo.asset_to_buy(db);
701  assert( asset_to_buy.buyback_account.valid() );
702 
703  const account_object& buyback_account = (*(asset_to_buy.buyback_account))(db);
704  asset_id_type next_asset = asset_id_type();
705 
706  if( !buyback_account.allowed_assets.valid() )
707  {
708  wlog( "skipping buyback account ${b} at block ${n} because allowed_assets does not exist", ("b", buyback_account)("n", db.head_block_num()) );
709  continue;
710  }
711 
712  while( true )
713  {
714  auto it = bal_idx.lower_bound( boost::make_tuple( buyback_account.id, next_asset ) );
715  if( it == bal_idx.end() )
716  break;
717  if( it->owner != buyback_account.id )
718  break;
719  asset_id_type asset_to_sell = it->asset_type;
720  share_type amount_to_sell = it->balance;
721  next_asset = asset_to_sell + 1;
722  if( asset_to_sell == asset_to_buy.id )
723  continue;
724  if( amount_to_sell == 0 )
725  continue;
726  if( buyback_account.allowed_assets->find( asset_to_sell ) == buyback_account.allowed_assets->end() )
727  {
728  wlog( "buyback account ${b} not selling disallowed holdings of asset ${a} at block ${n}", ("b", buyback_account)("a", asset_to_sell)("n", db.head_block_num()) );
729  continue;
730  }
731 
732  try
733  {
734  transaction_evaluation_state buyback_context(&db);
735  buyback_context.skip_fee_schedule_check = true;
736 
737  limit_order_create_operation create_vop;
738  create_vop.fee = asset( 0, asset_id_type() );
739  create_vop.seller = buyback_account.id;
740  create_vop.amount_to_sell = asset( amount_to_sell, asset_to_sell );
741  create_vop.min_to_receive = asset( 1, asset_to_buy.id );
742  create_vop.expiration = time_point_sec::maximum();
743  create_vop.fill_or_kill = false;
744 
745  limit_order_id_type order_id = db.apply_operation( buyback_context, create_vop ).get< object_id_type >();
746 
747  if( db.find( order_id ) != nullptr )
748  {
749  limit_order_cancel_operation cancel_vop;
750  cancel_vop.fee = asset( 0, asset_id_type() );
751  cancel_vop.order = order_id;
752  cancel_vop.fee_paying_account = buyback_account.id;
753 
754  db.apply_operation( buyback_context, cancel_vop );
755  }
756  }
757  catch( const fc::exception& e )
758  {
759  // we can in fact get here, e.g. if asset issuer of buy/sell asset blacklists/whitelists the buyback account
760  wlog( "Skipping buyback processing selling ${as} for ${ab} for buyback account ${b} at block ${n}; exception was ${e}",
761  ("as", asset_to_sell)("ab", asset_to_buy)("b", buyback_account)("n", db.head_block_num())("e", e.to_detail_string()) );
762  continue;
763  }
764  }
765  }
766  return;

800 void database::process_bids( const asset_bitasset_data_object& bad )
801 {
802  if( bad.is_prediction_market ) return;
803  if( bad.current_feed.settlement_price.is_null() ) return;
804 
805  asset_id_type to_revive_id = (asset( 0, bad.options.short_backing_asset ) * bad.settlement_price).asset_id;
806  const asset_object& to_revive = to_revive_id( *this );
807  const asset_dynamic_data_object& bdd = to_revive.dynamic_data( *this );
808 
809  const auto& bid_idx = get_index_type< collateral_bid_index >().indices().get<by_price>();
810  const auto start = bid_idx.lower_bound( boost::make_tuple( to_revive_id, price::max( bad.options.short_backing_asset, to_revive_id ), collateral_bid_id_type() ) );
811 
812  share_type covered = 0;
813  auto itr = start;
814  while( covered < bdd.current_supply && itr != bid_idx.end() && itr->inv_swan_price.quote.asset_id == to_revive_id )
815  {
816  const collateral_bid_object& bid = *itr;
817  asset total_collateral = bid.inv_swan_price.quote * bad.settlement_price;
818  total_collateral += bid.inv_swan_price.base;
819  price call_price = price::call_price( bid.inv_swan_price.quote, total_collateral, bad.current_feed.maintenance_collateral_ratio );
820  if( ~call_price >= bad.current_feed.settlement_price ) break;
821  covered += bid.inv_swan_price.quote.amount;
822  ++itr;
823  }
824  if( covered < bdd.current_supply ) return;
825 
826  const auto end = itr;
827  share_type to_cover = bdd.current_supply;
828  share_type remaining_fund = bad.settlement_fund;
829  for( itr = start; itr != end; )
830  {
831  const collateral_bid_object& bid = *itr;
832  ++itr;
833  share_type debt = bid.inv_swan_price.quote.amount;
834  share_type collateral = (bid.inv_swan_price.quote * bad.settlement_price).amount;
835  if( bid.inv_swan_price.quote.amount >= to_cover )
836  {
837  debt = to_cover;
838  collateral = remaining_fund;
839  }
840  to_cover -= debt;
841  remaining_fund -= collateral;
842  execute_bid( bid, debt, collateral, bad.current_feed );
843  }
844  FC_ASSERT( remaining_fund == 0 );
845  FC_ASSERT( to_cover == 0 );
846 
847  _cancel_bids_and_revive_mpa( to_revive, bad );
848 }
