42  public_key_type::public_key_type( const std::string& base58str )
43  {
44  // TODO: Refactor syntactic checks into static is_valid()
45  // to make public_key_type API more similar to address API
46  std::string prefix( GRAPHENE_ADDRESS_PREFIX );
47  const size_t prefix_len = prefix.size();
48  FC_ASSERT( base58str.size() > prefix_len );
49  FC_ASSERT( base58str.substr( 0, prefix_len ) == prefix , "", ("base58str", base58str) );
50  auto bin = fc::from_base58( base58str.substr( prefix_len ) );
51  auto bin_key = fc::raw::unpack<binary_key>(bin);
52  key_data = bin_key.data;
53  FC_ASSERT( fc::ripemd160::hash( key_data.data, key_data.size() )._hash[0] == bin_key.check );
54  };
