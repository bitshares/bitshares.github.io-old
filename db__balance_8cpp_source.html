53 void database::adjust_balance(account_id_type account, asset delta )
54 { try {
55  if( delta.amount == 0 )
56  return;
57 
58  auto& index = get_index_type<account_balance_index>().indices().get<by_account_asset>();
59  auto itr = index.find(boost::make_tuple(account, delta.asset_id));
60  if(itr == index.end())
61  {
62  FC_ASSERT( delta.amount > 0, "Insufficient Balance: ${a}'s balance of ${b} is less than required ${r}",
63  ("a",account(*this).name)
64  ("b",to_pretty_string(asset(0,delta.asset_id)))
65  ("r",to_pretty_string(-delta)));
66  create<account_balance_object>([account,&delta](account_balance_object& b) {
67  b.owner = account;
68  b.asset_type = delta.asset_id;
69  b.balance = delta.amount.value;
70  if( b.asset_type == asset_id_type() ) // CORE asset
71  b.maintenance_flag = true;
72  });
73  } else {
74  if( delta.amount < 0 )
75  FC_ASSERT( itr->get_balance() >= -delta, "Insufficient Balance: ${a}'s balance of ${b} is less than required ${r}", ("a",account(*this).name)("b",to_pretty_string(itr->get_balance()))("r",to_pretty_string(-delta)));
76  modify(*itr, [delta](account_balance_object& b) {
77  b.adjust_balance(delta);
78  });
79  }
80 
81 } FC_CAPTURE_AND_RETHROW( (account)(delta) ) }

83 optional< vesting_balance_id_type > database::deposit_lazy_vesting(
84  const optional< vesting_balance_id_type >& ovbid,
85  share_type amount, uint32_t req_vesting_seconds,
86  account_id_type req_owner,
87  bool require_vesting )
88 {
89  if( amount == 0 )
90  return optional< vesting_balance_id_type >();
91 
92  fc::time_point_sec now = head_block_time();
93 
94  while( true )
95  {
96  if( !ovbid.valid() )
97  break;
98  const vesting_balance_object& vbo = (*ovbid)(*this);
99  if( vbo.owner != req_owner )
100  break;
101  if( vbo.policy.which() != vesting_policy::tag< cdd_vesting_policy >::value )
102  break;
103  if( vbo.policy.get< cdd_vesting_policy >().vesting_seconds != req_vesting_seconds )
104  break;
105  modify( vbo, [&]( vesting_balance_object& _vbo )
106  {
107  if( require_vesting )
108  _vbo.deposit(now, amount);
109  else
110  _vbo.deposit_vested(now, amount);
111  } );
112  return optional< vesting_balance_id_type >();
113  }
114 
115  const vesting_balance_object& vbo = create< vesting_balance_object >( [&]( vesting_balance_object& _vbo )
116  {
117  _vbo.owner = req_owner;
118  _vbo.balance = amount;
119 
120  cdd_vesting_policy policy;
121  policy.vesting_seconds = req_vesting_seconds;
122  policy.coin_seconds_earned = require_vesting ? 0 : amount.value * policy.vesting_seconds;
123  policy.coin_seconds_earned_last_update = now;
124 
125  _vbo.policy = policy;
126  } );
127 
128  return vbo.id;
129 }

131 void database::deposit_cashback(const account_object& acct, share_type amount, bool require_vesting)
132 {
133  // If we don't have a VBO, or if it has the wrong maturity
134  // due to a policy change, cut it loose.
135 
136  if( amount == 0 )
137  return;
138 
139  if( acct.get_id() == GRAPHENE_COMMITTEE_ACCOUNT || acct.get_id() == GRAPHENE_WITNESS_ACCOUNT ||
140  acct.get_id() == GRAPHENE_RELAXED_COMMITTEE_ACCOUNT || acct.get_id() == GRAPHENE_NULL_ACCOUNT ||
141  acct.get_id() == GRAPHENE_TEMP_ACCOUNT )
142  {
143  // The blockchain's accounts do not get cashback; it simply goes to the reserve pool.
144  modify( get_core_dynamic_data(), [amount](asset_dynamic_data_object& d) {
145  d.current_supply -= amount;
146  });
147  return;
148  }
149 
150  optional< vesting_balance_id_type > new_vbid = deposit_lazy_vesting(
151  acct.cashback_vb,
152  amount,
153  get_global_properties().parameters.cashback_vesting_period_seconds,
154  acct.id,
155  require_vesting );
156 
157  if( new_vbid.valid() )
158  {
159  modify( acct, [&new_vbid]( account_object& _acct )
160  {
161  _acct.cashback_vb = *new_vbid;
162  } );
163  modify( acct.statistics( *this ), []( account_statistics_object& aso )
164  {
165  aso.has_cashback_vb = true;
166  } );
167  }
168 
169  return;
170 }
