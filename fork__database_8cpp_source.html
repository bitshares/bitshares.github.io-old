74 void fork_database::_push_block(const item_ptr& item)
75 {
76  if( _head ) // make sure the block is within the range that we are caching
77  {
78  FC_ASSERT( item->num > std::max<int64_t>( 0, int64_t(_head->num) - (_max_size) ),
79  "attempting to push a block that is too old",
80  ("item->num",item->num)("head",_head->num)("max_size",_max_size));
81  }
82 
83  if( _head && item->previous_id() != block_id_type() )
84  {
85  auto& index = _index.get<block_id>();
86  auto itr = index.find(item->previous_id());
87  GRAPHENE_ASSERT(itr != index.end(), unlinkable_block_exception, "block does not link to known chain");
88  item->prev = *itr;
89  }
90 
91  _index.insert(item);
92  if( !_head ) _head = item;
93  else if( item->num > _head->num )
94  {
95  _head = item;
96  uint32_t min_num = _head->num - std::min( _max_size, _head->num );
97 // ilog( "min block in fork DB ${n}, max_size: ${m}", ("n",min_num)("m",_max_size) );
98  auto& num_idx = _index.get<block_num>();
99  while( num_idx.size() && (*num_idx.begin())->num < min_num )
100  num_idx.erase( num_idx.begin() );
101 
102  _unlinked_index.get<block_num>().erase(_head->num - _max_size);
103  }
104  //_push_next( item );
105 }

113 void fork_database::_push_next( const item_ptr& new_item )
114 {
115  auto& prev_idx = _unlinked_index.get<by_previous>();
116 
117  auto itr = prev_idx.find( new_item->id );
118  while( itr != prev_idx.end() )
119  {
120  auto tmp = *itr;
121  prev_idx.erase( itr );
122  _push_block( tmp );
123 
124  itr = prev_idx.find( new_item->id );
125  }

128 void fork_database::set_max_size( uint32_t s )
129 {
130  _max_size = s;
131  if( !_head ) return;
132 
133  {
134  auto& by_num_idx = _index.get<block_num>();
135  auto itr = by_num_idx.begin();
136  while( itr != by_num_idx.end() )
137  {
138  if( (*itr)->num < std::max(int64_t(0),int64_t(_head->num) - _max_size) )
139  by_num_idx.erase(itr);
140  else
141  break;
142  itr = by_num_idx.begin();
143  }
144  }
145  {
146  auto& by_num_idx = _unlinked_index.get<block_num>();

198 pair<fork_database::branch_type,fork_database::branch_type>
199  fork_database::fetch_branch_from(block_id_type first, block_id_type second)const
200 { try {
201  // This function gets a branch (i.e. vector<fork_item>) leading
202  // back to the most recent common ancestor.
203  pair<branch_type,branch_type> result;
204  auto first_branch_itr = _index.get<block_id>().find(first);
205  FC_ASSERT(first_branch_itr != _index.get<block_id>().end());
206  auto first_branch = *first_branch_itr;
207 
208  auto second_branch_itr = _index.get<block_id>().find(second);
209  FC_ASSERT(second_branch_itr != _index.get<block_id>().end());
210  auto second_branch = *second_branch_itr;
211 
212 
213  while( first_branch->data.block_num() > second_branch->data.block_num() )
214  {
215  result.first.push_back(first_branch);
216  first_branch = first_branch->prev.lock();
217  FC_ASSERT(first_branch);
218  }
219  while( second_branch->data.block_num() > first_branch->data.block_num() )
220  {
221  result.second.push_back( second_branch );
222  second_branch = second_branch->prev.lock();
223  FC_ASSERT(second_branch);
224  }
225  while( first_branch->data.previous != second_branch->data.previous )
226  {
227  result.first.push_back(first_branch);
228  result.second.push_back(second_branch);
229  first_branch = first_branch->prev.lock();
230  FC_ASSERT(first_branch);
231  second_branch = second_branch->prev.lock();
232  FC_ASSERT(second_branch);
233  }
234  if( first_branch && second_branch )
235  {
236  result.first.push_back(first_branch);
237  result.second.push_back(second_branch);
238  }
239  return result;
240 } FC_CAPTURE_AND_RETHROW( (first)(second) ) }
