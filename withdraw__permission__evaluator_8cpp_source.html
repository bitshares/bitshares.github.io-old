59 void_result withdraw_permission_claim_evaluator::do_evaluate(const withdraw_permission_claim_evaluator::operation_type& op)
60 { try {
61  const database& d = db();
62  time_point_sec head_block_time = d.head_block_time();
63 
64  const withdraw_permission_object& permit = op.withdraw_permission(d);
65  FC_ASSERT(permit.expiration > head_block_time);
66  FC_ASSERT(permit.authorized_account == op.withdraw_to_account);
67  FC_ASSERT(permit.withdraw_from_account == op.withdraw_from_account);
68  if (head_block_time >= HARDFORK_23_TIME) {
69  FC_ASSERT(permit.period_start_time <= head_block_time);
70  }
71  FC_ASSERT(op.amount_to_withdraw <= permit.available_this_period( head_block_time ) );
72  FC_ASSERT(d.get_balance(op.withdraw_from_account, op.amount_to_withdraw.asset_id) >= op.amount_to_withdraw);
73 
74  const asset_object& _asset = op.amount_to_withdraw.asset_id(d);
75  if( _asset.is_transfer_restricted() )
76  {
77  FC_ASSERT( _asset.issuer == permit.authorized_account || _asset.issuer == permit.withdraw_from_account,
78  "Asset ${a} '${sym}' has transfer_restricted flag enabled",
79  ("a", _asset.id)("sym", _asset.symbol) );
80  }
81 
82  const account_object& to = permit.authorized_account(d);
83  FC_ASSERT( is_authorized_asset( d, to, _asset ),
84  "Account ${acct} '${name}' is unauthorized to transact asset ${a} '${sym}' due to whitelist / blacklist",
85  ("acct", to.id)("name", to.name)("a", _asset.id)("sym", _asset.symbol) );
86 
87  const account_object& from = op.withdraw_from_account(d);
88  bool from_is_authorized = ( is_authorized_asset( d, from, _asset ) );
89  if( head_block_time > HARDFORK_CORE_942_TIME ) // TODO remove this check after hard fork if things in `else` did not occur
90  {
91  FC_ASSERT( from_is_authorized,
92  "Account ${acct} '${name}' is unauthorized to withdraw asset ${a} '${sym}' due to whitelist / blacklist",
93  ("acct", from.id)("name", from.name)("a", _asset.id)("sym", _asset.symbol) );
94  }
95  else
96  {
97  if( !from_is_authorized )
98  wlog( "Unauthorized asset withdrawal (issue #942) occurred at block ${b}", ("b", d.head_block_num()) );
99  }
100 
101  return void_result();
102 } FC_CAPTURE_AND_RETHROW( (op) ) }

104 void_result withdraw_permission_claim_evaluator::do_apply(const withdraw_permission_claim_evaluator::operation_type& op)
105 { try {
106  database& d = db();
107 
108  const withdraw_permission_object& permit = d.get(op.withdraw_permission);
109  d.modify(permit, [&](withdraw_permission_object& p) {
110  auto periods = (d.head_block_time() - p.period_start_time).to_seconds() / p.withdrawal_period_sec;
111  p.period_start_time += periods * p.withdrawal_period_sec;
112  if( periods == 0 )
113  p.claimed_this_period += op.amount_to_withdraw.amount;
114  else
115  p.claimed_this_period = op.amount_to_withdraw.amount;
116  });
117 
118  d.adjust_balance(op.withdraw_from_account, -op.amount_to_withdraw);
119  d.adjust_balance(op.withdraw_to_account, op.amount_to_withdraw);
120 
121  return void_result();
122 } FC_CAPTURE_AND_RETHROW( (op) ) }
