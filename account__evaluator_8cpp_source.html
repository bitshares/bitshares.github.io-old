60 void verify_account_votes( const database& db, const account_options& options )
61 {
62  // ensure account's votes satisfy requirements
63  // NB only the part of vote checking that requires chain state is here,
64  // the rest occurs in account_options::validate()
65 
66  const auto& gpo = db.get_global_properties();
67  const auto& chain_params = gpo.parameters;
68 
69  FC_ASSERT( options.num_witness <= chain_params.maximum_witness_count,
70  "Voted for more witnesses than currently allowed (${c})", ("c", chain_params.maximum_witness_count) );
71  FC_ASSERT( options.num_committee <= chain_params.maximum_committee_count,
72  "Voted for more committee members than currently allowed (${c})", ("c", chain_params.maximum_committee_count) );
73 
74  FC_ASSERT( db.find_object(options.voting_account), "Invalid proxy account specified." );
75 
76  uint32_t max_vote_id = gpo.next_available_vote_id;
77  bool has_worker_votes = false;
78  for( auto id : options.votes )
79  {
80  FC_ASSERT( id < max_vote_id, "Can not vote for ${id} which does not exist.", ("id",id) );
81  has_worker_votes |= (id.type() == vote_id_type::worker);
82  }
83 
84  if( has_worker_votes && (db.head_block_time() >= HARDFORK_607_TIME) )
85  {
86  const auto& against_worker_idx = db.get_index_type<worker_index>().indices().get<by_vote_against>();
87  for( auto id : options.votes )
88  {
89  if( id.type() == vote_id_type::worker )
90  {
91  FC_ASSERT( against_worker_idx.find( id ) == against_worker_idx.end(),
92  "Can no longer vote against a worker." );
93  }
94  }
95  }
96  if ( db.head_block_time() >= HARDFORK_CORE_143_TIME ) {
97  const auto& approve_worker_idx = db.get_index_type<worker_index>().indices().get<by_vote_for>();
98  const auto& committee_idx = db.get_index_type<committee_member_index>().indices().get<by_vote_id>();
99  const auto& witness_idx = db.get_index_type<witness_index>().indices().get<by_vote_id>();
100  for ( auto id : options.votes ) {
101  switch ( id.type() ) {
102  case vote_id_type::committee:
103  FC_ASSERT( committee_idx.find(id) != committee_idx.end(),
104  "Can not vote for ${id} which does not exist.", ("id",id) );
105  break;
106  case vote_id_type::witness:
107  FC_ASSERT( witness_idx.find(id) != witness_idx.end(),
108  "Can not vote for ${id} which does not exist.", ("id",id) );
109  break;
110  case vote_id_type::worker:
111  FC_ASSERT( approve_worker_idx.find( id ) != approve_worker_idx.end(),
112  "Can not vote for ${id} which does not exist.", ("id",id) );
113  break;
114  default:
115  FC_THROW( "Invalid Vote Type: ${id}", ("id", id) );
116  break;
117  }
118  }
119  }

123 void_result account_create_evaluator::do_evaluate( const account_create_operation& op )
124 { try {
125  database& d = db();
126  if( d.head_block_time() < HARDFORK_516_TIME )
127  {
128  FC_ASSERT( !op.extensions.value.owner_special_authority.valid() );
129  FC_ASSERT( !op.extensions.value.active_special_authority.valid() );
130  }
131  if( d.head_block_time() < HARDFORK_599_TIME )
132  {
133  FC_ASSERT( !op.extensions.value.null_ext.valid() );
134  FC_ASSERT( !op.extensions.value.owner_special_authority.valid() );
135  FC_ASSERT( !op.extensions.value.active_special_authority.valid() );
136  FC_ASSERT( !op.extensions.value.buyback_options.valid() );
137  }
138 
139  FC_ASSERT( fee_paying_account->is_lifetime_member(), "Only Lifetime members may register an account." );
140  FC_ASSERT( op.referrer(d).is_member(d.head_block_time()), "The referrer must be either a lifetime or annual subscriber." );
141 
142  try
143  {
144  verify_authority_accounts( d, op.owner );
145  verify_authority_accounts( d, op.active );
146  }
147  GRAPHENE_RECODE_EXC( internal_verify_auth_max_auth_exceeded, account_create_max_auth_exceeded )
148  GRAPHENE_RECODE_EXC( internal_verify_auth_account_not_found, account_create_auth_account_not_found )
149 
150  if( op.extensions.value.owner_special_authority.valid() )
151  evaluate_special_authority( d, *op.extensions.value.owner_special_authority );
152  if( op.extensions.value.active_special_authority.valid() )
153  evaluate_special_authority( d, *op.extensions.value.active_special_authority );
154  if( op.extensions.value.buyback_options.valid() )
155  evaluate_buyback_account_options( d, *op.extensions.value.buyback_options );
156  verify_account_votes( d, op.options );
157 
158  auto& acnt_indx = d.get_index_type<account_index>();
159  if( op.name.size() )
160  {
161  auto current_account_itr = acnt_indx.indices().get<by_name>().find( op.name );
162  FC_ASSERT( current_account_itr == acnt_indx.indices().get<by_name>().end(),
163  "Account '${a}' already exists.", ("a",op.name) );
164  }
165 
166  return void_result();
167 } FC_CAPTURE_AND_RETHROW( (op) ) }

170 { try {
171 
172  database& d = db();
173  uint16_t referrer_percent = o.referrer_percent;
174  bool has_small_percent = (
175  (db().head_block_time() <= HARDFORK_453_TIME)
176  && (o.referrer != o.registrar )
177  && (o.referrer_percent != 0 )
178  && (o.referrer_percent <= 0x100)
179  );
180

182  {
183  if( referrer_percent >= 100 )
184  {
185  wlog( "between 100% and 0x100%: ${o}", ("o", o) );

189  referrer_percent = GRAPHENE_100_PERCENT;
190  }
191 
192  const auto& global_properties = d.get_global_properties();
193 
194  const auto& new_acnt_object = d.create<account_object>( [&o,&d,&global_properties,referrer_percent]( account_object& obj )
195  {
196  obj.registrar = o.registrar;
197  obj.referrer = o.referrer;
198  obj.lifetime_referrer = o.referrer(d).lifetime_referrer;
199 
200  const auto& params = global_properties.parameters;
201  obj.network_fee_percentage = params.network_percent_of_fee;
202  obj.lifetime_referrer_fee_percentage = params.lifetime_referrer_percent_of_fee;
203  obj.referrer_rewards_percentage = referrer_percent;
204 
205  obj.name = o.name;
206  obj.owner = o.owner;
207  obj.active = o.active;
208  obj.options = o.options;
209  obj.statistics = d.create<account_statistics_object>([&obj](account_statistics_object& s){
210  s.owner = obj.id;
211  s.name = obj.name;
212  s.is_voting = obj.options.is_voting();
213  }).id;
214 
215  if( o.extensions.value.owner_special_authority.valid() )
216  obj.owner_special_authority = *(o.extensions.value.owner_special_authority);
217  if( o.extensions.value.active_special_authority.valid() )
218  obj.active_special_authority = *(o.extensions.value.active_special_authority);
219  if( o.extensions.value.buyback_options.valid() )
220  {
221  obj.allowed_assets = o.extensions.value.buyback_options->markets;
222  obj.allowed_assets->emplace( o.extensions.value.buyback_options->asset_to_buy );
223  }
224  });
225 
226  /*
227  if( has_small_percent )
228  {
229  wlog( "Account affected by #453 registered in block ${n}: ${na} reg=${reg} ref=${ref}:${refp} ltr=${ltr}:${ltrp}",
230  ("n", db().head_block_num()) ("na", new_acnt_object.id)
231  ("reg", o.registrar) ("ref", o.referrer) ("ltr", new_acnt_object.lifetime_referrer)
232  ("refp", new_acnt_object.referrer_rewards_percentage) ("ltrp", new_acnt_object.lifetime_referrer_fee_percentage) );
233  wlog( "Affected account object is ${o}", ("o", new_acnt_object) );
234  }
235  */
236 
237  const auto& dynamic_properties = d.get_dynamic_global_properties();
238  d.modify(dynamic_properties, [](dynamic_global_property_object& p) {
239  ++p.accounts_registered_this_interval;
240  });
241 
242  if( dynamic_properties.accounts_registered_this_interval % global_properties.parameters.accounts_per_fee_scale == 0
243  && global_properties.parameters.account_fee_scale_bitshifts != 0 )
244  {
245  d.modify(global_properties, [](global_property_object& p) {
246  p.parameters.current_fees->get<account_create_operation>().basic_fee <<= p.parameters.account_fee_scale_bitshifts;
247  });
248  }
249 
250  if( o.extensions.value.owner_special_authority.valid()
251  || o.extensions.value.active_special_authority.valid() )
252  {
253  db().create< special_authority_object >( [&]( special_authority_object& sa )
254  {
255  sa.account = new_acnt_object.id;
256  } );
257  }
258 
259  if( o.extensions.value.buyback_options.valid() )

261  asset_id_type asset_to_buy = o.extensions.value.buyback_options->asset_to_buy;
262 
263  d.create< buyback_object >( [&]( buyback_object& bo )

269  {
270  a.buyback_account = new_acnt_object.id;
271  } );
272  }
273 
274  return new_acnt_object.id;
275 } FC_CAPTURE_AND_RETHROW((o)) }

314 void_result account_update_evaluator::do_apply( const account_update_operation& o )
315 { try {
316  database& d = db();
317 
318  bool sa_before = acnt->has_special_authority();
319 
320  // update account statistics
321  if( o.new_options.valid() && o.new_options->is_voting() != acnt->options.is_voting() )
322  {
323  d.modify( acnt->statistics( d ), []( account_statistics_object& aso )
324  {
325  aso.is_voting = !aso.is_voting;
326  } );
327  }
328 
329  // update account object
330  d.modify( *acnt, [&o](account_object& a){
331  if( o.owner )
332  {
333  a.owner = *o.owner;
334  a.top_n_control_flags = 0;
335  }
336  if( o.active )
337  {
338  a.active = *o.active;
339  a.top_n_control_flags = 0;
340  }
341  if( o.new_options ) a.options = *o.new_options;
342  if( o.extensions.value.owner_special_authority.valid() )
343  {
344  a.owner_special_authority = *(o.extensions.value.owner_special_authority);
345  a.top_n_control_flags = 0;
346  }
347  if( o.extensions.value.active_special_authority.valid() )
348  {
349  a.active_special_authority = *(o.extensions.value.active_special_authority);
350  a.top_n_control_flags = 0;
351  }
352  });
353 
354  bool sa_after = acnt->has_special_authority();
355 
356  if( sa_before && (!sa_after) )
357  {
358  const auto& sa_idx = d.get_index_type< special_authority_index >().indices().get<by_account>();
359  auto sa_it = sa_idx.find( o.account );
360  assert( sa_it != sa_idx.end() );
361  d.remove( *sa_it );
362  }
363  else if( (!sa_before) && sa_after )
364  {
365  d.create< special_authority_object >( [&]( special_authority_object& sa )
366  {
367  sa.account = o.account;
368  } );
369  }
370 
371  return void_result();
372 } FC_CAPTURE_AND_RETHROW( (o) ) }

428 void_result account_upgrade_evaluator::do_apply(const account_upgrade_evaluator::operation_type& o)
429 { try {
430  database& d = db();
431 
432  d.modify(*account, [&](account_object& a) {
433  if( o.upgrade_to_lifetime_member )
434  {
435  // Upgrade to lifetime member. I don't care what the account was before.
436  a.statistics(d).process_fees(a, d);
437  a.membership_expiration_date = time_point_sec::maximum();
438  a.referrer = a.registrar = a.lifetime_referrer = a.get_id();
439  a.lifetime_referrer_fee_percentage = GRAPHENE_100_PERCENT - a.network_fee_percentage;
440  } else if( a.is_annual_member(d.head_block_time()) ) {
441  // Renew an annual subscription that's still in effect.
442  FC_ASSERT( d.head_block_time() <= HARDFORK_613_TIME );
443  FC_ASSERT(a.membership_expiration_date - d.head_block_time() < fc::days(3650),
444  "May not extend annual membership more than a decade into the future.");
445  a.membership_expiration_date += fc::days(365);
446  } else {
447  // Upgrade from basic account.
448  FC_ASSERT( d.head_block_time() <= HARDFORK_613_TIME );
449  a.statistics(d).process_fees(a, d);
450  assert(a.is_basic_account(d.head_block_time()));
451  a.referrer = a.get_id();
452  a.membership_expiration_date = d.head_block_time() + fc::days(365);
453  }
454  });
455 
456  return {};
457 } FC_RETHROW_EXCEPTIONS( error, "Unable to upgrade account '${a}'", ("a",o.account_to_upgrade(db()).name) ) }
