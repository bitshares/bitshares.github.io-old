127 bool is_cheap_name( const string& n )
128 {
129  bool v = false;
130  for( auto c : n )
131  {
132  if( c >= '0' && c <= '9' ) return true;
133  if( c == '.' || c == '-' || c == '/' ) return true;
134  switch( c )
135  {
136  case 'a':
137  case 'e':
138  case 'i':
139  case 'o':
140  case 'u':
141  case 'y':
142  v = true;
143  }
144  }
145  if( !v )
146  return true;
147  return false;
148 }

180 void account_create_operation::validate()const
181 {
182  FC_ASSERT( fee.amount >= 0 );
183  FC_ASSERT( is_valid_name( name ) );
184  FC_ASSERT( referrer_percent <= GRAPHENE_100_PERCENT );
185  FC_ASSERT( owner.num_auths() != 0 );
186  FC_ASSERT( owner.address_auths.size() == 0 );
187  FC_ASSERT( active.num_auths() != 0 );
188  FC_ASSERT( active.address_auths.size() == 0 );
189  FC_ASSERT( !owner.is_impossible(), "cannot create an account with an imposible owner authority threshold" );
190  FC_ASSERT( !active.is_impossible(), "cannot create an account with an imposible active authority threshold" );
191  options.validate();
192  if( extensions.value.owner_special_authority.valid() )
193  validate_special_authority( *extensions.value.owner_special_authority );
194  if( extensions.value.active_special_authority.valid() )
195  validate_special_authority( *extensions.value.active_special_authority );
196  if( extensions.value.buyback_options.valid() )
197  {
198  FC_ASSERT( !(extensions.value.owner_special_authority.valid()) );
199  FC_ASSERT( !(extensions.value.active_special_authority.valid()) );
200  FC_ASSERT( owner == authority::null_authority() );
201  FC_ASSERT( active == authority::null_authority() );
202  size_t n_markets = extensions.value.buyback_options->markets.size();
203  FC_ASSERT( n_markets > 0 );
204  for( const asset_id_type m : extensions.value.buyback_options->markets )
205  {
206  FC_ASSERT( m != extensions.value.buyback_options->asset_to_buy );
207  }
208  }
209 }
