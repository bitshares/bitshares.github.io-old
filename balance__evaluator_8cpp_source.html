28 void_result balance_claim_evaluator::do_evaluate(const balance_claim_operation& op)
29 {
30  database& d = db();
31  balance = &op.balance_to_claim(d);
32 
33  GRAPHENE_ASSERT(
34  op.balance_owner_key == balance->owner ||
35  pts_address(op.balance_owner_key, false, 56) == balance->owner ||
36  pts_address(op.balance_owner_key, true, 56) == balance->owner ||
37  pts_address(op.balance_owner_key, false, 0) == balance->owner ||
38  pts_address(op.balance_owner_key, true, 0) == balance->owner,
39  balance_claim_owner_mismatch,
40  "Balance owner key was specified as '${op}' but balance's actual owner is '${bal}'",
41  ("op", op.balance_owner_key)
42  ("bal", balance->owner)
43  );
44 
45  FC_ASSERT(op.total_claimed.asset_id == balance->asset_type());
46 
47  if( balance->is_vesting_balance() )
48  {
49  GRAPHENE_ASSERT(
50  balance->vesting_policy->is_withdraw_allowed(
51  { balance->balance,
52  d.head_block_time(),
53  op.total_claimed } ),
54  balance_claim_invalid_claim_amount,
55  "Attempted to claim ${c} from a vesting balance with ${a} available",
56  ("c", op.total_claimed)("a", balance->available(d.head_block_time()))
57  );
58  GRAPHENE_ASSERT(
59  d.head_block_time() - balance->last_claim_date >= fc::days(1),
60  balance_claim_claimed_too_often,
61  "Genesis vesting balances may not be claimed more than once per day."
62  );
63  return {};
64  }
65 
66  FC_ASSERT(op.total_claimed == balance->balance);
67  return {};
68 }

74 void_result balance_claim_evaluator::do_apply(const balance_claim_operation& op)
75 {
76  database& d = db();
77 
78  if( balance->is_vesting_balance() && op.total_claimed < balance->balance )
79  d.modify(*balance, [&](balance_object& b) {
80  b.vesting_policy->on_withdraw({b.balance, d.head_block_time(), op.total_claimed});
81  b.balance -= op.total_claimed;
82  b.last_claim_date = d.head_block_time();
83  });
84  else
85  d.remove(*balance);
86 
87  d.adjust_balance(op.deposit_to_account, op.total_claimed);
88  return {};
89 }
